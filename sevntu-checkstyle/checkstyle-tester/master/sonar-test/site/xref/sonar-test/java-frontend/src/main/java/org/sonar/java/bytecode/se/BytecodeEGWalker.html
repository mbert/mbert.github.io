283   @CheckForNull
284   public MethodBehavior getMethodBehavior(String signature, SquidClassLoader classLoader) {
285     if (methodFromArray(signature)) {
286       // should not generate any method behavior
287       return null;
288     }
289     methodBehavior = behaviorCache.methodBehaviorForSymbol(signature);
290     if (!methodBehavior.isVisited()) {
291       try {
292         methodBehavior.visited();
293         execute(signature, classLoader);
294       } catch (ExplodedGraphWalker.MaximumStepsReachedException
295         | RelationalSymbolicValue.TransitiveRelationExceededException
296         | BytecodeAnalysisException e) {
297         LOG.debug("Dataflow analysis is incomplete for method {} : {}", signature, e.getMessage());
298       } catch (Exception e) {
299         throw new BytecodeAnalysisException("Failed dataflow analysis for " + signature, e);
300       }
301     }
302     return methodBehavior;
303   }
304 
305   private static boolean methodFromArray(String signature) {
306     return signature.substring(0, signature.indexOf('#')).endsWith("[]");
307   }

309   @VisibleForTesting
310   int maxSteps() {
311     return MAX_STEPS;
312   }
313 
314   private void execute(String signature, SquidClassLoader classLoader) {
315     BytecodeCFGMethodVisitor cfgVisitor = new BytecodeCFGMethodVisitor();
316     MethodLookup lookup = MethodLookup.lookup(signature, classLoader, cfgVisitor);
317     if (lookup == null) {
318       LOG.debug("Method body not found: {}", signature);
319       return;
320     }
321     methodBehavior.setDeclaredExceptions(lookup.declaredExceptions);
322     methodBehavior.setVarArgs(lookup.isVarArgs);
323     BytecodeCFG bytecodeCFG = cfgVisitor.getCfg();
324     if (bytecodeCFG == null) {
325       return;
326     }
327     exitBlock = bytecodeCFG.exitBlock();
328     steps = 0;
329     for (ProgramState startingState : startingStates(signature, ProgramState.EMPTY_STATE, lookup.isStatic)) {
330       enqueue(new ProgramPoint(bytecodeCFG.entry()), startingState);
331     }
332     while (!workList.isEmpty()) {
333       steps++;
334       if (steps > maxSteps()) {
335         throw new ExplodedGraphWalker.MaximumStepsReachedException("Too many steps resolving "+methodBehavior.signature());
336       }
337       // LIFO:
338       setNode(workList.removeFirst());
339       if (programPosition.block.successors().isEmpty()) {
340         endOfExecutionPath.add(node);
341         continue;
342       }
343 
344       if (programPosition.i < programPosition.block.elements().size()) {
345         // process block element
346         executeInstruction((Instruction) programPosition.block.elements().get(programPosition.i));
347       } else {
348         // process block exit, which is unconditional jump such as goto-statement or return-statement
349         handleBlockExit(programPosition);
350       }
351     }
352 
353     handleEndOfExecutionPath();
354     executeCheckEndOfExecution();
355     methodBehavior.completed();
356     // Cleanup:
357     workList = null;
358     node = null;
359     programState = null;
360     constraintManager = null;
361   }

363   @VisibleForTesting
364   void executeInstruction(Instruction instruction) {
365     if(!checkerDispatcher.executeCheckPreStatement(instruction)) {
366       return;
367     }
368     ProgramState.Pop pop;
369     SymbolicValue sv;
370     switch (instruction.opcode) {
371       case NOP:
372         break;
373       case ACONST_NULL:
374         programState = programState.stackValue(SymbolicValue.NULL_LITERAL);
375         break;
376       case ICONST_M1:
377       case ICONST_0:
378       case ICONST_1:
379       case ICONST_2:
380       case ICONST_3:
381       case ICONST_4:
382       case ICONST_5:
383       case LCONST_0:
384       case LCONST_1:
385       case FCONST_0:
386       case FCONST_1:
387       case FCONST_2:
388       case DCONST_0:
389       case DCONST_1:
390         sv = constraintManager.createSymbolicValue(instruction);
391         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
392         programState = programState.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
393         if (instruction.opcode == ICONST_1 || instruction.opcode == LCONST_1 || instruction.opcode == FCONST_1 || instruction.opcode == DCONST_1) {
394           programState = programState.addConstraint(sv, BooleanConstraint.TRUE);
395         }
396         if (instruction.opcode == ICONST_0 || instruction.opcode == LCONST_0 || instruction.opcode == FCONST_0  || instruction.opcode == DCONST_0  ) {
397           programState = programState.addConstraint(sv, BooleanConstraint.FALSE).addConstraint(sv, DivisionByZeroCheck.ZeroConstraint.ZERO);
398         } else {
399           programState = programState.addConstraint(sv, DivisionByZeroCheck.ZeroConstraint.NON_ZERO);
400         }
401         break;
402       case BIPUSH:
403       case SIPUSH:
404         sv = constraintManager.createSymbolicValue(instruction);
405         programState = programState.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL).addConstraint(sv,DivisionByZeroCheck.ZeroConstraint.NON_ZERO);
406         if(instruction.operand == 0) {
407           programState = programState.addConstraint(sv, BooleanConstraint.FALSE).addConstraint(sv, DivisionByZeroCheck.ZeroConstraint.ZERO);
408         } else if (instruction.operand == 1) {
409           programState = programState.addConstraint(sv, BooleanConstraint.TRUE);
410         }
411         break;
412       case LDC:
413         sv = constraintManager.createSymbolicValue(instruction);
414         programState = programState.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
415         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
416         break;
417       case ILOAD:
418       case LLOAD:
419       case FLOAD:
420       case DLOAD:
421       case ALOAD:
422         SymbolicValue value = programState.getValue(instruction.operand);
423         Preconditions.checkNotNull(value, "Loading a symbolic value unindexed");
424         programState = programState.stackValue(value);
425         break;
426       case IALOAD:
427       case LALOAD:
428       case FALOAD:
429       case DALOAD:
430       case AALOAD:
431       case BALOAD:
432       case CALOAD:
433       case SALOAD:
434         sv = constraintManager.createSymbolicValue(instruction);
435         programState = programState.unstackValue(2).state.stackValue(sv);
436         if (instruction.opcode != AALOAD) {
437           programState = programState.addConstraint(sv, ObjectConstraint.NOT_NULL);
438         }
439         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
440         break;
441       case ISTORE:
442       case LSTORE:
443       case FSTORE:
444       case DSTORE:
445       case ASTORE:
446         pop = popStack(1, instruction.opcode);
447         programState = pop.state.put(instruction.operand, pop.values.get(0));
448         break;
449       case IASTORE:
450       case LASTORE:
451       case FASTORE:
452       case DASTORE:
453       case AASTORE:
454       case BASTORE:
455       case CASTORE:
456       case SASTORE:
457         programState = programState.unstackValue(3).state;
458         break;
459       case POP:
460         programState = programState.unstackValue(1).state;
461         break;
462       case POP2:
463         sv = programState.peekValue();
464         Preconditions.checkNotNull(sv, "POP2 on empty stack");
465         pop = isDoubleOrLong(sv) ? popStack(1, instruction.opcode) : popStack(2, instruction.opcode);
466         programState = pop.state;
467         break;
468       case DUP:
469         sv = programState.peekValue();
470         Preconditions.checkNotNull(sv, "DUP on empty stack");
471         programState = programState.stackValue(sv);
472         break;
473       case DUP_X1:
474         pop = popStack(2, instruction.opcode);
475         programState = stackValues(pop, 0, 1, 0);
476         break;
477       case DUP_X2:
478         sv = programState.peekValue(1);
479         if (isDoubleOrLong(sv)) {
480           pop = popStack(2, instruction.opcode);
481           programState = stackValues(pop, 0, 1, 0);
482         } else {
483           pop = popStack(3, instruction.opcode);
484           programState = stackValues(pop, 0, 2, 1, 0);
485         }
486         break;
487       case DUP2:
488         sv = programState.peekValue();
489         Preconditions.checkNotNull(sv, "DUP2 needs at least 1 value on stack");
490         if (isDoubleOrLong(sv)) {
491           pop = popStack(1, instruction.opcode);
492           programState = stackValues(pop, 0, 0);
493         } else {
494           pop = popStack(2, instruction.opcode);
495           programState = stackValues(pop, 1, 0, 1, 0);
496         }
497         break;
498       case DUP2_X1:
499         sv = programState.peekValue();
500         Preconditions.checkNotNull(sv, "DUP2_X1 needs at least 1 value on stack");
501         if (isDoubleOrLong(sv)) {
502           pop = popStack(2, instruction.opcode);
503           programState = stackValues(pop, 0, 1, 0);
504         } else {
505           pop = popStack(3, instruction.opcode);
506           programState = stackValues(pop, 1, 0, 2, 1, 0);
507         }
508         break;
509       case DUP2_X2:
510         if (isDoubleOrLong(programState.peekValue()) && isDoubleOrLong(programState.peekValue(1))) {
511           pop = popStack(2, instruction.opcode);
512           programState = stackValues(pop, 0, 1, 0);
513         } else if (isDoubleOrLong(programState.peekValue(2))) {
514           pop = popStack(3, instruction.opcode);
515           programState = stackValues(pop, 1, 0, 2, 1, 0);
516         } else if (isDoubleOrLong(programState.peekValue())) {
517           pop = popStack(3, instruction.opcode);
518           programState = stackValues(pop, 0, 2, 1, 0);
519         } else {
520           pop = popStack(4, instruction.opcode);
521           programState = stackValues(pop, 1, 0, 3, 2, 1, 0);
522         }
523         break;
524       case SWAP:
525         pop = popStack(2, instruction.opcode);
526         programState = pop.state.stackValue(pop.values.get(0)).stackValue(pop.values.get(1));
527         break;
528       case IADD:
529       case LADD:
530       case FADD:
531       case DADD:
532       case ISUB:
533       case LSUB:
534       case FSUB:
535       case DSUB:
536       case IMUL:
537       case LMUL:
538       case FMUL:
539       case DMUL:
540       case IDIV:
541       case LDIV:
542       case FDIV:
543       case DDIV:
544       case IREM:
545       case LREM:
546       case FREM:
547       case DREM:
548       case ISHL:
549       case LSHL:
550       case ISHR:
551       case LSHR:
552       case IUSHR:
553       case LUSHR:
554         pop = popStack(2, instruction.opcode);
555         sv = constraintManager.createSymbolicValue(instruction);
556         programState = pop.state.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
557         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
558         break;
559       case INEG:
560       case LNEG:
561       case FNEG:
562       case DNEG:
563         pop = popStack(1, instruction.opcode);
564         sv = constraintManager.createSymbolicValue(instruction);
565         programState = pop.state.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
566         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
567         break;
568       case IAND:
569       case LAND:
570       case IOR:
571       case LOR:
572       case IXOR:
573       case LXOR:
574         pop = popStack(2, instruction.opcode);
575         sv = constraintManager.createBinarySymbolicValue(instruction, pop.valuesAndSymbols);
576         programState = pop.state.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
577         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
578         break;
579       case IINC:
580         int index = instruction.operand;
581         SymbolicValue existing = programState.getValue(index);
582         Preconditions.checkNotNull(existing, "Local variable " + index + " not found");
583         sv = constraintManager.createSymbolicValue(instruction);
584         programState = programState.put(index, sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
585         break;
586       case I2L:
587       case I2D:
588       case F2L:
589       case F2D:
590         sv = programState.peekValue();
591         Preconditions.checkNotNull(sv, "%s needs value on stack", instruction);
592         programState = setDoubleOrLong(sv, true);
593         break;
594       case L2I:
595       case L2F:
596       case D2I:
597       case D2F:
598         sv = programState.peekValue();
599         Preconditions.checkNotNull(sv, "%s needs value on stack", instruction);
600         programState = setDoubleOrLong(sv, false);
601         break;
602       case D2L:
603       case I2F:
604       case L2D:
605       case F2I:
606       case I2B:
607       case I2C:
608       case I2S:
609         break;
610       case LCMP:
611       case FCMPL:
612       case FCMPG:
613       case DCMPL:
614       case DCMPG:
615         pop = popStack(2, instruction.opcode);
616         sv = constraintManager.createSymbolicValue(instruction);
617         programState = pop.state.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
618         break;
619       case IRETURN:
620       case LRETURN:
621       case FRETURN:
622       case DRETURN:
623       case ARETURN:
624         programState.storeExitValue();
625         programState = programState.unstackValue(1).state;
626         break;
627       case RETURN:
628         // do nothing
629         break;
630       case GETSTATIC:
631         // TODO SONARJAVA-2510 associated symbolic value with symbol
632         sv = constraintManager.createSymbolicValue(instruction);
633         programState = programState.stackValue(sv);
634         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
635         break;
636       case PUTSTATIC:
637         pop = programState.unstackValue(1);
638         programState = pop.state;
639         break;
640       case GETFIELD:
641         pop = popStack(1, instruction.opcode);
642         sv = constraintManager.createSymbolicValue(instruction);
643         programState = pop.state.stackValue(sv);
644         programState = setDoubleOrLong(sv, instruction.isLongOrDoubleValue());
645         break;
646       case PUTFIELD:
647         pop = popStack(2, instruction.opcode);
648         programState = pop.state;
649         break;
650       case INVOKEVIRTUAL:
651       case INVOKESPECIAL:
652       case INVOKESTATIC:
653       case INVOKEINTERFACE:
654         if (handleMethodInvocation(instruction)) {
655           // when yields are available, do not execute post check on this node
656           return;
657         }
658         break;
659       case INVOKEDYNAMIC:
660         pop = popStack(instruction.arity(), instruction.opcode);
661         Preconditions.checkState(instruction.hasReturnValue(), "Lambda should always evaluate to target functional interface");
662         SymbolicValue lambdaTargetInterface = new SymbolicValue();
663         programState = pop.state.stackValue(lambdaTargetInterface).addConstraint(lambdaTargetInterface, ObjectConstraint.NOT_NULL);
664         break;
665       case NEW:
666         sv = constraintManager.createSymbolicValue(instruction);
667         programState = programState
668             .stackValue(sv)
669             .addConstraint(sv, ObjectConstraint.NOT_NULL)
670             .addConstraint(sv, new TypedConstraint(instruction.className));
671         break;
672       case ARRAYLENGTH:
673         pop = popStack(1, instruction.opcode);
674         sv = constraintManager.createSymbolicValue(instruction);
675         programState = pop.state.stackValue(sv);
676         break;
677       case NEWARRAY:
678       case ANEWARRAY:
679         pop = popStack(1, instruction.opcode);
680         sv = constraintManager.createSymbolicValue(instruction);
681         programState = pop.state.stackValue(sv).addConstraint(sv, ObjectConstraint.NOT_NULL);
682         break;
683       case ATHROW:
684         if (!(programState.peekValue() instanceof SymbolicValue.ExceptionalSymbolicValue)) {
685           // create exceptional SV if not already on top of the stack (e.g. throw new MyException(); )
686           pop = popStack(1, instruction.opcode);
687           sv = pop.values.get(0);
688           TypedConstraint typedConstraint = programState.getConstraint(sv, TypedConstraint.class);
689           Type type = typedConstraint != null ? typedConstraint.getType(semanticModel) : Symbols.unknownType;
690           programState = pop.state.stackValue(constraintManager.createExceptionalSymbolicValue(type));
691         }
692         programState.storeExitValue();
693         break;
694       case CHECKCAST:
695         Preconditions.checkState(programState.peekValue() != null, "CHECKCAST needs 1 value on stack");
696         break;
697       case INSTANCEOF:
698         pop = popStack(1, instruction.opcode);
699         SymbolicValue.InstanceOfSymbolicValue instanceOf = new SymbolicValue.InstanceOfSymbolicValue();
700         instanceOf.computedFrom(pop.valuesAndSymbols);
701         programState = pop.state.stackValue(instanceOf);
702         break;
703       case MONITORENTER:
704       case MONITOREXIT:
705         pop = popStack(1, instruction.opcode);
706         programState = pop.state;
707         break;
708       case MULTIANEWARRAY:
709         Instruction.MultiANewArrayInsn multiANewArrayInsn = (Instruction.MultiANewArrayInsn) instruction;
710         pop = popStack(multiANewArrayInsn.dim, instruction.opcode);
711         SymbolicValue arrayRef = new SymbolicValue();
712         programState = pop.state.stackValue(arrayRef).addConstraint(arrayRef, ObjectConstraint.NOT_NULL);
713         break;
714       default:
715         throw new IllegalStateException("Instruction not handled. " + instruction);
716     }
717     checkerDispatcher.executeCheckPostStatement(instruction);
718   }

750   private boolean handleMethodInvocation(Instruction instruction) {
751     boolean isStatic = instruction.opcode == Opcodes.INVOKESTATIC;
752     int arity = isStatic ? instruction.arity() : (instruction.arity() + 1);
753     ProgramState.Pop pop = programState.unstackValue(arity);
754     Preconditions.checkState(pop.values.size() == arity, "Arguments mismatch for INVOKE");
755     // TODO use constraintManager.createMethodSymbolicValue to create relational SV for equals
756     programState = pop.state;
757     SymbolicValue returnSV = instruction.hasReturnValue() ? constraintManager.createSymbolicValue(instruction) : null;
758     String signature = instruction.fieldOrMethod.completeSignature();
759     MethodBehavior methodInvokedBehavior = behaviorCache.get(signature);
760     enqueueUncheckedExceptions();
761     // FIXME : empty yields here should not happen, for now act as if behavior was not resolved.
762     if (methodInvokedBehavior != null && methodInvokedBehavior.isComplete() && !methodInvokedBehavior.yields().isEmpty()) {
763       List<SymbolicValue> stack = Lists.reverse(pop.values);
764       if (!isStatic) {
765         // remove "thisSV" from stack before trying to apply any yield, as it should not match with arguments
766         stack = stack.subList(1, stack.size());
767       }
768       List<SymbolicValue> arguments = stack;
769 
770       methodInvokedBehavior
771         .happyPathYields()
772         .forEach(yield -> yield.statesAfterInvocation(arguments, Collections.emptyList(), programState, () -> returnSV).forEach(ps -> {
773           checkerDispatcher.methodYield = yield;
774           if (ps.peekValue() != null) {
775             ps = setDoubleOrLong(ps, ps.peekValue(), instruction.isLongOrDoubleValue());
776           }
777           checkerDispatcher.addTransition(ps);
778           checkerDispatcher.methodYield = null;
779         }));
780       methodInvokedBehavior
781         .exceptionalPathYields()
782         .forEach(yield -> {
783           Type exceptionType = yield.exceptionType(semanticModel);
784           yield.statesAfterInvocation(
785             arguments, Collections.emptyList(), programState, () -> constraintManager.createExceptionalSymbolicValue(exceptionType)).forEach(ps -> {
786               ps.storeExitValue();
787               enqueueExceptionHandlers(exceptionType, ps);
788           });
789         });
790       return true;
791     }
792     if (methodInvokedBehavior != null) {
793       methodInvokedBehavior.getDeclaredExceptions().forEach(exception -> {
794         Type exceptionType = semanticModel.getClassType(exception);
795         ProgramState ps = programState.stackValue(constraintManager.createExceptionalSymbolicValue(exceptionType));
796         enqueueExceptionHandlers(exceptionType, ps);
797       });
798     }
799     if (instruction.hasReturnValue()) {
800       programState = programState.stackValue(returnSV);
801       programState = setDoubleOrLong(returnSV, instruction.isLongOrDoubleValue());
802     }
803     return false;
804   }

806   private void enqueueUncheckedExceptions() {
807     programPosition.block.successors()
808         .stream()
809         .map(BytecodeCFG.Block.class::cast)
810         .filter(this::isUncheckedExceptionCatchBlock)
811         .forEach(b -> enqueue(new ProgramPoint(b), stateWithException(programState, b)));
812   }

821     return programState.stackValue(sv);

824   private void enqueueExceptionHandlers(Type exceptionType, ProgramState ps) {
825     List<BytecodeCFG.Block> blocksCatchingException = programPosition.block.successors().stream()
826         .map(b -> (BytecodeCFG.Block) b)
827         .filter(BytecodeCFG.Block::isCatchBlock)
828         .filter(b -> isExceptionHandledByBlock(exceptionType, b))
829         .collect(Collectors.toList());
830     if (!blocksCatchingException.isEmpty()) {
831       blocksCatchingException.forEach(b -> enqueue(new ProgramPoint(b), ps));
832       if (isCatchExhaustive(exceptionType, blocksCatchingException)) {
833         return;
834       }
835     }
836     // exception was not handled or was handled only partially, enqueue exit block with exceptional SV
837     Preconditions.checkState(ps.peekValue() instanceof SymbolicValue.ExceptionalSymbolicValue,
838         "Exception shall be on top of the stack");
839     ps.storeExitValue();
840     enqueue(new ProgramPoint(exitBlock), ps);
841   }

843   private boolean isCatchExhaustive(Type exceptionType, List<BytecodeCFG.Block> blocksCatchingException) {
844     return blocksCatchingException.stream()
845         .filter(BytecodeCFG.Block::isCatchBlock)
846         .anyMatch(b -> b.isUncaughtException() || exceptionType.isSubtypeOf(b.getExceptionType(semanticModel)));
847   }

849   private boolean isExceptionHandledByBlock(Type exceptionType, BytecodeCFG.Block b) {
850     Type blockException = b.getExceptionType(semanticModel);
851     return b.isUncaughtException()
852       ||/*required as long as there is no real type tracking*/ exceptionType == null
853       || exceptionType.isSubtypeOf(blockException)
854       || blockException.isSubtypeOf(exceptionType);
855   }

857   @VisibleForTesting
858   void handleBlockExit(ProgramPoint programPosition) {
859     BytecodeCFG.Block block = (BytecodeCFG.Block) programPosition.block;
860     Instruction terminator = block.terminator();
861     if (terminator == null) {
862       enqueueHappyPath(programPosition);
863       return;
864     }
865     switch (terminator.opcode) {
866       case GOTO:
867         enqueueHappyPath(programPosition);
868         break;
869       case TABLESWITCH:
870       case LOOKUPSWITCH:
871         programState = programState.unstackValue(1).state;
872         enqueueHappyPath(programPosition);
873         break;
874       default:
875         handleBranching(terminator);
876     }
877   }

879   private void handleBranching(Instruction terminator) {
880     programState = branchingState(terminator, programState);
881     Pair<List<ProgramState>, List<ProgramState>> pair = constraintManager.assumeDual(programState);
882     ProgramPoint falsePP = new ProgramPoint(((BytecodeCFG.Block) programPosition.block).falseSuccessor());
883     ProgramPoint truePP = new ProgramPoint(((BytecodeCFG.Block) programPosition.block).trueSuccessor());
884     pair.a.forEach(s -> enqueue(falsePP, s));
885     pair.b.forEach(s -> enqueue(truePP, s));
886   }

888   @VisibleForTesting
889   ProgramState branchingState(Instruction terminator, ProgramState programState) {
890     ProgramState.Pop pop;
891     ProgramState ps;
892     List<ProgramState.SymbolicValueSymbol> symbolicValueSymbols;
893     switch (terminator.opcode) {
894       case IFEQ:
895       case IFNE:
896       case IFLT:
897       case IFGE:
898       case IFGT:
899       case IFLE:
900         pop = programState.unstackValue(1);
901         SymbolicValue svZero = new SymbolicValue();
902         symbolicValueSymbols = ImmutableList.of(
903             new ProgramState.SymbolicValueSymbol(svZero, null),
904             pop.valuesAndSymbols.get(0));
905         List<ProgramState> programStates = svZero.setConstraint(pop.state, DivisionByZeroCheck.ZeroConstraint.ZERO).stream()
906             .flatMap(s -> svZero.setConstraint(s, BooleanConstraint.FALSE).stream()).collect(Collectors.toList());
907         Preconditions.checkState(programStates.size() == 1);
908         ps = programStates.get(0);
909         break;
910       case IF_ICMPEQ:
911       case IF_ICMPNE:
912       case IF_ICMPLT:
913       case IF_ICMPGE:
914       case IF_ICMPGT:
915       case IF_ICMPLE:
916       case IF_ACMPEQ:
917       case IF_ACMPNE:
918         pop = programState.unstackValue(2);
919         symbolicValueSymbols = pop.valuesAndSymbols;
920         ps = pop.state;
921         break;
922       case IFNULL:
923       case IFNONNULL:
924         pop = programState.unstackValue(1);
925         symbolicValueSymbols = ImmutableList.of(
926             new ProgramState.SymbolicValueSymbol(SymbolicValue.NULL_LITERAL, null),
927             pop.valuesAndSymbols.get(0));
928         ps = pop.state;
929         break;
930       default:
931         throw new IllegalStateException("Unexpected terminator " + terminator);
932     }
933     return ps.stackValue(constraintManager.createBinarySymbolicValue(terminator, symbolicValueSymbols));
934   }

947   @VisibleForTesting
948   Iterable<ProgramState> startingStates(String signature, ProgramState currentState, boolean isStaticMethod) {
949     // TODO : deal with parameter annotations, equals methods etc.
950     int parameterIdx = 0;
951     ProgramState state = currentState;
952     if(!isStaticMethod) {
953       // Add a sv for "this"
954       SymbolicValue thisSV = constraintManager.createSymbolicValue((Instruction) null);
955       state = currentState.addConstraint(thisSV, ObjectConstraint.NOT_NULL)
956         .addConstraint(thisSV, new TypedConstraint(signature.substring(0, signature.indexOf('#')))).put(0, thisSV);
957       parameterIdx = 1;
958     }
959     org.objectweb.asm.Type[] argumentTypes = org.objectweb.asm.Type.getArgumentTypes(signature.substring(signature.indexOf('(')));
960     for (org.objectweb.asm.Type argumentType: argumentTypes) {
961       SymbolicValue sv = constraintManager.createSymbolicValue((Instruction) null);
962       methodBehavior.addParameter(sv);
963       state = state.put(parameterIdx, sv);
964       state = setDoubleOrLong(state, sv, argumentType.getSize() == 2);
965       parameterIdx += argumentType.getSize();
966     }
967     return Collections.singletonList(state);
968   }

977   void enqueue(ProgramPoint pp, ProgramState programState) {
978     int nbOfExecution = programState.numberOfTimeVisited(pp);
979     if (nbOfExecution > MAX_EXEC_PROGRAM_POINT) {
980       return;
981     }
982     ProgramState ps = programState.visitedPoint(pp, nbOfExecution + 1);
983     ExplodedGraph.Node cachedNode = explodedGraph.node(pp, ps);
984     cachedNode.addParent(node, null);
985     if (cachedNode.isNew()) {
986       workList.addFirst(cachedNode);
987     }
988   }

100   private void testIssueCreation(InputFile.Type onType, int expectedIssues) throws IOException {
101     MapSettings settings = new MapSettings();
102     NoSonarFilter noSonarFilter = mock(NoSonarFilter.class);
103     SensorContextTester context = createContext(onType).setRuntime(SonarRuntimeImpl.forSonarLint(Version.create(6, 7)));
104     DefaultFileSystem fs = context.fileSystem();
105     SonarComponents sonarComponents = createSonarComponentsMock(context);
106     DefaultJavaResourceLocator javaResourceLocator = new DefaultJavaResourceLocator(fs, new JavaClasspath(settings.asConfig(), fs));
107     PostAnalysisIssueFilter postAnalysisIssueFilter = new PostAnalysisIssueFilter(fs);
108     JavaSquidSensor jss = new JavaSquidSensor(sonarComponents, fs, javaResourceLocator, settings.asConfig(), noSonarFilter, postAnalysisIssueFilter);
109 
110     jss.execute(context);
111     verify(noSonarFilter, times(1)).noSonarInFile(fs.inputFiles().iterator().next(), Sets.newHashSet(96));
112     verify(sonarComponents, times(expectedIssues)).reportIssue(any(AnalyzerMessage.class));
113 
114     settings.setProperty(Java.SOURCE_VERSION, "wrongFormat");
115     jss.execute(context);
116 
117     settings.setProperty(Java.SOURCE_VERSION, "1.7");
118     jss.execute(context);
119   }

121   private static SensorContextTester createContext(InputFile.Type onType) throws IOException {
122     SensorContextTester context = SensorContextTester.create(new File("src/test/java/").getAbsoluteFile());
123     DefaultFileSystem fs = context.fileSystem();
124 
125     String effectiveKey = "org/sonar/plugins/java/JavaSquidSensorTest.java";
126     File file = new File(fs.baseDir(), effectiveKey);
127     DefaultInputFile inputFile = new TestInputFileBuilder("", effectiveKey).setLanguage("java").setModuleBaseDir(fs.baseDirPath())
128       .setType(onType)
129       .initMetadata(new String(Files.readAllBytes(file.toPath()), StandardCharsets.UTF_8))
130       .setCharset(StandardCharsets.UTF_8)
131       .build();
132     fs.add(inputFile);
133     return context;
134   }

153   @Test
154   public void verify_analysis_errors_are_collected_on_parse_error() throws Exception {
155     SensorContextTester context = createParseErrorContext();
156     context.settings().setProperty(SonarComponents.COLLECT_ANALYSIS_ERRORS_KEY, true);
157     executeJavaSquidSensor(context);
158 
159     String feedback = context.<String>measure("projectKey", "sonarjava_feedback").value();
160     Collection<AnalysisError> analysisErrors = new Gson().fromJson(feedback, new TypeToken<Collection<AnalysisError>>(){}.getType());
161     assertThat(analysisErrors).hasSize(1);
162     AnalysisError analysisError = analysisErrors.iterator().next();
163     assertThat(analysisError.getMessage()).startsWith("Parse error at line 6 column 1:");
164     assertThat(analysisError.getCause()).startsWith("com.sonar.sslr.api.RecognitionException: Parse error at line 6 column 1:");
165     assertThat(analysisError.getFilename()).endsWith("ParseError.java");
166     assertThat(analysisError.getKind()).isEqualTo(AnalysisError.Kind.PARSE_ERROR);
167   }

169   private SensorContextTester createParseErrorContext() throws IOException {
170     File file = new File("src/test/files/ParseError.java");
171     SensorContextTester context = SensorContextTester.create(file.getParentFile().getAbsoluteFile());
172 
173     DefaultInputFile defaultFile = new TestInputFileBuilder(file.getParentFile().getPath(), file.getName())
174       .setLanguage("java")
175       .initMetadata(new String(Files.readAllBytes(file.getAbsoluteFile().toPath()), StandardCharsets.UTF_8))
176       .setCharset(StandardCharsets.UTF_8)
177       .build();
178     context.fileSystem().add(defaultFile);
179     return context;
180   }

182   private void executeJavaSquidSensor(SensorContextTester context) {
183 
184     context.setRuntime(SonarRuntimeImpl.forSonarQube(Version.create(6, 7), SonarQubeSide.SCANNER));
185     // Mock visitor for metrics.
186     FileLinesContext fileLinesContext = mock(FileLinesContext.class);
187     FileLinesContextFactory fileLinesContextFactory = mock(FileLinesContextFactory.class);
188     when(fileLinesContextFactory.createFor(any(InputFile.class))).thenReturn(fileLinesContext);
189 
190     DefaultFileSystem fs = context.fileSystem().setWorkDir(tmp.getRoot().toPath());
191     JavaClasspath javaClasspath = mock(JavaClasspath.class);
192     JavaTestClasspath javaTestClasspath = mock(JavaTestClasspath.class);
193     SonarComponents sonarComponents = new SonarComponents(fileLinesContextFactory, fs, javaClasspath, javaTestClasspath, checkFactory);
194     DefaultJavaResourceLocator javaResourceLocator = mock(DefaultJavaResourceLocator.class);
195     NoSonarFilter noSonarFilter = mock(NoSonarFilter.class);
196     PostAnalysisIssueFilter postAnalysisIssueFilter = new PostAnalysisIssueFilter(fs);
197 
198     JavaSquidSensor jss = new JavaSquidSensor(sonarComponents, fs, javaResourceLocator, new MapSettings().asConfig(),noSonarFilter, postAnalysisIssueFilter);
199     jss.execute(context);
200   }

34    @Test
35    public void unknown_column_should_be_minus_one() {
36      XmlLocation loc;
37      int expectedValue = 1;
38  
39      loc = new XmlLocation(expectedValue, expectedValue);
40      assertThat(loc.column()).isEqualTo(-1);
41      assertThat(loc.line()).isEqualTo(expectedValue);
42      assertThat(loc.offset()).isEqualTo(expectedValue);
43  
44      loc = new XmlLocation(expectedValue, expectedValue, expectedValue);
45      assertThat(loc.column()).isEqualTo(expectedValue);
46      assertThat(loc.line()).isEqualTo(expectedValue);
47      assertThat(loc.offset()).isEqualTo(expectedValue);
48    }

50    @Test
51    public void should_fail_when_providing_invalid_arguments() {
52      int validValue = 1;
53      int invalidValue = -1;
54  
55      List<int[]> checks = Lists.newArrayList(
56        new int[] {validValue, invalidValue},
57        new int[] {invalidValue, validValue},
58        new int[] {validValue, validValue, invalidValue},
59        new int[] {validValue, invalidValue, validValue},
60        new int[] {invalidValue, validValue, validValue});
61  
62      for (int[] args : checks) {
63        try {
64          if (args.length == 2) {
65            new XmlLocation(args[0], args[1]);
66          } else {
67            new XmlLocation(args[0], args[1], args[2]);
68          }
69          Fail.fail("should have failed");
70        } catch (Exception e) {
71          assertThat(e).isInstanceOf(IllegalArgumentException.class);
72        }
73      }
74    }

86    @Test
87    public void test_equals() {
88      XmlLocation loc1 = new XmlLocation(1, 2, 3);
89      Object loc2 = new XmlLocation(1, 2, 3);
90      assertThat(loc1.equals(loc2)).isTrue();
91  
92      loc2 = new Object();
93      assertThat(loc1.equals(loc2)).isFalse();

95      loc2 = new XmlLocation(1, 3);

110   @Test
111   public void test_get_location_from_xml_stream_location() {
112     int line = 1;
113     int column = 2;
114     int offset = 3;
115     XmlLocation loc = XmlLocation.getLocation(fakeXmlLocation(line, column, offset));
116     assertThat(loc).isNotNull();
117     assertThat(loc.line()).isEqualTo(line);
118     assertThat(loc.column()).isEqualTo(column);
119     assertThat(loc.offset()).isEqualTo(offset);
120   }

122   @Test
123   public void start_location_should_be_correctly_calculated() throws Exception {
124     String text = "HelloWorld";
125     javax.xml.stream.Location endLocation = fakeXmlLocation(5, 20, 50);
126     XmlLocation location = XmlLocation.getStartLocation(text, endLocation);
127     assertThat(location).isNotNull();
128     assertThat(location.line()).isEqualTo(endLocation.getLineNumber());
129     assertThat(location.column()).isEqualTo(endLocation.getColumnNumber() - text.length());
130     assertThat(location.offset()).isEqualTo(endLocation.getCharacterOffset() - text.length());
131   }
132 
133   @Test
134   public void start_location_should_take_new_lines_into_account() throws Exception {
135     String text = "Hello\nWorld";
136     javax.xml.stream.Location endLocation = fakeXmlLocation(5, 20, 50);
137     XmlLocation location = XmlLocation.getStartLocation(text, endLocation);
138     assertThat(location).isNotNull();
139     assertThat(location.line()).isEqualTo(endLocation.getLineNumber() - 1);
140     // unknown column
141     assertThat(location.column()).isEqualTo(-1);
142     assertThat(location.offset()).isEqualTo(endLocation.getCharacterOffset() - 11);
143   }
144 
145   @Test
146   public void start_location_should_take_white_characters_into_account() throws Exception {
147     String text = "   HelloWorld       ";
148     javax.xml.stream.Location endLocation = fakeXmlLocation(5, 35, 50);
149     XmlLocation location = XmlLocation.getStartLocation(text, endLocation);
150     assertThat(location).isNotNull();
151     assertThat(location.line()).isEqualTo(endLocation.getLineNumber());
152     assertThat(location.column()).isEqualTo(15);
153     assertThat(location.offset()).isEqualTo(30);
154   }
155 
156   protected static javax.xml.stream.Location fakeXmlLocation(int line, int column, int offset) {
157     javax.xml.stream.Location loc = mock(javax.xml.stream.Location.class);
158     when(loc.getLineNumber()).thenReturn(line);
159     when(loc.getColumnNumber()).thenReturn(column);
160     when(loc.getCharacterOffset()).thenReturn(offset);
161     return loc;
162   }

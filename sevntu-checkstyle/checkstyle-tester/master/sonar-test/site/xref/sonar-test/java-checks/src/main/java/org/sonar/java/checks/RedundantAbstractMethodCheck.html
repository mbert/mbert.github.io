20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  import com.google.common.collect.ImmutableMultiset;
24  import org.sonar.check.Rule;
25  import org.sonar.java.resolve.JavaSymbol;
26  import org.sonar.java.resolve.JavaType;
27  import org.sonar.java.resolve.MethodJavaType;
28  import org.sonar.java.resolve.ParametrizedTypeJavaType;
29  import org.sonar.java.resolve.SymbolMetadataResolve;
30  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
31  import org.sonar.plugins.java.api.semantic.Symbol;
32  import org.sonar.plugins.java.api.semantic.SymbolMetadata.AnnotationInstance;
33  import org.sonar.plugins.java.api.semantic.Type;
34  import org.sonar.plugins.java.api.tree.MethodTree;
35  import org.sonar.plugins.java.api.tree.Tree;
36  
37  import java.util.List;
38  
39  @Rule(key = "S3038")
40  public class RedundantAbstractMethodCheck extends IssuableSubscriptionVisitor {
41  
42    @Override
43    public List<Tree.Kind> nodesToVisit() {
44      return ImmutableList.of(Tree.Kind.METHOD);
45    }
46  
47    @Override
48    public void visitNode(Tree tree) {
49      if (!hasSemantic()) {
50        return;
51      }
52      Symbol.MethodSymbol method = ((MethodTree) tree).symbol();
53      if (method.isAbstract() && method.owner().isAbstract()) {
54        checkMethod((JavaSymbol.MethodJavaSymbol) method);
55      }
56    }
57  
58    private void checkMethod(JavaSymbol.MethodJavaSymbol method) {
59      JavaSymbol.MethodJavaSymbol overridee = method.overriddenSymbol();
60      if (overridee != null && overridee.owner().isInterface() && !differentContract(method, overridee)) {
61        reportIssue(method.declaration(), "\"" + method.name() + "\" is defined in the \"" + overridee.owner().name() + "\" interface and can be removed from this class.");
62      }
63    }
64  
65    private static boolean differentContract(JavaSymbol.MethodJavaSymbol method, JavaSymbol.MethodJavaSymbol overridee) {
66      return removingParametrizedAspect(method, overridee)
67        || differentThrows(method, overridee)
68        || differentReturnType(method, overridee)
69        || differentParameters(method, overridee)
70        || differentAnnotations(method.metadata(), overridee.metadata());
71    }
72  
73    private static boolean removingParametrizedAspect(JavaSymbol.MethodJavaSymbol method, JavaSymbol.MethodJavaSymbol overridee) {
74      return !method.isParametrized() && overridee.isParametrized();
75    }
76  
77    private static boolean differentThrows(JavaSymbol.MethodJavaSymbol method, JavaSymbol.MethodJavaSymbol overridee) {
78      return !ImmutableMultiset.copyOf(method.thrownTypes()).equals(ImmutableMultiset.copyOf(overridee.thrownTypes()));
79    }
80  
81    private static boolean differentReturnType(JavaSymbol.MethodJavaSymbol method, JavaSymbol.MethodJavaSymbol overridee) {
82      JavaType methodResultType = resultType(method);
83      JavaType overrideeResultType = resultType(overridee);
84      return specializationOfReturnType(methodResultType.erasure(), overrideeResultType.erasure()) || useRawTypeOfParametrizedType(methodResultType, overrideeResultType);
85    }
86  
87    private static JavaType resultType(JavaSymbol.MethodJavaSymbol method) {
88      return ((MethodJavaType) method.type()).resultType();
89    }
90  
91    private static boolean specializationOfReturnType(JavaType methodResultType, JavaType overrideeResultType) {
92      return !methodResultType.isVoid()
93        && (methodResultType.isSubtypeOf(overrideeResultType) && !overrideeResultType.isSubtypeOf(methodResultType));
94    }
95  
96    private static boolean differentParameters(JavaSymbol.MethodJavaSymbol method, JavaSymbol.MethodJavaSymbol overridee) {
97      return useRawTypeOfParametrizedType(method.parameterTypes(), overridee.parameterTypes())
98        || differentAnnotations(method.getParameters().scopeSymbols(), overridee.getParameters().scopeSymbols());
99    }
100 
101   private static boolean useRawTypeOfParametrizedType(List<Type> methodParamTypes, List<Type> overrideeParamType) {
102     for (int i = 0; i < methodParamTypes.size(); i++) {
103       if (useRawTypeOfParametrizedType(methodParamTypes.get(i), overrideeParamType.get(i))) {
104         return true;
105       }
106     }
107     return false;
108   }
109 
110   private static boolean useRawTypeOfParametrizedType(Type methodParam, Type overrideeParam) {
111     return overrideeParam instanceof ParametrizedTypeJavaType && methodParam.equals(overrideeParam.erasure());
112   }
113 
114   private static boolean differentAnnotations(List<JavaSymbol> methodParamSymbols, List<JavaSymbol> overrideeParamSymbols) {
115     for (int i = 0; i < methodParamSymbols.size(); i++) {
116       if (differentAnnotations(methodParamSymbols.get(i).metadata(), overrideeParamSymbols.get(i).metadata())) {
117         return true;
118       }
119     }
120     return false;
121   }
122 
123   private static boolean differentAnnotations(SymbolMetadataResolve methodMetadata, SymbolMetadataResolve overrideeMetadata) {
124     for (AnnotationInstance annotation : methodMetadata.annotations()) {
125       Type type = annotation.symbol().type();
126       if (!type.is("java.lang.Override") && !overrideeMetadata.isAnnotatedWith(type.fullyQualifiedName())) {
127         return true;
128       }
129     }
130     return false;
131   }
132 }

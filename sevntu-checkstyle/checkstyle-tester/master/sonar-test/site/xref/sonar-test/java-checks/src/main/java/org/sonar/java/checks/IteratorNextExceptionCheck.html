20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  import java.util.List;
24  import org.sonar.check.Rule;
25  import org.sonar.java.matcher.MethodMatcher;
26  import org.sonar.java.matcher.TypeCriteria;
27  import org.sonar.java.model.ExpressionUtils;
28  import org.sonar.java.resolve.MethodJavaType;
29  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
30  import org.sonar.plugins.java.api.semantic.Symbol;
31  import org.sonar.plugins.java.api.semantic.Type;
32  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
33  import org.sonar.plugins.java.api.tree.ExpressionTree;
34  import org.sonar.plugins.java.api.tree.MethodInvocationTree;
35  import org.sonar.plugins.java.api.tree.MethodTree;
36  import org.sonar.plugins.java.api.tree.NewClassTree;
37  import org.sonar.plugins.java.api.tree.ThrowStatementTree;
38  import org.sonar.plugins.java.api.tree.Tree;
39  import org.sonar.plugins.java.api.tree.Tree.Kind;
40  
41  @Rule(key = "S2272")
42  public class IteratorNextExceptionCheck extends IssuableSubscriptionVisitor {
43  
44    private static final MethodMatcher NEXT_INVOCATION_MATCHER =
45      MethodMatcher.create()
46        .typeDefinition(TypeCriteria.subtypeOf("java.util.Iterator"))
47        .name("next")
48        .withoutParameter();
49  
50    @Override
51    public List<Kind> nodesToVisit() {
52      return ImmutableList.of(Tree.Kind.METHOD);
53    }
54  
55    @Override
56    public void visitNode(Tree tree) {
57      MethodTree methodTree = (MethodTree) tree;
58      if (hasSemantic() && isIteratorNextMethod(methodTree.symbol()) && methodTree.block() != null) {
59        NextMethodBodyVisitor visitor = new NextMethodBodyVisitor();
60        tree.accept(visitor);
61        if (!visitor.foundThrow) {
62          reportIssue(methodTree.simpleName(), "Add a \"NoSuchElementException\" for iteration beyond the end of the collection.");
63        }
64      }
65    }
66  
67    private static boolean isIteratorNextMethod(Symbol.MethodSymbol symbol) {
68      return "next".equals(symbol.name()) && symbol.parameterTypes().isEmpty() && isIterator(symbol.enclosingClass());
69    }
70  
71    private static boolean isIterator(Symbol.TypeSymbol typeSymbol) {
72      return typeSymbol.type().isSubtypeOf("java.util.Iterator");
73    }
74  
75    private static class NextMethodBodyVisitor extends BaseTreeVisitor {
76  
77      private boolean foundThrow = false;
78  
79      @Override
80      public void visitThrowStatement(ThrowStatementTree throwStatementTree) {
81        ExpressionTree expression = throwStatementTree.expression();
82        if (expression.is(Tree.Kind.NEW_CLASS)) {
83          NewClassTree newClassTree = (NewClassTree) expression;
84          if (newClassTree.symbolType().is("java.util.NoSuchElementException")) {
85            foundThrow = true;
86          }
87        }
88        super.visitThrowStatement(throwStatementTree);
89      }
90  
91      @Override
92      public void visitMethodInvocation(MethodInvocationTree methodInvocation) {
93        if (NEXT_INVOCATION_MATCHER.matches(methodInvocation) || throwsNoSuchElementException(methodInvocation)) {
94          foundThrow = true;
95        }
96        super.visitMethodInvocation(methodInvocation);
97      }
98  
99      private static boolean throwsNoSuchElementException(MethodInvocationTree methodInvocationTree) {
100       Symbol symbol = methodInvocationTree.symbol();
101       if (!symbol.isMethodSymbol()) {
102         return false;
103       }
104       if (throwsNoSuchElementException(((Symbol.MethodSymbol) symbol).thrownTypes())) {
105         return true;
106       }
107       MethodJavaType methodJavaType = (MethodJavaType) ExpressionUtils.methodName(methodInvocationTree).symbolType();
108       return throwsNoSuchElementException(methodJavaType.thrownTypes());
109     }
110 
111     private static boolean throwsNoSuchElementException(List<? extends Type> thrownTypes) {
112       return thrownTypes.stream().anyMatch(t -> t.is("java.util.NoSuchElementException"));
113     }
114 
115   }

20  package org.sonar.java.checks;
21  
22  import org.sonar.check.Rule;
23  import org.sonar.check.RuleProperty;
24  import org.sonar.plugins.java.api.JavaFileScanner;
25  import org.sonar.plugins.java.api.JavaFileScannerContext;
26  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
27  import org.sonar.plugins.java.api.tree.DoWhileStatementTree;
28  import org.sonar.plugins.java.api.tree.ForEachStatement;
29  import org.sonar.plugins.java.api.tree.ForStatementTree;
30  import org.sonar.plugins.java.api.tree.IfStatementTree;
31  import org.sonar.plugins.java.api.tree.StatementTree;
32  import org.sonar.plugins.java.api.tree.SwitchStatementTree;
33  import org.sonar.plugins.java.api.tree.SyntaxToken;
34  import org.sonar.plugins.java.api.tree.Tree;
35  import org.sonar.plugins.java.api.tree.TryStatementTree;
36  import org.sonar.plugins.java.api.tree.WhileStatementTree;
37  
38  import java.util.ArrayDeque;
39  import java.util.ArrayList;
40  import java.util.Deque;
41  import java.util.List;
42  
43  @Rule(key = "S134")
44  public class NestedIfStatementsCheck extends BaseTreeVisitor implements JavaFileScanner {
45  
46    private static final int DEFAULT_MAX = 3;
47  
48    @RuleProperty(
49      description = "Maximum allowed control flow statement nesting depth.",
50      defaultValue = "" + DEFAULT_MAX)
51    public int max = DEFAULT_MAX;
52  
53    private JavaFileScannerContext context;
54    private Deque<Tree> nestingLevel;
55  
56    @Override
57    public void scanFile(final JavaFileScannerContext context) {
58      this.context = context;
59      this.nestingLevel = new ArrayDeque<>();
60      scan(context.getTree());
61    }
62  
63    @Override
64    public void visitIfStatement(IfStatementTree tree) {
65      SyntaxToken ifKeyword = tree.ifKeyword();
66      checkNesting(ifKeyword);
67      nestingLevel.push(ifKeyword);
68      visit(tree);
69      nestingLevel.pop();
70    }
71  
72    @Override
73    public void visitForStatement(ForStatementTree tree) {
74      SyntaxToken forKeyword = tree.forKeyword();
75      checkNesting(forKeyword);
76      nestingLevel.push(forKeyword);
77      super.visitForStatement(tree);
78      nestingLevel.pop();
79    }
80  
81    @Override
82    public void visitForEachStatement(ForEachStatement tree) {
83      SyntaxToken forKeyword = tree.forKeyword();
84      checkNesting(forKeyword);
85      nestingLevel.push(forKeyword);
86      super.visitForEachStatement(tree);
87      nestingLevel.pop();
88    }
89  
90    @Override
91    public void visitWhileStatement(WhileStatementTree tree) {
92      SyntaxToken whileKeyword = tree.whileKeyword();
93      checkNesting(whileKeyword);
94      nestingLevel.push(whileKeyword);
95      super.visitWhileStatement(tree);
96      nestingLevel.pop();
97    }
98  
99    @Override
100   public void visitDoWhileStatement(DoWhileStatementTree tree) {
101     SyntaxToken doKeyword = tree.doKeyword();
102     checkNesting(doKeyword);
103     nestingLevel.push(doKeyword);
104     super.visitDoWhileStatement(tree);
105     nestingLevel.pop();
106   }
107 
108   @Override
109   public void visitSwitchStatement(SwitchStatementTree tree) {
110     SyntaxToken switchKeyword = tree.switchKeyword();
111     checkNesting(switchKeyword);
112     nestingLevel.push(switchKeyword);
113     super.visitSwitchStatement(tree);
114     nestingLevel.pop();
115   }
116 
117   @Override
118   public void visitTryStatement(TryStatementTree tree) {
119     SyntaxToken tryKeyword = tree.tryKeyword();
120     checkNesting(tryKeyword);
121     nestingLevel.push(tryKeyword);
122     scan(tree.block());
123     nestingLevel.pop();
124     scan(tree.resourceList());
125     scan(tree.catches());
126     scan(tree.finallyBlock());
127   }
128 
129   private void checkNesting(Tree tree) {
130     int size = nestingLevel.size();
131     if (size == max) {
132       List<JavaFileScannerContext.Location> secondary = new ArrayList<>(size);
133       for (Tree element : nestingLevel) {
134         secondary.add(new JavaFileScannerContext.Location("Nesting + 1", element));
135       }
136       context.reportIssue(this, tree, "Refactor this code to not nest more than " + max + " if/for/while/switch/try statements.", secondary, null);
137     }
138   }
139 
140   private void visit(IfStatementTree tree) {
141     scan(tree.condition());
142     scan(tree.thenStatement());
143 
144     StatementTree elseStatementTree = tree.elseStatement();
145     if (elseStatementTree != null && elseStatementTree.is(Tree.Kind.IF_STATEMENT)) {
146       visit((IfStatementTree) elseStatementTree);
147     } else {
148       scan(elseStatementTree);
149     }
150   }
151 
152 }

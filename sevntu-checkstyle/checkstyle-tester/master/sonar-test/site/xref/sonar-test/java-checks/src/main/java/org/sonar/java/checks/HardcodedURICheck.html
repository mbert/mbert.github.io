20  package org.sonar.java.checks;
21  
22  import org.sonar.check.Rule;
23  import org.sonar.java.matcher.MethodMatcher;
24  import org.sonar.java.matcher.MethodMatcherCollection;
25  import org.sonar.java.matcher.TypeCriteria;
26  import org.sonar.java.model.ExpressionUtils;
27  import org.sonar.java.model.LiteralUtils;
28  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
29  import org.sonar.plugins.java.api.tree.AssignmentExpressionTree;
30  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
31  import org.sonar.plugins.java.api.tree.BinaryExpressionTree;
32  import org.sonar.plugins.java.api.tree.ExpressionTree;
33  import org.sonar.plugins.java.api.tree.IdentifierTree;
34  import org.sonar.plugins.java.api.tree.LiteralTree;
35  import org.sonar.plugins.java.api.tree.MemberSelectExpressionTree;
36  import org.sonar.plugins.java.api.tree.NewClassTree;
37  import org.sonar.plugins.java.api.tree.Tree;
38  import org.sonar.plugins.java.api.tree.VariableTree;
39  
40  import javax.annotation.CheckForNull;
41  import javax.annotation.Nullable;
42  
43  import java.util.Arrays;
44  import java.util.List;
45  import java.util.regex.Pattern;
46  
47  @Rule(key = "S1075")
48  public class HardcodedURICheck extends IssuableSubscriptionVisitor {
49  
50    private static final String CONSTRUCTOR_NAME = "<init>";
51    private static final String JAVA_LANG_STRING = "java.lang.String";
52    private static final MethodMatcherCollection MATCHERS = MethodMatcherCollection.create(
53      MethodMatcher.create().typeDefinition("java.net.URI").name(CONSTRUCTOR_NAME).addParameter(JAVA_LANG_STRING),
54      MethodMatcher.create().typeDefinition("java.io.File").name(CONSTRUCTOR_NAME).addParameter(JAVA_LANG_STRING),
55      MethodMatcher.create().typeDefinition("java.io.File").name(CONSTRUCTOR_NAME).addParameter(TypeCriteria.anyType()).addParameter(JAVA_LANG_STRING));
56  
57    private static final String SCHEME = "[a-zA-Z][a-zA-Z\\+\\.\\-]+";
58    private static final String FOLDER_NAME = "[^/?%*:\\\\|\"<>]+";
59    private static final String URI_REGEX = String.format("^%s://.+", SCHEME);
60    private static final String LOCAL_URI = String.format("^(~/|/|//[\\w-]+/|%s:/)(%s/)*%s/?",
61      SCHEME, FOLDER_NAME, FOLDER_NAME);
62    private static final String BACKSLASH_LOCAL_URI = String.format("^(~\\\\\\\\|\\\\\\\\\\\\\\\\[\\w-]+\\\\\\\\|%s:\\\\\\\\)(%s\\\\\\\\)*%s(\\\\\\\\)?",
63      SCHEME, FOLDER_NAME, FOLDER_NAME);
64    private static final String DISK_URI = "^[A-Za-z]:(/|\\\\)";
65  
66    private static final Pattern URI_PATTERN = Pattern.compile(URI_REGEX + "|" + LOCAL_URI + "|" + DISK_URI + "|" + BACKSLASH_LOCAL_URI);
67    private static final Pattern VARIABLE_NAME_PATTERN = Pattern.compile("filename|path", Pattern.CASE_INSENSITIVE);
68    private static final Pattern PATH_DELIMETERS_PATTERN = Pattern.compile("\"/\"|\"//\"|\"\\\\\\\\\"|\"\\\\\\\\\\\\\\\\\"");
69  
70    @Override
71    public List<Tree.Kind> nodesToVisit() {
72      return Arrays.asList(Tree.Kind.NEW_CLASS, Tree.Kind.VARIABLE, Tree.Kind.ASSIGNMENT);
73    }
74  
75    @Override
76    public void visitNode(Tree tree) {
77      if (tree.is(Tree.Kind.NEW_CLASS)) {
78        checkNewClassTree((NewClassTree) tree);
79      } else if (tree.is(Tree.Kind.VARIABLE)) {
80        checkVariable((VariableTree) tree);
81      } else {
82        checkAssignment((AssignmentExpressionTree) tree);
83      }
84    }
85  
86    private void checkNewClassTree(NewClassTree nct) {
87      if (MATCHERS.anyMatch(nct)) {
88        nct.arguments().forEach(this::checkExpression);
89      }
90    }
91  
92    private void checkVariable(VariableTree tree) {
93      if (isFileNameVariable(tree.simpleName())) {
94        checkExpression(tree.initializer());
95      }
96    }
97  
98    private void checkAssignment(AssignmentExpressionTree tree) {
99      if (isFileNameVariable(getVariableIdentifier(tree)) && !isPartOfAnnotation(tree)) {
100       checkExpression(tree.expression());
101     }
102   }
103 
104   private static boolean isPartOfAnnotation(AssignmentExpressionTree tree) {
105     Tree parent = tree.parent();
106     while (parent != null) {
107       if (parent.is(Tree.Kind.ANNOTATION)) {
108         return true;
109       }
110       parent = parent.parent();
111     }
112     return false;
113   }
114 
115   private static boolean isFileNameVariable(@Nullable IdentifierTree variable) {
116     return variable != null && VARIABLE_NAME_PATTERN.matcher(variable.name()).find();
117   }
118 
119   private void checkExpression(@Nullable ExpressionTree expr) {
120     if (expr != null) {
121       if (isHardcodedURI(expr)) {
122         reportHardcodedURI(expr);
123       } else {
124         reportStringConcatenationWithPathDelimiter(expr);
125       }
126     }
127   }
128 
129   private static boolean isHardcodedURI(ExpressionTree expr) {
130     ExpressionTree newExpr = ExpressionUtils.skipParentheses(expr);
131     if (!newExpr.is(Tree.Kind.STRING_LITERAL)) {
132       return false;
133     }
134     String stringLiteral = LiteralUtils.trimQuotes(((LiteralTree) newExpr).value());
135     return URI_PATTERN.matcher(stringLiteral).find();
136   }
137 
138   private void reportHardcodedURI(ExpressionTree hardcodedURI) {
139     reportIssue(hardcodedURI, "Refactor your code to get this URI from a customizable parameter.");
140   }
141 
142   private void reportStringConcatenationWithPathDelimiter(ExpressionTree expr) {
143     expr.accept(new StringConcatenationVisitor());
144   }
145 
146   private class StringConcatenationVisitor extends BaseTreeVisitor {
147     @Override
148     public void visitBinaryExpression(BinaryExpressionTree tree) {
149       if (tree.is(Tree.Kind.PLUS)) {
150         checkPathDelimiter(tree.leftOperand());
151         checkPathDelimiter(tree.rightOperand());
152       }
153       super.visitBinaryExpression(tree);
154     }
155 
156     private void checkPathDelimiter(ExpressionTree expr) {
157       ExpressionTree newExpr = ExpressionUtils.skipParentheses(expr);
158       if (newExpr.is(Tree.Kind.STRING_LITERAL) && PATH_DELIMETERS_PATTERN.matcher(((LiteralTree) newExpr).value()).find()) {
159         reportIssue(newExpr, "Remove this hard-coded path-delimiter.");
160       }
161     }
162   }
163 
164   @CheckForNull
165   private static IdentifierTree getVariableIdentifier(AssignmentExpressionTree tree) {
166     ExpressionTree variable = ExpressionUtils.skipParentheses(tree.variable());
167     if (variable.is(Tree.Kind.IDENTIFIER)) {
168       return (IdentifierTree) variable;
169     } else if (variable.is(Tree.Kind.MEMBER_SELECT)) {
170       return ((MemberSelectExpressionTree) variable).identifier();
171     }
172     // ignore assignments in arrays
173     return null;
174   }
175 
176 }

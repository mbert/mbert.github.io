20  package org.sonar.java.checks;
21  
22  import org.sonar.check.Rule;
23  import org.sonar.plugins.java.api.JavaFileScanner;
24  import org.sonar.plugins.java.api.JavaFileScannerContext;
25  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
26  import org.sonar.plugins.java.api.tree.BinaryExpressionTree;
27  import org.sonar.plugins.java.api.tree.Tree;
28  
29  @Rule(key = "S2164")
30  public class MathOnFloatCheck extends BaseTreeVisitor implements JavaFileScanner {
31  
32    private JavaFileScannerContext context;
33  
34    @Override
35    public void scanFile(JavaFileScannerContext context) {
36      this.context = context;
37      scan(context.getTree());
38    }
39  
40    @Override
41    public void visitBinaryExpression(BinaryExpressionTree tree) {
42      if (tree.is(Tree.Kind.PLUS, Tree.Kind.MINUS, Tree.Kind.MULTIPLY, Tree.Kind.DIVIDE)) {
43        if (withinStringConcatenation(tree)) {
44          return;
45        }
46        if (tree.symbolType().is("float")) {
47          context.reportIssue(this, tree, "Use a \"double\" or \"BigDecimal\" instead.");
48          // do not look for other issues in sub-tree
49          return;
50        }
51      }
52      super.visitBinaryExpression(tree);
53    }
54  
55    private static boolean withinStringConcatenation(BinaryExpressionTree tree) {
56      Tree parent = tree.parent();
57      while (parent.is(Tree.Kind.PARENTHESIZED_EXPRESSION)) {
58        parent = parent.parent();
59      }
60      return parent.is(Tree.Kind.PLUS) && ((BinaryExpressionTree) parent).symbolType().is("java.lang.String");
61    }
62  
63  }

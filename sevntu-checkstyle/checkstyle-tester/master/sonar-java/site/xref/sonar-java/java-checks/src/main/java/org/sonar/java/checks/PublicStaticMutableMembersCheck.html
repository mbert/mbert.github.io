54  @Rule(key = "S2386")
55  public class PublicStaticMutableMembersCheck extends IssuableSubscriptionVisitor {
56  
57    private static final Set<String> ALWAYS_MUTABLE_TYPES = ImmutableSet.of(
58      "java.awt.Point",
59      "java.util.Date"
60    );
61  
62    private static final Set<String> MUTABLE_TYPES = ImmutableSet.<String>builder()
63      .addAll(ALWAYS_MUTABLE_TYPES)
64      .add("java.util.Collection")
65      .add("java.util.Map")
66      .build();
67  
68    private static final String DECORATE = "decorate";
69    // java.util and apache commons
70    private static final MethodMatcherCollection UNMODIFIABLE_METHOD_CALLS = MethodMatcherCollection.create()
71      .add(MethodMatcher.create().typeDefinition("java.util.Collections").name(NameCriteria.startsWith("singleton")).withAnyParameters())
72      .add(MethodMatcher.create().typeDefinition("java.util.Collections").name(NameCriteria.startsWith("empty")).withAnyParameters())
73      .add(MethodMatcher.create().typeDefinition(TypeCriteria.anyType()).name(NameCriteria.startsWith("unmodifiable")).withAnyParameters())
74        // apache commons 3.X
75      .add(MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf("org.apache.commons.collections.map.UnmodifiableMap")).name(DECORATE).withAnyParameters())
76      .add(MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf("org.apache.commons.collections.set.UnmodifiableSet")).name(DECORATE).withAnyParameters())
77      .add(MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf("org.apache.commons.collections.list.UnmodifiableList")).name(DECORATE).withAnyParameters())
78        // apache commons 4.X
79      .add(MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf("org.apache.commons.collections4.map.UnmodifiableMap")).name(DECORATE).withAnyParameters())
80      .add(MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf("org.apache.commons.collections4.set.UnmodifiableSet")).name(DECORATE).withAnyParameters())
81      .add(MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf("org.apache.commons.collections4.list.UnmodifiableList")).name(DECORATE).withAnyParameters());
82    private static final MethodMatcher ARRAYS_AS_LIST = MethodMatcher.create()
83      .typeDefinition("java.util.Arrays").name("asList").withAnyParameters();
84  
85    private static final Set<String> ACCEPTED_TYPES = ImmutableSet.of(
86      "com.google.common.collect.ImmutableMap",
87      "com.google.common.collect.ImmutableCollection"
88    );
89  
90    private static final Set<String> ACCEPTED_NEW_TYPES = Collections.singleton(
91      "org.apache.commons.collections4.list.UnmodifiableList"
92    );
93  
94    private static final Set<Symbol> IMMUTABLE_CANDIDATES = new HashSet<>();
95    private static final Multimap<Tree, Symbol> CLASS_IMMUTABLE_CANDIDATES = ArrayListMultimap.create();
96  
97    @Override
98    public List<Tree.Kind> nodesToVisit() {
99      return Arrays.asList(Tree.Kind.INTERFACE, Tree.Kind.CLASS, Tree.Kind.ENUM, Tree.Kind.ASSIGNMENT);
100   }
101 
102   @Override
103   public void leaveFile(JavaFileScannerContext context) {
104     CLASS_IMMUTABLE_CANDIDATES.clear();
105   }
106 
107   @Override
108   public void visitNode(Tree tree) {
109     if (tree.is(Tree.Kind.ASSIGNMENT)) {
110       checkAssignment((AssignmentExpressionTree) tree);
111     } else {
112       List<Tree> members = ((ClassTree) tree).members();
113       for (Tree member : members) {
114         if (member.is(Tree.Kind.VARIABLE)) {
115           preCheckVariable(tree, (VariableTree) member);
116         }
117       }
118     }
119   }
120 
121   private void preCheckVariable(Tree owner, VariableTree variableTree) {
122     Symbol symbol = variableTree.symbol();
123     if (symbol != null && isPublicStatic(symbol) && isForbiddenType(symbol.type())) {
124       if (isMutable(variableTree.initializer(), symbol)) {
125         String message = "Make this member \"protected\".";
126         if (owner.is(Tree.Kind.INTERFACE)) {
127           message = MessageFormat.format("Move \"{0}\" to a class and lower its visibility", variableTree.simpleName().name());
128         }
129         reportIssue(variableTree.simpleName(), message);
130       } else {
131         IMMUTABLE_CANDIDATES.add(symbol);
132         CLASS_IMMUTABLE_CANDIDATES.put(owner, symbol);
133       }
134     }
135   }
136 
137   private void checkAssignment(AssignmentExpressionTree node) {
138     ExpressionTree variable = ExpressionUtils.skipParentheses(node.variable());
139     if (variable.is(Tree.Kind.MEMBER_SELECT)) {
140       variable = ((MemberSelectExpressionTree) variable).identifier();
141     }
142     if (variable.is(Tree.Kind.IDENTIFIER)) {
143       IdentifierTree identifierTree = (IdentifierTree) variable;
144       Symbol symbol = identifierTree.symbol();
145       if (IMMUTABLE_CANDIDATES.contains(symbol) && isMutable(node.expression(), symbol)) {
146         reportIssue(identifierTree, "Make member \"" + symbol.name() + "\" \"protected\".");
147         IMMUTABLE_CANDIDATES.remove(symbol);
148       }
149     }
150   }
151 
152   @Override
153   public void leaveNode(Tree tree) {
154     // cleanup
155     if (tree.is(Tree.Kind.CLASS, Tree.Kind.ENUM)) {
156       IMMUTABLE_CANDIDATES.removeAll(CLASS_IMMUTABLE_CANDIDATES.get(tree));
157     }
158   }
159 
160   static boolean isMutable(@Nullable ExpressionTree initializer, Symbol symbol) {
161     Type type = symbol.type();
162     if (initializer == null) {
163       return ALWAYS_MUTABLE_TYPES.stream().anyMatch(type::isSubtypeOf);
164     }
165     if (symbol.isFinal() && isEmptyArray(initializer)) {
166       return false;
167     }
168     ExpressionTree expression = ExpressionUtils.skipParentheses(initializer);
169     if (expression.is(Tree.Kind.METHOD_INVOCATION)) {
170       return returnValueIsMutable((MethodInvocationTree) expression);
171     } else if (expression.is(Tree.Kind.NEW_CLASS)) {
172       return !isAcceptedType(expression.symbolType(), ACCEPTED_NEW_TYPES);
173     } else if (expression.is(Tree.Kind.IDENTIFIER)) {
174       Symbol assigned = ((IdentifierTree) expression).symbol();
175       return !IMMUTABLE_CANDIDATES.contains(assigned);
176     }
177     return true;
178   }
179 
180   private static boolean isEmptyArray(ExpressionTree expression) {
181     if (!expression.is(Tree.Kind.NEW_ARRAY)) {
182       return false;
183     }
184     NewArrayTree nat = (NewArrayTree) expression;
185     return hasEmptyInitializer(nat) || hasOnlyZeroDimensions(nat.dimensions());
186   }
187 
188   private static boolean hasEmptyInitializer(NewArrayTree newArrayTree) {
189     return newArrayTree.openBraceToken() != null && newArrayTree.initializers().isEmpty();
190   }
191 
192   private static boolean hasOnlyZeroDimensions(List<ArrayDimensionTree> dimensions) {
193     return !dimensions.isEmpty() && dimensions.stream().allMatch(PublicStaticMutableMembersCheck::isZeroDimension);
194   }
195 
196   private static boolean isZeroDimension(ArrayDimensionTree dim) {
197     ExpressionTree expression = dim.expression();
198     return expression != null && expression.is(Tree.Kind.INT_LITERAL) && "0".equals(((LiteralTree) expression).value());
199   }
200 
201   private static boolean returnValueIsMutable(MethodInvocationTree mit) {
202     if (isAcceptedTypeOrUnmodifiableMethodCall(mit)) {
203       return false;
204     } else if (ARRAYS_AS_LIST.matches(mit)) {
205       return !mit.arguments().isEmpty();
206     }
207     return true;
208   }
209 
210   private static boolean isAcceptedTypeOrUnmodifiableMethodCall(MethodInvocationTree mit) {
211     Type type = mit.symbolType();
212     return type.isUnknown() || isAcceptedType(type, ACCEPTED_TYPES) || UNMODIFIABLE_METHOD_CALLS.anyMatch(mit);
213   }
214 
215   private static boolean isAcceptedType(Type type, Set<String> accepted) {
216     for (String acceptedType : accepted) {
217       if (type.isSubtypeOf(acceptedType)) {
218         return true;
219       }
220     }
221     return false;
222   }
223 
224   static boolean isPublicStatic(Symbol symbol) {
225     return symbol.isStatic() && symbol.isPublic();
226   }
227 
228   static boolean isForbiddenType(final Type type) {
229     return type.isArray() || MUTABLE_TYPES.stream().anyMatch(type::isSubtypeOf);
230   }

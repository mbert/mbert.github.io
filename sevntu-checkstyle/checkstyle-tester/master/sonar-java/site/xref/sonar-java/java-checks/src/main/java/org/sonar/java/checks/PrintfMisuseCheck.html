47    private static final MethodMatcher TO_STRING = MethodMatcher.create().typeDefinition(TypeCriteria.anyType()).name("toString").withoutParameter();
48    private static final String JAVA_LANG_STRING = "java.lang.String";
49    private static final MethodMatcherCollection GET_LOGGER = MethodMatcherCollection.create(
50      MethodMatcher.create().typeDefinition("java.util.logging.Logger").name("getLogger").parameters(JAVA_LANG_STRING, JAVA_LANG_STRING),
51      MethodMatcher.create().typeDefinition("java.util.logging.Logger").name("getAnonymousLogger").parameters(JAVA_LANG_STRING)
52      );

55    @Override
56    protected void onMethodInvocationFound(MethodInvocationTree mit) {
57      boolean isMessageFormat = MESSAGE_FORMAT.matches(mit);
58      if (isMessageFormat && !mit.symbol().isStatic()) {
59        // only consider the static method
60        return;
61      }
62      if (!isMessageFormat) {
63        isMessageFormat = JAVA_UTIL_LOGGER.matches(mit);
64        if (isMessageFormat && (mit.arguments().get(mit.arguments().size() - 1).symbolType().isSubtypeOf("java.lang.Throwable") || hasResourceBundle(mit))) {
65          // ignore formatting issues when last argument is a throwable
66          return;
67        }
68      }
69      if(!isMessageFormat) {
70        isMessageFormat = LEVELS.contains(mit.symbol().name());
71        if (isMessageFormat && mit.arguments().get(mit.arguments().size() - 1).symbolType().isSubtypeOf("java.lang.Throwable")) {
72          // ignore formatting issues when last argument is a throwable
73          return;
74        }
75      }
76      checkFormatting(mit, isMessageFormat);
77    }

79    private static boolean hasResourceBundle(MethodInvocationTree mit) {
80      Tree id;
81      if (mit.methodSelect().is(Tree.Kind.MEMBER_SELECT)) {
82        id = ((MemberSelectExpressionTree) mit.methodSelect()).expression();
83      } else {
84        // defensive programming : cannot be reached : log methods are not static
85        return false;
86      }
87      if (id.is(Tree.Kind.MEMBER_SELECT)) {
88        id = ((MemberSelectExpressionTree) id).identifier();
89      }
90      if (id.is(Tree.Kind.IDENTIFIER)) {
91        Tree decl = ((IdentifierTree) id).symbol().declaration();
92        if (decl != null && decl.is(Tree.Kind.VARIABLE)) {
93          VariableTree var = ((VariableTree) decl);
94          ExpressionTree init = var.initializer();
95          if (init != null && init.is(Tree.Kind.METHOD_INVOCATION)) {
96            return GET_LOGGER.anyMatch((MethodInvocationTree) init);
97          }
98        }
99      }
100     return false;
101   }

103   private void checkFormatting(MethodInvocationTree mit, boolean isMessageFormat) {
104     if (mit.arguments().stream().map(ExpressionTree::symbolType).anyMatch(Type::isUnknown)) {
105       // method resolved but not all the parameters are
106       return;
107     }
108     ExpressionTree formatStringTree;
109     List<ExpressionTree> args;
110     // Check type of first argument:
111     if (mit.arguments().get(0).symbolType().is(JAVA_LANG_STRING)) {
112       formatStringTree = mit.arguments().get(0);
113       args = mit.arguments().subList(1, mit.arguments().size());
114     } else {
115       // format method with "Locale" first argument, skip that one.
116       formatStringTree = mit.arguments().get(1);
117       args = mit.arguments().subList(2, mit.arguments().size());
118     }
119     if (formatStringTree.is(Tree.Kind.STRING_LITERAL)) {
120       String formatString = LiteralUtils.trimQuotes(((LiteralTree) formatStringTree).value());
121       if (isMessageFormat) {
122         handleMessageFormat(mit, formatString, args);
123       } else {
124         handlePrintfFormat(mit, formatString, args);
125       }
126     } else if (isConcatenationOnSameLine(formatStringTree)) {
127       reportIssue(mit, "Format specifiers should be used instead of string concatenation.");
128     }
129   }

131   @Override
132   protected void handlePrintfFormat(MethodInvocationTree mit, String formatString, List<ExpressionTree> args) {
133     List<String> params = getParameters(formatString, mit);
134     if (usesMessageFormat(formatString, params)) {
135       reportIssue(mit, "Looks like there is a confusion with the use of java.text.MessageFormat, parameters will be simply ignored here");
136       return;
137     }
138     checkLineFeed(formatString, mit);
139     if (checkEmptyParams(mit, params)) {
140       return;
141     }
142     cleanupLineSeparator(params);
143     if (!params.isEmpty() && argIndexes(params).size() <= args.size()) {
144       verifyParameters(mit, args, params);
145     }
146   }
147 
148   private void verifyParameters(MethodInvocationTree mit, List<ExpressionTree> args, List<String> params) {
149     int index = 0;
150     List<ExpressionTree> unusedArgs = new ArrayList<>(args);
151     for (String rawParam : params) {
152       String param = rawParam;
153       int argIndex = index;
154       if (param.contains("$")) {
155         argIndex = getIndex(param) - 1;
156         if (argIndex == -1) {
157           return;
158         }
159         param = param.substring(param.indexOf('$') + 1);
160       } else if (param.charAt(0) == '<') {
161         //refers to previous argument
162         argIndex = Math.max(0, argIndex - 1);
163       }else {
164         index++;
165       }
166       ExpressionTree argExpressionTree = args.get(argIndex);
167       unusedArgs.remove(argExpressionTree);
168       Type argType = argExpressionTree.symbolType();
169       checkBoolean(mit, param, argType);
170     }
171     reportUnusedArgs(mit, args, unusedArgs);
172   }

174   @Override
175   protected void handleMessageFormat(MethodInvocationTree mit, String formatString, List<ExpressionTree> args) {
176     String newFormatString = cleanupDoubleQuote(formatString);
177     Set<Integer> indexes = getMessageFormatIndexes(newFormatString, mit);
178     List<ExpressionTree> newArgs = args;
179     if (newArgs.size() == 1) {
180       ExpressionTree firstArg = newArgs.get(0);
181       if (firstArg.symbolType().isArray()) {
182         if (isNewArrayWithInitializers(firstArg)) {
183           newArgs = ((NewArrayTree) firstArg).initializers();
184         } else {
185           // size is unknown
186           return;
187         }
188       }
189     }
190     if (checkEmptyParams(mit, indexes)) {
191       return;
192     }
193     checkToStringInvocation(newArgs);
194     verifyParameters(mit, newArgs, indexes);
195   }
196 
197   private boolean checkEmptyParams(MethodInvocationTree mit, Collection<?> params) {
198     if (params.isEmpty() && (!LEVELS.contains(mit.symbol().name()) || mit.arguments().size() > 1)) {
199       reportIssue(mit, "String contains no format specifiers.");
200       return true;
201     }
202     return false;
203   }
204 
205   private void checkToStringInvocation(List<ExpressionTree> args) {
206     args.stream()
207       .filter(arg -> arg.is(Tree.Kind.METHOD_INVOCATION) && TO_STRING.matches((MethodInvocationTree) arg))
208       .forEach(arg -> reportIssue(arg, getToStringMessage(arg)));
209   }
210 
211   private static String getToStringMessage(ExpressionTree arg) {
212     if (isInStringArrayInitializer(arg)) {
213       return "No need to call \"toString()\" method since an array of Objects can be used here.";
214     }
215     return "No need to call \"toString()\" method as formatting and string conversion is done by the Formatter.";
216   }
217 
218   private static boolean isInStringArrayInitializer(ExpressionTree arg) {
219     return Optional.of(arg)
220       .map(Tree::parent)
221       .filter(tree -> tree.is(Tree.Kind.LIST))
222       .map(Tree::parent)
223       .filter(tree -> tree.is(Tree.Kind.NEW_ARRAY))
224       .map(NewArrayTree.class::cast)
225       .map(ExpressionTree::symbolType)
226       .filter(Type::isArray)
227       .map(Type.ArrayType.class::cast)
228       .map(Type.ArrayType::elementType)
229       .filter(type -> type.is(JAVA_LANG_STRING))
230       .isPresent();
231   }
232 
233   private void verifyParameters(MethodInvocationTree mit, List<ExpressionTree> args, Set<Integer> indexes) {
234     List<ExpressionTree> unusedArgs = new ArrayList<>(args);
235     for (int index : indexes) {
236       if (index >= args.size()) {
237         return;
238       }
239       unusedArgs.remove(args.get(index));
240     }
241     reportUnusedArgs(mit, args, unusedArgs);
242   }
243 
244   private void reportUnusedArgs(MethodInvocationTree mit, List<ExpressionTree> args, List<ExpressionTree> unusedArgs) {
245     for (ExpressionTree unusedArg : unusedArgs) {
246       int i = args.indexOf(unusedArg);
247       String stringArgIndex = "first";
248       if (i == 1) {
249         stringArgIndex = "2nd";
250       } else if (i == 2) {
251         stringArgIndex = "3rd";
252       } else if (i >= 3) {
253         stringArgIndex = (i + 1) + "th";
254       }
255       reportIssue(mit, stringArgIndex + " argument is not used.");
256     }
257   }

260   private void checkBoolean(MethodInvocationTree mit, String param, Type argType) {
261     if (param.charAt(0) == 'b' && !(argType.is("boolean") || argType.is("java.lang.Boolean"))) {
262       reportIssue(mit, "Directly inject the boolean value.");
263     }
264   }
265 
266   private void checkLineFeed(String formatString, MethodInvocationTree mit) {
267     if (formatString.contains("\\n")) {
268       reportIssue(mit, "%n should be used in place of \\n to produce the platform-specific line separator.");
269     }
270   }
271 
272   private static boolean usesMessageFormat(String formatString, List<String> params) {
273     return params.isEmpty() && (formatString.contains("{0") || formatString.contains("{1"));
274   }

277   private static boolean isConcatenationOnSameLine(ExpressionTree formatStringTree) {
278     return formatStringTree.is(Tree.Kind.PLUS)
279       && operandsAreOnSameLine((BinaryExpressionTree) formatStringTree)
280       && ConstantUtils.resolveAsConstant(formatStringTree) == null;
281   }
282 
283   private static boolean operandsAreOnSameLine(BinaryExpressionTree formatStringTree) {
284     return formatStringTree.leftOperand().firstToken().line() == formatStringTree.rightOperand().firstToken().line();
285   }

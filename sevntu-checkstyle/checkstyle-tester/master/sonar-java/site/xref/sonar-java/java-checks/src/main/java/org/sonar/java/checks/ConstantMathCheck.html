41  @Rule(key = "S2185")
42  public class ConstantMathCheck extends IssuableSubscriptionVisitor {
43  
44    private static final String ABS = "abs";
45    private static final String CEIL = "ceil";
46    private static final String DOUBLE = "double";
47    private static final String FLOAT = "float";
48    private static final String FLOOR = "floor";
49    private static final String MATH_PACKAGE_NAME = "java.lang.Math";
50    private static final String ROUND = "round";
51  
52    private static final MethodMatcherCollection CONSTANT_WITH_LITERAL_METHODS = MethodMatcherCollection.create(
53      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(ABS).addParameter(DOUBLE),
54      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(ABS).addParameter(FLOAT),
55      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(ABS).addParameter("int"),
56      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(ABS).addParameter("long")
57    );
58  
59    private static final MethodMatcherCollection TRUNCATION_METHODS = MethodMatcherCollection.create(
60      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(CEIL).addParameter(DOUBLE),
61      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(CEIL).addParameter(FLOAT),
62      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(FLOOR).addParameter(DOUBLE),
63      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(FLOOR).addParameter(FLOAT),
64      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("rint").addParameter(DOUBLE),
65      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(ROUND).addParameter(DOUBLE),
66      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name(ROUND).addParameter(FLOAT)
67    );
68  
69    private static final MethodMatcherCollection CONSTANT_WITH_ZERO_METHODS = MethodMatcherCollection.create(
70      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("atan2").addParameter(DOUBLE).addParameter(DOUBLE),
71      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("cos").addParameter(DOUBLE),
72      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("cosh").addParameter(DOUBLE),
73      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("expm1").addParameter(DOUBLE),
74      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("sin").addParameter(DOUBLE),
75      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("sinh").addParameter(DOUBLE),
76      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("tan").addParameter(DOUBLE),
77      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("tanh").addParameter(DOUBLE),
78      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("toRadians").addParameter(DOUBLE)
79    );
80  
81    private static final MethodMatcherCollection CONSTANT_WITH_ZERO_OR_ONE_METHODS = MethodMatcherCollection.create(
82      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("acos").addParameter(DOUBLE),
83      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("asin").addParameter(DOUBLE),
84      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("atan").addParameter(DOUBLE),
85      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("cbrt").addParameter(DOUBLE),
86      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("exp").addParameter(DOUBLE),
87      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("log").addParameter(DOUBLE),
88      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("log10").addParameter(DOUBLE),
89      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("sqrt").addParameter(DOUBLE),
90      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("toDegrees").addParameter(DOUBLE),
91      MethodMatcher.create().typeDefinition(MATH_PACKAGE_NAME).name("exp").addParameter(DOUBLE)
92    );
93  
94    @Override
95    public List<Tree.Kind> nodesToVisit() {
96      return Arrays.asList(Tree.Kind.REMAINDER, Tree.Kind.METHOD_INVOCATION);
97    }
98  
99    @Override
100   public void visitNode(Tree tree) {
101     if (hasSemantic()) {
102       if (tree.is(Tree.Kind.REMAINDER)) {
103         BinaryExpressionTree remainderTree = (BinaryExpressionTree) tree;
104         if (isIntegralOne(remainderTree.rightOperand()) && isIntOrLong(remainderTree.leftOperand())) {
105           reportIssue(remainderTree.operatorToken(), "Remove this computation of % 1, which always evaluates to zero.");
106         }
107       } else {
108         MethodInvocationTree mit = (MethodInvocationTree) tree;
109         if (isConstantWithLiteral(mit) || isTruncation(mit) || isConstantWithZero(mit) || isConstantWithZeroOrOne(mit)) {
110           reportIssue(mit.methodSelect(), String.format("Remove this silly call to \"Math.%s\"", mit.symbol().name()));
111         }
112       }
113     }
114   }
115 
116   private static boolean isIntOrLong(ExpressionTree expression) {
117     JavaType type = (JavaType) expression.symbolType();
118     return isIntegral(type) || (type.isPrimitiveWrapper() && isIntegral(type.primitiveType()));
119   }
120 
121   private static boolean isTruncation(MethodInvocationTree methodTree) {
122     return TRUNCATION_METHODS.anyMatch(methodTree) && isCastFromIntegralToFloating(ExpressionUtils.skipParentheses(methodTree.arguments().get(0)));
123   }
124 
125   private static boolean isConstantWithLiteral(MethodInvocationTree methodTree) {
126     return CONSTANT_WITH_LITERAL_METHODS.anyMatch(methodTree) && isConstant(methodTree.arguments().get(0));
127   }
128 
129   private static boolean isConstantWithZero(MethodInvocationTree methodTree) {
130     return CONSTANT_WITH_ZERO_METHODS.anyMatch(methodTree) && isFloatingZero(methodTree.arguments().get(0));
131   }
132 
133   private static boolean isConstantWithZeroOrOne(MethodInvocationTree methodTree) {
134     return CONSTANT_WITH_ZERO_OR_ONE_METHODS.anyMatch(methodTree) && isFloatingZeroOrOne(methodTree.arguments().get(0));
135   }
136 
137   private static boolean isCastFromIntegralToFloating(ExpressionTree tree) {
138     Type resultType = tree.symbolType();
139     // explicit cast
140     if (tree.is(Tree.Kind.TYPE_CAST) && isIntegral(getInnerType(((TypeCastTree) tree).expression())) && (resultType.is(DOUBLE) || resultType.is(FLOAT))) {
141       return true;
142     }
143     // implicit cast
144     return isIntegral(resultType);
145   }
146 
147   private static boolean isConstant(ExpressionTree tree) {
148     return getInnerExpression(tree).is(Tree.Kind.CHAR_LITERAL, Tree.Kind.DOUBLE_LITERAL, Tree.Kind.FLOAT_LITERAL, Tree.Kind.INT_LITERAL, Tree.Kind.LONG_LITERAL);
149   }
150 
151   private static boolean isIntegral(Type type) {
152     return type.isPrimitive() && !type.is(DOUBLE) && !type.is(FLOAT);
153   }
154 
155   private static boolean isIntegralOne(ExpressionTree tree) {
156     Long value = LiteralUtils.longLiteralValue(tree);
157     return value != null && value == 1;
158   }
159 
160   private static ExpressionTree getInnerExpression(ExpressionTree tree) {
161     ExpressionTree result = ExpressionUtils.skipParentheses(tree);
162     while (result.is(Tree.Kind.TYPE_CAST)) {
163       result = ExpressionUtils.skipParentheses(((TypeCastTree) result).expression());
164     }
165     return result;
166   }
167 
168   private static Type getInnerType(ExpressionTree tree) {
169     return getInnerExpression(tree).symbolType();
170   }
171 
172   private static boolean isFloatingZero(ExpressionTree tree) {
173     Integer value = getFloatingZeroOrOne(tree);
174     return value != null && value == 0;
175   }
176 
177   private static boolean isFloatingZeroOrOne(ExpressionTree tree) {
178     return getFloatingZeroOrOne(tree) != null;
179   }
180 
181   @CheckForNull
182   private static Integer getFloatingZeroOrOne(ExpressionTree tree) {
183     ExpressionTree expressionTree = ExpressionUtils.skipParentheses(tree);
184     if (expressionTree.is(Tree.Kind.DOUBLE_LITERAL, Tree.Kind.FLOAT_LITERAL)) {
185       String value = ((LiteralTree) expressionTree).value();
186       if ("0.0".equals(value) || "0.0d".equalsIgnoreCase(value) || "0.0f".equalsIgnoreCase(value)) {
187         return 0;
188       } else if ("1.0".equals(value) || "1.0d".equalsIgnoreCase(value) || "1.0f".equalsIgnoreCase(value)) {
189         return 1;
190       }
191     }
192     return null;
193   }

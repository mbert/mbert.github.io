54  @Rule(key = "S2699")
55  public class AssertionsInTestsCheck extends BaseTreeVisitor implements JavaFileScanner {
56  
57    private static final Logger LOG = Loggers.get(AssertionsInTestsCheck.class);
58  
59    private static final TypeCriteria IO_RESTASSURED = TypeCriteria.is("io.restassured.response.ValidatableResponseOptions");
60    private static final TypeCriteria ANY_TYPE = TypeCriteria.anyType();
61    private static final NameCriteria ANY_NAME = NameCriteria.any();
62  
63    private static final Pattern ASSERTION_METHODS_PATTERN = Pattern.compile("(assert|verify|fail|should|check|expect|validate).*");
64    private static final Pattern TEST_METHODS_PATTERN = Pattern.compile("test.*|.*Test");
65  
66    private static final MethodMatcherCollection ASSERTION_INVOCATION_MATCHERS = MethodMatcherCollection.create(
67      // fest 1.x / 2.X
68      method(TypeCriteria.subtypeOf("org.fest.assertions.GenericAssert"), ANY_NAME).withAnyParameters(),
69      method(TypeCriteria.subtypeOf("org.fest.assertions.api.AbstractAssert"), ANY_NAME).withAnyParameters(),
70      // rest assured 2.0
71      method(IO_RESTASSURED, NameCriteria.is("body")).withAnyParameters(),
72      method(IO_RESTASSURED, NameCriteria.is("time")).withAnyParameters(),
73      method(IO_RESTASSURED, NameCriteria.startsWith("content")).withAnyParameters(),
74      method(IO_RESTASSURED, NameCriteria.startsWith("status")).withAnyParameters(),
75      method(IO_RESTASSURED, NameCriteria.startsWith("header")).withAnyParameters(),
76      method(IO_RESTASSURED, NameCriteria.startsWith("cookie")).withAnyParameters(),
77      method(IO_RESTASSURED, NameCriteria.startsWith("spec")).withAnyParameters(),
78      // assertJ
79      method(TypeCriteria.subtypeOf("org.assertj.core.api.AbstractAssert"), ANY_NAME).withAnyParameters(),
80      // spring
81      method("org.springframework.test.web.servlet.ResultActions", "andExpect").addParameter(ANY_TYPE),
82      // JMockit
83      method("mockit.Verifications", "<init>").withAnyParameters());
84  
85    private static final MethodMatcherCollection REACTIVE_X_TEST_METHODS = MethodMatcherCollection.create(
86      method(TypeCriteria.subtypeOf("rx.Observable"), NameCriteria.is("test")).withAnyParameters(),
87      method(TypeCriteria.subtypeOf("io.reactivex.Observable"), NameCriteria.is("test")).withAnyParameters());

89    @RuleProperty(
90      key = "customAssertionMethods",
91      description = "Comma-separated list of fully qualified method symbols that should be considered as assertion methods. " +
92        "The wildcard character can be used at the end of the method name.",
93      defaultValue = "")
94    public String customAssertionMethods = "";
95    private MethodMatcherCollection customAssertionMethodsMatcher = null;
96  
97    private final Map<Symbol, Boolean> assertionInMethod = new HashMap<>();
98    private JavaFileScannerContext context;
99  
100   @Override
101   public void scanFile(final JavaFileScannerContext context) {
102     this.context = context;
103     assertionInMethod.clear();
104     scan(context.getTree());
105     assertionInMethod.clear();
106   }
107 
108   @Override
109   public void visitMethod(MethodTree methodTree) {
110     if (ModifiersUtils.hasModifier(methodTree.modifiers(), Modifier.ABSTRACT)) {
111       return;
112     }
113 
114     if (isUnitTest(methodTree) && !expectAssertion(methodTree) && !isLocalMethodWithAssertion(methodTree.symbol())) {
115       context.reportIssue(this, methodTree.simpleName(), "Add at least one assertion to this test case.");
116     }
117   }
118 
119   private boolean isLocalMethodWithAssertion(Symbol symbol) {
120     if (!assertionInMethod.containsKey(symbol)) {
121       assertionInMethod.put(symbol, false);
122       Tree declaration = symbol.declaration();
123       if (declaration != null) {
124         AssertionVisitor assertionVisitor = new AssertionVisitor(getCustomAssertionMethodsMatcher());
125         declaration.accept(assertionVisitor);
126         assertionInMethod.put(symbol, assertionVisitor.hasAssertion);
127       }
128     }
129 
130     return assertionInMethod.get(symbol);
131   }
132 
133   private MethodMatcherCollection getCustomAssertionMethodsMatcher() {
134     if (customAssertionMethodsMatcher == null) {
135       String[] fullyQualifiedMethodSymbols = customAssertionMethods.split(",");
136       List<MethodMatcher> customMethodMatchers = new ArrayList<>(fullyQualifiedMethodSymbols.length);
137       for (String fullyQualifiedMethodSymbol : fullyQualifiedMethodSymbols) {
138         String[] methodMatcherParts = fullyQualifiedMethodSymbol.split("#");
139         if (methodMatcherParts.length == 2 && !isEmpty(methodMatcherParts[0].trim()) && !isEmpty(methodMatcherParts[1].trim())) {
140           String methodName = methodMatcherParts[1].trim();
141           NameCriteria nameCriteria;
142           if (methodName.endsWith("*")) {
143             nameCriteria = NameCriteria.startsWith(methodName.substring(0, methodName.length() - 1));
144           } else {
145             nameCriteria = NameCriteria.is(methodName);
146           }
147           customMethodMatchers.add(method(methodMatcherParts[0].trim(), nameCriteria).withAnyParameters());
148         } else {
149           LOG.warn("Unable to create a corresponding matcher for custom assertion method, please check the format of the following symbol: '{}'", fullyQualifiedMethodSymbol);
150         }
151       }
152 
153       customAssertionMethodsMatcher = MethodMatcherCollection.create(customMethodMatchers.toArray(new MethodMatcher[0]));
154     }
155 
156     return customAssertionMethodsMatcher;
157   }
158 
159   private static boolean expectAssertion(MethodTree methodTree) {
160     List<SymbolMetadata.AnnotationValue> annotationValues = methodTree.symbol().metadata().valuesForAnnotation("org.junit.Test");
161     if (annotationValues != null) {
162       for (SymbolMetadata.AnnotationValue annotationValue : annotationValues) {
163         if ("expected".equals(annotationValue.name())) {
164           return true;
165         }
166       }
167     }
168     return false;
169   }
170 
171   private static boolean isUnitTest(MethodTree methodTree) {
172     Symbol.MethodSymbol symbol = methodTree.symbol();
173     while (symbol != null) {
174       if (symbol.metadata().isAnnotatedWith("org.junit.Test")) {
175         return true;
176       }
177       symbol = symbol.overriddenSymbol();
178     }
179 
180     if (hasJUnit5TestAnnotation(methodTree)) {
181       // contrary to JUnit 4, JUnit 5 Test annotations are not inherited when method is overridden, so no need to check overridden symbols
182       return true;
183     }
184     Symbol.TypeSymbol enclosingClass = methodTree.symbol().enclosingClass();
185     return enclosingClass != null && enclosingClass.type().isSubtypeOf("junit.framework.TestCase") && methodTree.simpleName().name().startsWith("test");
186   }

188   private static MethodMatcher method(String typeDefinition, String methodName) {
189     return method(TypeCriteria.is(typeDefinition), NameCriteria.is(methodName));
190   }
191 
192   private static MethodMatcher method(String typeDefinition, NameCriteria nameCriteria) {
193     return MethodMatcher.create().typeDefinition(TypeCriteria.is(typeDefinition)).name(nameCriteria);
194   }
195 
196   private static MethodMatcher method(TypeCriteria typeDefinitionCriteria, NameCriteria nameCriteria) {
197     return MethodMatcher.create().typeDefinition(typeDefinitionCriteria).name(nameCriteria);
198   }

200   private class AssertionVisitor extends BaseTreeVisitor {
201     boolean hasAssertion = false;
202     private MethodMatcherCollection customMethodsMatcher;
203 
204     private AssertionVisitor(MethodMatcherCollection customMethodsMatcher) {
205       this.customMethodsMatcher = customMethodsMatcher;
206     }
207 
208     @Override
209     public void visitMethodInvocation(MethodInvocationTree mit) {
210       super.visitMethodInvocation(mit);
211       if (!hasAssertion && isAssertion(methodName(mit), mit.symbol())) {
212         hasAssertion = true;
213       }
214     }
215 
216     @Override
217     public void visitMethodReference(MethodReferenceTree methodReferenceTree) {
218       super.visitMethodReference(methodReferenceTree);
219       if (!hasAssertion && isAssertion(methodReferenceTree.method(), methodReferenceTree.method().symbol())) {
220         hasAssertion = true;
221       }
222     }
223 
224     @Override
225     public void visitNewClass(NewClassTree tree) {
226       super.visitNewClass(tree);
227       if (!hasAssertion && isAssertion(null, tree.constructorSymbol())) {
228         hasAssertion = true;
229       }
230     }
231 
232     private boolean isAssertion(@Nullable IdentifierTree method, Symbol methodSymbol) {
233       return matchesMethodPattern(method, methodSymbol)
234         || ASSERTION_INVOCATION_MATCHERS.anyMatch(methodSymbol)
235         || customMethodsMatcher.anyMatch(methodSymbol)
236         || isLocalMethodWithAssertion(methodSymbol);
237     }
238 
239     private boolean matchesMethodPattern(@Nullable IdentifierTree method, Symbol methodSymbol) {
240       if (method == null) {
241         return false;
242       }
243 
244       String methodName = method.name();
245       if (TEST_METHODS_PATTERN.matcher(methodName).matches()) {
246         return !REACTIVE_X_TEST_METHODS.anyMatch(methodSymbol);
247       }
248       return ASSERTION_METHODS_PATTERN.matcher(methodName).matches();
249     }
250   }

48  @Rule(key = "UselessImportCheck")
49  @RspecKey("S1128")
50  public class UselessImportCheck extends BaseTreeVisitor implements JavaFileScanner {
51  
52    private final Map<String, ImportTree> lineByImportReference = new HashMap<>();
53    private final Set<String> pendingImports = new HashSet<>();
54    private final Set<String> pendingReferences = new HashSet<>();
55  
56    private String currentPackage;
57    private JavaFileScannerContext context;
58  
59    @Override
60    public void scanFile(JavaFileScannerContext context) {
61      this.context = context;
62      CompilationUnitTree cut = context.getTree();
63  
64      if (cut.moduleDeclaration() != null) {
65        // skip module declarations as long as semantic is not resolved correctly.
66        // imports can be used for types used in module directive or annotations
67        return;
68      }
69  
70      ExpressionTree packageName = getPackageName(cut);
71  
72      pendingReferences.clear();
73      lineByImportReference.clear();
74      pendingImports.clear();
75  
76      currentPackage = ExpressionsHelper.concatenate(packageName);
77      for (ImportClauseTree importClauseTree : cut.imports()) {
78        ImportTree importTree = null;
79  
80        if (importClauseTree.is(Tree.Kind.IMPORT)) {
81          importTree = (ImportTree) importClauseTree;
82        }
83  
84        if (importTree == null) {
85          // discard empty statements, which can be part of imports
86          continue;
87        }
88  
89        reportIssue(importTree);
90      }
91      //check references
92      scan(cut);
93      //check references from comments.
94      new CommentVisitor().checkImportsFromComments(cut, pendingImports);
95      leaveFile();
96    }
97  
98    private void reportIssue(ImportTree importTree) {
99      String importName = ExpressionsHelper.concatenate((ExpressionTree) importTree.qualifiedIdentifier());
100     if ("java.lang.*".equals(importName)) {
101       context.reportIssue(this, importTree, "Remove this unnecessary import: java.lang classes are always implicitly imported.");
102     } else if (isImportFromSamePackage(importName)) {
103       context.reportIssue(this, importTree, "Remove this unnecessary import: same package classes are always implicitly imported.");
104     } else if (!isImportOnDemand(importName)) {
105       if (isJavaLangImport(importName)) {
106         context.reportIssue(this, importTree, "Remove this unnecessary import: java.lang classes are always implicitly imported.");
107       } else if (isDuplicatedImport(importName)) {
108         context.reportIssue(this, importTree, "Remove this duplicated import.");
109       } else if(importTree.isStatic()) {
110         checkSymbolUsage(importTree, importName);
111       } else {
112         lineByImportReference.put(importName, importTree);
113         pendingImports.add(importName);
114       }
115     }
116   }
117 
118   private void checkSymbolUsage(ImportTree importTree, String importName) {
119     IdentifierTree id = null;
120     if (importTree.qualifiedIdentifier().is(Tree.Kind.IDENTIFIER)) {
121       id = (IdentifierTree) importTree.qualifiedIdentifier();
122     } else if (importTree.qualifiedIdentifier().is(Tree.Kind.MEMBER_SELECT)) {
123       id = ((MemberSelectExpressionTree) importTree.qualifiedIdentifier()).identifier();
124     }
125     if (id != null) {
126       SemanticModel semanticModel = (SemanticModel) context.getSemanticModel();
127       if (semanticModel != null) {
128         Symbol symbol = semanticModel.getSymbol(importTree);
129         if (symbol != null) {
130           Symbol owner = symbol.owner();
131           // Exclude method symbols : they could be ambiguous or unresolved and lead to FP.
132           if (symbol.isVariableSymbol() && symbol.usages().stream().allMatch(identifierTree -> semanticModel.getEnclosingClass(identifierTree) == owner)) {
133             context.reportIssue(this, importTree, "Remove this unused import '" + importName + "'.");
134           }
135         }
136       }
137     }
138   }
139 
140   private static ExpressionTree getPackageName(CompilationUnitTree cut) {
141     return cut.packageDeclaration() != null ? cut.packageDeclaration().packageName() : null;
142   }
143 
144   private static boolean isImportOnDemand(String name) {
145     return name.endsWith("*");
146   }
147 
148   @Override
149   public void visitCompilationUnit(CompilationUnitTree tree) {
150     //do not scan imports and package name identifiers.
151     if (tree.packageDeclaration() != null) {
152       scan(tree.packageDeclaration().annotations());
153     }
154     scan(tree.types());
155   }
156 
157   @Override
158   public void visitIdentifier(IdentifierTree tree) {
159     scan(tree.annotations());
160     pendingReferences.add(tree.name());
161   }
162 
163   @Override
164   public void visitArrayType(ArrayTypeTree tree) {
165     scan(tree.annotations());
166     super.visitArrayType(tree);
167   }
168 
169   @Override
170   public void visitMemberSelectExpression(MemberSelectExpressionTree tree) {
171     scan(tree.annotations());
172     scan(tree.identifier().annotations());
173     pendingReferences.add(ExpressionsHelper.concatenate(tree));
174     //Don't visit identifiers of a member select expression.
175     if (!tree.expression().is(Tree.Kind.IDENTIFIER)) {
176       scan(tree.expression());
177     }
178   }
179 
180   private boolean isImportFromSamePackage(String reference) {
181     String importName = reference;
182     if (isImportOnDemand(reference)) {
183       //strip out .* to compare length with current package.
184       importName = reference.substring(0, reference.length() - 2);
185     }
186     return !currentPackage.isEmpty()
187       && (importName.equals(currentPackage)
188           || (reference.startsWith(currentPackage) && reference.charAt(currentPackage.length()) == '.') && reference.indexOf('.', currentPackage.length() + 1) == -1);
189   }
190 
191   private boolean isDuplicatedImport(String reference) {
192     return pendingImports.contains(reference);
193   }
194 
195   private static boolean isJavaLangImport(String reference) {
196     return reference.startsWith("java.lang.") && reference.indexOf('.', "java.lang.".length()) == -1;
197   }
198 
199   public void leaveFile() {
200     for (String reference : pendingReferences) {
201       updatePendingImports(reference);
202     }
203 
204     for (String pendingImport : pendingImports) {
205       context.reportIssue(this, lineByImportReference.get(pendingImport), "Remove this unused import '" + pendingImport + "'.");
206     }
207   }
208 
209   private void updatePendingImports(String reference) {
210     String firstClassReference = reference;
211     if (isFullyQualified(firstClassReference)) {
212       firstClassReference = extractFirstClassName(firstClassReference);
213     }
214     Iterator<String> it = pendingImports.iterator();
215     while (it.hasNext()) {
216       String pendingImport = it.next();
217       if (pendingImport.endsWith("." + firstClassReference)) {
218         it.remove();
219       }
220     }
221   }
222 
223   private static boolean isFullyQualified(String reference) {
224     return reference.indexOf('.') != -1;
225   }
226 
227   private static String extractFirstClassName(String reference) {
228     int firstIndexOfDot = reference.indexOf('.');
229     return firstIndexOfDot == -1 ? reference : reference.substring(0, firstIndexOfDot);
230   }
231 
232   private static class CommentVisitor extends SubscriptionVisitor {
233     private Set<String> pendingImports;
234 
235     @Override
236     public List<Tree.Kind> nodesToVisit() {
237       return Collections.singletonList(Tree.Kind.TRIVIA);
238     }
239 
240     public void checkImportsFromComments(CompilationUnitTree cut, Set<String> pendingImports) {
241       this.pendingImports = pendingImports;
242       scanTree(cut);
243     }
244 
245     @Override
246     public void visitTrivia(SyntaxTrivia syntaxTrivia) {
247       updatePendingImportsForComments(syntaxTrivia.comment());
248     }
249 
250     private void updatePendingImportsForComments(String comment) {
251       pendingImports.removeIf(pendingImport -> comment.contains(extractLastClassName(pendingImport)));
252     }
253 
254     private static String extractLastClassName(String reference) {
255       int lastIndexOfDot = reference.lastIndexOf('.');
256       return lastIndexOfDot == -1 ? reference : reference.substring(lastIndexOfDot + 1);
257     }

259   }

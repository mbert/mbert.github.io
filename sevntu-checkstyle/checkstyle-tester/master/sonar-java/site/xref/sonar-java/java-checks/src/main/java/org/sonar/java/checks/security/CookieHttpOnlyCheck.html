50  @Rule(key = "S3330")
51  public class CookieHttpOnlyCheck extends IssuableSubscriptionVisitor {
52    private final List<Symbol.VariableSymbol> compliantConstructorInitializations = new ArrayList<>();
53    private final List<Symbol.VariableSymbol> ignoredVariables = new ArrayList<>();
54    private final List<Symbol.VariableSymbol> variablesToReport = new ArrayList<>();
55    private final List<MethodInvocationTree> settersToReport = new ArrayList<>();
56    private final List<NewClassTree> newClassToReport = new ArrayList<>();

111   @Override
112   public void setContext(JavaFileScannerContext context) {
113     compliantConstructorInitializations.clear();
114     ignoredVariables.clear();
115     variablesToReport.clear();
116     settersToReport.clear();
117     newClassToReport.clear();
118     super.setContext(context);
119   }
120 
121   @Override
122   public void leaveFile(JavaFileScannerContext context) {
123     for (VariableSymbol var : variablesToReport) {
124       VariableTree declaration = var.declaration();
125       if (declaration != null) {
126         reportIssue(declaration.simpleName(), MESSAGE);
127       }
128     }
129     for (MethodInvocationTree mit : settersToReport) {
130       reportIssue(mit.arguments(), MESSAGE);
131     }
132     for (NewClassTree newClassTree : newClassToReport) {
133       reportIssue(newClassTree, MESSAGE);
134     }
135   }

137   @Override
138   public List<Tree.Kind> nodesToVisit() {
139     return Arrays.asList(
140       Tree.Kind.VARIABLE,
141       Tree.Kind.ASSIGNMENT,
142       Tree.Kind.METHOD_INVOCATION,
143       Tree.Kind.RETURN_STATEMENT);
144   }
145 
146   @Override
147   public void visitNode(Tree tree) {
148     if (hasSemantic()) {
149       if (tree.is(Tree.Kind.VARIABLE)) {
150         checkVariableDeclaration((VariableTree) tree);
151       } else if (tree.is(Tree.Kind.ASSIGNMENT)) {
152         checkAssignment((AssignmentExpressionTree) tree);
153       } else if (tree.is(Tree.Kind.METHOD_INVOCATION)) {
154         checkSetterInvocation((MethodInvocationTree) tree);
155       } else {
156         categorizeBasedOnConstructor((ReturnStatementTree) tree);
157       }
158     }
159   }
160 
161   private void checkAssignment(AssignmentExpressionTree assignment) {
162     checkCookieBuilder(assignment);
163     if (shouldVerify(assignment)) {
164       categorizeBasedOnConstructor((NewClassTree) assignment.expression(),
165         (VariableSymbol) ((IdentifierTree) assignment.variable()).symbol());
166     }
167   }
168 
169   private void checkVariableDeclaration(VariableTree declaration) {
170     checkCookieBuilder(declaration);
171     if (shouldVerify(declaration)) {
172       categorizeBasedOnConstructor((NewClassTree) declaration.initializer(),
173         (VariableSymbol) declaration.symbol());
174     }
175   }
176 
177   private void checkCookieBuilder(AssignmentExpressionTree assignment) {
178     if (assignment.expression().is(Tree.Kind.METHOD_INVOCATION)
179       && (assignment.variable().is(Tree.Kind.IDENTIFIER) || assignment.variable().is(Tree.Kind.MEMBER_SELECT))) {
180       MethodInvocationTree mit = (MethodInvocationTree) assignment.expression();
181       VariableSymbol variableSymbol = getVariableSymbol(assignment);
182       if (variableSymbol != null) {
183         addToIgnoredVariables(variableSymbol, mit);
184       }
185     }
186   }
187 
188   @CheckForNull
189   private static VariableSymbol getVariableSymbol(AssignmentExpressionTree assignment) {
190     VariableSymbol variableSymbol = null;
191     if (assignment.variable().is(Tree.Kind.IDENTIFIER)) {
192       Symbol reference = ((IdentifierTree) assignment.variable()).symbol();
193       if (reference.isVariableSymbol()) {
194         variableSymbol = (VariableSymbol) reference;
195       }
196     } else {
197       MemberSelectExpressionTree mse = (MemberSelectExpressionTree) assignment.variable();
198       if (mse.identifier().symbol().isVariableSymbol()) {
199         variableSymbol = (VariableSymbol) mse.identifier().symbol();
200       }
201     }
202     return variableSymbol;
203   }
204 
205   private void addToIgnoredVariables(VariableSymbol variableSymbol, MethodInvocationTree mit) {
206     if (PLAY_COOKIE_BUILDER.matches(mit) && isIgnoredCookieName(mit.arguments())) {
207       ignoredVariables.add(variableSymbol);
208     }
209   }
210 
211   private void checkCookieBuilder(VariableTree declaration) {
212     if (declaration.initializer() != null && declaration.initializer().is(Tree.Kind.METHOD_INVOCATION)) {
213       MethodInvocationTree mit = (MethodInvocationTree) declaration.initializer();
214       addToIgnoredVariables((VariableSymbol) declaration.symbol(), mit);
215     }
216   }
217 
218   private void categorizeBasedOnConstructor(ReturnStatementTree returnStatement) {
219     ExpressionTree returnedExpression = returnStatement.expression();
220     if (returnedExpression != null && returnedExpression.is(Tree.Kind.NEW_CLASS)) {
221       NewClassTree newClass = (NewClassTree) returnedExpression;
222       if (!isIgnoredCookieName(newClass.arguments()) && !isCompliantConstructorCall(newClass) && CLASSES.stream().anyMatch(newClass.symbolType()::isSubtypeOf)) {
223         newClassToReport.add(newClass);
224       }
225     }
226   }
227 
228   private void categorizeBasedOnConstructor(NewClassTree newClassTree, VariableSymbol variableSymbol) {
229     if (isIgnoredCookieName(newClassTree.arguments())) {
230       ignoredVariables.add(variableSymbol);
231     } else if (isCompliantConstructorCall(newClassTree)) {
232       compliantConstructorInitializations.add(variableSymbol);
233     } else {
234       variablesToReport.add(variableSymbol);
235     }
236   }
237 
238   private static boolean shouldVerify(VariableTree variableDeclaration) {
239     ExpressionTree initializer = variableDeclaration.initializer();
240     if (initializer != null && initializer.is(Tree.Kind.NEW_CLASS)) {
241       boolean isSupportedClass = CLASSES.stream().anyMatch(variableDeclaration.type().symbolType()::isSubtypeOf)
242         || CLASSES.stream().anyMatch(initializer.symbolType()::isSubtypeOf);
243       return variableDeclaration.symbol().owner().isMethodSymbol() && isSupportedClass;
244     }
245     return false;
246   }
247 
248   private static boolean shouldVerify(AssignmentExpressionTree assignment) {
249     if (assignment.expression().is(Tree.Kind.NEW_CLASS) && assignment.variable().is(Tree.Kind.IDENTIFIER)) {
250       IdentifierTree identifier = (IdentifierTree) assignment.variable();
251       boolean isMethodVariable = identifier.symbol().isVariableSymbol()
252         && identifier.symbol().owner().isMethodSymbol();
253       boolean isSupportedClass = CLASSES.stream().anyMatch(identifier.symbolType()::isSubtypeOf)
254         || CLASSES.stream().anyMatch(assignment.expression().symbolType()::isSubtypeOf);
255       return isMethodVariable && isSupportedClass;
256     }
257     return false;
258   }
259 
260   private static boolean isCompliantConstructorCall(NewClassTree newClassTree) {
261     if (CONSTRUCTORS_WITH_HTTP_ONLY_PARAM.stream().anyMatch(matcher -> matcher.matches(newClassTree))) {
262       Arguments arguments = newClassTree.arguments();
263       ExpressionTree lastArgument = arguments.get(arguments.size() - 1);
264       return LiteralUtils.isTrue(lastArgument);
265     } else {
266       return CONSTRUCTORS_WITH_GOOD_DEFAULT.stream().anyMatch(matcher -> matcher.matches(newClassTree));
267     }
268   }
269 
270   private static boolean isIgnoredCookieName(Arguments arguments) {
271     if (arguments.isEmpty()) {
272       return false;
273     }
274     ExpressionTree nameArgument = arguments.get(COOKIE_NAME_ARGUMENT);
275     String name = ConstantUtils.resolveAsStringConstant(nameArgument);
276     return name != null && IGNORED_COOKIE_NAMES.stream().anyMatch(cookieName -> name.toLowerCase(Locale.ENGLISH).contains(cookieName));
277   }

279   private void checkSetterInvocation(MethodInvocationTree mit) {
280     if (isExpectedSetter(mit)) {
281       if (mit.methodSelect().is(Tree.Kind.MEMBER_SELECT)) {
282         ExpressionTree expression = ((MemberSelectExpressionTree) mit.methodSelect()).expression();
283         boolean isCalledOnIdentifier = expression.is(Tree.Kind.IDENTIFIER);
284         boolean isCalledOnMemberSelect = expression.is(Tree.Kind.MEMBER_SELECT);
285         if (isCalledOnIdentifier || isCalledOnMemberSelect) {
286           updateIssuesToReport(mit);
287         } else if (!setterArgumentHasCompliantValue(mit.arguments())) {
288           // builder method
289           settersToReport.add(mit);
290         }
291       } else if (!setterArgumentHasCompliantValue(mit.arguments())) {
292         // sub-class method
293         settersToReport.add(mit);
294       }
295     }
296   }
297 
298   private static boolean isExpectedSetter(MethodInvocationTree mit) {
299     return mit.arguments().size() == 1
300       && mit.symbol().isMethodSymbol()
301       && CLASSES.stream().anyMatch(mit.symbol().owner().type()::isSubtypeOf)
302       && SETTER_NAMES.contains(getIdentifier(mit).name())
303       && isIgnoredBuilder(mit);
304   }
305 
306   private static boolean isIgnoredBuilder(MethodInvocationTree mit) {
307     if (!mit.symbol().owner().type().isSubtypeOf(ClassName.PLAY_COOKIE_BUILDER)) {
308       return true;
309     }
310     return getMethodChain(mit)
311       .filter(method -> "builder".contains(getIdentifier(method).name()))
312       .noneMatch(method -> isIgnoredCookieName(method.arguments()));
313   }
314 
315   private static Stream<MethodInvocationTree> getMethodChain(MethodInvocationTree mit) {
316     if (mit.methodSelect().is(Tree.Kind.MEMBER_SELECT)) {
317       ExpressionTree expressionTree = ((MemberSelectExpressionTree) mit.methodSelect()).expression();
318       if (expressionTree.is(Tree.Kind.METHOD_INVOCATION)) {
319         return Stream.concat(Stream.of(mit), getMethodChain((MethodInvocationTree) expressionTree));
320       }
321     }
322     return Stream.of(mit);
323   }
324 
325   private void updateIssuesToReport(MethodInvocationTree mit) {
326     MemberSelectExpressionTree mse = (MemberSelectExpressionTree) mit.methodSelect();
327     VariableSymbol reference;
328     if (mse.expression().is(Tree.Kind.IDENTIFIER)) {
329       reference = (VariableSymbol) ((IdentifierTree) mse.expression()).symbol();
330     } else {
331       reference = (VariableSymbol) ((MemberSelectExpressionTree) mse.expression()).identifier().symbol();
332     }
333     if (ignoredVariables.contains(reference)) {
334       // ignore XSRF-TOKEN cookies
335       return;
336     }
337     if (setterArgumentHasCompliantValue(mit.arguments())) {
338       variablesToReport.remove(reference);
339     } else if (compliantConstructorInitializations.contains(reference)) {
340       variablesToReport.add(reference);
341     } else if (!variablesToReport.contains(reference)) {
342       settersToReport.add(mit);
343     }
344   }
345 
346   private static boolean setterArgumentHasCompliantValue(Arguments arguments) {
347     ExpressionTree expressionTree = arguments.get(0);
348     Boolean booleanValue = ConstantUtils.resolveAsBooleanConstant(expressionTree);
349     return booleanValue == null || booleanValue;
350   }
351 
352   private static IdentifierTree getIdentifier(MethodInvocationTree mit) {
353     IdentifierTree id;
354     if (mit.methodSelect().is(Tree.Kind.IDENTIFIER)) {
355       id = (IdentifierTree) mit.methodSelect();
356     } else {
357       id = ((MemberSelectExpressionTree) mit.methodSelect()).identifier();
358     }
359     return id;
360   }

20  package org.sonar.java.checks;
21  
22  import org.sonar.check.Rule;
23  import org.sonar.java.matcher.MethodMatcher;
24  import org.sonar.java.matcher.TypeCriteria;
25  import org.sonar.java.model.ExpressionUtils;
26  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
27  import org.sonar.plugins.java.api.tree.BinaryExpressionTree;
28  import org.sonar.plugins.java.api.tree.ExpressionTree;
29  import org.sonar.plugins.java.api.tree.IdentifierTree;
30  import org.sonar.plugins.java.api.tree.LiteralTree;
31  import org.sonar.plugins.java.api.tree.MemberSelectExpressionTree;
32  import org.sonar.plugins.java.api.tree.MethodInvocationTree;
33  import org.sonar.plugins.java.api.tree.Tree;
34  
35  import java.util.Arrays;
36  import java.util.List;
37  
38  @Rule(key = "S3981")
39  public class CollectionSizeAndArrayLengthCheck extends IssuableSubscriptionVisitor {
40  
41    private static final MethodMatcher COLLECTION_SIZE = MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf("java.util.Collection")).name("size").withoutParameter();
42    private static final String COLLECTION_ISSUE_MSG = "The size of %s is always \">=0\", so update this test to use isEmpty().";
43    private static final String ARRAY_ISSUE_MSG = "The length of %s is always \">=0\", so update this test to either \"==0\" or \">0\".";
44  
45    @Override
46    public List<Tree.Kind> nodesToVisit() {
47      return Arrays.asList(
48        Tree.Kind.GREATER_THAN_OR_EQUAL_TO,
49        Tree.Kind.GREATER_THAN,
50        Tree.Kind.LESS_THAN,
51        Tree.Kind.LESS_THAN_OR_EQUAL_TO);
52    }
53  
54    @Override
55    public void visitNode(Tree tree) {
56      if (!hasSemantic()) {
57        return;
58      }
59      BinaryExpressionTree bet = (BinaryExpressionTree) tree;
60      ExpressionTree leftOperand = ExpressionUtils.skipParentheses(bet.leftOperand());
61      ExpressionTree rightOperand = ExpressionUtils.skipParentheses(bet.rightOperand());
62      boolean leftIsZero = isZero(leftOperand);
63      boolean rightIsZero = isZero(rightOperand);
64      if (!leftIsZero && !rightIsZero) {
65        return;
66      }
67      ExpressionTree testedValue = leftIsZero ? rightOperand : leftOperand;
68      if (testedValue.is(Tree.Kind.METHOD_INVOCATION)) {
69        checkCollectionSize((MethodInvocationTree) testedValue, bet, leftIsZero);
70      } else if (testedValue.is(Tree.Kind.MEMBER_SELECT)) {
71        checkArrayLength((MemberSelectExpressionTree) testedValue, bet, leftIsZero);
72      }
73    }
74  
75    private static boolean isZero(ExpressionTree expr) {
76      return expr.is(Tree.Kind.INT_LITERAL) && "0".equals(((LiteralTree) expr).value());
77    }
78  
79    private void checkCollectionSize(MethodInvocationTree testedValue, BinaryExpressionTree bet, boolean leftIsZero) {
80      if (!COLLECTION_SIZE.matches(testedValue)) {
81        return;
82      }
83      reportIssue(bet, leftIsZero, COLLECTION_ISSUE_MSG, collectionName(testedValue.methodSelect()));
84    }
85  
86    private static String collectionName(ExpressionTree sizeMethodSelect) {
87      if (sizeMethodSelect.is(Tree.Kind.MEMBER_SELECT)) {
88        ExpressionTree collectionAccess = ((MemberSelectExpressionTree) sizeMethodSelect).expression();
89        if (collectionAccess.is(Tree.Kind.IDENTIFIER)) {
90          return escape(((IdentifierTree) collectionAccess).name());
91        }
92      }
93      return "a collection";
94    }
95  
96    private static String escape(String s) {
97      return "\"" + s + "\"";
98    }
99  
100   private void reportIssue(BinaryExpressionTree bet, boolean leftIsZero, String message, String itemName) {
101     if ((leftIsZero && bet.is(Tree.Kind.GREATER_THAN, Tree.Kind.LESS_THAN_OR_EQUAL_TO))
102       || (!leftIsZero && bet.is(Tree.Kind.LESS_THAN, Tree.Kind.GREATER_THAN_OR_EQUAL_TO))) {
103       reportIssue(bet, String.format(message, itemName));
104     }
105   }
106 
107   private void checkArrayLength(MemberSelectExpressionTree testedValue, BinaryExpressionTree bet, boolean leftIsZero) {
108     if (!"length".equals(testedValue.identifier().name())) {
109       return;
110     }
111     ExpressionTree expression = testedValue.expression();
112     if (!expression.symbolType().isArray()) {
113       return;
114     }
115     reportIssue(bet, leftIsZero, ARRAY_ISSUE_MSG, arrayName(expression));
116   }
117 
118   private static String arrayName(ExpressionTree expr) {
119     if (expr.is(Tree.Kind.IDENTIFIER)) {
120       return escape(((IdentifierTree) expr).name());
121     }
122     if (expr.is(Tree.Kind.MEMBER_SELECT)) {
123       return escape(((MemberSelectExpressionTree) expr).identifier().name());
124     }
125     return "an array";
126   }

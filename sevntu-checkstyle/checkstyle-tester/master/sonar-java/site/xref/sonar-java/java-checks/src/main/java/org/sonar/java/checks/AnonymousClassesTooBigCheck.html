20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.Lists;
23  import org.sonar.check.Rule;
24  import org.sonar.check.RuleProperty;
25  import org.sonar.java.ast.visitors.LinesOfCodeVisitor;
26  import org.sonar.plugins.java.api.JavaFileScanner;
27  import org.sonar.plugins.java.api.JavaFileScannerContext;
28  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
29  import org.sonar.plugins.java.api.tree.EnumConstantTree;
30  import org.sonar.plugins.java.api.tree.LambdaExpressionTree;
31  import org.sonar.plugins.java.api.tree.NewClassTree;
32  import org.sonar.plugins.java.api.tree.SyntaxToken;
33  import org.sonar.plugins.java.api.tree.Tree;
34  
35  @Rule(key = "S1188")
36  public class AnonymousClassesTooBigCheck extends BaseTreeVisitor implements JavaFileScanner {
37  
38    private static final int DEFAULT_MAX = 20;
39  
40    @RuleProperty(key = "Max",
41      description = "Maximum allowed lines in an anonymous class/lambda",
42      defaultValue = "" + DEFAULT_MAX)
43    public int max = DEFAULT_MAX;
44  
45    private JavaFileScannerContext context;
46    /**
47     * Flag to skip check for class bodies of EnumConstants.
48     */
49    private boolean isEnumConstantBody;
50  
51    @Override
52    public void scanFile(JavaFileScannerContext context) {
53      this.context = context;
54      isEnumConstantBody = false;
55      scan(context.getTree());
56    }
57  
58    @Override
59    public void visitNewClass(NewClassTree tree) {
60      if (tree.classBody() != null && !isEnumConstantBody) {
61        int lines = getNumberOfLines(tree.classBody());
62        if (lines > max) {
63          context.reportIssue(this, tree.newKeyword(), tree.identifier(),
64            "Reduce this anonymous class number of lines from " + lines + " to at most " + max + ", or make it a named class.");
65        }
66      }
67      isEnumConstantBody = false;
68      super.visitNewClass(tree);
69    }
70  
71    @Override
72    public void visitEnumConstant(EnumConstantTree tree) {
73      isEnumConstantBody = true;
74      super.visitEnumConstant(tree);
75    }
76  
77    @Override
78    public void visitLambdaExpression(LambdaExpressionTree lambdaExpressionTree) {
79      int lines = getNumberOfLines(lambdaExpressionTree);
80      if (lines > max) {
81        SyntaxToken firstToken = lambdaExpressionTree.firstToken();
82        SyntaxToken lastSyntaxToken = lambdaExpressionTree.lastToken();
83        JavaFileScannerContext.Location lastTokenLocation = new JavaFileScannerContext.Location(lines + " lines", lastSyntaxToken);
84        context.reportIssue(this, firstToken, lambdaExpressionTree.arrowToken(),
85          "Reduce this lambda expression number of lines from " + lines + " to at most " + max + ".", Lists.newArrayList(lastTokenLocation), null);
86      }
87      super.visitLambdaExpression(lambdaExpressionTree);
88    }
89  
90    private static int getNumberOfLines(Tree tree) {
91      return new LinesOfCodeVisitor().linesOfCode(tree);
92    }
93  }

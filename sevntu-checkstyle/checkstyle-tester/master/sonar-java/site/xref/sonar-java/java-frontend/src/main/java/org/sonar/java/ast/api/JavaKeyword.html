20  package org.sonar.java.ast.api;
21  
22  import org.sonar.sslr.grammar.GrammarRuleKey;
23  
24  /**
25   * Keywords for java grammar.
26   */
27  public enum JavaKeyword implements GrammarRuleKey {
28  
29    ASSERT("assert"),
30    BREAK("break"),
31    CASE("case"),
32    CATCH("catch"),
33    CLASS("class"),
34    CONTINUE("continue"),
35    DEFAULT("default"),
36    DO("do"),
37    ELSE("else"),
38    ENUM("enum"),
39    EXTENDS("extends"),
40    FINALLY("finally"),
41    FINAL("final"),
42    FOR("for"),
43    IF("if"),
44    IMPLEMENTS("implements"),
45    IMPORT("import"),
46    INTERFACE("interface"),
47    INSTANCEOF("instanceof"),
48    NEW("new"),
49    PACKAGE("package"),
50    RETURN("return"),
51    STATIC("static"),
52    SUPER("super"),
53    SWITCH("switch"),
54    SYNCHRONIZED("synchronized"),
55    THIS("this"),
56    THROWS("throws"),
57    THROW("throw"),
58    TRY("try"),
59    VOID("void"),
60    WHILE("while"),
61    TRUE("true"),
62    FALSE("false"),
63    NULL("null"),
64    PUBLIC("public"),
65    PROTECTED("protected"),
66    PRIVATE("private"),
67    ABSTRACT("abstract"),
68    NATIVE("native"),
69    TRANSIENT("transient"),
70    VOLATILE("volatile"),
71    STRICTFP("strictfp"),
72    BYTE("byte"),
73    SHORT("short"),
74    CHAR("char"),
75    INT("int"),
76    LONG("long"),
77    FLOAT("float"),
78    DOUBLE("double"),
79    BOOLEAN("boolean");
80  
81    private final String value;
82  
83    JavaKeyword(String value) {
84      this.value = value;
85    }
86  
87    public String getName() {
88      return name();
89    }
90  
91    public String getValue() {
92      return value;
93    }
94  
95    /**
96     * keywords as String.
97     * @return an array containing all keywords as typed in Java
98     */
99    public static String[] keywordValues() {
100     JavaKeyword[] keywordsEnum = JavaKeyword.values();
101     String[] keywords = new String[keywordsEnum.length];
102     for (int i = 0; i < keywords.length; i++) {
103       keywords[i] = keywordsEnum[i].getValue();
104     }
105     return keywords;
106   }
107 
108 }

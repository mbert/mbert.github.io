41  /**
42   * Completes hierarchy of types.
43   */
44  public class SecondPass implements JavaSymbol.Completer {

46    private static final String CONSTRUCTOR_NAME = "<init>";
47  
48    private final SemanticModel semanticModel;
49    private final Symbols symbols;
50    private final TypeAndReferenceSolver typeAndReferenceSolver;
51    private final ParametrizedTypeCache parametrizedTypeCache;
52  
53    public SecondPass(SemanticModel semanticModel, Symbols symbols, ParametrizedTypeCache parametrizedTypeCache, TypeAndReferenceSolver typeAndReferenceSolver) {
54      this.semanticModel = semanticModel;
55      this.symbols = symbols;
56      this.parametrizedTypeCache = parametrizedTypeCache;
57      this.typeAndReferenceSolver = typeAndReferenceSolver;
58    }
59  
60    @Override
61    public void complete(JavaSymbol symbol) {
62      if (symbol.kind == JavaSymbol.TYP) {
63        complete((JavaSymbol.TypeJavaSymbol) symbol);
64      } else if (symbol.kind == JavaSymbol.MTH) {
65        complete((JavaSymbol.MethodJavaSymbol) symbol);
66      } else if (symbol.kind == JavaSymbol.VAR) {
67        complete((JavaSymbol.VariableJavaSymbol) symbol);
68      } else {
69        throw new IllegalArgumentException();
70      }
71    }
72  
73    private void complete(JavaSymbol.TypeJavaSymbol symbol) {
74      Resolve.Env env = semanticModel.getEnv(symbol);
75      ClassJavaType type = (ClassJavaType) symbol.type;
76  
77      if (!symbol.isFlag(Flags.ANNOTATION)) {
78        // JLS8 15.8.3 If this is a class or interface (default methods), enter symbol for "this"
79        symbol.members.enter(new JavaSymbol.VariableJavaSymbol(Flags.FINAL, "this", symbol.type, symbol));
80      }
81  
82      if ("".equals(symbol.name)) {
83        // Anonymous Class Declaration
84        // FIXME(Godin): This case avoids NPE which occurs because semanticModel has no associations for anonymous classes.
85        type.interfaces = Collections.emptyList();
86        return;
87      }
88  
89      ClassTree tree = symbol.declaration;
90      completeTypeParameters(tree.typeParameters(), env);
91  
92      //Interfaces
93      ImmutableList.Builder<JavaType> interfaces = ImmutableList.builder();
94      tree.superInterfaces().stream().map(interfaceTree -> resolveType(env, interfaceTree)).filter(Objects::nonNull).forEach(interfaces::add);
95  
96      if (tree.is(Tree.Kind.ANNOTATION_TYPE)) {
97        // JLS8 9.6: The direct superinterface of every annotation type is java.lang.annotation.Annotation.
98        // (Godin): Note that "extends" and "implements" clauses are forbidden by grammar for annotation types
99        interfaces.add(symbols.annotationType);
100     }
101 
102     if (tree.is(Tree.Kind.ENUM, Tree.Kind.INTERFACE) && symbol.owner.isKind(JavaSymbol.TYP)) {
103       // JSL8 8.9: A nested enum type is implicitly static. It is permitted for the declaration of a nested 
104       // enum type to redundantly specify the static modifier.
105       symbol.flags |= Flags.STATIC;
106     }
107 
108     type.interfaces = interfaces.build();
109 
110     populateSuperclass(symbol, env, type);
111 
112     if ((symbol.flags() & Flags.INTERFACE) == 0) {
113       symbol.members.enter(new JavaSymbol.VariableJavaSymbol(Flags.FINAL, "super", type.supertype, symbol));
114     } else {
115       // JLS9 - 15.12.1 : Used in form 'TypeName.super.foo()', where 'TypeName' is an interface. To support invocation
116       // of default methods from super-interfaces, 'TypeName' may also refer to a direct super-interface of the current
117       // class or interface. The method being invoked ('foo()') has to be searched in that super-interface.
118       symbol.members.enter(new JavaSymbol.VariableJavaSymbol(Flags.FINAL, "super", type, symbol));
119       // Note: The above "super" symbol will always be qualified when referenced. e.g. A.super.hashCode()
120       // because it's a compilation error to use unqualified "super" in default method. e.g. super.hashCode()
121       // Note: interface/class can extend/implement multiple interfaces containing default methods with the same
122       // signature. Mentioning the super-interfaces explicitly removes any ambiguity.
123     }
124 
125     // Register default constructor
126     if (tree.is(Tree.Kind.CLASS, Tree.Kind.ENUM) && symbol.lookupSymbols(CONSTRUCTOR_NAME).isEmpty()) {
127       List<JavaType> argTypes = Collections.emptyList();
128       if (!symbol.isStatic()) {
129         // JLS8 - 8.8.1 & 8.8.9 : constructors of inner class have an implicit first arg of its directly enclosing class type
130         JavaSymbol owner = symbol.owner();
131         if (!owner.isPackageSymbol()) {
132           argTypes = Collections.singletonList(Objects.requireNonNull(owner.enclosingClass().type));
133         }
134       }
135       JavaSymbol.MethodJavaSymbol defaultConstructor = new JavaSymbol.MethodJavaSymbol(symbol.flags & Flags.ACCESS_FLAGS, CONSTRUCTOR_NAME, symbol);
136       MethodJavaType defaultConstructorType = new MethodJavaType(argTypes, null, Collections.emptyList(), symbol);
137       defaultConstructor.setMethodType(defaultConstructorType);
138       defaultConstructor.parameters = new Scope(defaultConstructor);
139       symbol.members.enter(defaultConstructor);
140     }
141   }

143   private void populateSuperclass(JavaSymbol.TypeJavaSymbol symbol, Resolve.Env env, ClassJavaType type) {
144     ClassTree tree = symbol.declaration;
145     Tree superClassTree = tree.superClass();
146     if (superClassTree != null) {
147       type.supertype = resolveType(env, superClassTree);
148       checkHierarchyCycles(symbol.type);
149     } else if (tree.is(Tree.Kind.ENUM)) {
150       // JLS8 8.9: The direct superclass of an enum type E is Enum<E>.
151       Scope enumParameters = ((JavaSymbol.TypeJavaSymbol) symbols.enumType.symbol()).typeParameters();
152       TypeVariableJavaType enumParameter = (TypeVariableJavaType) enumParameters.lookup("E").get(0).type();
153       type.supertype = parametrizedTypeCache.getParametrizedTypeType(symbols.enumType.symbol, new TypeSubstitution().add(enumParameter, type));
154     } else if (tree.is(Tree.Kind.CLASS, Tree.Kind.INTERFACE)) {
155       // For CLASS JLS8 8.1.4: the direct superclass of the class type C<F1,...,Fn> is
156       // the type given in the extends clause of the declaration of C
157       // if an extends clause is present, or Object otherwise.
158       // For INTERFACE JLS8 9.1.3: While every class is an extension of class Object, there is no single interface of which all interfaces are
159       // extensions.
160       // but we can call object method on any interface type.
161       type.supertype = symbols.objectType;
162     }
163   }

165   private void completeTypeParameters(TypeParameters typeParameters, Resolve.Env env) {
166     for (TypeParameterTree typeParameterTree : typeParameters) {
167       List<JavaType> bounds = new ArrayList<>();
168       if(typeParameterTree.bounds().isEmpty()) {
169         bounds.add(symbols.objectType);
170       } else {
171         for (Tree boundTree : typeParameterTree.bounds()) {
172           bounds.add(resolveType(env, boundTree));
173         }
174       }
175       ((TypeVariableJavaType) semanticModel.getSymbol(typeParameterTree).type()).bounds = bounds;
176     }
177   }
178 
179   private static void checkHierarchyCycles(JavaType baseType) {
180     Set<ClassJavaType> types = new HashSet<>();
181     ClassJavaType type = (ClassJavaType) baseType;
182     while (type != null) {
183       if (!types.add(type)) {
184         throw new IllegalStateException("Cycling class hierarchy detected with symbol : " + baseType.symbol.name + ".");
185       }
186       type = (ClassJavaType) type.symbol.getSuperclass();
187     }
188   }

190   private void complete(JavaSymbol.MethodJavaSymbol symbol) {
191     MethodTree methodTree = symbol.declaration;
192     Resolve.Env env = semanticModel.getEnv(symbol);
193     completeTypeParameters(methodTree.typeParameters(), env);
194     ImmutableList.Builder<JavaType> thrownTypes = ImmutableList.builder();
195     for (TypeTree throwClause : methodTree.throwsClauses()) {
196       JavaType thrownType = resolveType(env, throwClause);
197       if (thrownType != null) {
198         thrownTypes.add(thrownType);
199       }
200     }
201 
202     JavaType returnType = null;
203     List<JavaType> argTypes = new ArrayList<>();
204     // no return type for constructor
205     if (!CONSTRUCTOR_NAME.equals(symbol.name)) {
206       returnType = resolveType(env, methodTree.returnType());
207       if (returnType != null) {
208         symbol.returnType = returnType.symbol;
209       }
210     } else if (!symbol.enclosingClass().isStatic()) {
211       JavaSymbol owner = symbol.enclosingClass().owner();
212       if (!owner.isPackageSymbol()) {
213         // JLS8 - 8.8.1 & 8.8.9 : constructors of inner class have an implicit first arg of its directly enclosing class type
214         argTypes.add(owner.enclosingClass().type);
215       }
216     }
217     List<VariableTree> parametersTree = methodTree.parameters();
218     List<JavaSymbol> scopeSymbols = symbol.parameters.scopeSymbols();
219     for(int i = 0; i < parametersTree.size(); i += 1) {
220       VariableTree variableTree = parametersTree.get(i);
221       JavaSymbol param = scopeSymbols.get(i);
222       if (variableTree.simpleName().name().equals(param.getName())) {
223         param.complete();
224         argTypes.add(param.getType());
225       }
226       if(((VariableTreeImpl)variableTree).isVararg()) {
227         symbol.flags |= Flags.VARARGS;
228       }
229     }
230     MethodJavaType methodType = new MethodJavaType(argTypes, returnType, thrownTypes.build(), (JavaSymbol.TypeJavaSymbol) symbol.owner);
231     symbol.setMethodType(methodType);
232   }
233 
234   private void complete(JavaSymbol.VariableJavaSymbol symbol) {
235     VariableTree variableTree = symbol.declaration;
236     Resolve.Env env = semanticModel.getEnv(symbol);
237     if (variableTree.is(Tree.Kind.ENUM_CONSTANT)) {
238       symbol.type = env.enclosingClass.type;
239     } else {
240       symbol.type = resolveType(env, variableTree.type());
241     }
242   }

244   private JavaType resolveType(Resolve.Env env, Tree tree) {
245     Preconditions.checkArgument(checkTypeOfTree(tree), "Kind of tree unexpected " + tree.kind());
246     //FIXME(benzonico) as long as Variables share the same node type, (int i,j; or worse : int i[], j[];) check nullity to respect invariance.
247     if (((AbstractTypedTree) tree).isTypeSet()) {
248       return (JavaType) ((AbstractTypedTree) tree).symbolType();
249     }
250     typeAndReferenceSolver.env = env;
251     typeAndReferenceSolver.resolveAs(tree, JavaSymbol.TYP, env);
252     typeAndReferenceSolver.env = null;
253     return (JavaType) ((AbstractTypedTree) tree).symbolType();
254   }
255 
256   private static boolean checkTypeOfTree(Tree tree) {
257     return tree.is(Tree.Kind.MEMBER_SELECT,
258       Tree.Kind.IDENTIFIER,
259       Tree.Kind.PARAMETERIZED_TYPE,
260       Tree.Kind.ARRAY_TYPE,
261       Tree.Kind.UNION_TYPE,
262       Tree.Kind.PRIMITIVE_TYPE,
263       Tree.Kind.VAR_TYPE,
264       Tree.Kind.INFERED_TYPE);
265   }

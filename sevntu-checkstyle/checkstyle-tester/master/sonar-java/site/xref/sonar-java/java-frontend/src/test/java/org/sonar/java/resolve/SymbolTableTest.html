79      JavaType arrayErasure = typeSymbol.members().lookup("arrayErasure").get(0).type;
80      assertThat(arrayErasure.isTagged(JavaType.ARRAY)).isTrue();
81      assertThat(arrayErasure.erasure().isTagged(JavaType.ARRAY)).isTrue();
82      assertThat(((ArrayJavaType) arrayErasure.erasure()).elementType().symbol.getName()).isEqualTo("CharSequence");
83  
84      IdentifierTree tree = result.referenceTree(20, 7);
85      JavaType symbolType = (JavaType) tree.symbolType();
86      assertThat(symbolType).isInstanceOf(ParametrizedTypeJavaType.class);

88      assertThat(ptt.symbol.getName()).isEqualTo("C");
89      assertThat(ptt.typeSubstitution.size()).isEqualTo(1);
90      assertThat(ptt.typeSubstitution.substitutedType(ptt.typeSubstitution.typeVariables().iterator().next()).symbol.getName()).isEqualTo("String");
91

95      JavaSymbol.MethodJavaSymbol method2 = (JavaSymbol.MethodJavaSymbol) typeSymbol.members().lookup("method2").get(0);

98      assertThat(method2.parameterTypes().get(0)).isSameAs(PTypeVariableType);

353     Tree parent = referenceTree.parent();

357     return (NewClassTree) parent;

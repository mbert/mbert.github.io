22  import com.google.common.annotations.Beta;

24  /**
25   * Binary expression.
26   *
27   * JLS from 15.17 to 15.24
28   *
29   * <pre>
30   *   {@link #leftOperand()} {@link Tree.Kind#MULTIPLY *} {@link #rightOperand()}
31   *   {@link #leftOperand()} {@link Tree.Kind#DIVIDE /} {@link #rightOperand()}
32   *   {@link #leftOperand()} {@link Tree.Kind#REMAINDER %} {@link #rightOperand()}
33   *   {@link #leftOperand()} {@link Tree.Kind#PLUS +} {@link #rightOperand()}
34   *   {@link #leftOperand()} {@link Tree.Kind#MINUS -} {@link #rightOperand()}
35   *   {@link #leftOperand()} {@link Tree.Kind#LEFT_SHIFT <<} {@link #rightOperand()}
36   *   {@link #leftOperand()} {@link Tree.Kind#RIGHT_SHIFT >>} {@link #rightOperand()}
37   *   {@link #leftOperand()} {@link Tree.Kind#UNSIGNED_RIGHT_SHIFT >>>} {@link #rightOperand()}
38   *   {@link #leftOperand()} {@link Tree.Kind#LESS_THAN <} {@link #rightOperand()}
39   *   {@link #leftOperand()} {@link Tree.Kind#GREATER_THAN >} {@link #rightOperand()}
40   *   {@link #leftOperand()} {@link Tree.Kind#LESS_THAN_OR_EQUAL_TO <=} {@link #rightOperand()}
41   *   {@link #leftOperand()} {@link Tree.Kind#GREATER_THAN_OR_EQUAL_TO >=} {@link #rightOperand()}
42   *   {@link #leftOperand()} {@link Tree.Kind#EQUAL_TO ==} {@link #rightOperand()}
43   *   {@link #leftOperand()} {@link Tree.Kind#NOT_EQUAL_TO !=} {@link #rightOperand()}
44   *   {@link #leftOperand()} {@link Tree.Kind#AND &} {@link #rightOperand()}
45   *   {@link #leftOperand()} {@link Tree.Kind#XOR ^} {@link #rightOperand()}
46   *   {@link #leftOperand()} {@link Tree.Kind#OR |} {@link #rightOperand()}
47   *   {@link #leftOperand()} {@link Tree.Kind#CONDITIONAL_AND &&} {@link #rightOperand()}
48   *   {@link #leftOperand()} {@link Tree.Kind#CONDITIONAL_OR ||} {@link #rightOperand()}
49   * </pre>
50   *
51   * @since Java 1.3
52   */
53  @Beta
54  public interface BinaryExpressionTree extends ExpressionTree {
55  
56    ExpressionTree leftOperand();
57  
58    SyntaxToken operatorToken();
59  
60    ExpressionTree rightOperand();
61  
62  }

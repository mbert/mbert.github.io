145   public Symbols(BytecodeCompleter bytecodeCompleter) {
146     defaultPackage = new JavaSymbol.PackageJavaSymbol("", rootPackage);
147 
148     predefClass = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, "", rootPackage);
149     predefClass.members = new Scope(predefClass);
150     ((ClassJavaType) predefClass.type).interfaces = Collections.emptyList();
151 
152     // TODO should have type "noType":
153     noSymbol = new JavaSymbol.TypeJavaSymbol(0, "", rootPackage);
154 
155     methodClass = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, "", noSymbol);
156 
157     // builtin types
158     byteType = initType(JavaType.BYTE, "byte");
159     charType = initType(JavaType.CHAR, "char");
160     shortType = initType(JavaType.SHORT, "short");
161     intType = initType(JavaType.INT, "int");
162     longType = initType(JavaType.LONG, "long");
163     floatType = initType(JavaType.FLOAT, "float");
164     doubleType = initType(JavaType.DOUBLE, "double");
165     booleanType = initType(JavaType.BOOLEAN, "boolean");
166     nullType = initType(JavaType.BOT, "<nulltype>");
167     voidType = initType(JavaType.VOID, "void");
168 
169     bytecodeCompleter.init(this);
170 
171     // predefined types for java lang
172     JavaSymbol.PackageJavaSymbol javalang = bytecodeCompleter.enterPackage("java.lang");
173     // define a star import scope to let resolve types to java.lang when needed.
174     javalang.members = new Scope.StarImportScope(javalang, bytecodeCompleter);
175     javalang.members.enter(javalang);
176 
177     objectType = bytecodeCompleter.loadClass("java.lang.Object").type;
178     classType = bytecodeCompleter.loadClass("java.lang.Class").type;
179     stringType = bytecodeCompleter.loadClass("java.lang.String").type;
180     cloneableType = bytecodeCompleter.loadClass("java.lang.Cloneable").type;
181     serializableType = bytecodeCompleter.loadClass("java.io.Serializable").type;
182     annotationType = bytecodeCompleter.loadClass("java.lang.annotation.Annotation").type;
183     enumType = bytecodeCompleter.loadClass("java.lang.Enum").type;
184 
185     unboundedWildcard = new WildCardType(objectType, WildCardType.BoundType.UNBOUNDED);
186 
187     // Associate boxed types
188     boxedTypes = HashBiMap.create();
189     boxedTypes.put(byteType, bytecodeCompleter.loadClass("java.lang.Byte").type);
190     boxedTypes.put(charType, bytecodeCompleter.loadClass("java.lang.Character").type);
191     boxedTypes.put(shortType, bytecodeCompleter.loadClass("java.lang.Short").type);
192     boxedTypes.put(intType, bytecodeCompleter.loadClass("java.lang.Integer").type);
193     boxedTypes.put(longType, bytecodeCompleter.loadClass("java.lang.Long").type);
194     boxedTypes.put(floatType, bytecodeCompleter.loadClass("java.lang.Float").type);
195     boxedTypes.put(doubleType, bytecodeCompleter.loadClass("java.lang.Double").type);
196     boxedTypes.put(booleanType, bytecodeCompleter.loadClass("java.lang.Boolean").type);
197 
198     for (Entry<JavaType, JavaType> entry : boxedTypes.entrySet()) {
199       entry.getKey().primitiveWrapperType = entry.getValue();
200       entry.getValue().primitiveType = entry.getKey();
201     }
202 
203     arrayClass = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, "Array", noSymbol);
204     ClassJavaType arrayClassType = (ClassJavaType) arrayClass.type;
205     arrayClassType.supertype = objectType;
206     arrayClassType.interfaces = ImmutableList.of(cloneableType, serializableType);
207     // clone method return type is handled during method resolution.
208     arrayClass.members = new Scope(arrayClass);
209     arrayClass.members().enter(new JavaSymbol.VariableJavaSymbol(Flags.PUBLIC | Flags.FINAL, "length", intType, arrayClass));
210 
211     // java.lang.Synthetic is a virtual annotation added by ASM to workaround a bug in javac on inner classes parameter numbers.
212     // Predefining this type avoids to look it up in classpath where it will not be found. We rely on this to detect synthetic parameters on some enum constructor for instance.
213     JavaSymbol.TypeJavaSymbol syntheticAnnotation = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC | Flags.ANNOTATION, "Synthetic", javalang);
214     javalang.members.enter(syntheticAnnotation);
215     bytecodeCompleter.registerClass(syntheticAnnotation);
216     enterOperators();
217   }

219   /**
220    * Registers builtin types as symbols, so that they can be found as an usual identifiers.
221    */
222   private JavaType initType(int tag, String name) {
223     JavaSymbol.TypeJavaSymbol symbol = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, name, rootPackage);
224     symbol.members = new Scope(symbol);
225     predefClass.members.enter(symbol);
226     ((ClassJavaType) symbol.type).interfaces = Collections.emptyList();
227     symbol.type.tag = tag;
228     return symbol.type;
229   }

231   /**
232    * Registers operators as methods, so that they can be found as an usual methods.
233    */
234   private void enterOperators() {
235     for (String op : new String[] {"+", "-", "*", "/", "%"}) {
236       for (JavaType type : Arrays.asList(doubleType, floatType, longType, intType)) {
237         enterBinop(op, type, type, type);
238       }
239     }
240     for (String op : new String[] {"&", "|", "^"}) {
241       for (JavaType type : Arrays.asList(booleanType, longType, intType)) {
242         enterBinop(op, type, type, type);
243       }
244     }
245     for (String op : new String[] {"<<", ">>", ">>>"}) {
246       enterBinop(op, longType, longType, longType);
247       enterBinop(op, intType, longType, intType);
248       enterBinop(op, longType, intType, longType);
249       enterBinop(op, intType, intType, intType);
250     }
251     for (String op : new String[] {"<", ">", ">=", "<="}) {
252       for (JavaType type : Arrays.asList(doubleType, floatType, longType, intType)) {
253         enterBinop(op, type, type, booleanType);
254       }
255     }
256     for (String op : new String[] {"==", "!="}) {
257       for (JavaType type : Arrays.asList(objectType, booleanType, doubleType, floatType, longType, intType)) {
258         enterBinop(op, type, type, booleanType);
259       }
260     }
261     enterBinop("&&", booleanType, booleanType, booleanType);
262     enterBinop("||", booleanType, booleanType, booleanType);
263 
264     // string concatenation
265     for (JavaType type : Arrays.asList(nullType, objectType, booleanType, doubleType, floatType, longType, intType)) {
266       enterBinop("+", stringType, type, stringType);
267       enterBinop("+", type, stringType, stringType);
268     }
269     enterBinop("+", stringType, stringType, stringType);
270   }
271 
272   private void enterBinop(String name, JavaType left, JavaType right, JavaType result) {
273     JavaType type = new MethodJavaType(ImmutableList.of(left, right), result, Collections.emptyList(), methodClass);
274     JavaSymbol symbol = new JavaSymbol.MethodJavaSymbol(Flags.PUBLIC | Flags.STATIC, name, type, predefClass);
275     predefClass.members.enter(symbol);
276   }

278   public JavaType getPrimitiveFromDescriptor(char descriptor) {
279     switch (descriptor) {
280       case 'S':
281         return shortType;
282       case 'I':
283         return intType;
284       case 'C':
285         return charType;
286       case 'Z':
287         return booleanType;
288       case 'B':
289         return byteType;
290       case 'J':
291         return longType;
292       case 'F':
293         return floatType;
294       case 'D':
295         return doubleType;
296       case 'V':
297         return voidType;
298       default:
299         throw new IllegalStateException("Descriptor '" + descriptor + "' cannot be mapped to a primitive type");
300     }
301   }

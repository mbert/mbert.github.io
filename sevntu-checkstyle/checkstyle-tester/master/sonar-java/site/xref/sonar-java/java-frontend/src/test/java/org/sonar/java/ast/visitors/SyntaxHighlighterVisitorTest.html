105    * Java 9 modules introduces restricted keywords only used in their context

107   @Test
108   public void test_restricted_keywords_within_module() throws Exception {
109     this.eol = "\n";
110     InputFile inputFile = generateTestFile("src/test/files/highlighter/ModuleExample.java");
111     scan(inputFile);
112 
113     String componentKey = inputFile.key();
114     assertThatHasBeenHighlighted(componentKey, 1, 1, 3, 4, TypeOfText.COMMENT);
115     assertThatHasBeenHighlighted(componentKey, 4, 1, 4, 7, TypeOfText.KEYWORD); // import
116     assertThatHasBeenHighlighted(componentKey, 6, 1, 8, 4, TypeOfText.STRUCTURED_COMMENT);
117     assertThatHasBeenHighlighted(componentKey, 9, 1, 9, 6, TypeOfText.ANNOTATION); // @Beta
118     assertThatHasBeenHighlighted(componentKey, 10, 1, 10, 5, TypeOfText.KEYWORD); // open
119     assertThatHasBeenHighlighted(componentKey, 10, 6, 10, 12, TypeOfText.KEYWORD); // module
120     assertThatHasBeenHighlighted(componentKey, 11, 3, 11, 11, TypeOfText.KEYWORD); // requires
121     assertThatHasNotBeenHighlighted(componentKey, 11, 12, 11, 22); // transitive as identifier
122     assertThatHasBeenHighlighted(componentKey, 12, 3, 12, 11, TypeOfText.KEYWORD); // requires
123     assertThatHasBeenHighlighted(componentKey, 12, 12, 12, 18, TypeOfText.KEYWORD); // static
124     assertThatHasNotBeenHighlighted(componentKey, 12, 19, 12, 29); // transitive as identifier
125     assertThatHasBeenHighlighted(componentKey, 13, 3, 13, 11, TypeOfText.KEYWORD); // requires
126     assertThatHasBeenHighlighted(componentKey, 13, 12, 13, 18, TypeOfText.KEYWORD); // static
127     assertThatHasBeenHighlighted(componentKey, 13, 19, 13, 29, TypeOfText.KEYWORD); // transitive as keyword
128     assertThatHasBeenHighlighted(componentKey, 14, 3, 14, 10, TypeOfText.KEYWORD); // exports
129     assertThatHasBeenHighlighted(componentKey, 14, 19, 14, 21, TypeOfText.KEYWORD); // to
130     assertThatHasBeenHighlighted(componentKey, 15, 3, 15, 8, TypeOfText.KEYWORD); // opens
131     assertThatHasBeenHighlighted(componentKey, 15, 17, 15, 19, TypeOfText.KEYWORD); // to
132     assertThatHasBeenHighlighted(componentKey, 16, 3, 16, 7, TypeOfText.KEYWORD); // uses
133     assertThatHasBeenHighlighted(componentKey, 17, 3, 17, 11, TypeOfText.KEYWORD); // provides
134     assertThatHasBeenHighlighted(componentKey, 17, 26, 17, 30, TypeOfText.KEYWORD); // with
135 
136     // usages of restricted keywords in module name and package names
137     assertThatHasBeenHighlighted(componentKey, 18, 3, 18, 60, TypeOfText.COMMENT);
138     assertThatHasBeenHighlighted(componentKey, 19, 3, 19, 10, TypeOfText.KEYWORD); // exports
139     assertThatHasNotBeenHighlighted(componentKey, 19, 28, 19, 34); // 'module' used in package name
140     assertThatHasBeenHighlighted(componentKey, 20, 3, 20, 11, TypeOfText.KEYWORD); // provides
141     assertThatHasBeenHighlighted(componentKey, 20, 34, 20, 38, TypeOfText.KEYWORD); // with
142     assertThatHasNotBeenHighlighted(componentKey, 20, 45, 20, 49); // 'with' used in package name
143     assertThatHasNotBeenHighlighted(componentKey, 20, 50, 20, 52); // 'to' used in package name
144     assertThatHasNotBeenHighlighted(componentKey, 20, 53, 20, 60); // 'exports' used in package name
145     assertThatHasNotBeenHighlighted(componentKey, 20, 61, 20, 67); // 'module' used in package name
146   }

150     this.eol = "\n";

157     assertThatHasBeenHighlighted(componentKey, 6, 1, 8, 4, TypeOfText.STRUCTURED_COMMENT);
158     assertThatHasBeenHighlighted(componentKey, 9, 1, 9, 6, TypeOfText.ANNOTATION); // @Beta
159     assertThatHasBeenHighlighted(componentKey, 10, 1, 10, 9, TypeOfText.KEYWORD); // abstract
160     assertThatHasBeenHighlighted(componentKey, 10, 10, 10, 15, TypeOfText.KEYWORD); // class
161     assertThatHasNotBeenHighlighted(componentKey, 10, 16, 10, 22); // module
162     assertThatHasNotBeenHighlighted(componentKey, 11, 10, 11, 14); // open
163     assertThatHasNotBeenHighlighted(componentKey, 11, 16, 11, 26); // transitive
164     assertThatHasBeenHighlighted(componentKey, 13, 3, 13, 7, TypeOfText.KEYWORD); // void
165     assertThatHasNotBeenHighlighted(componentKey, 13, 8, 13, 16); // requires
166     assertThatHasNotBeenHighlighted(componentKey, 13, 24, 13, 31); // exports
167     assertThatHasNotBeenHighlighted(componentKey, 13, 40, 13, 45); // opens
168     assertThatHasBeenHighlighted(componentKey, 14, 5, 14, 8, TypeOfText.KEYWORD); // int
169     assertThatHasNotBeenHighlighted(componentKey, 14, 9, 14, 11); // to
170     assertThatHasBeenHighlighted(componentKey, 15, 5, 15, 11, TypeOfText.KEYWORD); // double
171     assertThatHasNotBeenHighlighted(componentKey, 15, 12, 15, 16); // with
172     assertThatHasNotBeenHighlighted(componentKey, 16, 12, 16, 16); // uses
173     assertThatHasNotBeenHighlighted(componentKey, 17, 5, 17, 13); // provides
174     assertThatHasBeenHighlighted(componentKey, 20, 3, 20, 11, TypeOfText.KEYWORD); // abstract
175     assertThatHasBeenHighlighted(componentKey, 20, 12, 20, 16, TypeOfText.KEYWORD); // void
176     assertThatHasNotBeenHighlighted(componentKey, 20, 17, 20, 25); // provides
177   }

195   private void scan(InputFile inputFile) {
196     JavaSquid squid = new JavaSquid(new JavaVersionImpl(), null, null, null, null, new JavaCheck[] {syntaxHighlighterVisitor});
197     squid.scan(Collections.singletonList(inputFile), Collections.emptyList());
198   }

200   private InputFile generateDefaultTestFile() throws IOException {
201     return generateTestFile("src/test/files/highlighter/Example.java");
202   }
203 
204   private InputFile generateTestFile(String sourceFile) throws IOException {
205     File source = new File(sourceFile);
206     File target = temp.newFile().getAbsoluteFile();
207     String content = Files.asCharSource(source, StandardCharsets.UTF_8)
208       .read()
209       .replaceAll("\\r\\n", "\n")
210       .replaceAll("\\r", "\n")
211       .replaceAll("\\n", eol);
212     Files.asCharSink(target, StandardCharsets.UTF_8).write(content);
213     return TestUtils.inputFile(target);
214   }

216   private void verifyHighlighting(InputFile inputFile) throws IOException {
217     String componentKey = inputFile.key();
218     assertThatHasBeenHighlighted(componentKey, 1, 1, 3, 4, TypeOfText.COMMENT);
219     assertThatHasBeenHighlighted(componentKey, 5, 1, 7, 4, TypeOfText.STRUCTURED_COMMENT);
220     assertThatHasBeenHighlighted(componentKey, 8, 1, 8, 18, TypeOfText.ANNOTATION);
221     assertThatHasBeenHighlighted(componentKey, 8, 19, 8, 27, TypeOfText.STRING);
222     assertThatHasBeenHighlighted(componentKey, 9, 1, 9, 6, TypeOfText.KEYWORD);
223     assertThatHasBeenHighlighted(componentKey, 11, 3, 11, 24, TypeOfText.ANNOTATION);
224     assertThatHasBeenHighlighted(componentKey, 12, 3, 12, 6, TypeOfText.KEYWORD);
225     assertThatHasBeenHighlighted(componentKey, 13, 5, 13, 11, TypeOfText.KEYWORD);
226     assertThatHasBeenHighlighted(componentKey, 13, 12, 13, 14, TypeOfText.CONSTANT);
227     assertThatHasBeenHighlighted(componentKey, 18, 1, 18, 18, TypeOfText.COMMENT);
228     assertThatHasBeenHighlighted(componentKey, 19, 1, 19, 11, TypeOfText.KEYWORD);
229     assertThatHasBeenHighlighted(componentKey, 20, 21, 20, 28, TypeOfText.KEYWORD);
230     assertThatHasBeenHighlighted(componentKey, 20, 29, 20, 30, TypeOfText.CONSTANT);
231     assertThatHasBeenHighlighted(componentKey, 23, 1, 23, 10, TypeOfText.KEYWORD);
232     assertThatHasBeenHighlighted(componentKey, 24, 3, 24, 7, TypeOfText.KEYWORD);
233   }

235   private void assertThatHasBeenHighlighted(String componentKey, int startLine, int startColumn, int endLine, int endColumn, TypeOfText expected) {
236     assertThat(context.highlightingTypeAt(componentKey, startLine, startColumn - 1)).hasSize(1).contains(expected);
237     // -1 because of offset (column start at 0) and -1 to be within the range.
238     assertThat(context.highlightingTypeAt(componentKey, endLine, endColumn - 1 - 1)).hasSize(1).contains(expected);
239   }
240 
241   private void assertThatHasNotBeenHighlighted(String componentKey, int startLine, int startColumn, int endLine, int endColumn) {
242     assertThat(context.highlightingTypeAt(componentKey, startLine, startColumn - 1)).isEmpty();
243     // -1 because of offset (column start at 0) and -1 to be within the range.
244     assertThat(context.highlightingTypeAt(componentKey, endLine, endColumn - 1 - 1)).isEmpty();
245   }

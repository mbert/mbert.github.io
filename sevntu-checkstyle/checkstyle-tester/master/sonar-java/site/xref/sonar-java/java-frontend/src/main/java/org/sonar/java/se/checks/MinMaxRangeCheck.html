48  @Rule(key = "S3065")
49  public class MinMaxRangeCheck extends SECheck {
50  
51    private static final String UPPER = "upper";
52    private static final String LOWER = "lower";
53    private static final String FLOW_MESSAGE = "Returns the %s bound.";
54    private static final String ISSUE_MESSAGE = "Change these chained %s methods invocations, as final results will always be the %s bound.";
55  
56    private static final MethodMatcher MIN_MAX_MATCHER = MethodMatcher.create()
57      .typeDefinition("java.lang.Math")
58      .name(name -> "min".equals(name) || "max".equals(name))
59      .addParameter(TypeCriteria.anyType()).addParameter(TypeCriteria.anyType());
60  
61    private enum Operation {
62      MIN,
63      MAX
64    }
65  
66    private static class NumericalConstraint implements Constraint {
67      private final Number value;
68  
69      NumericalConstraint(Number value) {
70        this.value = value;
71      }
72  
73      @Override
74      public String toString() {
75        return "Number(" + value + ")";
76      }
77    }
78  
79    private static class MinMaxRangeConstraint implements Constraint {
80      private final Operation op;
81      private final MethodInvocationTree syntaxNode;
82  
83      MinMaxRangeConstraint(MethodInvocationTree syntaxNode) {
84        this.syntaxNode = syntaxNode;
85        this.op = "min".equals(syntaxNode.symbol().name()) ? Operation.MIN : Operation.MAX;
86      }
87  
88      @Override
89      public String toString() {
90        return "Range_" + op.name();
91      }
92    }
93  
94    private static class MinMaxValue {
95      private final Number value;
96      private final Operation op;
97      private final MethodInvocationTree syntaxNode;
98  
99      MinMaxValue(NumericalConstraint numericalConstraint, MinMaxRangeConstraint minMaxRangeConstraint) {
100       value = numericalConstraint.value;
101       op = minMaxRangeConstraint.op;
102       syntaxNode = minMaxRangeConstraint.syntaxNode;
103     }
104 
105     @CheckForNull
106     static MinMaxValue fromConstraints(@Nullable ConstraintsByDomain constraints) {
107       if (constraints == null) {
108         return null;
109       }
110       Constraint minMaxRangeConstraint = constraints.get(MinMaxRangeConstraint.class);
111       Constraint numericalConstraint = constraints.get(NumericalConstraint.class);
112       if (minMaxRangeConstraint != null && numericalConstraint != null) {
113         return new MinMaxValue((NumericalConstraint) numericalConstraint, (MinMaxRangeConstraint) minMaxRangeConstraint);
114       }
115       return null;
116     }
117   }
118 
119   @Override
120   public ProgramState checkPreStatement(CheckerContext context, Tree syntaxNode) {
121     ProgramState programState = context.getState();
122     if (!syntaxNode.is(Tree.Kind.METHOD_INVOCATION)) {
123       return programState;
124     }
125     MethodInvocationTree mit = (MethodInvocationTree) syntaxNode;
126     if (!MIN_MAX_MATCHER.matches(mit)) {
127       return programState;
128     }
129     List<SymbolicValue> args = programState.peekValues(2);
130     for (SymbolicValue arg : args) {
131       MinMaxRangeConstraint minMaxConstraint = programState.getConstraint(arg, MinMaxRangeConstraint.class);
132       NumericalConstraint numericalConstraint = programState.getConstraint(arg, NumericalConstraint.class);
133       if (minMaxConstraint == null && numericalConstraint != null) {
134         programState = programState.addConstraint(arg, new MinMaxRangeConstraint(mit));
135       }
136     }
137     return programState;
138   }
139 
140   @Override
141   public ProgramState checkPostStatement(CheckerContext context, Tree syntaxNode) {
142     // TODO handle float and double
143     switch (syntaxNode.kind()) {
144       case INT_LITERAL:
145         return handleNumericalLiteral(context, LiteralUtils.intLiteralValue((ExpressionTree) syntaxNode));
146       case LONG_LITERAL:
147         return handleNumericalLiteral(context, LiteralUtils.longLiteralValue((ExpressionTree) syntaxNode));
148       case UNARY_MINUS:
149       case UNARY_PLUS:
150         return handleNumericalLiteral(context, (UnaryExpressionTree) syntaxNode);
151       case IDENTIFIER:
152         return handleNumericalConstant(context, (IdentifierTree) syntaxNode);
153       case METHOD_INVOCATION:
154         return handleMinMaxInvocation(context, (MethodInvocationTree) syntaxNode);
155       default:
156         return context.getState();
157     }
158   }

160   private static ProgramState handleNumericalConstant(CheckerContext context, IdentifierTree syntaxNode) {
161     ProgramState programState = context.getState();
162     Symbol identifier = syntaxNode.symbol();
163     if (!isNumericalConstant(identifier)) {
164       return programState;
165     }
166     SymbolicValue constant = programState.getValue(identifier);
167     if (constant == null) {
168       return programState;
169     }
170     NumericalConstraint numericalConstraint = programState.getConstraint(constant, NumericalConstraint.class);
171     if (numericalConstraint == null) {
172       return ((JavaSymbol.VariableJavaSymbol) identifier).constantValue()
173         .filter(Number.class::isInstance)
174         .map(Number.class::cast)
175         .map(value -> programState.addConstraint(constant, new NumericalConstraint(value)))
176         .orElse(programState);
177     }
178     return programState;
179   }

181   private ProgramState handleMinMaxInvocation(CheckerContext context, MethodInvocationTree syntaxNode) {
182     if (!MIN_MAX_MATCHER.matches(syntaxNode)) {
183       return context.getState();
184     }
185     ProgramState programState = context.getState();
186     ProgramState psBeforeInvocation = context.getNode().programState;
187 
188     List<SymbolicValue> args = psBeforeInvocation.peekValues(2);
189     List<ConstraintsByDomain> constraintsByArgs = args.stream().map(programState::getConstraints).collect(Collectors.toList());
190 
191     checkRangeInconsistencies(context, syntaxNode, constraintsByArgs);
192 
193     return context.getState();
194   }
195 
196   private void checkRangeInconsistencies(CheckerContext context, MethodInvocationTree syntaxNode, List<ConstraintsByDomain> constraintsByArgs) {
197     MinMaxValue arg0MinMaxValue = MinMaxValue.fromConstraints(constraintsByArgs.get(0));
198     MinMaxValue arg1MinMaxValue = MinMaxValue.fromConstraints(constraintsByArgs.get(1));
199 
200     if (arg0MinMaxValue != null && arg1MinMaxValue != null && arg0MinMaxValue.op != arg1MinMaxValue.op) {
201       // bounds have been inverted
202       Number upperBound = arg0MinMaxValue.op == Operation.MIN ? arg0MinMaxValue.value : arg1MinMaxValue.value;
203       Number lowerBound = arg0MinMaxValue.op == Operation.MAX ? arg0MinMaxValue.value : arg1MinMaxValue.value;
204 
205       // all the used values are going to be Numbers
206       @SuppressWarnings("unchecked")
207       int comparedValue = ((Comparable<Number>) lowerBound).compareTo(upperBound);
208 
209       if (comparedValue > 0) {
210         String issueMessage;
211         String secondOpMessage;
212         String firstOpMessage;
213         if ("min".equals(syntaxNode.symbol().name())) {
214           issueMessage = String.format(ISSUE_MESSAGE, "min/max", LOWER);
215           firstOpMessage = String.format(FLOW_MESSAGE, UPPER);
216           secondOpMessage = String.format(FLOW_MESSAGE, LOWER);
217         } else {
218           issueMessage = String.format(ISSUE_MESSAGE, "max/min", UPPER);
219           firstOpMessage = String.format(FLOW_MESSAGE, LOWER);
220           secondOpMessage = String.format(FLOW_MESSAGE, UPPER);
221         }
222         MethodInvocationTree flowTree = syntaxNode == arg0MinMaxValue.syntaxNode ? arg1MinMaxValue.syntaxNode : arg0MinMaxValue.syntaxNode;
223         Set<Flow> flow = Collections.singleton(Flow.builder()
224           .add(new JavaFileScannerContext.Location(secondOpMessage, syntaxNode))
225           .add(new JavaFileScannerContext.Location(firstOpMessage, flowTree))
226           .build());
227         context.reportIssue(syntaxNode, this, issueMessage, flow);
228       }
229     }
230   }

232   private static ProgramState handleNumericalLiteral(CheckerContext context, @Nullable Number value) {
233     ProgramState programState = context.getState();
234     if (value == null) {
235       return programState;
236     }
237     return programState.addConstraint(programState.peekValue(), new NumericalConstraint(value));
238   }
239 
240   private static ProgramState handleNumericalLiteral(CheckerContext context, UnaryExpressionTree syntaxNode) {
241     ProgramState previousPS = context.getNode().programState;
242     NumericalConstraint knownNumericalConstraint = previousPS.getConstraint(previousPS.peekValue(), NumericalConstraint.class);
243 
244     ProgramState programState = context.getState();
245     if (knownNumericalConstraint == null) {
246       return programState;
247     }
248     if (syntaxNode.is(Tree.Kind.UNARY_PLUS)) {
249       return programState.addConstraint(programState.peekValue(), knownNumericalConstraint);
250     }
251     Number value = knownNumericalConstraint.value;
252     if (value instanceof Integer) {
253       value = -1 * value.intValue();
254     } else if (value instanceof Long) {
255       value = -1L * value.longValue();
256     } else if (value instanceof Float) {
257       value = -1.0f * value.floatValue();
258     } else {
259       value = -1.0 * value.doubleValue();
260     }
261     return programState.addConstraint(programState.peekValue(), new NumericalConstraint(value));
262   }
263 
264   private static boolean isNumericalConstant(@Nullable Symbol symbol) {
265     return symbol != null && isConstant(symbol) && isNumericalPrimitive(symbol);
266   }
267 
268   private static boolean isNumericalPrimitive(Symbol symbol) {
269     Type type = symbol.type();
270     return type.isPrimitive() && type.isNumerical();
271   }
272 
273   private static boolean isConstant(Symbol symbol) {
274     return symbol.isVariableSymbol() && symbol.isStatic() && symbol.isFinal();
275   }

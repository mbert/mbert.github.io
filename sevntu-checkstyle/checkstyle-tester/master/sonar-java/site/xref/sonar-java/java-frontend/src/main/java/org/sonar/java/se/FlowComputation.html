96    private static Set<SymbolicValue> computedFrom(@Nullable SymbolicValue symbolicValue) {
97      if (symbolicValue == null) {
98        return Collections.emptySet();
99      }
100     Set<SymbolicValue> result = new HashSet<>();
101     result.add(symbolicValue);
102     symbolicValue.computedFrom().forEach(sv -> result.addAll(computedFrom(sv)));
103     return result;
104   }

131   private static Set<Flow> flow(ExplodedGraph.Node currentNode, Set<SymbolicValue> symbolicValues, Predicate<Constraint> addToFlow,
132     Predicate<Constraint> terminateTraversal, List<Class<? extends Constraint>> domains, Set<Symbol> symbols, boolean skipExceptionMessages) {
133     Set<SymbolicValue> allSymbolicValues = symbolicValues.stream()
134       .map(FlowComputation::computedFrom)
135       .flatMap(Set::stream)
136       .collect(Collectors.toSet());
137 
138     PSet<Symbol> trackedSymbols = PCollections.emptySet();
139     for (Symbol symbol: symbols) {
140       trackedSymbols = trackedSymbols.add(symbol);
141     }
142 
143     if (symbols.isEmpty()) {
144       for (SymbolicValue symbolicValue : symbolicValues) {
145         for (Symbol symbol : symbolicValue.computedFromSymbols()) {
146           trackedSymbols = trackedSymbols.add(symbol);
147         }
148       }
149     }
150     FlowComputation flowComputation = new FlowComputation(allSymbolicValues, addToFlow, terminateTraversal, domains, skipExceptionMessages);
151     return flowComputation.run(currentNode, trackedSymbols);
152   }

158   private Set<Flow> run(final ExplodedGraph.Node node, PSet<Symbol> trackedSymbols) {
159     Set<Flow> flows = new HashSet<>();
160     Deque<ExecutionPath> workList = new ArrayDeque<>();
161     SameConstraints sameConstraints = new SameConstraints(node, trackedSymbols, domains);
162     node.edges().stream().flatMap(e -> startPath(e, trackedSymbols, sameConstraints)).forEach(workList::push);
163     int flowSteps = 0;
164     Set<ExecutionPath> visited = new HashSet<>(workList);
165     while (!workList.isEmpty()) {
166       ExecutionPath path = workList.pop();
167       if (path.finished) {
168         flows.add(path.flow);
169       } else {
170         path.lastEdge.parent.edges().stream()
171           .filter(path::notVisited)
172           .flatMap(path::addEdge)
173           .forEach(ep -> {
174             if(visited.add(ep)) {
175               workList.push(ep);
176             }
177           });
178       }
179       flowSteps++;
180       if(flowSteps == MAX_FLOW_STEPS) {
181         LOG.debug("Flow was not able to complete");
182         break;
183       }
184     }
185     return flows;
186   }

192   private static class SameConstraints {
193     private final List<Class<? extends Constraint>> domains;
194     private final Node node;
195     private PSet<Symbol> symbolsHavingAlwaysSameConstraints;
196 
197     SameConstraints(ExplodedGraph.Node startNode, PSet<Symbol> trackedSymbols, List<Class<? extends Constraint>> domains) {
198       this.domains = domains;
199       this.node = startNode;
200       this.symbolsHavingAlwaysSameConstraints = PCollections.emptySet();
201 
202       findSymbolsHavingAlwaysSameConstraints(trackedSymbols);
203     }
204 
205     SameConstraints(SameConstraints knownSameConstraints, PSet<Symbol> newTrackedSymbols) {
206       this.domains = knownSameConstraints.domains;
207       this.node = knownSameConstraints.node;
208       this.symbolsHavingAlwaysSameConstraints = knownSameConstraints.symbolsHavingAlwaysSameConstraints;
209 
210       findSymbolsHavingAlwaysSameConstraints(newTrackedSymbols);
211     }

213     private void findSymbolsHavingAlwaysSameConstraints(PSet<Symbol> trackedSymbols) {
214       trackedSymbols.forEach(symbol -> {
215         if (!symbolsHavingAlwaysSameConstraints.contains(symbol) && hasAlwaysSameConstraints(symbol)) {
216           symbolsHavingAlwaysSameConstraints = symbolsHavingAlwaysSameConstraints.add(symbol);
217         }
218       });
219     }
220 
221     private boolean hasAlwaysSameConstraints(Symbol symbol) {
222       return domains.stream().allMatch(domain -> sameConstraintWhenSameProgramPoint(node, symbol, domain));
223     }
224 
225     private static boolean sameConstraintWhenSameProgramPoint(ExplodedGraph.Node currentNode, Symbol symbol, Class<? extends Constraint> domain) {
226       ProgramState programState = currentNode.programState;
227       SymbolicValue sv = programState.getValue(symbol);
228       if (sv == null) {
229         return false;
230       }
231       Constraint constraint = programState.getConstraint(sv, domain);
232       if (constraint == null) {
233         return false;
234       }
235       Collection<Node> siblingNodes = currentNode.siblings();
236       return siblingNodes.stream()
237         .map(node -> node.programState)
238         .allMatch(ps -> {
239           SymbolicValue siblingSV = ps.getValue(symbol);
240           if (siblingSV == null) {
241             return false;
242           }
243           Constraint siblingConstraint = ps.getConstraint(siblingSV, domain);
244           return constraint.equals(siblingConstraint);
245         });
246     }
247 
248     public boolean hasAlwaysSameConstraint(@Nullable Symbol symbol) {
249       return symbol != null && symbolsHavingAlwaysSameConstraints.contains(symbol);
250     }
251   }

271     @Override

273       if (this == o) {
274         return true;
275       }
276       if (o == null || getClass() != o.getClass()) {
277         return false;
278       }
279       ExecutionPath that = (ExecutionPath) o;
280       return Objects.equals(trackedSymbols, that.trackedSymbols) &&

290     Stream<ExecutionPath> addEdge(ExplodedGraph.Edge edge) {
291       Flow.Builder flowBuilder = Flow.builder();
292       flowBuilder.addAll(flow);
293 
294       Flow laFlow = learnedAssociation(edge)
295         .map(la -> flowFromLearnedAssociation(la, edge.parent))
296         .orElse(Flow.empty());
297       flowBuilder.addAll(laFlow);
298 
299       PSet<Symbol> newTrackSymbols = newTrackedSymbols(edge);
300       SameConstraints newSameConstraints = newTrackSymbols == trackedSymbols ? sameConstraints : new SameConstraints(sameConstraints, newTrackSymbols);
301 
302       if (!skipExceptionMessages) {
303         flowFromThrownException(edge).ifPresent(loc -> {
304           flowBuilder.setAsExceptional();
305           flowBuilder.add(loc);
306         });
307         flowFromCaughtException(edge).ifPresent(loc -> {
308           flowBuilder.setAsExceptional();
309           flowBuilder.add(loc);
310         });
311       }
312 
313       Set<LearnedConstraint> learnedConstraints = learnedConstraints(edge);
314       Flow lcFlow = flowFromLearnedConstraints(edge, filterRedundantObjectDomain(learnedConstraints));
315       flowBuilder.addAll(lcFlow);
316 
317       boolean endOfPath = visitedAllParents(edge) || shouldTerminate(learnedConstraints);
318 
319       if (endOfPath) {
320         flowBuilder.addAll(flowForNullableMethodParameters(edge.parent));
321       }
322 
323       Flow currentFlow = flowBuilder.build();
324       Set<Flow> yieldsFlows = flowFromYields(edge);
325       if (yieldsFlows.isEmpty()) {
326         return Stream.of(new ExecutionPath(edge, visited.add(edge), newTrackSymbols, newSameConstraints, Flow.of(currentFlow), endOfPath));
327       }
328       return yieldsFlows.stream()
329         .map(yieldFlow -> Flow.builder().addAll(currentFlow).addAll(yieldFlow).build())
330         .map(f -> new ExecutionPath(edge, visited.add(edge), newTrackSymbols, newSameConstraints, f, endOfPath));
331     }

333     private Optional<JavaFileScannerContext.Location> flowFromThrownException(ExplodedGraph.Edge edge) {
334       SymbolicValue peekValue = edge.child.programState.peekValue();
335       if (peekValue instanceof SymbolicValue.ExceptionalSymbolicValue
336         && (isMethodInvocationNode(edge.parent) || isDivByZeroExceptionalYield(edge))) {
337         Type type = ((SymbolicValue.ExceptionalSymbolicValue) peekValue).exceptionType();
338         String msg = String.format("%s is thrown.", exceptionName(type));
339         return Optional.of(location(edge.parent, msg));
340       }
341       return Optional.empty();
342     }
343 
344     private boolean isDivByZeroExceptionalYield(ExplodedGraph.Edge edge) {
345       Tree tree = edge.parent.programPoint.syntaxTree();
346       return tree != null && tree.is(Tree.Kind.DIVIDE, Tree.Kind.DIVIDE_ASSIGNMENT);
347     }

349     private Optional<JavaFileScannerContext.Location> flowFromCaughtException(ExplodedGraph.Edge edge) {
350       ProgramPoint programPoint = edge.parent.programPoint;
351       if (((CFG.Block) programPoint.block).isCatchBlock() && programPoint.i == 0) {
352         VariableTree catchVariable = ((VariableTree) programPoint.syntaxTree());
353         SymbolicValue.CaughtExceptionSymbolicValue caughtSv = ((SymbolicValue.CaughtExceptionSymbolicValue) edge.child.programState.getValue(catchVariable.symbol()));
354         Preconditions.checkNotNull(caughtSv, "Caught exception not found in program state");
355         Type exceptionType = caughtSv.exception().exceptionType();
356         return Optional.of(location(edge.parent, String.format("%s is caught.", exceptionName(exceptionType))));
357       }
358       return Optional.empty();
359     }

361     private String exceptionName(@Nullable Type type) {
362       if (type == null || type.isUnknown()) {
363         return "Exception";
364       }
365       return "'" + type.name() + "'";
366     }

368     private Set<LearnedConstraint> filterRedundantObjectDomain(Set<LearnedConstraint> learnedConstraints) {
369       Map<SymbolicValue, Long> constraintsBySV = learnedConstraints.stream()
370         .collect(Collectors.groupingBy(LearnedConstraint::symbolicValue, Collectors.counting()));
371       return learnedConstraints.stream()
372         .flatMap(lc -> isConstraintFromObjectDomain(lc.constraint()) && constraintsBySV.get(lc.symbolicValue()) > 1 ? Stream.empty() : Stream.of(lc))
373         .collect(Collectors.toSet());
374     }

380     private Flow flowForNullableMethodParameters(ExplodedGraph.Node node) {
381       if (!node.edges().isEmpty() || !domains.contains(ObjectConstraint.class)) {
382         return Flow.empty();
383       }
384       Flow.Builder flowBuilder = Flow.builder();
385       trackedSymbols.forEach(symbol -> {
386         SymbolicValue sv = node.programState.getValue(symbol);
387         if (sv == null) {
388           return;
389         }
390         ObjectConstraint startConstraint = node.programState.getConstraint(sv, ObjectConstraint.class);
391         if (startConstraint != null && isMethodParameter(symbol)) {
392           String msg = IMPLIES_CAN_BE_MSG;
393           if (ObjectConstraint.NOT_NULL == startConstraint) {
394             msg = "Implies '%s' can not be %s.";
395           }
396           flowBuilder.add(new JavaFileScannerContext.Location(String.format(msg, symbol.name(), "null"), ((VariableTree) symbol.declaration()).simpleName()));
397         }
398       });
399       return flowBuilder.build();
400     }

402     private boolean isMethodParameter(Symbol symbol) {
403       Symbol owner = symbol.owner();
404       return owner.isMethodSymbol() && ((Symbol.MethodSymbol) owner).declaration().parameters().contains(symbol.declaration());
405     }

407     private Flow flowFromLearnedConstraints(ExplodedGraph.Edge edge, Set<LearnedConstraint> learnedConstraints) {
408       // FIXME SONARJAVA-2303 calling distinct is temporary workaround to avoid duplicated messages when constraint is reported on relational SV and also on its operand
409       Flow.Builder flowBuilder = Flow.builder();
410       learnedConstraints.stream()
411         .map(lc -> learnedConstraintFlow(lc, edge))
412         .flatMap(Flow::stream)
413         .distinct()
414         .forEach(flowBuilder::add);
415       return flowBuilder.build();
416     }

428     private Flow flowFromLearnedAssociation(LearnedAssociation learnedAssociation, ExplodedGraph.Node node) {
429       ProgramState programState = node.programState;
430       Preconditions.checkState(programState != null, "Learned association with null state in parent node of the edge.");
431       Symbol rhsSymbol = symbolFromStack(learnedAssociation.symbolicValue(), programState);
432       if (rhsSymbol != null) {
433         return Flow.of(location(node, String.format(IMPLIES_SAME_VALUE, learnedAssociation.symbol().name(), rhsSymbol.name())));
434       }
435       Flow.Builder flowBuilder = Flow.builder();
436       ConstraintsByDomain allConstraints = programState.getConstraints(learnedAssociation.symbolicValue());
437       Collection<Constraint> constraints = filterByDomains(allConstraints);
438       boolean isPrimitive = learnedAssociation.symbol.type().isPrimitive();
439       for (Constraint constraint : constraints) {
440         if (isPrimitive && constraint == ObjectConstraint.NOT_NULL) {
441           // don't add message about booleans being NOT_NULL, it's obvious
442           continue;
443         }
444         String symbolName = learnedAssociation.symbol().name();
445         String msg;
446         if (assigningNullFromTernary(node) || (assigningFromMethodInvocation(node) && assignedFromYieldWithUncertainResult(constraint, node))) {
447           msg = IMPLIES_CAN_BE_MSG;
448         } else {
449           msg = IMPLIES_IS_MSG;
450         }
451         flowBuilder.add(location(node, String.format(msg, symbolName, constraint.valueAsString())));
452       }
453       return flowBuilder.build();
454     }

456     private Collection<Constraint> filterByDomains(@Nullable ConstraintsByDomain allConstraints) {
457       if (allConstraints == null) {
458         return Collections.emptySet();
459       }
460       Map<Class<? extends  Constraint>, Constraint> constraints = new LinkedHashMap<>();
461       for (Class<? extends Constraint> domain : domains) {
462         Constraint constraint = allConstraints.get(domain);
463         if (constraint != null) {
464           constraints.put(domain, constraint);
465         }
466       }
467       if (constraints.size() > 1) {
468         constraints.remove(ObjectConstraint.class);
469       }
470       return constraints.values();
471     }

473     private boolean assigningNullFromTernary(Node node) {
474       ExpressionTree expr = getInitializer(node);
475       return isTernaryWithNullBranch(expr);
476     }

478     @CheckForNull
479     private ExpressionTree getInitializer(Node node) {
480       Tree tree = node.programPoint.syntaxTree();
481       ExpressionTree expr;
482       switch (tree.kind()) {
483         case VARIABLE:
484           expr = ((VariableTree) tree).initializer();
485           break;
486         case ASSIGNMENT:
487           expr = ((AssignmentExpressionTree) tree).expression();
488           break;
489         default:
490           expr = null;
491       }
492       return expr;
493     }

495     private boolean isTernaryWithNullBranch(@Nullable ExpressionTree expressionTree) {
496       if (expressionTree == null) {
497         return false;
498       }
499       ExpressionTree expr = ExpressionUtils.skipParentheses(expressionTree);
500       if (expr.is(Tree.Kind.CONDITIONAL_EXPRESSION)) {
501         ConditionalExpressionTree cet = (ConditionalExpressionTree) expr;
502         return ExpressionUtils.isNullLiteral(cet.trueExpression()) ^ ExpressionUtils.isNullLiteral(cet.falseExpression());
503       }
504       return false;
505     }

520     private PSet<Symbol> newTrackedSymbols(ExplodedGraph.Edge edge) {
521       Optional<LearnedAssociation> learnedAssociation = learnedAssociation(edge);
522       return learnedAssociation.map(la -> {
523         PSet<Symbol> newTrackedSymbols = trackedSymbols.remove(la.symbol);
524         ProgramState programState = edge.parent.programState;
525         Symbol symbol = symbolFromStack(la.symbolicValue(), programState);
526         if (symbol != null) {
527           newTrackedSymbols = newTrackedSymbols.add(symbol);
528         } else {
529           for (Symbol s : la.symbolicValue().computedFromSymbols()) {
530             newTrackedSymbols = newTrackedSymbols.add(s);
531           }
532         }
533         return newTrackedSymbols;
534       }).orElse(trackedSymbols);
535     }

539       if (programState != null && programState.peekValue() == symbolicValue) {
540         return programState.peekValueSymbol().symbol;
541       }
542       return null;
543     }

553     Set<LearnedConstraint> learnedConstraints(ExplodedGraph.Edge edge) {
554       Set<LearnedConstraint> learnedConstraints = edge.learnedConstraints();
555       ImmutableSet.Builder<LearnedConstraint> lcByDomainBuilder = ImmutableSet.builder();
556       // guarantee that we will keep the same domain order when reporting
557       for (Class<? extends Constraint> domain : domains) {
558         learnedConstraints.stream()
559           .filter(lc -> symbolicValues.contains(lc.symbolicValue()) && hasConstraintForDomain(lc, domain))
560           .forEach(lcByDomainBuilder::add);
561       }
562 
563       return lcByDomainBuilder.build();
564     }

567       return domain.isAssignableFrom(lc.constraint.getClass());

570     private Flow learnedConstraintFlow(LearnedConstraint learnedConstraint, ExplodedGraph.Edge edge) {
571       Constraint constraint = learnedConstraint.constraint();
572       if (!addToFlow.test(constraint)) {
573         return Flow.empty();
574       }
575       if (constraint == ObjectConstraint.NOT_NULL && learnedConstraint.sv instanceof BinarySymbolicValue) {
576         // don't report on binary expression to be not null, it's obvious
577         return Flow.empty();
578       }
579       ExplodedGraph.Node parent = edge.parent;
580       Tree nodeTree = parent.programPoint.syntaxTree();
581       if (isMethodInvocationNode(parent)) {
582         return methodInvocationFlow(constraint, edge);
583       }
584       if (nodeTree.is(Tree.Kind.NEW_CLASS)) {
585         return Flow.of(location(parent, String.format("Constructor implies '%s'.", constraint.valueAsString())));
586       }
587       Symbol finalField = learnedConstraintOnInitializedFinalField(nodeTree);
588       if (finalField != null) {
589         // constraints on final fields are set on the fly by the EGW when encountering them for the first time
590         String msg = String.format(IMPLIES_IS_MSG, finalField.name(), constraint.valueAsString());
591         return Flow.of(new JavaFileScannerContext.Location(msg, ((VariableTree) finalField.declaration()).initializer()));
592       }
593 
594       Symbol trackedSymbol = getSymbol(parent.programState, learnedConstraint.sv);
595       String name = trackedSymbol != null ? trackedSymbol.name() : SyntaxTreeNameFinder.getName(nodeTree);
596       if (name == null) {
597         // unable to deduce name of element on which we learn a constraint. Nothing is reported
598         return Flow.empty();
599       }
600       String msg;
601       if (ObjectConstraint.NULL == constraint && !sameConstraints.hasAlwaysSameConstraint(trackedSymbol)) {
602         msg = IMPLIES_CAN_BE_MSG;
603       } else {
604         msg = IMPLIES_IS_MSG;
605       }
606       return Flow.of(location(parent, String.format(msg, name, constraint.valueAsString())));
607     }
608 
609     @CheckForNull
610     private Symbol getSymbol(ProgramState programState, SymbolicValue sv) {
611       SymbolicValue peekValue = programState.peekValue();
612       if (sv.equals(peekValue)) {
613         return programState.peekValueSymbol().symbol;
614       }
615       if (peekValue instanceof BinarySymbolicValue) {
616         BinarySymbolicValue bsv = (BinarySymbolicValue) peekValue;
617         if (sv.equals(bsv.getRightOp())) {
618           return bsv.rightSymbol();
619         }
620         return bsv.leftSymbol();
621       }
622       return null;
623     }
624 
625     @CheckForNull
626     private Symbol learnedConstraintOnInitializedFinalField(Tree syntaxTree) {
627       Symbol result = null;
628       if (syntaxTree.is(Tree.Kind.IDENTIFIER)) {
629         result = ((IdentifierTree) syntaxTree).symbol();
630       } else if (syntaxTree.is(Tree.Kind.MEMBER_SELECT)) {
631         MemberSelectExpressionTree mset = (MemberSelectExpressionTree) syntaxTree;
632         if (ExpressionUtils.isSelectOnThisOrSuper(mset)) {
633           result = mset.identifier().symbol();
634         }
635       }
636       if (isFinalFieldWithInitializer(result)) {
637         return result;
638       }
639       return null;
640     }

642     private boolean isFinalFieldWithInitializer(@Nullable Symbol symbol) {
643       if (symbol != null && symbol.isVariableSymbol() && symbol.owner().isTypeSymbol() && symbol.isFinal()) {
644         VariableTree declaration = ((Symbol.VariableSymbol) symbol).declaration();
645         return declaration != null && declaration.initializer() != null;
646       }
647       return false;
648     }

650     private Flow methodInvocationFlow(Constraint learnedConstraint, ExplodedGraph.Edge edge) {
651       ExplodedGraph.Node parent = edge.parent;
652       MethodInvocationTree mit = (MethodInvocationTree) parent.programPoint.syntaxTree();
653       Flow.Builder flowBuilder = Flow.builder();
654       SymbolicValue returnSV = edge.child.programState.peekValue();
655       if (symbolicValues.contains(returnSV)) {
656         flowBuilder.add(methodInvocationReturnMessage(learnedConstraint, edge, mit.symbol().name()));
657       }
658       SymbolicValue invocationTarget = parent.programState.peekValue(mit.arguments().size());
659       if (symbolicValues.contains(invocationTarget)) {
660         String invocationTargetName = SyntaxTreeNameFinder.getName(mit.methodSelect());
661         flowBuilder.add(location(parent, String.format(IMPLIES_IS_MSG, invocationTargetName, learnedConstraint.valueAsString())));
662       }
663 
664       List<Integer> argumentIndices = correspondingArgumentIndices(symbolicValues, parent);
665       argumentIndices.stream()
666         .map(mit.arguments()::get)
667         .map(argTree -> {
668           String message = String.format(IMPLIES_IS_MSG, SyntaxTreeNameFinder.getName(argTree), learnedConstraint.valueAsString());
669           return new JavaFileScannerContext.Location(message, argTree);
670         })
671         .forEach(flowBuilder::add);
672 
673       return flowBuilder.build();
674     }

676     private JavaFileScannerContext.Location methodInvocationReturnMessage(Constraint constraint, ExplodedGraph.Edge edge, String methodName) {
677       String msg;
678       if (isConstraintOnlyPossibleResult(constraint, edge)) {
679         msg = String.format("'%s()' returns %s.", methodName, constraint.valueAsString());
680       } else {
681         msg = String.format("'%s()' can return %s.", methodName, constraint.valueAsString());
682       }
683       return location(edge.parent, msg);
684     }

686     private boolean isConstraintOnlyPossibleResult(Constraint constraint, ExplodedGraph.Edge edge) {
687       Set<MethodYield> selectedYields = edge.yields();
688       if (selectedYields.isEmpty()) {
689         // not based on x-procedural analysis, so certainty of constraint is not be guaranteed
690         return false;
691       }
692       MethodBehavior methodBehavior = selectedYields.iterator().next().methodBehavior();
693       return methodBehavior.happyPathYields()
694         .map(HappyPathYield::resultConstraint)
695         .allMatch(resultConstraint -> resultConstraint != null && constraint.equals(resultConstraint.get(constraint.getClass())));
696     }

698     private Set<Flow> flowFromYields(ExplodedGraph.Edge edge) {
699       Set<MethodYield> methodYields = edge.yields();
700       if (methodYields.isEmpty()) {
701         return Collections.emptySet();
702       }
703 
704       List<Integer> argumentIndices = correspondingArgumentIndices(symbolicValues, edge.parent);
705 
706       MethodInvocationTree mit = (MethodInvocationTree) edge.parent.programPoint.syntaxTree();
707       // computes flow messages for arguments being passed to the called method
708       Flow passedArgumentsMessages = flowsForPassedArguments(argumentIndices, mit);
709       // computes flow messages for arguments changing name within called method
710       Flow changingNameArgumentsMessages = flowsForArgumentsChangingName(argumentIndices, mit);
711 
712       SymbolicValue returnSV = edge.child.programState.peekValue();
713       if (symbolicValues.contains(returnSV)) {
714         // to retrieve flow for return value
715         argumentIndices.add(-1);
716       }
717       if (argumentIndices.isEmpty()) {
718         // no need to compute any flow on yields : no arg nor return value are corresponding to tracked SVs
719         return Collections.emptySet();
720       }
721       return methodYields.stream()
722         .map(y -> y.flow(argumentIndices, domains))
723         .flatMap(Set::stream)
724         .filter(f -> !f.isEmpty())
725         .map(flowFromYield -> Flow.builder()
726           .addAll(flowFromYield)
727           .addAll(changingNameArgumentsMessages)
728           .addAll(passedArgumentsMessages)
729           .build())
730         .collect(Collectors.toSet());
731     }

733     private boolean isMethodInvocationNode(ExplodedGraph.Node node) {
734       // ProgramPoint#syntaxTree will not always return the correct tree, so we need to go to ProgramPoint#block directly
735       ProgramPoint pp = node.programPoint;
736       if (pp.i < pp.block.elements().size()) {
737         Tree tree = ((CFG.Block) pp.block).elements().get(pp.i);
738         return tree.is(Tree.Kind.METHOD_INVOCATION);
739       }
740       return false;
741     }

743     private List<Integer> correspondingArgumentIndices(Set<SymbolicValue> candidates, ExplodedGraph.Node invocationNode) {
744       MethodInvocationTree mit = (MethodInvocationTree) invocationNode.programPoint.syntaxTree();
745       List<SymbolicValue> arguments = argumentsUsedForMethodInvocation(invocationNode, mit);
746       return IntStream.range(0, arguments.size())
747         .filter(i -> candidates.contains(arguments.get(i)))
748         .boxed().collect(Collectors.toList());
749     }

751     private List<SymbolicValue> argumentsUsedForMethodInvocation(ExplodedGraph.Node invocationNode, MethodInvocationTree mit) {
752       List<SymbolicValue> values = invocationNode.programState.peekValues(mit.arguments().size());
753       return Lists.reverse(values);
754     }

757       return new JavaFileScannerContext.Location(message, node.programPoint.syntaxTree());

761   public static Flow flowsForPassedArguments(List<Integer> argumentIndices, MethodInvocationTree mit) {
762     String methodName = mit.symbol().name();
763     Flow.Builder flowBuilder = Flow.builder();
764     argumentIndices.stream()
765       .map(index -> getArgumentIdentifier(mit, index))
766       .filter(Objects::nonNull)
767       .map(identifierTree -> new JavaFileScannerContext.Location(String.format("'%s' is passed to '%s()'.", identifierTree.name(), methodName), identifierTree))
768       .forEach(flowBuilder::add);
769     return flowBuilder.build().reverse();
770   }

772   public static Flow flowsForArgumentsChangingName(List<Integer> argumentIndices, MethodInvocationTree mit) {
773 
774     JavaSymbol.MethodJavaSymbol methodSymbol = (JavaSymbol.MethodJavaSymbol) mit.symbol();
775     MethodTree declaration = methodSymbol.declaration();
776     if (declaration == null) {
777       return Flow.empty();
778     }
779     Flow.Builder flowBuilder = Flow.builder();
780     List<VariableTree> methodParameters = declaration.parameters();
781 
782     for (Integer argumentIndex : argumentIndices) {
783       // do not consider varargs part
784       if (methodSymbol.isVarArgs() && argumentIndex >= methodParameters.size() - 1) {
785         break;
786       }
787       IdentifierTree argumentName = getArgumentIdentifier(mit, argumentIndex);
788       if (argumentName != null) {
789         IdentifierTree parameterIdentifier = methodParameters.get(argumentIndex).simpleName();
790         String identifierName = parameterIdentifier.name();
791         if (!argumentName.name().equals(identifierName)) {
792           flowBuilder.add(new JavaFileScannerContext.Location(String.format(IMPLIES_SAME_VALUE, identifierName, argumentName.name()), parameterIdentifier));
793         }
794       }
795     }
796     return flowBuilder.build().reverse();
797   }

799   @CheckForNull
800   public static IdentifierTree getArgumentIdentifier(MethodInvocationTree mit, int index) {
801     Arguments arguments = mit.arguments();
802     if (index < 0 || index > arguments.size()) {
803       throw new IllegalArgumentException("index must be within arguments range.");
804     }
805     ExpressionTree expr = ExpressionUtils.skipParentheses(arguments.get(index));
806     switch (expr.kind()) {
807       case MEMBER_SELECT:
808         return ((MemberSelectExpressionTree) expr).identifier();
809       case IDENTIFIER:
810         return ((IdentifierTree) expr);
811       default:
812         return null;
813     }
814   }

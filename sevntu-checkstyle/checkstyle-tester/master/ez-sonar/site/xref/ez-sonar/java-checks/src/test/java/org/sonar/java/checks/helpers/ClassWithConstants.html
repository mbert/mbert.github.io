20  package org.sonar.java.checks.helpers;
21  
22  import static java.lang.Boolean.TRUE;
23  
24  // used by ConstantUtilsTest
25  public class ClassWithConstants {
26  
27    public static final String CONSTANT1 = "abc";
28    public static final String CONSTANT2 = CONSTANT1 + "def";
29    public static final int INT_CONSTANT1 = 42;
30    public static final long LONG_CONSTANT1 = 99L;
31    public static final Object OBJECT_CONSTANT = new Object();
32    public static final boolean BOOLEAN_CONSTANT = false;
33  
34    public void literals(String param) {
35      System.out.println("hello");
36      System.out.println(true);
37      System.out.println(43);
38      System.out.println(77L);
39      System.out.println(param);
40      System.out.println(1_000);
41      System.out.println(0x99567L);
42    }
43  
44    public void identifiers(String param) {
45      System.out.println(CONSTANT1);
46      System.out.println(CONSTANT2);
47      System.out.println(INT_CONSTANT1);
48      System.out.println(LONG_CONSTANT1);
49      System.out.println(TRUE);
50      System.out.println(BOOLEAN_CONSTANT);
51    }
52  
53    public void parentheses(String parentheses) {
54      System.out.println(((CONSTANT1)));
55      System.out.println(((INT_CONSTANT1)));
56    }
57  
58    public void memberSelect(String parentheses) {
59      System.out.println(ClassWithConstants.CONSTANT1);
60      System.out.println(Boolean.TRUE);
61      System.out.println(Boolean.FALSE);
62    }
63  
64    public void plus(String param) {
65      System.out.println("hello " + CONSTANT1);
66      System.out.println("hello " + param);
67      System.out.println(param + "hello ");
68      System.out.println("hello" + INT_CONSTANT1);
69      System.out.println(INT_CONSTANT1 + "hello");
70      System.out.println(INT_CONSTANT1 + 1);
71      System.out.println(LONG_CONSTANT1 + 1);
72      System.out.println(2 + LONG_CONSTANT1);
73      System.out.println(3L + LONG_CONSTANT1);
74    }
75  
76    public void other() {
77      System.out.println(String.valueOf(1));
78      System.out.println(OBJECT_CONSTANT);
79    }
80  
81  }

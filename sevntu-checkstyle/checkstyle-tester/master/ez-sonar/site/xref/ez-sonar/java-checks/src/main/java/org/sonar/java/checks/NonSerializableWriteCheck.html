20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  import com.google.common.collect.Lists;
24  import org.sonar.check.Rule;
25  import org.sonar.java.matcher.MethodMatcher;
26  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
27  import org.sonar.plugins.java.api.JavaFileScannerContext;
28  import org.sonar.plugins.java.api.semantic.Symbol;
29  import org.sonar.plugins.java.api.tree.ExpressionTree;
30  import org.sonar.plugins.java.api.tree.IdentifierTree;
31  import org.sonar.plugins.java.api.tree.InstanceOfTree;
32  import org.sonar.plugins.java.api.tree.MethodInvocationTree;
33  import org.sonar.plugins.java.api.tree.Tree;
34  import org.sonar.plugins.java.api.tree.Tree.Kind;
35  
36  import java.util.List;
37  
38  @Rule(key = "S2118")
39  public class NonSerializableWriteCheck extends IssuableSubscriptionVisitor {
40  
41    private static final MethodMatcher WRITE_OBJECT_MATCHER = MethodMatcher.create()
42      .typeDefinition("java.io.ObjectOutputStream")
43      .name("writeObject")
44      .addParameter("java.lang.Object");
45  
46    private final List<Symbol> testedSymbols = Lists.newArrayList();
47  
48    @Override
49    public List<Kind> nodesToVisit() {
50      return ImmutableList.of(Tree.Kind.METHOD_INVOCATION, Tree.Kind.INSTANCE_OF);
51    }
52  
53    @Override
54    public void scanFile(JavaFileScannerContext context) {
55      testedSymbols.clear();
56      super.scanFile(context);
57    }
58  
59    @Override
60    public void visitNode(Tree tree) {
61      if (hasSemantic()) {
62        if (tree.is(Tree.Kind.METHOD_INVOCATION)) {
63          visitMethodInvocation((MethodInvocationTree) tree);
64        } else {
65          visitInstanceOf((InstanceOfTree) tree);
66        }
67      }
68    }
69  
70    private void visitInstanceOf(InstanceOfTree instanceOfTree) {
71      ExpressionTree expression = instanceOfTree.expression();
72      if (expression.is(Tree.Kind.IDENTIFIER) && instanceOfTree.type().symbolType().is("java.io.Serializable")) {
73        testedSymbols.add(((IdentifierTree) expression).symbol());
74      }
75    }
76  
77    // If we met a test such as "x instanceof Serializable", we suppose that symbol x is Serializable
78    private boolean isTestedSymbol(ExpressionTree tree) {
79      if (tree.is(Tree.Kind.IDENTIFIER)) {
80        Symbol symbol = ((IdentifierTree) tree).symbol();
81        return testedSymbols.contains(symbol);
82      }
83      return false;
84    }
85  
86    private void visitMethodInvocation(MethodInvocationTree methodInvocation) {
87      if (WRITE_OBJECT_MATCHER.matches(methodInvocation)) {
88        ExpressionTree argument = methodInvocation.arguments().get(0);
89        if (!isAcceptableType(argument.symbolType()) && !isTestedSymbol(argument)) {
90          reportIssue(argument, "Make the \"" + argument.symbolType().fullyQualifiedName() + "\" class \"Serializable\" or don't write it.");
91        }
92      }
93    }
94  
95    private static boolean isAcceptableType(org.sonar.plugins.java.api.semantic.Type argType) {
96      return argType.isSubtypeOf("java.io.Serializable")
97        || argType.is("java.lang.Object")
98        || argType.isPrimitive()
99        || !argType.isClass();
100   }

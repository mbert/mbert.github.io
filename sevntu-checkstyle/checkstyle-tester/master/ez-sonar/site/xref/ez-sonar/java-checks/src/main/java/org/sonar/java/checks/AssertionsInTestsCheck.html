111   private final Deque<Boolean> methodContainsAssertion = new ArrayDeque<>();
112   private final Deque<Boolean> inUnitTest = new ArrayDeque<>();
113   private JavaFileScannerContext context;
114 
115   @Override
116   public void scanFile(final JavaFileScannerContext context) {
117     this.context = context;
118     inUnitTest.push(false);
119     methodContainsAssertion.push(false);
120     scan(context.getTree());
121     methodContainsAssertion.pop();
122     inUnitTest.pop();
123   }
124 
125   @Override
126   public void visitMethod(MethodTree methodTree) {
127     if (ModifiersUtils.hasModifier(methodTree.modifiers(), Modifier.ABSTRACT)) {
128       return;
129     }
130     boolean isUnitTest = isUnitTest(methodTree);
131     inUnitTest.push(isUnitTest);
132     methodContainsAssertion.push(false);
133     super.visitMethod(methodTree);
134     Boolean containsAssertion = methodContainsAssertion.pop();
135     inUnitTest.pop();
136     if (isUnitTest && !expectAssertion(methodTree) && !containsAssertion) {
137       context.reportIssue(this, methodTree.simpleName(), "Add at least one assertion to this test case.");
138     }
139   }
140 
141   @Override
142   public void visitMethodInvocation(MethodInvocationTree mit) {
143     super.visitMethodInvocation(mit);
144     if (!methodContainsAssertion.peek() && inUnitTest.peek() && ASSERTION_INVOCATION_MATCHERS.anyMatch(mit)) {
145       methodContainsAssertion.pop();
146       methodContainsAssertion.push(true);
147     }
148   }
149 
150   private static boolean expectAssertion(MethodTree methodTree) {
151     List<SymbolMetadata.AnnotationValue> annotationValues = methodTree.symbol().metadata().valuesForAnnotation("org.junit.Test");
152     if (annotationValues != null) {
153       for (SymbolMetadata.AnnotationValue annotationValue : annotationValues) {
154         if ("expected".equals(annotationValue.name())) {
155           return true;
156         }
157       }
158     }
159     return false;
160   }
161 
162   private static boolean isUnitTest(MethodTree methodTree) {
163     Symbol.MethodSymbol symbol = methodTree.symbol();
164     while (symbol != null) {
165       if (symbol.metadata().isAnnotatedWith("org.junit.Test")) {
166         return true;
167       }
168       symbol = symbol.overriddenSymbol();
169     }
170     Symbol.TypeSymbol enclosingClass = methodTree.symbol().enclosingClass();
171     return enclosingClass != null && enclosingClass.type().isSubtypeOf("junit.framework.TestCase") && methodTree.simpleName().name().startsWith("test");
172   }

174   private static MethodMatcher method(String typeDefinition, String methodName) {
175     return method(TypeCriteria.is(typeDefinition), NameCriteria.is(methodName));
176   }
177 
178   private static MethodMatcher method(TypeCriteria typeDefinitionCriteria, String methodName) {
179     return method(typeDefinitionCriteria, NameCriteria.is(methodName));
180   }
181 
182   private static MethodMatcher method(String typeDefinition, NameCriteria nameCriteria) {
183     return MethodMatcher.create().typeDefinition(TypeCriteria.is(typeDefinition)).name(nameCriteria);
184   }
185 
186   private static MethodMatcher method(TypeCriteria typeDefinitionCriteria, NameCriteria nameCriteria) {
187     return MethodMatcher.create().typeDefinition(typeDefinitionCriteria).name(nameCriteria);
188   }

20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  
24  import org.sonar.check.Rule;
25  import org.sonar.java.JavaVersionAwareVisitor;
26  import org.sonar.java.matcher.MethodMatcher;
27  import org.sonar.java.matcher.MethodMatcherCollection;
28  import org.sonar.java.matcher.TypeCriteria;
29  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
30  import org.sonar.plugins.java.api.JavaVersion;
31  import org.sonar.plugins.java.api.semantic.Symbol;
32  import org.sonar.plugins.java.api.semantic.Type;
33  import org.sonar.plugins.java.api.tree.ClassTree;
34  import org.sonar.plugins.java.api.tree.IdentifierTree;
35  import org.sonar.plugins.java.api.tree.MethodTree;
36  import org.sonar.plugins.java.api.tree.Tree;
37  
38  import java.util.List;
39  
40  @Rule(key = "S1609")
41  public class SAMAnnotatedCheck extends IssuableSubscriptionVisitor implements JavaVersionAwareVisitor {
42  
43    private static final MethodMatcherCollection OBJECT_METHODS = MethodMatcherCollection.create(
44      methodMatcherWithName("equals", "java.lang.Object"),
45      methodMatcherWithName("getClass"),
46      methodMatcherWithName("hashcode"),
47      methodMatcherWithName("notify"),
48      methodMatcherWithName("notifyAll"),
49      methodMatcherWithName("toString"),
50      methodMatcherWithName("wait"),
51      methodMatcherWithName("wait", "long"),
52      methodMatcherWithName("wait", "long", "int"));
53  
54    @Override
55    public boolean isCompatibleWithJavaVersion(JavaVersion version) {
56      return version.isJava8Compatible();
57    }
58  
59    @Override
60    public List<Tree.Kind> nodesToVisit() {
61      return ImmutableList.of(Tree.Kind.INTERFACE);
62    }
63  
64    @Override
65    public void visitNode(Tree tree) {
66      if (!hasSemantic()) {
67        return;
68      }
69      ClassTree classTree = (ClassTree) tree;
70      if (hasOneAbstractMethod(classTree.symbol()) && !isAnnotated(classTree)) {
71        IdentifierTree simpleName = classTree.simpleName();
72        reportIssue(
73          simpleName,
74          "Annotate the \"" + simpleName.name() + "\" interface with the @FunctionalInterface annotation" + context.getJavaVersion().java8CompatibilityMessage());
75      }
76    }
77  
78    private static boolean isAnnotated(ClassTree tree) {
79      return tree.symbol().metadata().isAnnotatedWith("java.lang.FunctionalInterface");
80    }
81  
82    private static boolean hasOneAbstractMethod(Symbol.TypeSymbol symbol) {
83      return numberOfAbstractMethod(symbol) == 1 && noAbstractMethodInParentInterfaces(symbol.interfaces());
84    }
85  
86    private static boolean noAbstractMethodInParentInterfaces(List<Type> interfaces) {
87      return interfaces.stream()
88        .map(Type::symbol)
89        .noneMatch(symbol -> numberOfAbstractMethod(symbol) > 0 || !noAbstractMethodInParentInterfaces(symbol.interfaces()));
90    }
91  
92    private static long numberOfAbstractMethod(Symbol symbol) {
93      if (!symbol.isTypeSymbol()) {
94        // unknown interface
95        return Integer.MAX_VALUE;
96      }
97  
98      Symbol.TypeSymbol interfaceSymbol = (Symbol.TypeSymbol) symbol;
99      return interfaceSymbol.memberSymbols().stream()
100       .filter(Symbol::isMethodSymbol)
101       .filter(member -> {
102         Symbol.MethodSymbol methodSymbol = (Symbol.MethodSymbol) member;
103         return isNotObjectMethod(methodSymbol) && methodSymbol.isAbstract();
104       }).count();
105   }
106 
107 
108   private static boolean isNotObjectMethod(Symbol.MethodSymbol method) {
109     MethodTree declaration = method.declaration();
110     return declaration == null || !OBJECT_METHODS.anyMatch(declaration);
111   }
112 
113   private static MethodMatcher methodMatcherWithName(String name, String... parameters) {
114     return MethodMatcher.create().typeDefinition(TypeCriteria.anyType()).name(name).parameters(parameters);
115   }
116 }

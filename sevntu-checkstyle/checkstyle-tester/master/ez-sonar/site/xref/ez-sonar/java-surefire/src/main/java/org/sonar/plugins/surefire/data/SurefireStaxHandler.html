20  package org.sonar.plugins.surefire.data;
21  
22  import java.text.ParseException;
23  import java.util.Locale;
24  import javax.xml.stream.XMLStreamException;
25  import org.apache.commons.lang.StringUtils;
26  import org.codehaus.staxmate.in.ElementFilter;
27  import org.codehaus.staxmate.in.SMEvent;
28  import org.codehaus.staxmate.in.SMHierarchicCursor;
29  import org.codehaus.staxmate.in.SMInputCursor;
30  import org.sonar.api.utils.ParsingUtils;
31  
32  public class SurefireStaxHandler {
33  
34    private final UnitTestIndex index;
35  
36    public SurefireStaxHandler(UnitTestIndex index) {
37      this.index = index;
38    }
39  
40    public void stream(SMHierarchicCursor rootCursor) throws XMLStreamException {
41      SMInputCursor testSuite = rootCursor.constructDescendantCursor(new ElementFilter("testsuite"));
42      SMEvent testSuiteEvent;
43      for (testSuiteEvent = testSuite.getNext(); testSuiteEvent != null; testSuiteEvent = testSuite.getNext()) {
44        if (testSuiteEvent.compareTo(SMEvent.START_ELEMENT) == 0) {
45          String testSuiteClassName = testSuite.getAttrValue("name");
46          if (StringUtils.contains(testSuiteClassName, "$")) {
47            // test suites for inner classes are ignored
48            return;
49          }
50          parseTestCase(testSuiteClassName, testSuite.childCursor(new ElementFilter("testcase")));
51        }
52      }
53    }
54  
55    private void parseTestCase(String testSuiteClassName, SMInputCursor testCase) throws XMLStreamException {
56      for (SMEvent event = testCase.getNext(); event != null; event = testCase.getNext()) {
57        if (event.compareTo(SMEvent.START_ELEMENT) == 0) {
58          String testClassName = getClassname(testCase, testSuiteClassName);
59          UnitTestClassReport classReport = index.index(testClassName);
60          parseTestCase(testCase, testSuiteClassName, classReport);
61        }
62      }
63    }
64  
65    private static String getClassname(SMInputCursor testCaseCursor, String defaultClassname) throws XMLStreamException {
66      String testClassName = testCaseCursor.getAttrValue("classname");
67      if(StringUtils.isNotBlank(testClassName) && testClassName.endsWith(")")) {
68        testClassName = testClassName.substring(0, testClassName.indexOf('('));
69      }
70      return StringUtils.defaultIfBlank(testClassName, defaultClassname);
71    }
72  
73    private static void parseTestCase(SMInputCursor testCaseCursor, String testSuiteClassName, UnitTestClassReport report) throws XMLStreamException {
74      report.add(parseTestResult(testCaseCursor, testSuiteClassName));
75    }
76  
77    private static void setStackAndMessage(UnitTestResult result, SMInputCursor stackAndMessageCursor) throws XMLStreamException {
78      result.setMessage(stackAndMessageCursor.getAttrValue("message"));
79      String stack = stackAndMessageCursor.collectDescendantText();
80      result.setStackTrace(stack);
81    }
82  
83    private static UnitTestResult parseTestResult(SMInputCursor testCaseCursor, String testSuiteClassName) throws XMLStreamException {
84      UnitTestResult detail = new UnitTestResult();
85      String name = getTestCaseName(testCaseCursor);
86      detail.setName(name);
87      detail.setTestSuiteClassName(testSuiteClassName);
88  
89      String status = UnitTestResult.STATUS_OK;
90      String time = testCaseCursor.getAttrValue("time");
91      Long duration = null;
92  
93      SMInputCursor childNode = testCaseCursor.descendantElementCursor();
94      if (childNode.getNext() != null) {
95        String elementName = childNode.getLocalName();
96        if ("skipped".equals(elementName)) {
97          status = UnitTestResult.STATUS_SKIPPED;
98          // bug with surefire reporting wrong time for skipped tests
99          duration = 0L;
100 
101       } else if ("failure".equals(elementName)) {
102         status = UnitTestResult.STATUS_FAILURE;
103         setStackAndMessage(detail, childNode);
104 
105       } else if ("error".equals(elementName)) {
106         status = UnitTestResult.STATUS_ERROR;
107         setStackAndMessage(detail, childNode);
108       }
109     }
110     while (childNode.getNext() != null) {
111       // make sure we loop till the end of the elements cursor
112     }
113     if (duration == null) {
114       duration = getTimeAttributeInMS(time);
115     }
116     detail.setDurationMilliseconds(duration);
117     detail.setStatus(status);
118     return detail;
119   }
120 
121   private static long getTimeAttributeInMS(String value) throws XMLStreamException {
122     // hardcoded to Locale.ENGLISH see http://jira.codehaus.org/browse/SONAR-602
123     try {
124       Double time = ParsingUtils.parseNumber(value, Locale.ENGLISH);
125       return !Double.isNaN(time) ? (long) ParsingUtils.scaleValue(time * 1000, 3) : 0L;
126     } catch (ParseException e) {
127       throw new XMLStreamException(e);
128     }
129   }
130 
131   private static String getTestCaseName(SMInputCursor testCaseCursor) throws XMLStreamException {
132     String classname = testCaseCursor.getAttrValue("classname");
133     String name = testCaseCursor.getAttrValue("name");
134     if (StringUtils.contains(classname, "$")) {
135       return StringUtils.substringAfter(classname, "$") + "/" + name;
136     }
137     return name;
138   }

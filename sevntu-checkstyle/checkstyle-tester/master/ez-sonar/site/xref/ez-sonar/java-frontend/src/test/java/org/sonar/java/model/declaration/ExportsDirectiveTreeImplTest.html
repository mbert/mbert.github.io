47  public class ExportsDirectiveTreeImplTest {
48  
49    private final ActionParser<Tree> p = JavaParser.createParser();
50  
51    private CompilationUnitTree createTree(String... lines) {
52      CompilationUnitTree compilationUnitTree = (CompilationUnitTree) p.parse(Arrays.stream(lines).collect(Collectors.joining("\n")));
53      SemanticModel.createFor(compilationUnitTree, new SquidClassLoader(Collections.emptyList()));
54      return compilationUnitTree;
55    }
56  
57    private ExportsDirectiveTree exportsDirective(String exportsDirective) {
58      CompilationUnitTree compilationUnitTree = createTree("module org.foo {\n  " + exportsDirective + "\n}");
59      SemanticModel.createFor(compilationUnitTree, new SquidClassLoader(Collections.emptyList()));
60      return (ExportsDirectiveTree) compilationUnitTree.moduleDeclaration().moduleDirectives().get(0);
61    }
62  
63    @Test
64    public void simple_exports() {
65      ExportsDirectiveTree exports = exportsDirective("exports foo;");
66  
67      assertThat(exports.kind()).isEqualTo(Tree.Kind.EXPORTS_DIRECTIVE);
68      assertThat(exports.directiveKeyword().text()).isEqualTo("exports");
69      assertThat(((IdentifierTree) exports.packageName()).name()).isEqualTo("foo");
70      assertThat(exports.toKeyword()).isNull();
71      assertThat(exports.moduleNames()).isEmpty();
72      assertThat(exports.semicolonToken().text()).isEqualTo(";");
73    }
74  
75    @Test
76    public void exports_with_modules() {
77      ExportsDirectiveTree exports = exportsDirective("exports org.foo to com.module1, module2;");
78  
79      assertThat(exports.kind()).isEqualTo(Tree.Kind.EXPORTS_DIRECTIVE);
80      assertThat(exports.directiveKeyword().text()).isEqualTo("exports");
81      ExpressionTree packageName = exports.packageName();
82      assertThat(packageName.is(Tree.Kind.MEMBER_SELECT)).isTrue();
83      MemberSelectExpressionTree mset = (MemberSelectExpressionTree) packageName;
84      assertThat(((IdentifierTree) mset.expression()).name()).isEqualTo("org");
85      assertThat(mset.identifier().name()).isEqualTo("foo");
86      assertThat(exports.toKeyword().text()).isEqualTo("to");
87  
88      ListTree<ModuleNameTree> moduleNames = exports.moduleNames();
89      assertThat(moduleNames).hasSize(2);
90      assertThat(moduleNames.get(0).stream().map(IdentifierTree::name)).containsExactly("com", "module1");
91      assertThat(moduleNames.separators()).hasSize(1);
92      assertThat(moduleNames.separators().iterator().next().text()).isEqualTo(",");
93      assertThat(moduleNames.get(1).stream().map(IdentifierTree::name)).containsExactly("module2");
94  
95      assertThat(exports.semicolonToken().text()).isEqualTo(";");
96    }
97  
98    @Test
99    public void test_BaseTreeVisitor() {
100     CompilationUnitTree cut = createTree(
101       "@org.foo.Bar",
102       "open module com.greetings {",
103       "  exports foo;",
104       "  requires org.gul;",
105       "  exports org.bar to com.module1, module2;",
106       "}");
107     ExportsDirectiveVisitor moduleDeclarationVisitor = new ExportsDirectiveVisitor();
108     cut.accept(moduleDeclarationVisitor);
109     assertThat(moduleDeclarationVisitor.visited).isTrue();
110     assertThat(moduleDeclarationVisitor.directives).hasSize(2);
111     assertThat(moduleDeclarationVisitor.identifiers).containsExactly("com", "greetings", "foo", "com", "module1", "module2");
112   }
113 
114   private static class ExportsDirectiveVisitor extends BaseTreeVisitor {
115 
116     boolean visited = false;
117     List<ExportsDirectiveTree> directives = new ArrayList<>();
118     List<String> identifiers = new ArrayList<>();
119 
120     @Override
121     public void visitRequiresDirectiveTree(RequiresDirectiveTree tree) {
122       // skip requires directives
123     }
124 
125     @Override
126     public void visitExportsDirectiveTree(ExportsDirectiveTree tree) {
127       visited = true;
128       directives.add(tree);
129       super.visitExportsDirectiveTree(tree);
130     }
131 
132     @Override
133     public void visitMemberSelectExpression(MemberSelectExpressionTree tree) {
134       // skip member selects
135     }
136 
137     @Override
138     public void visitIdentifier(IdentifierTree tree) {
139       identifiers.add(tree.name());
140     }
141   }

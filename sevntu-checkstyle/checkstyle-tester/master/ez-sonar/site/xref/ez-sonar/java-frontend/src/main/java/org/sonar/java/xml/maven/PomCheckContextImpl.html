20  package org.sonar.java.xml.maven;
21  
22  import com.google.common.annotations.VisibleForTesting;
23  
24  import org.sonar.java.AnalyzerMessage;
25  import org.sonar.java.AnalyzerMessage.TextSpan;
26  import org.sonar.java.SonarComponents;
27  import org.sonar.java.xml.XmlCheckContextImpl;
28  import org.sonar.maven.model.LocatedTree;
29  import org.sonar.maven.model.XmlLocation;
30  import org.sonar.maven.model.maven2.MavenProject;
31  import org.sonar.plugins.java.api.JavaCheck;
32  import org.w3c.dom.Document;
33  
34  import javax.xml.xpath.XPath;
35  
36  import java.io.File;
37  import java.util.Collections;
38  import java.util.List;
39  
40  public class PomCheckContextImpl extends XmlCheckContextImpl implements PomCheckContext {
41  
42    private final MavenProject project;
43  
44    public PomCheckContextImpl(MavenProject project, Document document, File file, XPath xPath, SonarComponents sonarComponents) {
45      super(document, file, xPath, sonarComponents);
46      this.project = project;
47    }
48  
49    @Override
50    public MavenProject getMavenProject() {
51      return project;
52    }
53  
54    @Override
55    public void reportIssue(PomCheck check, LocatedTree tree, String message) {
56      getSonarComponents().addIssue(getFile(), check, tree.startLocation().line(), message, null);
57    }
58  
59    @Override
60    public void reportIssue(PomCheck check, int line, String message, List<Location> secondary) {
61      File file = getFile();
62      AnalyzerMessage analyzerMessage = new AnalyzerMessage(check, file, line, message, 0);
63      for (Location location : secondary) {
64        AnalyzerMessage secondaryLocation = getSecondaryAnalyzerMessage(check, file, location);
65        analyzerMessage.flows.add(Collections.singletonList(secondaryLocation));
66      }
67      getSonarComponents().reportIssue(analyzerMessage);
68    }
69  
70    @VisibleForTesting
71    AnalyzerMessage getSecondaryAnalyzerMessage(JavaCheck check, File file, Location location) {
72      TextSpan ts;
73      if (location.onLine()) {
74        ts = textSpanForLine(file, location.line);
75      } else {
76        XmlLocation startLocation = location.tree.startLocation();
77        XmlLocation endLocation = location.tree.endLocation();
78        ts = new TextSpan(
79          startLocation.line(), offsetFromColumn(startLocation.column()),
80          endLocation.line(), offsetFromColumn(endLocation.column()));
81        if (ts.isEmpty()) {
82          ts = textSpanForLine(file, startLocation.line());
83        }
84      }
85      return new AnalyzerMessage(check, file, ts, location.msg, 0);
86    }
87  
88    private TextSpan textSpanForLine(File file, int line) {
89      String lineAsText = getSonarComponents().fileLines(file).get(line - 1);
90      return new TextSpan(line, 0, line, lineAsText.length());
91    }
92  
93    private static int offsetFromColumn(int column) {
94      // unknown column are marked as -1,
95      // first column is equivalent to an offset of 0
96      return column > 0 ? (column - 1) : 0;
97    }
98  }

40  public class PomParserTest {
41    private static final File UNRELATED_XML_FILE = new File("src/test/files/xml/parsing.xml");
42    private static final File SIMPLE_POM_FILE = new File("src/test/files/xml/maven/simple-project/pom.xml");
43    private static final File COMPLEX_ELEMENT_POM_FILE = new File("src/test/files/xml/maven/complex-element/pom.xml");
44    private static final File PARSE_ISSUE_POM_FILE = new File("src/test/files/xml/maven/parse-issue/pom.xml");

46    @Test
47    public void should_parse_simple_pom() throws Exception {
48      MavenProject project = PomParser.parseXML(SIMPLE_POM_FILE);
49      assertThat(project).isNotNull();
50  
51      assertThat(project.getProperties().getElements()).hasSize(3);
52      assertThat(project.getDependencies().getDependencies()).hasSize(4);
53      assertThat(project.getBuild().getPlugins().getPlugins()).hasSize(1);
54      assertThat(project.getScm()).isNull();
55      assertThat(project.getParent()).isNull();
56    }
57  
58    @Test
59    public void should_not_parse_unrelated_xml_files() throws Exception {
60      assertThat(PomParser.parseXML(UNRELATED_XML_FILE)).isNull();
61    }
62  
63    @Test
64    public void should_parse_incompletly_pom_with_complex_elements() throws Exception {
65      MavenProject project = PomParser.parseXML(COMPLEX_ELEMENT_POM_FILE);
66      Configuration config = project.getBuild().getPlugins().getPlugins().get(0).getExecutions().getExecutions().get(0).getConfiguration();
67      checkPosition(config, 16, 13, 24, 13);
68      assertThat(config).isNotNull();
69  
70      List<Element> values = config.getElements();
71      assertThat(values).hasSize(2);
72    }

78    }

80    @Test
81    public void should_not_parse() {
82      MavenProject project = PomParser.parseXML(new File("."));
83      assertThat(project).isNull();
84    }

86    @Test
87    public void should_retrieve_attributes() throws Exception {
88      MavenProject project = PomParser.parseXML(SIMPLE_POM_FILE);
89  
90      checkAttribute(project.getModelVersion(), "4.0.0", 3, 17, 3, 22);
91      checkAttribute(project.getGroupId(), "org.sonarsource.java", 5, 12, 5, 32);
92      checkAttribute(project.getArtifactId(), "simple-project", 6, 15, 6, 29);
93      checkAttribute(project.getPackaging(), "jar", 10, 14, 10, 17);
94      checkAttribute(project.getName(), "simple-project", 12, 9, 12, 23);
95      checkAttribute(project.getUrl(), "http://maven.apache.org", 13, 8, 13, 31);
96  
97      // starting column is unknown
98      checkAttribute(project.getVersion(), "1.0-SNAPSHOT", 7, -1, 9, 10);
99      // empty property
100     checkAttribute(project.getDescription(), "", 14, 16, 14, 16);
101   }

103   @Test
104   public void should_retrieve_properties() throws Exception {
105     MavenProject project = PomParser.parseXML(SIMPLE_POM_FILE);
106 
107     Properties properties = project.getProperties();
108     checkPosition(properties, 19, 1, 24, 1);
109     List<Element> values = properties.getElements();
110     assertThat(values).hasSize(3);
111   }

113   @Test
114   public void should_retrieve_dependencies_from_dependency_management() {
115     MavenProject project = PomParser.parseXML(SIMPLE_POM_FILE);
116     DependencyManagement dependencyManagement = project.getDependencyManagement();
117     checkPosition(dependencyManagement, 26, 3, 36, 3);
118     Dependency dependency = dependencyManagement.getDependencies().getDependencies().get(0);
119     checkAttribute(dependency.getGroupId(), "fake", 29, 18, 29, 22);
120     checkAttribute(dependency.getArtifactId(), "mock", 30, 21, 30, 25);
121     checkAttribute(dependency.getVersion(), "4.0", 31, 18, 31, 21);
122     checkAttribute(dependency.getScope(), "system", 32, 16, 32, 22);
123     checkAttribute(dependency.getSystemPath(), "hello", 33, 21, 33, 26);
124   }
125 
126   @Test
127   public void should_retrieve_configurations() throws Exception {
128     MavenProject project = PomParser.parseXML(SIMPLE_POM_FILE);
129     List<Plugin> plugins = project.getBuild().getPlugins().getPlugins();
130     Configuration configuration = plugins.get(0).getExecutions().getExecutions().get(0).getConfiguration();
131     checkPosition(configuration, 88, 13, 90, 13);
132     List<Element> values = configuration.getElements();
133     assertThat(values).hasSize(1);
134   }
135 
136   private static void checkAttribute(LocatedAttribute attribute, @Nullable String value, int startLine, int startColumn, int endLine, int endColumn) {
137     String attributeValue = attribute.getValue();
138     if (value == null) {
139       assertThat(attributeValue).isNull();
140     } else {
141       assertThat(attributeValue.trim()).isEqualTo(value);
142     }
143     checkPosition(attribute, startLine, startColumn, endLine, endColumn);
144   }
145 
146   private static void checkPosition(LocatedTree tree, int startLine, int startColumn, int endLine, int endColumn) {
147     assertThat(tree.startLocation()).overridingErrorMessage("start location should have been correctly parsed").isNotNull();
148     assertThat(tree.startLocation().line()).isEqualTo(startLine);
149     assertThat(tree.startLocation().column()).isEqualTo(startColumn);
150     assertThat(tree.endLocation()).overridingErrorMessage("end location should have been correctly parsed").isNotNull();
151     assertThat(tree.endLocation().line()).isEqualTo(endLine);
152     assertThat(tree.endLocation().column()).isEqualTo(endColumn);
153   }

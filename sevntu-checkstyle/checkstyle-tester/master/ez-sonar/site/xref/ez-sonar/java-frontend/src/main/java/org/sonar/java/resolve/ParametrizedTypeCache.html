20  package org.sonar.java.resolve;
21  
22  import com.google.common.collect.Maps;
23  
24  import org.sonar.java.resolve.WildCardType.BoundType;
25  
26  import java.util.EnumMap;
27  import java.util.List;
28  import java.util.Map;
29  
30  public class ParametrizedTypeCache {
31  
32    private Map<JavaSymbol, Map<TypeSubstitution, ParametrizedTypeJavaType>> typeCache = Maps.newHashMap();
33    private Map<JavaType, Map<WildCardType.BoundType, WildCardType>> wildcardCache = Maps.newHashMap();
34    private TypeSubstitutionSolver typeSubstitutionSolver;
35  
36    public JavaType getParametrizedTypeType(JavaSymbol.TypeJavaSymbol symbol, TypeSubstitution typeSubstitution) {
37      if (symbol.getType().isUnknown()) {
38        return symbol.getType();
39      }
40      if (typeCache.get(symbol) == null) {
41        Map<TypeSubstitution, ParametrizedTypeJavaType> map = Maps.newHashMap();
42        typeCache.put(symbol, map);
43      }
44      TypeSubstitution newSubstitution = typeSubstitution;
45      if (newSubstitution.size() == 0) {
46        newSubstitution = identitySubstitution(symbol.typeVariableTypes);
47      }
48      if (typeCache.get(symbol).get(newSubstitution) == null) {
49        typeCache.get(symbol).put(newSubstitution, new ParametrizedTypeJavaType(symbol, newSubstitution, typeSubstitutionSolver));
50      }
51      return typeCache.get(symbol).get(newSubstitution);
52    }
53  
54    private static TypeSubstitution identitySubstitution(List<TypeVariableJavaType> typeVariables) {
55      TypeSubstitution result = new TypeSubstitution();
56      for (TypeVariableJavaType typeVar : typeVariables) {
57        result.add(typeVar, typeVar);
58      }
59      return result;
60    }
61  
62    public JavaType getWildcardType(JavaType bound, BoundType boundType) {
63      Map<WildCardType.BoundType, WildCardType> map = wildcardCache.computeIfAbsent(bound, b -> new EnumMap<>(WildCardType.BoundType.class));
64      return map.computeIfAbsent(boundType, bt -> new WildCardType(bound, bt));
65    }
66  
67    public void setTypeSubstitutionSolver(TypeSubstitutionSolver typeSubstitutionSolver) {
68      this.typeSubstitutionSolver = typeSubstitutionSolver;
69    }
70  }

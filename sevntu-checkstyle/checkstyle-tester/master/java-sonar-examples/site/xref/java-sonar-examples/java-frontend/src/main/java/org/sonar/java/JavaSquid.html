46  public class JavaSquid {
47  
48    private static final Logger LOG = Loggers.get(JavaSquid.class);
49  
50    private final JavaAstScanner astScanner;
51    private final JavaAstScanner astScannerForTests;
52  
53    public JavaSquid(JavaVersion javaVersion,
54      @Nullable SonarComponents sonarComponents, @Nullable Measurer measurer,
55      JavaResourceLocator javaResourceLocator, @Nullable SonarJavaIssueFilter postAnalysisIssueFilter, JavaCheck... visitors) {
56      this(javaVersion, false, sonarComponents, measurer, javaResourceLocator, postAnalysisIssueFilter, visitors);
57    }
58  
59    public JavaSquid(JavaVersion javaVersion, boolean xFileEnabled,
60                     @Nullable SonarComponents sonarComponents, @Nullable Measurer measurer,
61                     JavaResourceLocator javaResourceLocator, @Nullable SonarJavaIssueFilter postAnalysisIssueFilter, JavaCheck... visitors) {
62  
63      List<JavaCheck> commonVisitors = Lists.newArrayList(javaResourceLocator);
64      if (postAnalysisIssueFilter != null) {
65        commonVisitors.add(postAnalysisIssueFilter);
66      }
67  
68      Iterable<JavaCheck> codeVisitors = Iterables.concat(commonVisitors, Arrays.asList(visitors));
69      Collection<JavaCheck> testCodeVisitors = Lists.newArrayList(commonVisitors);
70      if (measurer != null) {
71        Iterable<JavaCheck> measurers = Collections.singletonList(measurer);
72        codeVisitors = Iterables.concat(measurers, codeVisitors);
73        testCodeVisitors.add(measurer.new TestFileMeasurer());
74      }
75      List<File> classpath = Lists.newArrayList();
76      List<File> testClasspath = Lists.newArrayList();
77      if (sonarComponents != null) {
78        if(!sonarComponents.isSonarLintContext()) {
79          codeVisitors = Iterables.concat(codeVisitors, Arrays.asList(new FileLinesVisitor(sonarComponents), new SyntaxHighlighterVisitor(sonarComponents)));
80          testCodeVisitors.add(new SyntaxHighlighterVisitor(sonarComponents));
81        }
82        classpath = sonarComponents.getJavaClasspath();
83        testClasspath = sonarComponents.getJavaTestClasspath();
84        testCodeVisitors.addAll(sonarComponents.testCheckClasses());
85      }
86  
87      //AstScanner for main files
88      ActionParser<Tree> parser = JavaParser.createParser();
89      astScanner = new JavaAstScanner(parser, sonarComponents);
90      astScanner.setVisitorBridge(createVisitorBridge(codeVisitors, classpath, javaVersion, sonarComponents, SymbolicExecutionMode.getMode(visitors, xFileEnabled)));
91  
92      //AstScanner for test files
93      astScannerForTests = new JavaAstScanner(parser, sonarComponents);
94      astScannerForTests.setVisitorBridge(createVisitorBridge(testCodeVisitors, testClasspath, javaVersion, sonarComponents, SymbolicExecutionMode.DISABLED));
95  
96    }
97  
98    private static VisitorsBridge createVisitorBridge(
99      Iterable<JavaCheck> codeVisitors, List<File> classpath, JavaVersion javaVersion, @Nullable SonarComponents sonarComponents, SymbolicExecutionMode symbolicExecutionMode) {
100     VisitorsBridge visitorsBridge = new VisitorsBridge(codeVisitors, classpath, sonarComponents, symbolicExecutionMode);
101     visitorsBridge.setJavaVersion(javaVersion);
102     return visitorsBridge;
103   }
104 
105 
106   public void scan(Collection<File> sourceFiles, Collection<File> testFiles) {
107     scanSources(sourceFiles);
108     scanTests(testFiles);
109   }
110 
111   private void scanSources(Collection<File> sourceFiles) {
112     Profiler profiler = Profiler.create(LOG).startInfo("Java Main Files AST scan");
113     astScanner.scan(sourceFiles);
114     profiler.stopInfo();
115   }
116 
117   private void scanTests(Collection<File> testFiles) {
118     Profiler profiler = Profiler.create(LOG).startInfo("Java Test Files AST scan");
119     astScannerForTests.scan(testFiles);
120     profiler.stopInfo();
121   }

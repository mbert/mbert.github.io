293   private static final String EXP_REGEXP = "(?:[Ee][+-]?+[0-9_]++)";
294   private static final String BINARY_EXP_REGEXP = "(?:[Pp][+-]?+[0-9_]++)";
295   private static final String FLOATING_LITERAL_WITHOUT_SUFFIX_REGEXP = "(?:" +
296   // Decimal
297     "[0-9][0-9_]*+\\.([0-9_]++)?+" + EXP_REGEXP + "?+" +
298     "|" + "\\.[0-9][0-9_]*+" + EXP_REGEXP + "?+" +
299     "|" + "[0-9][0-9_]*+" + EXP_REGEXP +
300     // Hexadecimal
301     "|" + "0[xX][0-9_a-fA-F]++\\.[0-9_a-fA-F]*+" + BINARY_EXP_REGEXP +
302     "|" + "0[xX][0-9_a-fA-F]++" + BINARY_EXP_REGEXP +
303     ")";
304 
305   private static final String INTEGER_LITERAL_REGEXP = "(?:" +
306   // Hexadecimal
307     "0[xX][0-9_a-fA-F]++" +
308     // Binary (Java 7)
309     "|" + "0[bB][01_]++" +
310     // Decimal and Octal
311     "|" + "[0-9][0-9_]*+" +
312     ")";

314   public static LexerlessGrammarBuilder createGrammarBuilder() {
315     LexerlessGrammarBuilder b = LexerlessGrammarBuilder.create();
316 
317     punctuators(b);
318     keywords(b);
319 
320     literals(b);
321 
322     b.setRootRule(COMPILATION_UNIT);
323 
324     return b;
325   }

327   private static void punctuators(LexerlessGrammarBuilder b) {
328     punctuator(b, AT, "@");
329     punctuator(b, AND, "&", b.nextNot(b.firstOf("=", "&")));
330     punctuator(b, ANDAND, "&&");
331     punctuator(b, ANDEQU, "&=");
332     punctuator(b, BANG, "!", b.nextNot("="));
333     punctuator(b, BSR, ">>>", b.nextNot("="));
334     punctuator(b, BSREQU, ">>>=");
335     punctuator(b, COLON, ":");
336     punctuator(b, DBLECOLON, "::");
337     punctuator(b, COMMA, ",");
338     punctuator(b, DEC, "--");
339     punctuator(b, DIV, "/", b.nextNot("="));
340     punctuator(b, DIVEQU, "/=");
341     punctuator(b, DOT, ".");
342     punctuator(b, ELLIPSIS, "...");
343     punctuator(b, EQU, "=", b.nextNot("="));
344     punctuator(b, EQUAL, "==");
345     punctuator(b, GE, ">=");
346     punctuator(b, GT, ">", b.nextNot(b.firstOf("=", ">")));
347     punctuator(b, HAT, "^", b.nextNot("="));
348     punctuator(b, HATEQU, "^=");
349     punctuator(b, INC, "++");
350     punctuator(b, LBRK, "[");
351     punctuator(b, LT, "<", b.nextNot(b.firstOf("=", "<")));
352     punctuator(b, LE, "<=");
353     punctuator(b, LPAR, "(");
354     punctuator(b, LWING, "{");
355     punctuator(b, MINUS, "-", b.nextNot(b.firstOf("-", "=")));
356     punctuator(b, MINUSEQU, "-=");
357     punctuator(b, MOD, "%", b.nextNot("="));
358     punctuator(b, MODEQU, "%=");
359     punctuator(b, NOTEQUAL, "!=");
360     punctuator(b, OR, "|", b.nextNot(b.firstOf("=", "|")));
361     punctuator(b, OREQU, "|=");
362     punctuator(b, OROR, "||");
363     punctuator(b, PLUS, "+", b.nextNot(b.firstOf("=", "+")));
364     punctuator(b, PLUSEQU, "+=");
365     punctuator(b, QUERY, "?");
366     punctuator(b, RBRK, "]");
367     punctuator(b, RPAR, ")");
368     punctuator(b, RWING, "}");
369     punctuator(b, SEMI, ";");
370     punctuator(b, SL, "<<", b.nextNot("="));
371     punctuator(b, SLEQU, "<<=");
372     punctuator(b, SR, ">>", b.nextNot(b.firstOf("=", ">")));
373     punctuator(b, SREQU, ">>=");
374     punctuator(b, STAR, "*", b.nextNot("="));
375     punctuator(b, STAREQU, "*=");
376     punctuator(b, TILDA, "~");
377 
378     punctuator(b, LPOINT, "<");
379     punctuator(b, RPOINT, ">");
380 
381     punctuator(b, ARROW, "->");
382   }

384   private static void keywords(LexerlessGrammarBuilder b) {
385     b.rule(LETTER_OR_DIGIT).is(javaIdentifierPart(b));
386     for (JavaKeyword tokenType : JavaKeyword.values()) {
387       b.rule(tokenType).is(tokenType.getValue(), b.nextNot(LETTER_OR_DIGIT), SPACING);
388     }
389     String[] keywords = JavaKeyword.keywordValues();
390     Arrays.sort(keywords);
391     ArrayUtils.reverse(keywords);
392     b.rule(KEYWORD).is(
393       b.firstOf(
394         keywords[0],
395         keywords[1],
396         ArrayUtils.subarray(keywords, 2, keywords.length)),
397       b.nextNot(LETTER_OR_DIGIT));
398 
399     // restricted keywords introduced with java 9 (JLS9 - §3.9)
400     for (JavaRestrictedKeyword tokenType : JavaRestrictedKeyword.values()) {
401       b.rule(tokenType).is(tokenType.getValue(), b.nextNot(LETTER_OR_DIGIT), SPACING);
402     }
403     String[] restrictedKeywords = JavaRestrictedKeyword.restrictedKeywordValues();
404     Arrays.sort(restrictedKeywords);
405     ArrayUtils.reverse(restrictedKeywords);
406     b.rule(RESTRICTED_KEYWORD).is(
407       b.firstOf(
408         restrictedKeywords[0],
409         restrictedKeywords[1],
410         ArrayUtils.subarray(restrictedKeywords, 2, restrictedKeywords.length)),
411       b.nextNot(LETTER_OR_DIGIT));
412 
413     // special identifier introduced for java10: 'var'
414     for (JavaSpecialIdentifier tokenType : JavaSpecialIdentifier.values()) {
415       b.rule(tokenType).is(tokenType.getValue(), b.nextNot(LETTER_OR_DIGIT), SPACING);
416     }
417   }

427   /**
428    * 3.10. Literals
429    */
430   private static void literals(LexerlessGrammarBuilder b) {
431     b.rule(SPACING).is(
432       b.skippedTrivia(whitespace(b)),
433       b.zeroOrMore(
434         b.commentTrivia(b.firstOf(inlineComment(b), multilineComment(b))),
435         b.skippedTrivia(whitespace(b))));
436 
437     b.rule(EOF).is(b.token(GenericTokenType.EOF, b.endOfInput()));
438 
439     b.rule(CHARACTER_LITERAL).is(characterLiteral(b), SPACING);
440     b.rule(JavaTokenType.STRING_LITERAL).is(stringLiteral(b), SPACING);
441 
442     b.rule(FLOAT_LITERAL).is(b.regexp(FLOATING_LITERAL_WITHOUT_SUFFIX_REGEXP + "[fF]|[0-9][0-9_]*+[fF]"), SPACING);
443     b.rule(DOUBLE_LITERAL).is(b.regexp(FLOATING_LITERAL_WITHOUT_SUFFIX_REGEXP + "[dD]?+|[0-9][0-9_]*+[dD]"), SPACING);
444 
445     b.rule(LONG_LITERAL).is(b.regexp(INTEGER_LITERAL_REGEXP + "[lL]"), SPACING);
446     b.rule(INTEGER_LITERAL).is(b.regexp(INTEGER_LITERAL_REGEXP), SPACING);
447 
448     b.rule(JavaTokenType.IDENTIFIER).is(
449       b.firstOf(
450         b.next(ENUM),
451         b.nextNot(KEYWORD)),
452       javaIdentifier(b),
453       SPACING);
454   }

456   private static Object characterLiteral(LexerlessGrammarBuilder b) {
457     return b.sequence(b.next("'"), b.regexp("'([^'\\\\]*+(\\\\[\\s\\S])?+)*+'"));
458   }
459 
460   private static Object stringLiteral(LexerlessGrammarBuilder b) {
461     return b.sequence(b.next("\""), b.regexp("\"([^\"\\\\]*+(\\\\[\\s\\S])?+)*+\""));
462   }
463 
464   private static Object whitespace(LexerlessGrammarBuilder b) {
465     return b.regexp("\\s*+");
466   }
467 
468   private static Object inlineComment(LexerlessGrammarBuilder b) {
469     return b.regexp("//[^\\n\\r]*+");
470   }
471 
472   private static Object multilineComment(LexerlessGrammarBuilder b) {
473     return b.regexp("/\\*[\\s\\S]*?\\*\\/");
474   }
475 
476   private static Object javaIdentifier(LexerlessGrammarBuilder b) {
477     return b.regexp("\\p{javaJavaIdentifierStart}++\\p{javaJavaIdentifierPart}*+");
478   }
479 
480   private static Object javaIdentifierPart(LexerlessGrammarBuilder b) {
481     return b.regexp("\\p{javaJavaIdentifierPart}");
482   }

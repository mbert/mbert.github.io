20  package org.sonar.java;
21  
22  import java.util.List;
23  import java.util.stream.Collectors;
24  import javax.annotation.Nullable;
25  import org.sonar.api.batch.fs.InputComponent;
26  import org.sonar.api.batch.fs.InputFile;
27  import org.sonar.api.batch.fs.TextRange;
28  import org.sonar.api.batch.sensor.SensorContext;
29  import org.sonar.api.batch.sensor.issue.NewIssue;
30  import org.sonar.api.batch.sensor.issue.NewIssueLocation;
31  import org.sonar.api.rule.RuleKey;
32  
33  public final class JavaIssue {
34    private final NewIssue newIssue;
35  
36    public JavaIssue(NewIssue newIssue) {
37      this.newIssue = newIssue;
38    }
39  
40    public static JavaIssue create(SensorContext context, RuleKey ruleKey, @Nullable Double effortToFix) {
41      NewIssue newIssue = context.newIssue()
42        .forRule(ruleKey)
43        .gap(effortToFix);
44      return new JavaIssue(newIssue);
45    }
46  
47    public JavaIssue setPrimaryLocationOnComponent(InputComponent fileOrProject, String message) {
48      newIssue.at(
49        newIssue.newLocation()
50          .on(fileOrProject)
51          .message(message));
52      return this;
53    }
54  
55    public JavaIssue setPrimaryLocation(InputFile file, String message, int startLine, int startLineOffset, int endLine, int endLineOffset) {
56      NewIssueLocation newIssueLocation;
57      if (startLineOffset == -1) {
58        newIssueLocation = newIssue.newLocation()
59          .on(file)
60          .at(file.selectLine(startLine))
61          .message(message);
62      } else {
63        newIssueLocation = newIssue.newLocation()
64          .on(file)
65          .at(file.newRange(startLine, startLineOffset, endLine, endLineOffset))
66          .message(message);
67      }
68      newIssue.at(newIssueLocation);
69      return this;
70    }
71  
72    public JavaIssue addSecondaryLocation(InputFile file, int startLine, int startLineOffset, int endLine, int endLineOffset, String message) {
73      newIssue.addLocation(
74        newIssue.newLocation()
75          .on(file)
76          .at(file.newRange(startLine, startLineOffset, endLine, endLineOffset))
77          .message(message));
78      return this;
79    }
80  
81    public JavaIssue addFlow(InputFile file, List<List<AnalyzerMessage>> flows) {
82      for (List<AnalyzerMessage> flow : flows) {
83        newIssue.addFlow(flow.stream()
84          .map(am -> newIssue.newLocation()
85            .on(file)
86            .at(range(file, am.primaryLocation()))
87            .message(am.getMessage()))
88          .collect(Collectors.toList()));
89      }
90      return this;
91    }
92  
93    private static TextRange range(InputFile file, AnalyzerMessage.TextSpan textSpan) {
94      return file.newRange(textSpan.startLine, textSpan.startCharacter, textSpan.endLine, textSpan.endCharacter);
95    }
96  
97  
98  
99    public void save() {
100     newIssue.save();
101   }
102 
103 }

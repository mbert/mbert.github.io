126   @Override
127   public void visitMethod(MethodTree tree) {
128     //skip return type, and throw clauses : visited in second pass.
129     scan(tree.modifiers());
130     completeMetadata((JavaSymbol.MethodJavaSymbol) tree.symbol(), tree.modifiers().annotations());
131     scan(tree.typeParameters());
132     // revisits the parameters to resolve their annotations.
133     scan(tree.parameters());
134     scan(tree.defaultValue());
135     scan(tree.block());
136   }

138   @Override
139   public void visitClass(ClassTree tree) {
140     //skip superclass and interfaces : visited in second pass.
141     scan(tree.modifiers());
142     completeMetadata((JavaSymbol) tree.symbol(), tree.modifiers().annotations());
143     scan(tree.typeParameters());
144     scan(tree.members());
145   }

147   private static void completeMetadata(JavaSymbol symbol, List<AnnotationTree> annotations) {
148     for (AnnotationTree tree : annotations) {
149       AnnotationInstanceResolve annotationInstance = new AnnotationInstanceResolve((JavaSymbol.TypeJavaSymbol) tree.symbolType().symbol());
150       symbol.metadata().addAnnotation(annotationInstance);
151       Arguments arguments = tree.arguments();
152       if (arguments.size() > 1 || (!arguments.isEmpty() && arguments.get(0).is(Tree.Kind.ASSIGNMENT))) {
153         for (ExpressionTree expressionTree : arguments) {
154           AssignmentExpressionTree aet = (AssignmentExpressionTree) expressionTree;
155           // TODO: Store more precise value than the expression (real value in case of literal, symbol for enums, array of values, solve constants?)
156           annotationInstance.addValue(new AnnotationValueResolve(((IdentifierTree) aet.variable()).name(), aet.expression()));
157         }
158       } else {
159         // Constant
160         addConstantValue(tree, annotationInstance);
161       }
162     }
163   }
164 
165   private static void addConstantValue(AnnotationTree tree, AnnotationInstanceResolve annotationInstance) {
166     Collection<Symbol> scopeSymbols = tree.annotationType().symbolType().symbol().memberSymbols();
167     for (ExpressionTree expressionTree : tree.arguments()) {
168       String name = "";
169       for (Symbol scopeSymbol : scopeSymbols) {
170         if(scopeSymbol.isMethodSymbol()) {
171           name = scopeSymbol.name();
172           break;
173         }
174       }
175       annotationInstance.addValue(new AnnotationValueResolve(name, expressionTree));
176     }
177   }

201   @Override
202   public void visitMethodInvocation(MethodInvocationTree tree) {
203     MethodInvocationTreeImpl mit = (MethodInvocationTreeImpl) tree;
204     Resolve.Env methodEnv = semanticModel.getEnv(tree);
205     if(mit.isTypeSet() && mit.symbol().isMethodSymbol()) {
206       TypeSubstitution typeSubstitution = inferedSubstitution(mit);
207       List<JavaType> argTypes = getParameterTypes(tree.arguments());
208       JavaSymbol.MethodJavaSymbol methodSymbol = (JavaSymbol.MethodJavaSymbol) mit.symbol();
209       List<JavaType> formals = methodSymbol.parameterTypes().stream().map(t -> (JavaType) t).collect(Collectors.toList());
210       List<JavaType> inferedArgTypes = resolve.resolveTypeSubstitution(formals, typeSubstitution);
211       int size = inferedArgTypes.size();
212       IntStream.range(0, argTypes.size()).forEach(
213         i -> {
214           JavaType arg = argTypes.get(i);
215           Type formal = inferedArgTypes.get(Math.min(i, size - 1));
216           if (formal != arg) {
217             AbstractTypedTree argTree = (AbstractTypedTree) mit.arguments().get(i);
218             argTree.setInferedType(formal);
219             argTree.accept(this);
220           }
221         }
222       );
223       List<JavaType> typeParamTypes = getParameterTypes(tree.typeArguments());
224       JavaType resultType = ((MethodJavaType) mit.symbol().type()).resultType;
225       // if result type is a type var defined by the method we are solving, use the target type.
226       if(resultType.symbol.owner == mit.symbol()) {
227         resultType = (JavaType) mit.symbolType();
228       }
229       inferReturnTypeFromInferedArgs(tree, methodEnv, argTypes, typeParamTypes, resultType, typeSubstitution);
230       return;
231     }
232     scan(tree.arguments());
233     scan(tree.typeArguments());
234     List<JavaType> argTypes = getParameterTypes(tree.arguments());
235     List<JavaType> typeParamTypes = getParameterTypes(tree.typeArguments());
236     Resolve.Resolution resolution = resolveMethodSymbol(tree.methodSelect(), methodEnv, argTypes, typeParamTypes);
237     JavaSymbol symbol;
238     JavaType returnType;
239     if(resolution == null) {
240       returnType = symbols.deferedType(mit);
241       symbol = Symbols.unknownSymbol;
242     } else {
243       symbol = resolution.symbol();
244       returnType = resolution.type();
245       if(symbol.isMethodSymbol()) {
246         MethodJavaType methodType = (MethodJavaType) resolution.type();
247         returnType = methodType.resultType;
248       }
249     }
250     mit.setSymbol(symbol);
251     if(returnType != null && returnType.isTagged(JavaType.DEFERRED)) {
252       ((DeferredType) returnType).setTree(mit);
253     }
254     registerType(tree, returnType);
255     if(resolution != null) {
256       inferArgumentTypes(argTypes, resolution);
257       inferReturnTypeFromInferedArgs(tree, methodEnv, argTypes, typeParamTypes, returnType, new TypeSubstitution());
258     }
259   }

261   private void inferReturnTypeFromInferedArgs(MethodInvocationTree tree, Resolve.Env methodEnv, List<JavaType> argTypes, List<JavaType> typeParamTypes,
262                                               JavaType returnType, TypeSubstitution typeSubstitution) {
263     List<JavaType> parameterTypes = getParameterTypes(tree.arguments());
264     if(!parameterTypes.equals(argTypes)) {
265       IdentifierTree identifier = null;
266       Resolution resolution = null;
267       Tree methodSelect = tree.methodSelect();
268       if (methodSelect.is(Tree.Kind.MEMBER_SELECT)) {
269         MemberSelectExpressionTree mset = (MemberSelectExpressionTree) methodSelect;
270         JavaType type = getType(mset.expression());
271         if(type.isTagged(JavaType.DEFERRED)) {
272           throw new IllegalStateException("type of arg should not be defered anymore ??");
273         }
274         identifier = mset.identifier();
275         resolution = resolve.findMethod(methodEnv, type, identifier.name(), parameterTypes, typeParamTypes);
276       } else if (methodSelect.is(Tree.Kind.IDENTIFIER)) {
277         identifier = (IdentifierTree) methodSelect;
278         resolution = resolve.findMethod(methodEnv, identifier.name(), parameterTypes, typeParamTypes);
279       }
280       if(resolution != null && returnType != resolution.type() && resolution.symbol().isMethodSymbol()) {
281         MethodJavaType methodType = (MethodJavaType) resolution.type();
282         if(!methodType.resultType.isTagged(JavaType.DEFERRED)) {
283           registerType(tree, resolve.applySubstitution(methodType.resultType, typeSubstitution));
284           // update type associated to identifier as it may have been inferred deeper when re-resolving method with new parameter types
285           registerType(identifier, methodType);
286         }
287       }
288     } else {
289       registerType(tree, resolve.applySubstitution(returnType, typeSubstitution));
290     }
291   }

293   private static TypeSubstitution inferedSubstitution(MethodInvocationTreeImpl mit) {
294     JavaSymbol.MethodJavaSymbol methodSymbol = (JavaSymbol.MethodJavaSymbol) mit.symbol();
295     JavaType methodReturnedType = (JavaType) mit.symbolType();
296     TypeSubstitution typeSubstitution = new TypeSubstitution();
297     if(methodReturnedType.isTagged(JavaType.PARAMETERIZED)) {
298       JavaType resultType = ((MethodJavaType) methodSymbol.type).resultType;
299       if(resultType.isTagged(JavaType.PARAMETERIZED)) {
300         typeSubstitution =((ParametrizedTypeJavaType) resultType).typeSubstitution.combine(((ParametrizedTypeJavaType) methodReturnedType).typeSubstitution);
301       } else if(resultType.isTagged(JavaType.TYPEVAR)) {
302         typeSubstitution.add((TypeVariableJavaType) resultType, methodReturnedType);
303       }
304     }
305     return typeSubstitution;
306   }

308   private void setInferedType(Type infered, DeferredType deferredType) {
309     AbstractTypedTree inferedExpression = deferredType.tree();
310     Type newType = infered;
311     if (inferedExpression.is(Tree.Kind.NEW_CLASS)) {
312       Type newClassType = ((NewClassTree) inferedExpression).identifier().symbolType();
313       if(((JavaType) newClassType).isParameterized()) {
314         newType = resolve.resolveTypeSubstitutionWithDiamondOperator((ParametrizedTypeJavaType) newClassType, (JavaType) infered);
315       }
316     }
317     inferedExpression.setInferedType(newType);
318     inferedExpression.accept(this);
319     if (inferedExpression.is(Tree.Kind.VAR_TYPE)) {
320       // change type of the variable
321       JavaSymbol.VariableJavaSymbol variableSymbol = ((VariableTreeImpl) inferedExpression.parent()).getSymbol();
322       variableSymbol.type = (JavaType) newType;
323     }
324   }

326   private static List<JavaType> getParameterTypes(@Nullable List<? extends Tree> args) {
327     if(args == null) {
328       return new ArrayList<>();
329     }
330     return args.stream().map(e ->
331       ((AbstractTypedTree) e).isTypeSet() ?
332         (JavaType) ((AbstractTypedTree) e).symbolType() : Symbols.unknownType).collect(Collectors.toList());
333   }

335   @CheckForNull
336   private Resolve.Resolution resolveMethodSymbol(Tree methodSelect, Resolve.Env methodEnv, List<JavaType> argTypes, List<JavaType> typeParamTypes) {
337     Resolve.Resolution resolution;
338     IdentifierTree identifier;
339     if (methodSelect.is(Tree.Kind.MEMBER_SELECT)) {
340       MemberSelectExpressionTree mset = (MemberSelectExpressionTree) methodSelect;
341       resolveAs(mset.expression(), JavaSymbol.TYP | JavaSymbol.VAR);
342       JavaType type = getType(mset.expression());
343       if(type.isTagged(JavaType.DEFERRED)) {
344         return null;
345       }
346       identifier = mset.identifier();
347       resolution = resolve.findMethod(methodEnv, type, identifier.name(), argTypes, typeParamTypes);
348     } else if (methodSelect.is(Tree.Kind.IDENTIFIER)) {
349       identifier = (IdentifierTree) methodSelect;
350       resolution = resolve.findMethod(methodEnv, identifier.name(), argTypes, typeParamTypes);
351     } else {
352       throw new IllegalStateException("Method select in method invocation is not of the expected type " + methodSelect);
353     }
354     registerType(identifier, resolution.type());
355     associateReference(identifier, resolution.symbol());
356     return resolution;
357   }

359   private void resolveAs(@Nullable Tree tree, int kind) {
360     if (tree == null) {
361       return;
362     }
363     if (env == null) {
364       resolveAs(tree, kind, semanticModel.getEnv(tree));
365     } else {
366       resolveAs(tree, kind, env);
367     }
368   }

374   public JavaSymbol resolveAs(Tree tree, int kind, Resolve.Env resolveEnv, boolean associateReference) {
375     if (tree.is(Tree.Kind.IDENTIFIER, Tree.Kind.MEMBER_SELECT)) {
376       JavaSymbol resolvedSymbol;
377       IdentifierTree identifierTree;
378       if (tree.is(Tree.Kind.MEMBER_SELECT)) {
379         MemberSelectExpressionTree mse = (MemberSelectExpressionTree) tree;
380         if (JavaKeyword.CLASS.getValue().equals(mse.identifier().name())) {
381           // resolve type of expression xxx.class
382           return resolveClassType(tree, resolveEnv, mse);
383         }
384         identifierTree = mse.identifier();
385 
386         List<AnnotationTree> identifierAnnotations = identifierTree.annotations();
387         scan(identifierAnnotations);
388         completeMetadata((JavaSymbol) identifierTree.symbol(), identifierAnnotations);
389 
390         Resolve.Resolution res = getSymbolOfMemberSelectExpression(mse, kind, resolveEnv);
391         resolvedSymbol = res.symbol();
392         JavaType resolvedType = resolve.resolveTypeSubstitution(res.type(), getType(mse.expression()));
393         registerType(identifierTree, resolvedType);
394         registerType(tree, resolvedType);
395       } else {
396         identifierTree = (IdentifierTree) tree;
397         Resolve.Resolution resolution = resolve.findIdent(resolveEnv, identifierTree.name(), kind);
398         resolvedSymbol = resolution.symbol();
399 
400         // Resolve annotations which can be present (for instance) when declaring types: "List<@MyAnnotation MyClass>"
401         // but do not associate it with the symbol of the identifier (that would be a non-sense to associate '@MyAnnotation'
402         // with the 'MyClass' symbol type)
403         scan(identifierTree.annotations());
404 
405         JavaType type = resolution.type();
406         if(kind == JavaSymbol.TYP && type.isParameterized()) {
407           type = type.erasure();
408         }
409         registerType(tree, type);
410       }
411       if(associateReference) {
412         associateReference(identifierTree, resolvedSymbol);
413       }
414       return resolvedSymbol;
415     }
416     tree.accept(this);
417     JavaType type = getType(tree);
418     if (tree.is(Tree.Kind.INFERED_TYPE, Tree.Kind.VAR_TYPE)) {
419       type = symbols.deferedType((AbstractTypedTree) tree);
420       registerType(tree, type);
421     }
422     if (type == null) {
423       throw new IllegalStateException("Type not resolved " + tree);
424     }
425     return type.symbol;
426   }

428   private JavaSymbol resolveClassType(Tree tree, Resolve.Env resolveEnv, MemberSelectExpressionTree mse) {
429     resolveAs(mse.expression(), JavaSymbol.TYP, resolveEnv);
430     // member select ending with .class
431     JavaType expressionType = getType(mse.expression());
432     if (expressionType.isPrimitive()) {
433       expressionType = expressionType.primitiveWrapperType();
434     }
435     TypeSubstitution typeSubstitution = new TypeSubstitution();
436     typeSubstitution.add(symbols.classType.getSymbol().typeVariableTypes.get(0), expressionType);
437     JavaType parametrizedClassType = parametrizedTypeCache.getParametrizedTypeType(symbols.classType.symbol, typeSubstitution);
438     registerType(tree, parametrizedClassType);
439     return parametrizedClassType.symbol;
440   }

442   private Resolve.Resolution getSymbolOfMemberSelectExpression(MemberSelectExpressionTree mse, int kind, Resolve.Env resolveEnv) {
443     int expressionKind = JavaSymbol.TYP;
444     if ((kind & JavaSymbol.VAR) != 0) {
445       expressionKind |= JavaSymbol.VAR;
446     }
447     if ((kind & JavaSymbol.TYP) != 0) {
448       expressionKind |= JavaSymbol.PCK;
449     }
450     //TODO: resolveAs result is only used here, should probably be refactored
451     JavaSymbol site = resolveAs(mse.expression(), expressionKind, resolveEnv);
452     if (site.kind == JavaSymbol.VAR) {
453       return resolve.findIdentInType(resolveEnv, getType(mse.expression()).symbol, mse.identifier().name(), JavaSymbol.VAR);
454     }
455     if (site.kind == JavaSymbol.TYP) {
456       return resolve.findIdentInType(resolveEnv, (JavaSymbol.TypeJavaSymbol) site, mse.identifier().name(), kind);
457     }
458     if (site.kind == JavaSymbol.PCK) {
459       return Resolve.Resolution.resolution(resolve.findIdentInPackage(site, mse.identifier().name(), kind));
460     }
461     return resolve.unresolved();
462   }

495   private JavaType parametrizedTypeWithTypeArguments(JavaSymbol.TypeJavaSymbol symbol, TypeArguments typeArguments) {
496     TypeSubstitution typeSubstitution = new TypeSubstitution();
497     if (typeArguments.size() <= symbol.typeVariableTypes.size()) {
498       for (int i = 0; i < typeArguments.size(); i++) {
499         typeSubstitution.add(symbol.typeVariableTypes.get(i), getType(typeArguments.get(i)));
500       }
501     }
502     return parametrizedTypeCache.getParametrizedTypeType(symbol, typeSubstitution);
503   }

505   @Override
506   public void visitWildcard(WildcardTree tree) {
507     if (tree.is(Tree.Kind.UNBOUNDED_WILDCARD)) {
508       registerType(tree, symbols.unboundedWildcard);
509     } else {
510       resolveAs(tree.bound(), JavaSymbol.TYP);
511       JavaType bound = getType(tree.bound());
512       WildCardType.BoundType boundType = tree.is(Tree.Kind.SUPER_WILDCARD) ? WildCardType.BoundType.SUPER : WildCardType.BoundType.EXTENDS;
513       registerType(tree, parametrizedTypeCache.getWildcardType(bound, boundType));
514     }
515   }

517   @Override
518   public void visitConditionalExpression(ConditionalExpressionTree tree) {
519     if(((ConditionalExpressionTreeImpl) tree).isTypeSet()) {
520       JavaType trueType = getType(tree.trueExpression());
521       if(trueType.isTagged(JavaType.DEFERRED)) {
522         setInferedType(tree.symbolType(), (DeferredType) trueType);
523       }
524       JavaType falseType = getType(tree.falseExpression());
525       if(falseType.isTagged(JavaType.DEFERRED)) {
526         setInferedType(tree.symbolType(), (DeferredType) falseType);
527       }
528     } else {
529       resolveAs(tree.condition(), JavaSymbol.VAR);
530       resolveAs(tree.trueExpression(), JavaSymbol.VAR);
531       resolveAs(tree.falseExpression(), JavaSymbol.VAR);
532 
533       registerType(tree, resolve.conditionalExpressionType(tree, (JavaType) tree.trueExpression().symbolType(),(JavaType) tree.falseExpression().symbolType()));
534     }
535   }

549   @Override
550   public void visitLambdaExpression(LambdaExpressionTree tree) {
551     LambdaExpressionTreeImpl lambdaExpressionTree = (LambdaExpressionTreeImpl) tree;
552     if (lambdaExpressionTree.isTypeSet()) {
553       // type should be tied to a SAM interface
554       JavaType lambdaType = (JavaType) lambdaExpressionTree.symbolType();
555       List<JavaType> samMethodArgs = resolve.findSamMethodArgs(lambdaType);
556       for (int i = 0; i < samMethodArgs.size(); i++) {
557         VariableTree param = lambdaExpressionTree.parameters().get(i);
558         if (param.type().is(Tree.Kind.INFERED_TYPE)) {
559           JavaType inferedType = samMethodArgs.get(i);
560           if(inferedType.isTagged(JavaType.WILDCARD)) {
561             // JLS8 18.5.3
562             inferedType = ((WildCardType) inferedType).bound;
563           }
564           ((AbstractTypedTree) param.type()).setInferedType(inferedType);
565           ((JavaSymbol.VariableJavaSymbol) param.symbol()).type = inferedType;
566         }
567       }
568       super.visitLambdaExpression(tree);
569       if(lambdaType.isUnknown() || lambdaType.isTagged(JavaType.DEFERRED)) {
570         return;
571       }
572       refineLambdaType(lambdaExpressionTree, lambdaType);
573     } else {
574       registerType(tree, symbols.deferedType(lambdaExpressionTree));
575     }
576   }

578   private void refineLambdaType(LambdaExpressionTreeImpl lambdaExpressionTree, JavaType lambdaType) {
579     Optional<JavaSymbol.MethodJavaSymbol> samMethod = resolve.getSamMethod(lambdaType);
580     if (!samMethod.isPresent()) {
581       return;
582     }
583     JavaType samReturnType = (JavaType) samMethod.get().returnType().type();
584     JavaType capturedReturnType = resolve.resolveTypeSubstitution(samReturnType, lambdaType);
585     if (capturedReturnType.is("void") || !lambdaType.isParameterized()) {
586       return;
587     }
588     JavaType refinedReturnType = capturedReturnType;
589     if (lambdaExpressionTree.body().is(Tree.Kind.BLOCK)) {
590       LambdaBlockReturnVisitor lambdaBlockReturnVisitor = new LambdaBlockReturnVisitor();
591       lambdaExpressionTree.body().accept(lambdaBlockReturnVisitor);
592       if(!lambdaBlockReturnVisitor.types.isEmpty()) {
593         refinedReturnType = (JavaType) resolve.leastUpperBound(lambdaBlockReturnVisitor.types);
594       }
595     } else {
596       refinedReturnType = (JavaType) ((AbstractTypedTree) lambdaExpressionTree.body()).symbolType();
597     }
598     refineType(lambdaExpressionTree, lambdaType, capturedReturnType, refinedReturnType);
599   }

601   private void refineType(AbstractTypedTree expression, JavaType expressionType, JavaType capturedReturnType, JavaType refinedReturnType) {
602     if (refinedReturnType != capturedReturnType) {
603       // found a lambda return type different from the one infered : update infered type
604       if (expressionType.isTagged(JavaType.PARAMETERIZED)) {
605         ParametrizedTypeJavaType functionType = (ParametrizedTypeJavaType) resolve.functionType((ParametrizedTypeJavaType) expressionType);
606         TypeSubstitution typeSubstitution = ((ParametrizedTypeJavaType) expressionType).typeSubstitution;
607         typeSubstitution.substitutionEntries().stream()
608           .filter(e -> e.getValue() == capturedReturnType)
609           .map(Map.Entry::getKey)
610           .findFirst()
611           .ifPresent(t -> {
612             if(refinedReturnType instanceof DeferredType) {
613               setInferedType(functionType.typeSubstitution.substitutedType(t), (DeferredType) refinedReturnType);
614             } else  {
615               TypeSubstitution refinedSubstitution = new TypeSubstitution(typeSubstitution).add(t, refinedReturnType);
616               JavaType refinedType = parametrizedTypeCache.getParametrizedTypeType(expressionType.symbol, refinedSubstitution);
617               expression.setType(refinedType);
618             }
619           });
620       } else {
621         expression.setType(refinedReturnType);
622       }
623     }
624   }

628     super.visitReturnStatement(tree);
629     ExpressionTree expression = tree.expression();

637         }
638       }
639       Type infered;

641         infered = ((MethodTree) parent).returnType().symbolType();
642       } else {
643         infered = ((LambdaExpressionTree) parent).symbolType();
644       }
645       setInferedType(infered, (DeferredType) expression.symbolType());
646     }

780   @Override
781   public void visitNewClass(NewClassTree tree) {
782     NewClassTreeImpl newClassTreeImpl = (NewClassTreeImpl) tree;
783     if (newClassTreeImpl.isTypeSet()) {
784       return;
785     }
786     List<JavaType> typeArgumentsTypes = ImmutableList.of();
787     if (tree.typeArguments() != null) {
788       resolveAs((List<Tree>) tree.typeArguments(), JavaSymbol.TYP);
789       typeArgumentsTypes = tree.typeArguments().stream().map(this::getType).collect(Collectors.toList());
790     }
791     resolveAs((List<ExpressionTree>) tree.arguments(), JavaSymbol.VAR);
792     List<JavaType> parameterTypes = getParameterTypes(tree.arguments());
793     Resolve.Env newClassEnv = semanticModel.getEnv(tree);
794     ExpressionTree enclosingExpression = tree.enclosingExpression();
795 
796     TypeTree typeTree = tree.identifier();
797     IdentifierTree constructorIdentifier = newClassTreeImpl.getConstructorIdentifier();
798     JavaType identifierType = resolveIdentifierType(newClassEnv, enclosingExpression, typeTree, constructorIdentifier.name());
799     JavaSymbol.TypeJavaSymbol constructorIdentifierSymbol = (JavaSymbol.TypeJavaSymbol) identifierType.symbol();
800     constructorIdentifierSymbol.addUsage(constructorIdentifier);
801     parameterTypes = addImplicitOuterClassParameter(parameterTypes, constructorIdentifierSymbol);
802     Resolution resolution = resolveConstructorSymbol(constructorIdentifier, identifierType, newClassEnv, parameterTypes, typeArgumentsTypes);
803     JavaType constructedType = identifierType;
804     if (resolution.symbol().isMethodSymbol()) {
805       constructedType = ((MethodJavaType) resolution.type()).resultType;
806       if (constructedType.isTagged(JavaType.DEFERRED)) {
807         Tree parent = newClassTreeImpl.parent();
808         if (parent.is(Tree.Kind.MEMBER_SELECT)) {
809           constructedType = resolve.parametrizedTypeWithErasure((ParametrizedTypeJavaType) identifierType);
810         } else {
811           // will be resolved by type inference
812           ((DeferredType) constructedType).setTree(newClassTreeImpl);
813         }
814       } else if (identifierType.symbol().isInterface()) {
815         // constructor of interface always resolve to 'Object' no-arg constructor
816         registerType(typeTree, identifierType);
817       } else {
818         registerType(typeTree, resolution.type());
819       }
820     }
821     ClassTree classBody = tree.classBody();
822     if (classBody != null) {
823       constructedType = getAnonymousClassType(identifierType, constructedType, classBody);
824     }
825     registerType(tree, constructedType);
826   }

828   private JavaType getAnonymousClassType(JavaType identifierType, JavaType constructedType, ClassTree classBody) {
829     JavaType parentType = (constructedType.isTagged(JavaType.DEFERRED) || identifierType.symbol().isInterface()) ? identifierType : constructedType;
830     ClassJavaType anonymousClassType = (ClassJavaType) classBody.symbol().type();
831     if (parentType.getSymbol().isInterface()) {
832       anonymousClassType.interfaces = ImmutableList.of(parentType);
833       anonymousClassType.supertype = symbols.objectType;
834     } else {
835       anonymousClassType.supertype = parentType;
836       anonymousClassType.interfaces = ImmutableList.of();
837     }
838     anonymousClassType.symbol.members.enter(new JavaSymbol.VariableJavaSymbol(Flags.FINAL, "super", anonymousClassType.supertype, anonymousClassType.symbol));
839     scan(classBody);
840     return anonymousClassType;
841   }

952   private JavaType getIteratedObjectType(JavaType type, Set<String> knownTypes) {
953     if (type.isUnknown()) {
954       return Symbols.unknownType;
955     }
956     String fullyQualifiedName = type.fullyQualifiedName();
957     if (knownTypes.contains(fullyQualifiedName)) {
958       // already visited, early return to avoid loops in type hierarchy
959       return Symbols.unknownType;
960     }
961     knownTypes.add(fullyQualifiedName);
962     if (type.is("java.lang.Iterable")) {
963       if (!type.isParameterized()) {
964         // raw type
965         return symbols.objectType;
966       }
967       ParametrizedTypeJavaType ptjt = (ParametrizedTypeJavaType) type;
968       return ptjt.substitution(ptjt.typeParameters().get(0));
969     }
970     return type.directSuperTypes().stream()
971       .map(superType -> getIteratedObjectType(superType, knownTypes))
972       .filter(resolved -> !resolved.isUnknown())
973       .findFirst().orElse(Symbols.unknownType);
974   }

976   /**
977    * Upward projection: JLS10 - §14.4.1, §4.10.5, §5.1.10
978    *
979    * Find a close supertype of a type, where that supertype does not mention certain synthetic type variables
980    * (type variables introduced by the compiler during capture conversion or inference variable resolution).
981    *
982    * NOTE: Capture conversions or intersection types are not implemented as proper types in SonarJava, so
983    * approximating to type of the expression, except for wildcards, which are the closest thing we have from
984    * capture conversions.
985    *
986    * @param initializerType the type computed for the initializer of a local variable, to be projected upward
987    * @return the upward projection of the initializer type
988    */
989   private JavaType upwardProjection(JavaType initializerType) {
990 
991     if (initializerType.isTagged(JavaType.WILDCARD)) {
992       // JLS10 - §5.1.10: the result is the upward projection of the upper bound of T.
993       WildCardType type = (WildCardType) initializerType;
994       switch (type.boundType) {
995         case UNBOUNDED:
996           return symbols.objectType;
997         case EXTENDS:
998           return type.bound;
999         case SUPER:
1000           // the highest upper bound of the the bound
1001           return symbols.objectType;
1002       }
1003     }
1004     if (initializerType.isTagged(JavaType.TYPEVAR)) {
1005       return initializerType.erasure();
1006     }
1007     // identity
1008     return initializerType;
1009   }

1070   /**
1071    * JLS8 4.9 Intersection types
1072    */
1073   private JavaType intersectionType(JavaType type, Set<JavaType> interfaces, Tree tree) {
1074     if (interfaces.isEmpty()) {
1075       return type;
1076     }
1077     List<String> names = new ArrayList<>();
1078     names.add(type.name());
1079     names.addAll(interfaces.stream().map(JavaType::name).collect(Collectors.toSet()));
1080     JavaSymbol.TypeJavaSymbol typeJavaSymbol = new JavaSymbol.TypeJavaSymbol(0, "<intersectionType of "+names.stream().collect(Collectors.joining(" & ")) + ">",
1081       semanticModel.getEnv(tree).packge);
1082 
1083     typeJavaSymbol.members = new Scope(typeJavaSymbol);
1084     IntersectionType intersectionType = new IntersectionType(typeJavaSymbol);
1085     typeJavaSymbol.type = intersectionType;
1086     Set<JavaType> superInterfaces =  new HashSet<>(interfaces);
1087     if(type.symbol.isInterface()) {
1088       superInterfaces.add(type);
1089       intersectionType.supertype = symbols.objectType;
1090     } else {
1091       intersectionType.supertype = type;
1092     }
1093     intersectionType.interfaces = ImmutableList.<JavaType>builder().addAll(superInterfaces).build();
1094     return intersectionType;
1095   }

1126     if (((AbstractTypedTree) tree.annotationType()).isTypeSet()) {
1127       //FIXME: annotation type is set, so we skip this annotation as it was already visited.
1128       // This handle the case where type and its annotation is shared between two variables.
1129       return;
1130     }
1131     resolveAs(tree.annotationType(), JavaSymbol.TYP);
1132     Arguments arguments = tree.arguments();
1133     if (arguments.size() > 1 || (!arguments.isEmpty() && arguments.get(0).is(Tree.Kind.ASSIGNMENT))) {
1134       // resolve by identifying correct identifier in assignment.
1135       for (ExpressionTree expressionTree : arguments) {
1136         AssignmentExpressionTree aet = (AssignmentExpressionTree) expressionTree;
1137         IdentifierTree variable = (IdentifierTree) aet.variable();
1138         JavaSymbol identInType = resolve.findMethod(semanticModel.getEnv(tree), getType(tree.annotationType()), variable.name(), ImmutableList.<JavaType>of()).symbol();
1139         associateReference(variable, identInType);
1140         JavaType type = identInType.type;
1141         if (type == null) {
1142           type = Symbols.unknownType;
1143         }
1144         registerType(variable, type);
1145         resolveAs(aet.expression(), JavaSymbol.VAR);
1146       }
1147     } else {
1148       for (ExpressionTree expressionTree : arguments) {
1149         resolveAs(expressionTree, JavaSymbol.VAR);
1150       }
1151     }
1152     registerType(tree, getType(tree.annotationType()));
1153   }

1168   public void visitMethodReference(MethodReferenceTree methodReferenceTree) {
1169     MethodReferenceTreeImpl methodRefTree = (MethodReferenceTreeImpl) methodReferenceTree;
1170     if (methodRefTree.isTypeSet() && methodReferenceTree.typeArguments() == null) {
1171       resolve.getSamMethod((JavaType) methodRefTree.symbolType()).ifPresent(samMethod -> resolveMethodReference(samMethod, methodRefTree));
1172     } else {
1173       // TODO : SONARJAVA-1663 : consider type arguments for method resolution and substitution

1180   private void resolveMethodReference(JavaSymbol.MethodJavaSymbol samMethod, MethodReferenceTreeImpl methodRefTree) {
1181     JavaType methodRefType = (JavaType) methodRefTree.symbolType();
1182     JavaType samReturnType = (JavaType) samMethod.returnType().type();
1183     List<JavaType> samMethodArgs = resolve.findSamMethodArgs(methodRefType);
1184     if (JavaKeyword.NEW.getValue().equals(methodRefTree.method().name())) {
1185       Type constructorType = ((AbstractTypedTree) methodRefTree.expression()).symbolType();
1186       samMethodArgs = addImplicitOuterClassParameter(samMethodArgs, (JavaSymbol.TypeJavaSymbol) constructorType.symbol());
1187     }
1188     Resolution resolution = resolve.findMethodReference(semanticModel.getEnv(methodRefTree), samMethodArgs, methodRefTree);
1189     JavaSymbol methodSymbol = resolution.symbol();
1190     if (methodSymbol.isMethodSymbol()) {
1191       IdentifierTree methodIdentifier = methodRefTree.method();
1192       addMethodRefReference(methodIdentifier, methodSymbol);
1193       setMethodRefType(methodRefTree, methodRefType, resolution.type());
1194 
1195       JavaType capturedReturnType = resolve.resolveTypeSubstitution(samReturnType, methodRefType);
1196       JavaType refinedReturnType = ((MethodJavaType) methodIdentifier.symbolType()).resultType();
1197       if ("<init>".equals(methodSymbol.name)) {
1198         refinedReturnType = refinedTypeForConstructor(capturedReturnType, refinedReturnType);
1199       }
1200       if (refinedReturnType instanceof DeferredType) {
1201         ((DeferredType) refinedReturnType).setTree((AbstractTypedTree) methodRefTree.method());
1202       }
1203       refineType(methodRefTree, methodRefType, capturedReturnType, refinedReturnType);
1204     } else {
1205       handleNewArray(methodRefTree, methodRefType, samReturnType);
1206     }
1207   }

1219   private JavaType refinedTypeForConstructor(JavaType capturedReturnType, JavaType refinedReturnType) {
1220     JavaType sanitizedCaptured = capturedReturnType;
1221     JavaType refinedConstructorType = refinedReturnType;
1222     if (refinedConstructorType.symbol().isTypeSymbol() && !((JavaSymbol.TypeJavaSymbol) refinedConstructorType.symbol()).typeParameters().scopeSymbols().isEmpty()) {
1223       refinedConstructorType = parametrizedTypeCache.getParametrizedTypeType(refinedConstructorType.symbol, new TypeSubstitution());
1224     }
1225     if (sanitizedCaptured.isTagged(JavaType.TYPEVAR)) {
1226       sanitizedCaptured = ((TypeVariableJavaType) sanitizedCaptured).bounds.get(0);
1227     }
1228     if (refinedConstructorType.isParameterized()) {
1229       refinedConstructorType = resolve.resolveTypeSubstitutionWithDiamondOperator((ParametrizedTypeJavaType) refinedConstructorType, sanitizedCaptured);
1230     }
1231     return refinedConstructorType;
1232   }

1234   private void handleNewArray(MethodReferenceTree methodReferenceTree, JavaType methodRefType, JavaType samReturnType) {
1235     JavaType expressionType = getType(methodReferenceTree.expression());

1237       JavaType capturedReturnType = resolve.resolveTypeSubstitution(samReturnType, methodRefType);
1238       refineType((MethodReferenceTreeImpl) methodReferenceTree, methodRefType, capturedReturnType, expressionType);
1239     }

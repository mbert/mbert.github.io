448   private static class FakeVisitor extends IssuableSubscriptionVisitor implements IssueWithFlowBuilder {
449 
450     ListMultimap<Integer, String> issues = LinkedListMultimap.create();
451     ListMultimap<Integer, AnalyzerMessage> preciseIssues = LinkedListMultimap.create();
452     List<String> issuesOnFile = Lists.newLinkedList();
453     private AnalyzerMessage issueWithFlow;
454 
455     private FakeVisitor withDefaultIssues() {
456       AnalyzerMessage withMultipleLocation = new AnalyzerMessage(this, new File("a"), new AnalyzerMessage.TextSpan(10, 9, 10, 10), "message4", 3);
457       withMultipleLocation.flows.add(Collections.singletonList(new AnalyzerMessage(this, new File("a"), 3, "no message", 0)));
458       withMultipleLocation.flows.add(Collections.singletonList(new AnalyzerMessage(this, new File("a"), 4, "no message", 0)));
459       return this.withIssue(1, "message")
460         .withIssue(3, "message1")
461         .withIssue(7, "message2")
462         .withIssue(8, "message3")
463         .withIssue(8, "message3")
464         .withPreciseIssue(withMultipleLocation)
465         .withPreciseIssue(new AnalyzerMessage(this, new File("a"), 11, "no message", 4))
466         .withPreciseIssue(new AnalyzerMessage(this, new File("a"), 12, "message12", 0))
467         .withPreciseIssue(new AnalyzerMessage(this, new File("a"), new AnalyzerMessage.TextSpan(14, 5, 15, 11), "message12", 0))
468         .withIssue(17, "message17");
469     }
470 
471     private FakeVisitor withPreciseIssue(AnalyzerMessage message) {
472       preciseIssues.put(message.getLine(), message);
473       return this;
474     }
475 
476     private FakeVisitor withIssue(int line, String message) {
477       issues.put(line, message);
478       return this;
479     }
480 
481     private FakeVisitor withoutIssue(int line) {
482       issues.removeAll(line);
483       preciseIssues.removeAll(line);
484       return this;
485     }
486 
487     private FakeVisitor withIssueOnFile(String message) {
488       issuesOnFile.add(message);
489       return this;
490     }
491 
492     private FakeVisitor issueWithFlow(int line) {
493       return issueWithFlow(null, new AnalyzerMessage.TextSpan(line));
494     }
495 
496     private FakeVisitor issueWithFlow(int line, @Nullable String message, int startColumn, int endLine, int endColumn) {
497       return issueWithFlow(message, new AnalyzerMessage.TextSpan(line, startColumn, endLine, endColumn));
498     }
499 
500     private FakeVisitor issueWithFlow(@Nullable String message, AnalyzerMessage.TextSpan location) {
501       Preconditions.checkState(issueWithFlow == null, "Finish previous issueWithFlow by calling #add");
502       issueWithFlow = new AnalyzerMessage(this, new File("f"), location, message, 0);
503       return this;
504     }
505 
506     private FakeVisitor flow() {
507       Preconditions.checkNotNull(issueWithFlow, "Finish previous issueWithFlow by calling #add");
508       issueWithFlow.flows.add(new LinkedList<>());
509       return this;
510     }
511 
512     private FakeVisitor flow(int line1, @Nullable String msg1, int line2, @Nullable String msg2) {
513       flow();
514       flowItem(line1, msg1);
515       return flowItem(line2, msg2);
516     }
517 
518     private FakeVisitor flowItem(int line, @Nullable String msg) {
519       return flowItem(msg, new AnalyzerMessage.TextSpan(line));
520     }
521 
522     private FakeVisitor flowItem(int line, @Nullable String msg, int startColumn, int endColumn) {
523       return flowItem(msg, new AnalyzerMessage.TextSpan(line, startColumn, line, endColumn));
524     }
525 
526     private FakeVisitor flowItem(@Nullable String msg, AnalyzerMessage.TextSpan textSpan) {
527       List<List<AnalyzerMessage>> flows = issueWithFlow.flows;
528       Preconditions.checkState(!flows.isEmpty(), "Call #flow first to create a flow");
529       AnalyzerMessage flowItem = new AnalyzerMessage(this, new File("f"), textSpan, msg, 0);
530       flows.get(flows.size() - 1).add(flowItem);
531       return this;
532     }
533 
534     private FakeVisitor add() {
535       // flows are in reverse order the same way as real checks report in reverse order
536       issueWithFlow.flows.forEach(Collections::reverse);
537       withPreciseIssue(issueWithFlow);
538       issueWithFlow = null;
539       return this;
540     }
541 
542     @Override
543     public List<Tree.Kind> nodesToVisit() {
544       return ImmutableList.of();
545     }
546 
547     @Override
548     public void setContext(JavaFileScannerContext context) {
549       super.setContext(context);
550       for (Integer line : issues.keySet()) {
551         for (String message : issues.get(line)) {
552           addIssue(line, message);
553         }
554       }
555       for (AnalyzerMessage analyzerMessage : preciseIssues.values()) {
556         Double messageCost = analyzerMessage.getCost();
557         Integer cost = messageCost != null ? messageCost.intValue() : null;
558         List<JavaFileScannerContext.Location> secLocations = new ArrayList<>();
559         if (!analyzerMessage.flows.isEmpty()) {
560           List<List<JavaFileScannerContext.Location>> flows = analyzerMessage.flows.stream()
561             .map(FakeVisitor::messagesToLocations)
562             .collect(toList());
563           context.reportIssueWithFlow(this, mockTree(analyzerMessage), analyzerMessage.getMessage(), flows, null);
564         } else {
565           reportIssue(mockTree(analyzerMessage), analyzerMessage.getMessage(), secLocations, cost);
566         }
567       }
568       for (String message : issuesOnFile) {
569         addIssueOnFile(message);
570       }
571     }
572 
573     private static List<JavaFileScannerContext.Location> messagesToLocations(List<AnalyzerMessage> flow) {
574       return flow.stream().map(m -> new JavaFileScannerContext.Location(m.getMessage(), mockTree(m))).collect(toList());
575     }
576 
577     private static Tree mockTree(final AnalyzerMessage analyzerMessage) {
578       AnalyzerMessage.TextSpan textSpan = analyzerMessage.primaryLocation();
579       if (textSpan.onLine()) {
580         return new InternalSyntaxToken(textSpan.startLine, 0, "mock", Lists.<SyntaxTrivia>newArrayList(), 0, 0, false);
581       }
582       return new ReturnStatementTreeImpl(
583         new InternalSyntaxToken(textSpan.startLine, textSpan.startCharacter - 1, "", Lists.<SyntaxTrivia>newArrayList(), 0, 0, false),
584         null,
585         new InternalSyntaxToken(textSpan.endLine, textSpan.endCharacter - 1, "", Lists.<SyntaxTrivia>newArrayList(), 0, 0, false));
586     }

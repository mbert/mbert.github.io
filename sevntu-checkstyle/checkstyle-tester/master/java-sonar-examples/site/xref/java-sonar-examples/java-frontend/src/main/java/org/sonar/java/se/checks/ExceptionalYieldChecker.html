50  public class ExceptionalYieldChecker {
51  
52    private final String message;
53  
54    ExceptionalYieldChecker(String message) {
55      this.message = message;
56    }
57  
58    void reportOnExceptionalYield(ExplodedGraph.Node node, SECheck check) {
59      node.edges().stream().forEach(edge -> edge.yields().stream()
60        .filter(yield -> yield.generatedByCheck(check))
61        .forEach(yield -> reportIssue(edge.parent(), (ExceptionalCheckBasedYield) yield, check))
62      );
63    }
64  
65    private void reportIssue(ExplodedGraph.Node node, ExceptionalCheckBasedYield yield, SECheck check) {
66      MethodInvocationTree mit = (MethodInvocationTree) node.programPoint.syntaxTree();
67      ExpressionTree methodSelect = mit.methodSelect();
68      String methodName = mit.symbol().name();
69  
70      Tree reportTree = methodSelect;
71      if (methodSelect.is(Tree.Kind.MEMBER_SELECT)) {
72        reportTree = ((MemberSelectExpressionTree) methodSelect).identifier();
73      }
74  
75      JavaFileScannerContext.Location methodInvocationMessage;
76      int parameterCausingExceptionIndex = yield.parameterCausingExceptionIndex();
77      IdentifierTree identifierTree = FlowComputation.getArgumentIdentifier(mit, parameterCausingExceptionIndex);
78      if (identifierTree != null) {
79        methodInvocationMessage = new JavaFileScannerContext.Location(String.format("'%s' is passed to '%s()'.", identifierTree.name(), methodName), identifierTree);
80      } else {
81        methodInvocationMessage = new JavaFileScannerContext.Location(String.format("'%s()' is invoked.", methodName), reportTree);
82      }
83  
84      Flow argumentChangingNameFlows = flowsForArgumentsChangingName(yield, mit);
85      Set<Flow> argumentsFlows = flowsForMethodArguments(node, mit, parameterCausingExceptionIndex);
86      Set<Flow> exceptionFlows = yield.exceptionFlows();
87  
88      ImmutableSet.Builder<Flow> flows = ImmutableSet.builder();
89      for (Flow argumentFlow : argumentsFlows) {
90        for (Flow exceptionFlow : exceptionFlows) {
91          flows.add(Flow.builder()
92            .addAll(exceptionFlow)
93            .addAll(argumentChangingNameFlows)
94            .add(methodInvocationMessage)
95            .addAll(argumentFlow)
96            .build());
97        }
98      }
99  
100     check.reportIssue(reportTree, String.format(message, methodName), flows.build());
101   }
102 
103   private static Set<Flow> flowsForMethodArguments(ExplodedGraph.Node node, MethodInvocationTree mit, int parameterCausingExceptionIndex) {
104     ProgramState programState = node.programState;
105     List<ProgramState.SymbolicValueSymbol> arguments = Lists.reverse(programState.peekValuesAndSymbols(mit.arguments().size()));
106     SymbolicValue parameterCausingExceptionSV = arguments.get(parameterCausingExceptionIndex).symbolicValue();
107 
108     Set<SymbolicValue> argSymbolicValues = new LinkedHashSet<>();
109     Set<Symbol> argSymbols = new LinkedHashSet<>();
110     arguments.stream()
111       .filter(svs -> parameterCausingExceptionSV == svs.symbolicValue() || hasConstraintOtherThanNonNull(svs, programState))
112       .forEach(svs -> {
113         argSymbolicValues.add(svs.symbolicValue());
114         Symbol symbol = svs.symbol();
115         if (symbol != null) {
116           argSymbols.add(symbol);
117         }
118       });
119 
120     List<Class<? extends Constraint>> domains = domainsFromArguments(programState, argSymbolicValues);
121     return FlowComputation.flow(node, argSymbolicValues, c -> true, c -> false, domains, argSymbols);
122   }
123 
124   private static boolean hasConstraintOtherThanNonNull(ProgramState.SymbolicValueSymbol svs, ProgramState ps) {
125     SymbolicValue sv = svs.symbolicValue();
126     ConstraintsByDomain constraints = ps.getConstraints(sv);
127     return constraints != null && !hasOnlyNonNullConstraint(constraints);
128   }
129 
130   private static boolean hasOnlyNonNullConstraint(ConstraintsByDomain constraints) {
131     return constraints.domains().count() == 1 && constraints.get(ObjectConstraint.class) == ObjectConstraint.NOT_NULL;
132   }
133 
134   private static List<Class<? extends Constraint>> domainsFromArguments(ProgramState programState, Collection<SymbolicValue> arguments) {
135     return arguments.stream()
136       .map(programState::getConstraints)
137       .filter(Objects::nonNull)
138       .flatMap(ConstraintsByDomain::domains)
139       .distinct()
140       .collect(Collectors.toList());
141   }
142 
143   private static Flow flowsForArgumentsChangingName(ExceptionalCheckBasedYield yield, MethodInvocationTree mit) {
144     return FlowComputation.flowsForArgumentsChangingName(Collections.singletonList(yield.parameterCausingExceptionIndex()), mit);
145   }

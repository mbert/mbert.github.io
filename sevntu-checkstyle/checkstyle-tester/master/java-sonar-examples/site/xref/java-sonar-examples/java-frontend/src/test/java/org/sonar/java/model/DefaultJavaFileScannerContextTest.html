211   @Test
212   public void report_issue_on_tree_with_secondary() {
213     ClassTree tree = (ClassTree) compilationUnitTree.types().get(0);
214     Tree firstMember = tree.members().get(0);
215     Tree secondMember = tree.members().get(1);
216 
217     ArrayList<Location> secondary = new ArrayList<>();
218     secondary.add(new JavaFileScannerContext.Location("secondary", firstMember));
219     secondary.add(new JavaFileScannerContext.Location("secondary", secondMember));
220 
221     context.reportIssue(CHECK, tree.simpleName(), "msg", secondary, null);
222 
223     assertThat(reportedMessage.getMessage()).isEqualTo("msg");
224     assertThat(reportedMessage.getCost()).isNull();
225     assertThat(reportedMessage.flows).hasSize(2);
226 
227     assertMessagePosition(reportedMessage, 1, 6, 1, 7);
228     List<AnalyzerMessage> secondaries = reportedMessage.flows.stream().map(flow -> flow.get(0)).collect(Collectors.toList());
229     assertThat(secondaries).hasSize(2);
230     assertMessagePosition(secondaries.get(0), 2, 2, 2, 13);
231     assertMessagePosition(secondaries.get(1), 3, 2, 3, 15);
232   }

236     ClassTree tree = (ClassTree) compilationUnitTree.types().get(0);
237     VariableTree firstMember = (VariableTree) tree.members().get(0);
238     VariableTree secondMember = (VariableTree) tree.members().get(1);
239 
240     context.reportIssue(CHECK, firstMember.simpleName(), secondMember.equalToken(), "msg");
241 
242     assertThat(reportedMessage.getMessage()).isEqualTo("msg");
243     assertThat(reportedMessage.getCost()).isNull();
244     assertThat(reportedMessage.flows).isEmpty();
245 
246     assertMessagePosition(reportedMessage, 2, 6, 3, 10);
247   }

249   private static void assertMessagePosition(AnalyzerMessage message, int startLine, int startColumn, int endLine, int endColumn) {
250     TextSpan location = message.primaryLocation();
251     assertThat(location.startLine).isEqualTo(startLine);
252     assertThat(location.startCharacter).isEqualTo(startColumn);
253     assertThat(location.endLine).isEqualTo(endLine);
254     assertThat(location.endCharacter).isEqualTo(endColumn);
255   }

257   private static SonarComponents createSonarComponentsMock() {
258     SonarComponents sonarComponents = mock(SonarComponents.class);
259     doAnswer(invocation -> {
260       reportedMessage = (AnalyzerMessage) invocation.getArguments()[0];
261       return null;
262     }).when(sonarComponents).reportIssue(any(AnalyzerMessage.class));
263 
264     doAnswer(invocation -> {
265       Integer cost = invocation.getArgument(4);
266       reportedMessage = new AnalyzerMessage(invocation.getArgument(1),
267           invocation.getArgument(0),
268           null,
269           invocation.getArgument(3),
270           cost != null ? cost : 0);
271       return null;
272     }).when(sonarComponents).addIssue(any(File.class), any(JavaCheck.class), anyInt(), anyString(), any());
273 
274     doAnswer(invocation -> new ArrayList<>()).when(sonarComponents).fileLines(eq(JAVA_FILE));
275     doAnswer(invocation -> "content").when(sonarComponents).fileContent(eq(JAVA_FILE));
276 
277     doAnswer(invocation -> WORK_DIR).when(sonarComponents).workDir();
278     File moduleBaseDir = new File("");
279     SensorContextTester context = SensorContextTester.create(moduleBaseDir);
280     doAnswer(invocation -> context.fileSystem()).when(sonarComponents).getFileSystem();
281 
282     return sonarComponents;
283   }

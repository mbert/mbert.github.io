20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  import org.apache.commons.lang.StringUtils;
24  import org.sonar.check.Rule;
25  import org.sonar.java.matcher.MethodMatcher;
26  import org.sonar.java.model.declaration.VariableTreeImpl;
27  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
28  import org.sonar.plugins.java.api.semantic.Symbol;
29  import org.sonar.plugins.java.api.semantic.Type;
30  import org.sonar.plugins.java.api.tree.ExpressionTree;
31  import org.sonar.plugins.java.api.tree.IdentifierTree;
32  import org.sonar.plugins.java.api.tree.MemberSelectExpressionTree;
33  import org.sonar.plugins.java.api.tree.MethodInvocationTree;
34  import org.sonar.plugins.java.api.tree.NewClassTree;
35  import org.sonar.plugins.java.api.tree.Tree;
36  import org.sonar.plugins.java.api.tree.Tree.Kind;
37  import org.sonar.plugins.java.api.tree.VariableTree;
38  
39  import java.util.List;
40  
41  @Rule(key = "S2130")
42  public class StringToPrimitiveConversionCheck extends IssuableSubscriptionVisitor {
43  
44    private final List<PrimitiveCheck> primitiveChecks = buildPrimitiveChecks();
45  
46    @Override
47    public List<Kind> nodesToVisit() {
48      return ImmutableList.of(Tree.Kind.VARIABLE, Tree.Kind.METHOD_INVOCATION);
49    }
50  
51    @Override
52    public void visitNode(Tree tree) {
53      if (hasSemantic()) {
54        if (tree.is(Tree.Kind.VARIABLE)) {
55          VariableTreeImpl variableTree = (VariableTreeImpl) tree;
56          Type variableType = variableTree.type().symbolType();
57          PrimitiveCheck primitiveCheck = getPrimitiveCheck(variableType);
58          ExpressionTree initializer = variableTree.initializer();
59          if (primitiveCheck != null && initializer != null) {
60            primitiveCheck.checkInstantiation(initializer);
61          }
62        } else {
63          MethodInvocationTree methodInvocationTree = (MethodInvocationTree) tree;
64          for (PrimitiveCheck primitiveCheck : primitiveChecks) {
65            primitiveCheck.checkMethodInvocation(methodInvocationTree);
66          }
67        }
68      }
69    }
70  
71    private PrimitiveCheck getPrimitiveCheck(Type type) {
72      if (!type.isPrimitive()) {
73        return null;
74      }
75      for (PrimitiveCheck primitiveCheck : primitiveChecks) {
76        if (type.isPrimitive(primitiveCheck.tag)) {
77          return primitiveCheck;
78        }
79      }
80      return null;
81    }
82  
83    private List<PrimitiveCheck> buildPrimitiveChecks() {
84      return ImmutableList.of(
85        new PrimitiveCheck("int", "Integer", Type.Primitives.INT),
86        new PrimitiveCheck("boolean", "Boolean", Type.Primitives.BOOLEAN),
87        new PrimitiveCheck("byte", "Byte", Type.Primitives.BYTE),
88        new PrimitiveCheck("double", "Double", Type.Primitives.DOUBLE),
89        new PrimitiveCheck("float", "Float", Type.Primitives.FLOAT),
90        new PrimitiveCheck("long", "Long", Type.Primitives.LONG),
91        new PrimitiveCheck("short", "Short", Type.Primitives.SHORT));
92    }
93  
94    private class PrimitiveCheck {
95      private final String primitiveName;
96      private final String className;
97      private final Type.Primitives tag;
98      private final String message;
99      private final MethodMatcher unboxingInvocationMatcher;
100     private final MethodMatcher valueOfInvocationMatcher;
101 
102     private PrimitiveCheck(String primitiveName, String className, Type.Primitives tag) {
103       this.primitiveName = primitiveName;
104       this.className = className;
105       this.tag = tag;
106       this.message = "Use \"" + parseMethodName() + "\" for this string-to-" + primitiveName + " conversion.";
107       this.unboxingInvocationMatcher = MethodMatcher.create()
108         .typeDefinition("java.lang." + className)
109         .name(primitiveName + "Value")
110         .withoutParameter();
111       this.valueOfInvocationMatcher = MethodMatcher.create()
112         .typeDefinition("java.lang." + className)
113         .name("valueOf")
114         .addParameter("java.lang.String");
115     }
116 
117     private void checkMethodInvocation(MethodInvocationTree methodInvocationTree) {
118       if (unboxingInvocationMatcher.matches(methodInvocationTree)) {
119         MemberSelectExpressionTree methodSelect = (MemberSelectExpressionTree) methodInvocationTree.methodSelect();
120         checkInstantiation(methodSelect.expression());
121       }
122     }
123 
124     private void checkInstantiation(ExpressionTree expression) {
125       if (isBadlyInstantiated(expression)) {
126         reportIssue(expression, message);
127       }
128     }
129 
130     private boolean isBadlyInstantiated(ExpressionTree expression) {
131       boolean result = false;
132       if (expression.is(Tree.Kind.NEW_CLASS)) {
133         result = isStringBasedConstructor((NewClassTree) expression);
134       } else if (expression.is(Tree.Kind.METHOD_INVOCATION)) {
135         result = valueOfInvocationMatcher.matches((MethodInvocationTree) expression);
136       } else if (expression.is(Tree.Kind.IDENTIFIER)) {
137         IdentifierTree identifier = (IdentifierTree) expression;
138         Symbol reference = identifier.symbol();
139         if (reference.isVariableSymbol() && reference.usages().size() == 1) {
140           Symbol.VariableSymbol variableSymbol = (Symbol.VariableSymbol) reference;
141           result = isBadlyInstantiatedVariable(variableSymbol);
142         }
143       }
144       return result;
145     }
146 
147     private boolean isBadlyInstantiatedVariable(Symbol.VariableSymbol variableSymbol) {
148       VariableTree variableTree = variableSymbol.declaration();
149       if (variableTree != null) {
150         ExpressionTree initializer = variableTree.initializer();
151         if (initializer != null) {
152           return isBadlyInstantiated(initializer);
153         }
154       }
155       return false;
156     }
157 
158     private boolean isStringBasedConstructor(NewClassTree newClassTree) {
159       List<ExpressionTree> arguments = newClassTree.arguments();
160       return arguments.get(0).symbolType().is("java.lang.String");
161     }
162 
163     private String parseMethodName() {
164       return className + ".parse" + StringUtils.capitalize(primitiveName);
165     }
166   }

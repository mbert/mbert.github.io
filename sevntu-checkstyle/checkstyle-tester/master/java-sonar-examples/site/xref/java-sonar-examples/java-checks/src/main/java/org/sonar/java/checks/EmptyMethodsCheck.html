20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  
24  import org.sonar.check.Rule;
25  import org.sonar.java.model.ModifiersUtils;
26  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
27  import org.sonar.plugins.java.api.tree.BlockTree;
28  import org.sonar.plugins.java.api.tree.ClassTree;
29  import org.sonar.plugins.java.api.tree.MethodTree;
30  import org.sonar.plugins.java.api.tree.Modifier;
31  import org.sonar.plugins.java.api.tree.StatementTree;
32  import org.sonar.plugins.java.api.tree.Tree;
33  import org.sonar.plugins.java.api.tree.Tree.Kind;
34  
35  import java.util.List;
36  import java.util.stream.Collectors;
37  
38  @Rule(key = "S1186")
39  public class EmptyMethodsCheck extends IssuableSubscriptionVisitor {
40  
41    @Override
42    public List<Kind> nodesToVisit() {
43      return ImmutableList.of(Tree.Kind.CLASS, Tree.Kind.ENUM);
44    }
45  
46    @Override
47    public void visitNode(Tree tree) {
48      ClassTree classTree = (ClassTree) tree;
49      if (!ModifiersUtils.hasModifier(classTree.modifiers(), Modifier.ABSTRACT)) {
50        List<Tree> members = classTree.members();
51        checkMethods(members);
52        checkSingleNoArgPublicConstructor(members);
53      }
54    }
55  
56    private void checkMethods(List<Tree> members) {
57      members.stream()
58        .filter(member -> member.is(Tree.Kind.METHOD))
59        .map(MethodTree.class::cast)
60        .forEach(this::checkMethod);
61    }
62  
63    private void checkSingleNoArgPublicConstructor(List<Tree> members) {
64      List<MethodTree> constructors = members.stream()
65        .filter(member -> member.is(Tree.Kind.CONSTRUCTOR))
66        .map(MethodTree.class::cast)
67        .collect(Collectors.toList());
68      if (constructors.size() == 1 && isPublicNoArgConstructor(constructors.get(0))) {
69        checkMethod(constructors.get(0));
70      }
71    }
72  
73    private static boolean isPublicNoArgConstructor(MethodTree constructor) {
74      return ModifiersUtils.hasModifier(constructor.modifiers(), Modifier.PUBLIC) && constructor.parameters().isEmpty();
75    }
76  
77    private void checkMethod(MethodTree methodTree) {
78      BlockTree block = methodTree.block();
79      if (block != null && isEmpty(block) && !containsComment(block)) {
80        reportIssue(methodTree.simpleName(), "Add a nested comment explaining why this method is empty, throw an UnsupportedOperationException or complete the implementation.");
81      }
82    }
83  
84    private static boolean isEmpty(BlockTree block) {
85      List<StatementTree> body = block.body();
86      return body.isEmpty() || body.stream().allMatch(stmt -> stmt.is(Kind.EMPTY_STATEMENT));
87    }
88  
89    private static boolean containsComment(BlockTree block) {
90      return !block.closeBraceToken().trivias().isEmpty();
91    }
92  
93  }

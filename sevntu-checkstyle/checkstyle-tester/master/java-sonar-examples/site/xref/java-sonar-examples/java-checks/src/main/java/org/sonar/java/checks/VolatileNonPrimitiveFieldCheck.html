20  package org.sonar.java.checks;
21  
22  import java.util.Arrays;
23  import java.util.List;
24  import org.sonar.check.Rule;
25  import org.sonar.java.model.ModifiersUtils;
26  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
27  import org.sonar.plugins.java.api.semantic.Type;
28  import org.sonar.plugins.java.api.tree.ClassTree;
29  import org.sonar.plugins.java.api.tree.Modifier;
30  import org.sonar.plugins.java.api.tree.Tree;
31  import org.sonar.plugins.java.api.tree.VariableTree;
32  
33  @Rule(key = "S3077")
34  public class VolatileNonPrimitiveFieldCheck extends IssuableSubscriptionVisitor {
35  
36    private static final String REF_MESSAGE = "Remove the \"volatile\" keyword from this field.";
37    private static final String ARRAY_MESSAGE = "Use an \"Atomic%sArray\" instead.";
38  
39    @Override
40    public List<Tree.Kind> nodesToVisit() {
41      return Arrays.asList(Tree.Kind.CLASS, Tree.Kind.ENUM);
42    }
43  
44    @Override
45    public void visitNode(Tree tree) {
46      ((ClassTree) tree).members()
47        .stream()
48        .filter(m -> m.is(Tree.Kind.VARIABLE))
49        .map(m -> ((VariableTree) m))
50        .filter(v -> ModifiersUtils.hasModifier(v.modifiers(), Modifier.VOLATILE))
51        .filter(v -> !v.type().symbolType().isPrimitive())
52        .forEach(v -> reportIssue(ModifiersUtils.getModifier(v.modifiers(), Modifier.VOLATILE), v.type(), getMessage(v)));
53    }
54  
55    private static String getMessage(VariableTree variableTree) {
56      Type varType = variableTree.type().symbolType();
57      if (varType.isArray()) {
58        String atomicType = "Reference";
59        Type elementType = ((Type.ArrayType) varType).elementType();
60        if (elementType.isPrimitive(Type.Primitives.LONG)) {
61          atomicType = "Long";
62        } else if (elementType.isPrimitive(Type.Primitives.INT)) {
63          atomicType = "Integer";
64        }
65        return String.format(ARRAY_MESSAGE, atomicType);
66      }
67      return REF_MESSAGE;
68    }
69  }

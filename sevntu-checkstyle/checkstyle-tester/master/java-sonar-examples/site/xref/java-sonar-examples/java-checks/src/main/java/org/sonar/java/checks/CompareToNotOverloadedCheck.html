20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  import java.util.List;
24  import org.sonar.check.Rule;
25  import org.sonar.java.resolve.ClassJavaType;
26  import org.sonar.java.resolve.ParametrizedTypeJavaType;
27  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
28  import org.sonar.plugins.java.api.tree.MethodTree;
29  import org.sonar.plugins.java.api.tree.Tree;
30  
31  @Rule(key = "S4351")
32  public class CompareToNotOverloadedCheck extends IssuableSubscriptionVisitor {
33  
34    @Override
35    public List<Tree.Kind> nodesToVisit() {
36      return ImmutableList.of(Tree.Kind.METHOD);
37    }
38  
39    @Override
40    public void visitNode(Tree tree) {
41      MethodTree methodTree = (MethodTree) tree;
42      if (hasSemantic() && isCompareToMethod(methodTree) && Boolean.FALSE.equals(methodTree.isOverriding())) {
43        ClassJavaType ownerType = (ClassJavaType) methodTree.symbol().owner().type();
44        ownerType.superTypes().stream().filter(supertype -> supertype.is("java.lang.Comparable")).findFirst().ifPresent(
45          comparableType -> {
46            String name = "Object";
47            if (comparableType.isParameterized()) {
48              ParametrizedTypeJavaType ptjt = (ParametrizedTypeJavaType) comparableType;
49              name = ptjt.substitution(ptjt.typeParameters().get(0)).symbol().name();
50            }
51            reportIssue(methodTree.parameters().get(0), "Refactor this method so that its argument is of type '" + name + "'.");
52          });
53      }
54    }
55  
56    private static boolean isCompareToMethod(MethodTree tree) {
57      return "compareTo".equals(tree.simpleName().name()) && tree.parameters().size() == 1;
58    }
59  
60  }

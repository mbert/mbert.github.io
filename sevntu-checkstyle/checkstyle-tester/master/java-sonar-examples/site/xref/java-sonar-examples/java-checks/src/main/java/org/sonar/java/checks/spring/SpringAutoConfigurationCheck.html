20  package org.sonar.java.checks.spring;
21  
22  import java.util.Arrays;
23  import java.util.Collections;
24  import java.util.HashSet;
25  import java.util.List;
26  import java.util.Set;
27  import org.sonar.check.Rule;
28  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
29  import org.sonar.plugins.java.api.tree.AnnotationTree;
30  import org.sonar.plugins.java.api.tree.Arguments;
31  import org.sonar.plugins.java.api.tree.AssignmentExpressionTree;
32  import org.sonar.plugins.java.api.tree.ClassTree;
33  import org.sonar.plugins.java.api.tree.ExpressionTree;
34  import org.sonar.plugins.java.api.tree.NewArrayTree;
35  import org.sonar.plugins.java.api.tree.Tree;
36  
37  @Rule(key = "S4604")
38  public class SpringAutoConfigurationCheck extends IssuableSubscriptionVisitor {
39  
40    private static final List<String> ANNOTATIONS = Arrays.asList(
41      "org.springframework.boot.autoconfigure.SpringBootApplication",
42      "org.springframework.boot.autoconfigure.EnableAutoConfiguration");
43  
44    private static final Set<String> EXCLUDE_ELEMENTS = new HashSet<>(Arrays.asList("exclude", "excludeName"));
45  
46    @Override
47    public List<Tree.Kind> nodesToVisit() {
48      return Collections.singletonList(Tree.Kind.CLASS);
49    }
50  
51    @Override
52    public void visitNode(Tree tree) {
53      if (!hasSemantic()) {
54        return;
55      }
56      ((ClassTree) tree).modifiers().annotations().stream()
57        .filter(SpringAutoConfigurationCheck::isAutoConfiguration)
58        .filter(annotation -> !hasExclude(annotation.arguments()))
59        .forEach(annotation -> reportIssue(annotation, "Exclude from the auto-configuration mechanism the beans you don't need."));
60    }
61  
62    private static boolean isAutoConfiguration(AnnotationTree annotationTree) {
63      return ANNOTATIONS.stream().anyMatch(annotationTree.annotationType().symbolType()::is);
64    }
65  
66    private static boolean hasExclude(Arguments arguments) {
67      return arguments.stream()
68        .filter(arg -> arg.is(Tree.Kind.ASSIGNMENT))
69        .map(arg -> (AssignmentExpressionTree) arg)
70        .anyMatch(SpringAutoConfigurationCheck::isExcludeElement);
71    }
72  
73    private static boolean isExcludeElement(AssignmentExpressionTree assignment) {
74      ExpressionTree expression = assignment.expression();
75      boolean isExcludeElement = EXCLUDE_ELEMENTS.contains(assignment.variable().toString());
76      boolean arrayNotEmpty = expression.is(Tree.Kind.NEW_ARRAY) && !((NewArrayTree) expression).initializers().isEmpty();
77      return isExcludeElement && arrayNotEmpty;
78    }
79  }

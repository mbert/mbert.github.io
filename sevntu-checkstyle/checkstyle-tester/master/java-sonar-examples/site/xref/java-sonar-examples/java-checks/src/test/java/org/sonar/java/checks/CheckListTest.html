74      int count = 0;
75      List<File> files = (List<File>) FileUtils.listFiles(new File("src/main/java/org/sonar/java/checks/"), new String[] {"java"}, true);

84    private static class CustomRulesDefinition implements RulesDefinition {
85  
86      @Override
87      public void define(Context context) {
88        String language = "java";
89        NewRepository repository = context
90          .createRepository(CheckList.REPOSITORY_KEY, language)
91          .setName("SonarQube");
92  
93        List<Class> checks = CheckList.getChecks();
94        new RulesDefinitionAnnotationLoader().load(repository, checks.toArray(new Class[checks.size()]));
95  
96        for (NewRule rule : repository.rules()) {
97          try {
98            rule.setName("Artificial Name (set via JSON files, no need to test it)");
99            rule.setMarkdownDescription(ARTIFICIAL_DESCRIPTION);
100         } catch (IllegalStateException e) {
101           // it means that the html description was already set in Rule annotation
102           fail("Description of " + rule.key() + " should be in separate file");
103         }
104       }
105       repository.done();
106     }
107   }

114   /**
115    * Enforces that each check has test, name and description.
116    */
117   @Test
118   public void test() {
119     List<Class> checks = CheckList.getChecks();
120     Map<String, String> keyMap = new HashMap<>();
121     for (Class cls : checks) {
122       String testName = '/' + cls.getName().replace('.', '/') + "Test.class";
123       String simpleName = cls.getSimpleName();
124       // Handle legacy keys.
125       Rule ruleAnnotation = AnnotationUtils.getAnnotation(cls, Rule.class);
126       String key = getKey(cls, ruleAnnotation);
127       keyMap.put(ruleAnnotation.key(), key);
128       if (SE_CHEKS.contains(simpleName)) {
129         continue;
130       }
131       assertThat(getClass().getResource(testName))
132         .overridingErrorMessage("No test for " + simpleName)
133         .isNotNull();
134     }
135 
136     Set<String> keys = Sets.newHashSet();
137     Set<String> names = Sets.newHashSet();
138     CustomRulesDefinition definition = new CustomRulesDefinition();
139     RulesDefinition.Context context = new RulesDefinition.Context();
140     definition.define(context);
141     List<RulesDefinition.Rule> rules = context.repository(CheckList.REPOSITORY_KEY).rules();
142     for (RulesDefinition.Rule rule : rules) {
143       assertThat(keys).as("Duplicate key " + rule.key()).doesNotContain(rule.key());
144       keys.add(rule.key());
145       names.add(rule.name());
146       assertThat(getClass().getResource("/org/sonar/l10n/java/rules/" + CheckList.REPOSITORY_KEY + "/" + keyMap.get(rule.key()) + "_java.html"))
147         .overridingErrorMessage("No description for " + rule.key()+ " " +keyMap.get(rule.key()))
148         .isNotNull();
149       assertThat(getClass().getResource("/org/sonar/l10n/java/rules/" + CheckList.REPOSITORY_KEY + "/" + keyMap.get(rule.key()) + "_java.json"))
150         .overridingErrorMessage("No json metadata file for " + rule.key()+ " " +keyMap.get(rule.key()))
151         .isNotNull();
152 
153       assertThat(rule.htmlDescription()).isNull();
154       assertThat(rule.markdownDescription()).isEqualTo(ARTIFICIAL_DESCRIPTION);
155 
156       for (RulesDefinition.Param param : rule.params()) {
157         assertThat(param.description()).overridingErrorMessage(rule.key() + " missing description for param " + param.key()).isNotEmpty();
158       }
159     }
160   }

162   private static String getKey(Class cls, Rule ruleAnnotation) {
163     String key = ruleAnnotation.key();
164     RspecKey rspecKeyAnnotation = AnnotationUtils.getAnnotation(cls, RspecKey.class);
165     if (rspecKeyAnnotation != null) {
166       return rspecKeyAnnotation.value();
167     }
168     return key;
169   }

171   @Test
172   public void enforce_CheckList_registration() {
173     List<File> files = (List<File>) FileUtils.listFiles(new File("src/main/java/org/sonar/java/checks/"), new String[]{"java"}, false);
174     List<Class> checks = CheckList.getChecks();
175     for (File file : files) {
176       String name = file.getName();
177       if (name.endsWith("Check.java")) {
178         String className = name.substring(0, name.length() - 5);
179         try {
180           Class aClass = Class.forName("org.sonar.java.checks." + className);
181           assertThat(checks).as(className + " is not declared in CheckList").contains(aClass);
182         } catch (ClassNotFoundException e) {
183           Throwables.propagate(e);
184         }
185       }
186     }
187   }

189   @Test
190   public void rules_targeting_tests_should_have_tests_tag() throws Exception {
191     Set<Class> testChecks = new HashSet<>(CheckList.getJavaTestChecks());
192 
193     for (Class cls : CheckList.getChecks()) {
194       String key = getKey(cls, AnnotationUtils.getAnnotation(cls, Rule.class));
195       URL metadataURL = getClass().getResource("/org/sonar/l10n/java/rules/" + CheckList.REPOSITORY_KEY + "/" + key + "_java.json");
196       File metadataFile = new File(metadataURL.toURI());
197       assertThat(metadataFile).exists();
198       try (FileReader jsonReader = new FileReader(metadataFile)) {
199         DummyMetatada metatada = gson.fromJson(jsonReader, DummyMetatada.class);
200         if (testChecks.contains(cls)) {
201           assertThat(metatada.tags).as("Rule " + key + " is targeting tests sources and should contain the 'tests' tag.").contains("tests");
202         } else {
203           assertThat(metatada.tags).as("Rule " + key + " is targeting main sources and should not contain the 'tests' tag.").doesNotContain("tests");
204         }
205       }
206     }
207   }

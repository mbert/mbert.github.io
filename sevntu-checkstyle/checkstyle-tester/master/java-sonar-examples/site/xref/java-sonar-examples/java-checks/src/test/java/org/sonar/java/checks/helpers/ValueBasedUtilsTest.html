20  package org.sonar.java.checks.helpers;
21  
22  import org.junit.Test;
23  import org.sonar.java.ast.parser.JavaParser;
24  import org.sonar.java.bytecode.loader.SquidClassLoader;
25  import org.sonar.java.resolve.SemanticModel;
26  import org.sonar.plugins.java.api.tree.ClassTree;
27  import org.sonar.plugins.java.api.tree.CompilationUnitTree;
28  import org.sonar.plugins.java.api.tree.SyntaxTrivia;
29  import org.sonar.plugins.java.api.tree.Tree;
30  import org.sonar.plugins.java.api.tree.VariableTree;
31  
32  import java.io.File;
33  import java.util.Collections;
34  import java.util.List;
35  
36  import static org.assertj.core.api.Assertions.assertThat;
37  
38  public class ValueBasedUtilsTest {
39  
40    @Test
41    public void testIsValueBased() throws Exception {
42      File file = new File("src/test/files/checks/helpers/ValueBasedUtilsTest.java");
43      CompilationUnitTree tree = (CompilationUnitTree) JavaParser.createParser().parse(file);
44      SemanticModel.createFor(tree, new SquidClassLoader(Collections.emptyList()));
45  
46      List<Tree> members = ((ClassTree) tree.types().get(0)).members();
47      members.stream()
48        .forEach(member -> checkMember((VariableTree)member));
49    }
50  
51    private static void checkMember(VariableTree member) {
52      boolean expected = getCommentValue(member);
53      boolean found = ValueBasedUtils.isValueBased(member.type().symbolType());
54      assertThat(found).as("Wrong value for field '" + member.symbol().name() + "'").isEqualTo(expected);
55    }
56  
57    private static boolean getCommentValue(VariableTree member) {
58      SyntaxTrivia trivia = member.firstToken().trivias().get(0);
59      String value = trivia.comment().substring(16);
60      return Boolean.valueOf(value);
61    }

63  }

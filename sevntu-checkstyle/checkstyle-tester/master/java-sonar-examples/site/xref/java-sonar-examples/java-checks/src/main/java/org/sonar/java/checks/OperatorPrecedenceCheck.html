45  import java.util.Deque;
46  import java.util.EnumSet;

51  @Rule(key = "S864")
52  public class OperatorPrecedenceCheck extends BaseTreeVisitor implements JavaFileScanner {
53  
54    private static final Table<Tree.Kind, Tree.Kind, Boolean> OPERATORS_RELATION_TABLE;
55  
56    private static final Set<Tree.Kind> ARITHMETIC_OPERATORS = EnumSet.of(
57      Tree.Kind.MINUS,
58      Tree.Kind.REMAINDER,
59      Tree.Kind.MULTIPLY,
60      Tree.Kind.PLUS
61      );
62  
63    private static final Set<Tree.Kind> EQUALITY_RELATIONAL_OPERATORS = EnumSet.of(
64      Tree.Kind.EQUAL_TO,
65      Tree.Kind.GREATER_THAN,
66      Tree.Kind.GREATER_THAN_OR_EQUAL_TO,
67      Tree.Kind.LESS_THAN,
68      Tree.Kind.LESS_THAN_OR_EQUAL_TO,
69      Tree.Kind.NOT_EQUAL_TO
70      );
71  
72    private static final Set<Tree.Kind> SHIFT_OPERATORS = EnumSet.of(
73      Tree.Kind.LEFT_SHIFT,
74      Tree.Kind.RIGHT_SHIFT,
75      Tree.Kind.UNSIGNED_RIGHT_SHIFT
76      );
77  
78    private static final Tree.Kind[] CONDITIONAL_EXCLUSIONS = new Tree.Kind[]{
79        Tree.Kind.METHOD_INVOCATION, Tree.Kind.IDENTIFIER, Tree.Kind.MEMBER_SELECT,
80        Tree.Kind.PARENTHESIZED_EXPRESSION, Tree.Kind.TYPE_CAST, Tree.Kind.NEW_CLASS,
81        Tree.Kind.ARRAY_ACCESS_EXPRESSION, Tree.Kind.NEW_ARRAY, Tree.Kind.METHOD_REFERENCE
82    };
83  
84    static {
85      OPERATORS_RELATION_TABLE = HashBasedTable.create();
86      put(ARITHMETIC_OPERATORS, Iterables.concat(SHIFT_OPERATORS, EnumSet.of(Tree.Kind.AND, Tree.Kind.XOR, Tree.Kind.OR)));
87      put(SHIFT_OPERATORS, Iterables.concat(ARITHMETIC_OPERATORS, EnumSet.of(Tree.Kind.AND, Tree.Kind.XOR, Tree.Kind.OR)));
88      put(EnumSet.of(Tree.Kind.AND), Iterables.concat(ARITHMETIC_OPERATORS, SHIFT_OPERATORS, EnumSet.of(Tree.Kind.XOR, Tree.Kind.OR)));
89      put(EnumSet.of(Tree.Kind.XOR), Iterables.concat(ARITHMETIC_OPERATORS, SHIFT_OPERATORS, EnumSet.of(Tree.Kind.AND, Tree.Kind.OR)));
90      put(EnumSet.of(Tree.Kind.OR), Iterables.concat(ARITHMETIC_OPERATORS, SHIFT_OPERATORS, EnumSet.of(Tree.Kind.AND, Tree.Kind.XOR)));
91      put(EnumSet.of(Tree.Kind.CONDITIONAL_AND), EnumSet.of(Tree.Kind.CONDITIONAL_OR));
92      put(EnumSet.of(Tree.Kind.CONDITIONAL_OR), EnumSet.of(Tree.Kind.CONDITIONAL_AND));
93    }
94  
95    private JavaFileScannerContext context;
96    private Deque<Tree.Kind> stack = new LinkedList<>();
97    private Set<Integer> reportedLines = new HashSet<>();
98  
99    private static void put(Iterable<Tree.Kind> firstSet, Iterable<Tree.Kind> secondSet) {
100     for (Tree.Kind first : firstSet) {
101       for (Tree.Kind second : secondSet) {
102         OPERATORS_RELATION_TABLE.put(first, second, true);
103       }
104     }
105   }
106 
107   @Override
108   public void scanFile(JavaFileScannerContext context) {
109     this.context = context;
110     reportedLines.clear();
111     scan(context.getTree());
112     reportedLines.clear();
113   }
114 
115   @Override
116   public void visitAnnotation(AnnotationTree tree) {
117     stack.push(null);
118     for (ExpressionTree argument : tree.arguments()) {
119       if (argument.is(Tree.Kind.ASSIGNMENT)) {
120         scan(((AssignmentExpressionTree) argument).expression());
121       } else {
122         scan(argument);
123       }
124     }
125     stack.pop();
126   }
127 
128   @Override
129   public void visitArrayAccessExpression(ArrayAccessExpressionTree tree) {
130     scan(tree.expression());
131     stack.push(null);
132     scan(tree.dimension());
133     stack.pop();
134   }
135 
136   @Override
137   public void visitBinaryExpression(BinaryExpressionTree tree) {
138     Tree.Kind peek = stack.peek();
139     Tree.Kind kind = tree.kind();
140     if (requiresParenthesis(peek, kind)) {
141       raiseIssue(tree.operatorToken().line(), tree);
142     }
143     stack.push(kind);
144     super.visitBinaryExpression(tree);
145     stack.pop();
146   }
147 
148   private static boolean requiresParenthesis(Tree.Kind kind1, Tree.Kind kind2) {
149     return BooleanUtils.isTrue(OPERATORS_RELATION_TABLE.get(kind1, kind2));
150   }
151 
152   @Override
153   public void visitIfStatement(IfStatementTree tree) {
154     super.visitIfStatement(tree);
155     ExpressionTree condition = tree.condition();
156     if (condition.is(Tree.Kind.ASSIGNMENT) && EQUALITY_RELATIONAL_OPERATORS.contains(((AssignmentExpressionTree) condition).expression().kind())) {
157       raiseIssue(((AssignmentExpressionTree) condition).operatorToken().line(), tree);
158     }
159   }
160 
161   @Override
162   public void visitMethodInvocation(MethodInvocationTree tree) {
163     scan(tree.methodSelect());
164     scan(tree.typeArguments());
165     for (ExpressionTree argument : tree.arguments()) {
166       stack.push(null);
167       scan(argument);
168       stack.pop();
169     }
170   }
171 
172   @Override
173   public void visitNewArray(NewArrayTree tree) {
174     stack.push(null);
175     super.visitNewArray(tree);
176     stack.pop();
177   }
178 
179   @Override
180   public void visitNewClass(NewClassTree tree) {
181     stack.push(null);
182     super.visitNewClass(tree);
183     stack.pop();
184   }
185 
186   @Override
187   public void visitParenthesized(ParenthesizedTree tree) {
188     stack.push(null);
189     super.visitParenthesized(tree);
190     stack.pop();
191   }
192 
193   @Override
194   public void visitConditionalExpression(ConditionalExpressionTree tree) {
195     checkConditionalOperand(tree.trueExpression());
196     checkConditionalOperand(tree.falseExpression());
197     super.visitConditionalExpression(tree);
198   }
199 
200   private void checkConditionalOperand(ExpressionTree tree) {
201     if(!(tree.is(CONDITIONAL_EXCLUSIONS) || tree instanceof LiteralTree || tree instanceof UnaryExpressionTree)) {
202       raiseIssue(tree.firstToken().line(), tree);
203     }
204   }
205 
206   private void raiseIssue(int line, Tree tree) {
207     if (reportedLines.add(line)) {
208       context.reportIssue(this, tree, "Add parentheses to make the operator precedence explicit.");
209     }
210   }

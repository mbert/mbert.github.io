59    private static final int LOGS_NUMBER_LINES = 200;
60    private static final Logger LOG = LoggerFactory.getLogger(SemanticTest.class);

62    @ClassRule
63    public static final Orchestrator ORCHESTRATOR;
64  
65    static {
66      OrchestratorBuilder orchestratorBuilder = Orchestrator.builderEnv()
67        .setSonarVersion(System.getProperty("sonar.runtimeVersion", "LATEST_RELEASE[6.7]"))
68        .addPlugin(MavenLocation.of("org.sonarsource.sonar-lits-plugin","sonar-lits-plugin", "0.8.0.1209"))
69        .addPlugin(FileLocation.byWildcardMavenFilename(new File("../../../sonar-java-plugin/target"), "sonar-java-plugin-*.jar"))
70        .addPlugin(FileLocation.of(debuggingPlugin()))
71        .restoreProfileAtStartup(FileLocation.of("src/test/resources/profile-java-semantic.xml"));
72      ORCHESTRATOR = orchestratorBuilder.build();
73    }
74  
75    @ClassRule
76    public static TemporaryFolder TMP_DUMP_OLD_FOLDER = new TemporaryFolder();
77    private static Path effectiveDumpOldFolder;
78  
79    private static final Path RESULTS_FOLDER = Paths.get("src/test/resources/");
80  
81    /**
82     * Contains the key of the rules from debugging plugin used for semantic tests
83     */
84    private static final List<String> RULE_KEYS = Arrays.asList(
85      "UnknownConstructorCall",
86      "UnknownMethodInvocations");
87  
88    @BeforeClass
89    public static void prepare_analysis() {
90      effectiveDumpOldFolder = TMP_DUMP_OLD_FOLDER.getRoot().toPath().toAbsolutePath();
91      Try.of(() -> Files.list(RESULTS_FOLDER)).getOrElseThrow(Throwables::propagate)
92        .filter(p -> p.toFile().isDirectory())
93        .forEach(srcProjectDir -> copyDumpSubset(srcProjectDir, effectiveDumpOldFolder.resolve(srcProjectDir.getFileName())));
94    }
95  
96    private static void copyDumpSubset(Path srcProjectDir, Path dstProjectDir) {
97      Try.of(() -> Files.createDirectory(dstProjectDir)).getOrElseThrow(Throwables::propagate);
98  
99      RULE_KEYS.stream()
100       .map(ruleKey -> srcProjectDir.resolve("debug-" + ruleKey + ".json"))
101       .filter(p -> p.toFile().exists())
102       .forEach(srcJsonFile -> Try.of(() -> Files.copy(srcJsonFile, dstProjectDir.resolve(srcJsonFile.getFileName()), StandardCopyOption.REPLACE_EXISTING))
103         .getOrElseThrow(Throwables::propagate));
104   }

122   private static void test_project(String projectKey, String projectName) throws IOException {
123     test_project(projectKey, null, projectName);
124   }
125 
126   private static void test_project(String projectKey, @Nullable String path, String projectName) throws IOException {
127     prepareProject(projectKey, projectName);
128     String pomLocation = "../../sources/" + (path != null ? path + "/" : "") + projectName + "/pom.xml";
129     MavenBuild mavenBuild = MavenBuild.create().setPom(FileLocation.of(pomLocation).getFile()).setCleanPackageSonarGoals().addArgument("-DskipTests");
130     executeBuildWithCommonProperties(mavenBuild, projectName);
131   }
132 
133   private static void prepareProject(String projectKey, String projectName) {
134     ORCHESTRATOR.getServer().provisionProject(projectKey, projectName);
135     ORCHESTRATOR.getServer().associateProjectToQualityProfile(projectKey, "java", "java-semantic");
136   }
137 
138   private static void executeBuildWithCommonProperties(Build<?> build, String projectName) throws IOException {
139     build.setProperty("sonar.cpd.exclusions", "**/*")
140       .setProperty("sonar.skipPackageDesign", "true")
141       .setProperty("dump.old", effectiveDumpOldFolder.resolve(projectName).toString())
142       .setProperty("dump.new", FileLocation.of("target/actual/" + projectName).getFile().getAbsolutePath())
143       .setProperty("lits.differences", litsDifferencesPath(projectName));
144     BuildResult buildResult = ORCHESTRATOR.executeBuild(build);
145     if (buildResult.isSuccess()) {
146       assertNoDifferences(projectName);
147     } else {
148       dumpServerLogs();
149     }
150   }
151 
152   private static void dumpServerLogs() throws IOException {
153     Server server = ORCHESTRATOR.getServer();
154     LOG.error("::::::::::::::::::::::::::::::::::: DUMPING SERVER LOGS :::::::::::::::::::::::::::::::::::");
155     dumpServerLogLastLines(server.getAppLogs());
156     dumpServerLogLastLines(server.getCeLogs());
157     dumpServerLogLastLines(server.getEsLogs());
158     dumpServerLogLastLines(server.getWebLogs());
159   }
160 
161   private static void dumpServerLogLastLines(File logFile) throws IOException {
162     List<String> logs = Files.readAllLines(logFile.toPath());
163     int nbLines = logs.size();
164     if (nbLines > LOGS_NUMBER_LINES) {
165       logs = logs.subList(nbLines - LOGS_NUMBER_LINES, nbLines);
166     }
167     LOG.error("=================================== START " + logFile.getName() + " ===================================");
168     LOG.error(System.lineSeparator() + logs.stream().collect(Collectors.joining(System.lineSeparator())));
169     LOG.error("===================================== END " + logFile.getName() + " ===================================");
170   }
171 
172   private static String litsDifferencesPath(String projectName) {
173     return FileLocation.of("target/" + projectName + "_differences").getFile().getAbsolutePath();
174   }
175 
176   private static void assertNoDifferences(String projectName) throws IOException {
177     String differences = new String(Files.readAllBytes(Paths.get(litsDifferencesPath(projectName))), StandardCharsets.UTF_8);
178     Assertions.assertThat(differences).isEmpty();
179   }

181   private static File debuggingPlugin() {
182     return Iterables.getOnlyElement(Arrays.asList(new File("../java-debugging-plugin/target/").listFiles(new FilenameFilter() {
183       @Override
184       public boolean accept(File dir, String name) {
185         return name.endsWith(".jar") && !name.endsWith("-sources.jar");
186       }
187     })));
188   }

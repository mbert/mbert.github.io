44  import static org.sonar.java.se.ProgramState.isField;

46  /**
47   * Current implementation raises the issue only for the fields used in one method
48   */
49  @Rule(key = "S1450")
50  public class PrivateFieldUsedLocallyCheck extends IssuableSubscriptionVisitor {
51  
52    private static final String MESSAGE = "Remove the \"%s\" field and declare it as a local variable in the relevant methods.";
53  
54    @Override
55    public List<Kind> nodesToVisit() {
56      return ImmutableList.of(Kind.CLASS);
57    }
58  
59    @Override
60    public void visitNode(Tree tree) {
61      TypeSymbol classSymbol = ((ClassTree) tree).symbol();
62      Set<Symbol> fieldsReadOnAnotherInstance = FieldsReadOnAnotherInstanceVisitor.getFrom(tree);
63  
64      classSymbol.memberSymbols().stream()
65        .filter(PrivateFieldUsedLocallyCheck::isPrivateField)
66        .filter(s -> !(s.isFinal() && s.isStatic()))
67        .filter(s -> !hasAnnotation(s))
68        .filter(s -> !s.usages().isEmpty())
69        .filter(s -> !fieldsReadOnAnotherInstance.contains(s))
70        .forEach(s -> checkPrivateField(s, classSymbol));
71    }
72  
73    private static boolean hasAnnotation(Symbol s) {
74      return !s.metadata().annotations().isEmpty();
75    }
76  
77    private void checkPrivateField(Symbol privateFieldSymbol, TypeSymbol classSymbol) {
78      MethodTree methodWhereUsed = usedInOneMethodOnly(privateFieldSymbol, classSymbol);
79  
80      if (methodWhereUsed != null && !isLiveInMethodEntry(privateFieldSymbol, methodWhereUsed)) {
81        IdentifierTree declarationIdentifier = ((VariableTree) privateFieldSymbol.declaration()).simpleName();
82        String message = String.format(MESSAGE, privateFieldSymbol.name());
83        reportIssue(declarationIdentifier, message);
84      }
85    }
86  
87    private static boolean isLiveInMethodEntry(Symbol privateFieldSymbol, MethodTree methodTree) {
88      CFG cfg = CFG.build(methodTree);
89      LiveVariables liveVariables = LiveVariables.analyzeWithFields(cfg);
90      return liveVariables.getIn(cfg.entryBlock()).contains(privateFieldSymbol);
91    }
92  
93    private static boolean isPrivateField(Symbol memberSymbol) {
94      return memberSymbol.isPrivate() && memberSymbol.isVariableSymbol();
95    }
96  
97    /**
98     * If private field used in several methods then returns null, otherwise returns the method where it's used
99     */
100   @CheckForNull
101   private static MethodTree usedInOneMethodOnly(Symbol privateFieldSymbol, TypeSymbol classSymbol) {
102     MethodTree method = null;
103 
104     for (IdentifierTree usageIdentifier : privateFieldSymbol.usages()) {
105       Tree containingClassOrMethod = containingClassOrMethod(usageIdentifier);
106 
107       if (noContainerOrClassContainer(containingClassOrMethod)
108         || !((MethodTree) containingClassOrMethod).symbol().owner().equals(classSymbol)
109         || (method != null && !method.equals(containingClassOrMethod))) {
110         return null;
111 
112       } else {
113         method = (MethodTree)containingClassOrMethod;
114 
115       }
116     }
117 
118     return method;
119   }
120 
121   private static boolean noContainerOrClassContainer(@Nullable Tree containingClassOrMethod) {
122     return containingClassOrMethod == null || containingClassOrMethod.is(Kind.CLASS, Kind.INTERFACE, Kind.ENUM, Kind.ANNOTATION_TYPE);
123   }
124 
125   private static Tree containingClassOrMethod(IdentifierTree usageIdentifier) {
126     Tree parent = usageIdentifier;
127     do {
128       parent = parent.parent();
129     } while (!parent.is(Kind.METHOD, Kind.CLASS, Kind.INTERFACE, Kind.ENUM, Kind.ANNOTATION_TYPE));
130 
131     return parent;
132   }
133 
134   private static class FieldsReadOnAnotherInstanceVisitor extends BaseTreeVisitor {
135 
136     private Set<Symbol> fieldsReadOnAnotherInstance = new HashSet<>();
137 
138     static Set<Symbol> getFrom(Tree classTree) {
139       FieldsReadOnAnotherInstanceVisitor fieldsReadOnAnotherInstanceVisitor = new FieldsReadOnAnotherInstanceVisitor();
140       fieldsReadOnAnotherInstanceVisitor.scan(classTree);
141       return fieldsReadOnAnotherInstanceVisitor.fieldsReadOnAnotherInstance;
142     }
143 
144     @Override
145     public void visitMemberSelectExpression(MemberSelectExpressionTree tree) {
146       Symbol symbol = tree.identifier().symbol();
147       if (isField(symbol) && !symbol.isStatic()) {
148 
149         if (tree.expression().is(Kind.IDENTIFIER)) {
150           String objectName = ((IdentifierTree) tree.expression()).name();
151 
152           if (!"this".equals(objectName)) {
153             fieldsReadOnAnotherInstance.add(symbol);
154           }
155 
156         } else {
157           fieldsReadOnAnotherInstance.add(symbol);
158         }
159       }
160 
161       super.visitMemberSelectExpression(tree);
162     }
163   }

86    @BeforeClass
87    public static void prepare_quality_profiles() {
88      ImmutableMap<String, ImmutableMap<String, String>> rulesParameters = ImmutableMap.<String, ImmutableMap<String, String>>builder()
89        .put(
90          "IndentationCheck",
91          ImmutableMap.of("indentationLevel", "4"))
92        .put(
93          "S1451",
94          ImmutableMap.of(
95            "headerFormat",
96            "\n/*\n" +
97              " * Copyright (c) 1998, 2006, Oracle and/or its affiliates. All rights reserved.\n" +
98            " * ORACLE PROPRIETARY/CONFIDENTIAL. Use is subject to license terms."))
99        .build();
100     ImmutableSet<String> disabledRules = ImmutableSet.of(
101       "CallToDeprecatedMethod",
102       "CycleBetweenPackages",
103       // disable because it generates too many issues, performance reasons
104       "LeftCurlyBraceStartLineCheck"
105       );
106     Set<String> activatedRuleKeys = new HashSet<>();
107     ProfileGenerator.generate(orchestrator, rulesParameters, disabledRules, SUBSET_OF_ENABLED_RULES, activatedRuleKeys);
108     instantiateTemplateRule("S2253", "stringToCharArray", "className=\"java.lang.String\";methodName=\"toCharArray\"", activatedRuleKeys);
109     instantiateTemplateRule("S4011", "longDate", "className=\"java.util.Date\";argumentTypes=\"long\"", activatedRuleKeys);
110     instantiateTemplateRule("S124", "commentRegexTest", "regularExpression=\"(?i).*TODO\\(user\\).*\";message=\"bad user\"", activatedRuleKeys);
111     instantiateTemplateRule("S3417", "doNotUseCommonsCollections", "dependencyName=\"commons-collections:*\";", activatedRuleKeys);
112     instantiateTemplateRule("S3417", "doNotUseJunitBefore4", "dependencyName=\"junit:junit\";version=\"*-3.9.9\"", activatedRuleKeys);
113     instantiateTemplateRule("S3546", "InstancesOfNewControllerClosedWithDone",
114       "factoryMethod=\"org.sonar.api.server.ws.WebService$Context#createController\";closingMethod=\"org.sonar.api.server.ws.WebService$NewController#done\"", activatedRuleKeys);
115     instantiateTemplateRule("S3546", "JsonWriterNotClosed",
116       "factoryMethod=\"org.sonar.api.server.ws.Response#newJsonWriter\";closingMethod=\"org.sonar.api.utils.text.JsonWriter#close\"", activatedRuleKeys);
117 
118     SUBSET_OF_ENABLED_RULES.stream()
119       .filter(ruleKey -> !activatedRuleKeys.contains(ruleKey))
120       .forEach(ruleKey -> Fail.fail("Specified rule does not exist: " + ruleKey));
121 
122     prepareDumpOldFolder();
123   }

125   private static void prepareDumpOldFolder() {
126     Path allRulesFolder = Paths.get("src/test/resources");
127     if (SUBSET_OF_ENABLED_RULES.isEmpty()) {
128       effectiveDumpOldFolder = allRulesFolder.toAbsolutePath();
129     } else {
130       effectiveDumpOldFolder = TMP_DUMP_OLD_FOLDER.getRoot().toPath().toAbsolutePath();
131       Try.of(() -> Files.list(allRulesFolder)).orElseThrow(Throwables::propagate)
132         .filter(p -> p.toFile().isDirectory())
133         .forEach(srcProjectDir -> copyDumpSubset(srcProjectDir, effectiveDumpOldFolder.resolve(srcProjectDir.getFileName())));
134     }
135   }
136 
137   private static void copyDumpSubset(Path srcProjectDir, Path dstProjectDir) {
138     Try.of(() -> Files.createDirectory(dstProjectDir)).orElseThrow(Throwables::propagate);
139     SUBSET_OF_ENABLED_RULES.stream()
140       .map(ruleKey -> srcProjectDir.resolve("squid-" + ruleKey + ".json"))
141       .filter(p -> p.toFile().exists())
142       .forEach(srcJsonFile -> Try.of(() -> Files.copy(srcJsonFile, dstProjectDir.resolve(srcJsonFile.getFileName()), StandardCopyOption.REPLACE_EXISTING))
143         .orElseThrow(Throwables::propagate));
144   }

230   private static void executeBuildWithCommonProperties(Build<?> build, String projectName, boolean buildQuietly) throws IOException {
231     build.setProperty("sonar.cpd.skip", "true")
232       .setProperty("sonar.import_unknown_files", "true")
233       .setProperty("sonar.skipPackageDesign", "true")
234       .setProperty("sonar.analysis.mode", "preview")
235       .setProperty("sonar.issuesReport.html.enable", "true")
236       .setProperty("sonar.issuesReport.html.location", htmlReportPath(projectName))
237       .setProperty("dump.old", effectiveDumpOldFolder.resolve(projectName).toString())
238       .setProperty("dump.new", FileLocation.of("target/actual/" + projectName).getFile().getAbsolutePath())
239       .setProperty("lits.differences", litsDifferencesPath(projectName))
240       .setProperty("sonar.java.xfile", "true")
241       .setProperty("sonar.java.failOnException", "true");
242     BuildResult buildResult;
243     if (buildQuietly) {
244       // if build fail, ruling job is not violently interrupted, allowing time to dump SQ logs
245       buildResult = orchestrator.executeBuildQuietly(build);
246     } else {
247       buildResult = orchestrator.executeBuild(build);
248     }
249     if (buildResult.isSuccess()) {
250       assertNoDifferences(projectName);
251     } else {
252       dumpServerLogs();
253       Fail.fail("Build failure for project: " + projectName);
254     }
255   }

257   private static void dumpServerLogs() throws IOException {
258     Server server = orchestrator.getServer();
259     LOG.error("::::::::::::::::::::::::::::::::::: DUMPING SERVER LOGS :::::::::::::::::::::::::::::::::::");
260     dumpServerLogLastLines(server.getAppLogs());
261     dumpServerLogLastLines(server.getCeLogs());
262     dumpServerLogLastLines(server.getEsLogs());
263     dumpServerLogLastLines(server.getWebLogs());
264   }

266   private static void dumpServerLogLastLines(File logFile) throws IOException {
267     if (!logFile.exists()) {
268       return;
269     }
270     List<String> logs = Files.readAllLines(logFile.toPath());
271     int nbLines = logs.size();
272     if (nbLines > LOGS_NUMBER_LINES) {
273       logs = logs.subList(nbLines - LOGS_NUMBER_LINES, nbLines);
274     }
275     LOG.error("=================================== START " + logFile.getName() + " ===================================");
276     LOG.error(System.lineSeparator() + logs.stream().collect(Collectors.joining(System.lineSeparator())));
277     LOG.error("===================================== END " + logFile.getName() + " ===================================");
278   }

289     String differences = new String(Files.readAllBytes(Paths.get(litsDifferencesPath(projectName))), StandardCharsets.UTF_8);

293   private static void instantiateTemplateRule(String ruleTemplateKey, String instantiationKey, String params, Set<String> activatedRuleKeys) {
294     if (!SUBSET_OF_ENABLED_RULES.isEmpty() && !SUBSET_OF_ENABLED_RULES.contains(instantiationKey)) {
295       return;
296     }
297     activatedRuleKeys.add(instantiationKey);
298     SonarClient sonarClient = orchestrator.getServer().adminWsClient();
299     sonarClient.post("/api/rules/create", ImmutableMap.<String, Object>builder()
300       .put("name", instantiationKey)
301       .put("markdown_description", instantiationKey)
302       .put("severity", "INFO")
303       .put("status", "READY")
304       .put("template_key", "squid:" + ruleTemplateKey)
305       .put("custom_key", instantiationKey)
306       .put("prevent_reactivation", "true")
307       .put("params", "name=\"" + instantiationKey + "\";key=\"" + instantiationKey + "\";markdown_description=\"" + instantiationKey + "\";" + params)
308       .build());
309     String post = sonarClient.get("api/qualityprofiles/search");
310     String profileKey = null;
311     Map map = GSON.fromJson(post, Map.class);
312     for (Map qp : ((List<Map>) map.get("profiles"))) {
313       if ("rules".equals(qp.get("name"))) {
314         profileKey = (String) qp.get("key");
315         break;
316       }
317     }
318     if (StringUtils.isEmpty(profileKey)) {
319       LOG.error("Could not retrieve profile key : Template rule " + ruleTemplateKey + " has not been activated");
320     } else {
321       String activateRuleResponse = sonarClient.post("api/qualityprofiles/activate_rule", ImmutableMap.of(
322         "profile_key", profileKey,
323         "rule_key", "squid:" + instantiationKey,
324         "severity", "INFO",
325         "params", ""));
326       LOG.warn(activateRuleResponse);
327     }
328   }

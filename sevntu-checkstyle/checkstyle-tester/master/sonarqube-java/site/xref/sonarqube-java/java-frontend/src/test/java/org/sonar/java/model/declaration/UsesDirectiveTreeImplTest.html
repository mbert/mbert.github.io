44  public class UsesDirectiveTreeImplTest {
45    
46    private final ActionParser<Tree> p = JavaParser.createParser();
47  
48    private CompilationUnitTree createTree(String... lines) {
49      CompilationUnitTree compilationUnitTree = (CompilationUnitTree) p.parse(Arrays.stream(lines).collect(Collectors.joining("\n")));
50      SemanticModel.createFor(compilationUnitTree, new SquidClassLoader(Collections.emptyList()));
51      return compilationUnitTree;
52    }
53  
54    private UsesDirectiveTree moduleDirective(String exportsDirective) {
55      CompilationUnitTree compilationUnitTree = createTree("module org.foo {\n  " + exportsDirective + "\n}");
56      SemanticModel.createFor(compilationUnitTree, new SquidClassLoader(Collections.emptyList()));
57      return (UsesDirectiveTree) compilationUnitTree.moduleDeclaration().moduleDirectives().get(0);
58    }
59  
60    @Test
61    public void simple_uses() {
62      UsesDirectiveTree exports = moduleDirective("uses foo.MyInterface;");
63  
64      assertThat(exports.kind()).isEqualTo(Tree.Kind.USES_DIRECTIVE);
65      assertThat(exports.directiveKeyword().text()).isEqualTo("uses");
66      TypeTree typeName = exports.typeName();
67      assertThat(typeName.is(Tree.Kind.MEMBER_SELECT)).isTrue();
68      assertThat(((MemberSelectExpressionTree) typeName).identifier().name()).isEqualTo("MyInterface");
69      assertThat(exports.semicolonToken().text()).isEqualTo(";");
70    }
71  
72    @Test
73    public void test_BaseTreeVisitor() {
74      CompilationUnitTree cut = createTree(
75        "open module com.greetings {",
76        "  uses org.bar.MyInterface;",
77        "  uses org.gul.MyOtherInterface;",
78        "}");
79      UsesDirectiveVisitor moduleDeclarationVisitor = new UsesDirectiveVisitor();
80      cut.accept(moduleDeclarationVisitor);
81      assertThat(moduleDeclarationVisitor.visited).isTrue();
82      assertThat(moduleDeclarationVisitor.directives).hasSize(2);
83      assertThat(moduleDeclarationVisitor.identifiers).containsExactly("com", "greetings", "MyInterface", "MyOtherInterface");
84    }
85  
86    private static class UsesDirectiveVisitor extends BaseTreeVisitor {
87  
88      boolean visited = false;
89      List<UsesDirectiveTree> directives = new ArrayList<>();
90      List<String> identifiers = new ArrayList<>();
91  
92      @Override
93      public void visitUsesDirective(UsesDirectiveTree tree) {
94        visited = true;
95        directives.add(tree);
96        super.visitUsesDirective(tree);
97      }
98  
99  
100     @Override
101     public void visitMemberSelectExpression(MemberSelectExpressionTree tree) {
102       // skip all the rest
103       scan(tree.identifier());
104     }
105 
106     @Override
107     public void visitIdentifier(IdentifierTree tree) {
108       identifiers.add(tree.name());
109     }
110   }

112 }

91    public VisitorsBridge(Iterable visitors, List<File> projectClasspath, @Nullable SonarComponents sonarComponents, SymbolicExecutionMode symbolicExecutionMode) {
92      this.allScanners = new ArrayList<>();
93      for (Object visitor : visitors) {
94        if (visitor instanceof JavaFileScanner) {
95          allScanners.add((JavaFileScanner) visitor);
96        }
97      }
98      this.executableScanners = allScanners.stream().filter(isIssuableSubscriptionVisitor.negate()).collect(Collectors.toList());
99      this.scannerRunner = new ScannerRunner(allScanners);
100     this.sonarComponents = sonarComponents;
101     this.classLoader = ClassLoaderBuilder.create(projectClasspath);
102     this.symbolicExecutionEnabled = symbolicExecutionMode.isEnabled();
103     this.behaviorCache = new BehaviorCache(classLoader, symbolicExecutionMode.isCrossFileEnabled());
104   }

106   public void setJavaVersion(JavaVersion javaVersion) {
107     this.javaVersion = javaVersion;
108     List<JavaFileScanner> scannersForJavaVersion = executableScanners(allScanners, javaVersion);
109     this.executableScanners = scannersForJavaVersion.stream().filter(isIssuableSubscriptionVisitor.negate()).collect(Collectors.toList());
110     this.scannerRunner = new ScannerRunner(scannersForJavaVersion);
111   }

113   public void visitFile(@Nullable Tree parsedTree) {
114     semanticModel = null;
115     CompilationUnitTree tree = new JavaTree.CompilationUnitTreeImpl(null, new ArrayList<>(), new ArrayList<>(), null, null);
116     boolean fileParsed = parsedTree != null;
117     if (fileParsed && parsedTree.is(Tree.Kind.COMPILATION_UNIT)) {
118       tree = (CompilationUnitTree) parsedTree;
119       if (isNotJavaLangOrSerializable(PackageUtils.packageName(tree.packageDeclaration(), "/"))) {
120         try {
121           semanticModel = SemanticModel.createFor(tree, classLoader);
122         } catch (Exception e) {
123           LOG.error("Unable to create symbol table for : " + currentFile.getAbsolutePath(), e);
124           addAnalysisError(e, currentFile.getPath(), AnalysisError.Kind.SEMANTIC_ERROR);
125           sonarComponents.reportAnalysisError(currentFile, e.getMessage());
126           return;
127         }
128         createSonarSymbolTable(tree);
129       } else {
130         SemanticModel.handleMissingTypes(tree);
131       }
132     }
133     JavaFileScannerContext javaFileScannerContext = createScannerContext(tree, semanticModel, sonarComponents, fileParsed);
134     // Symbolic execution checks
135     if (symbolicExecutionEnabled && isNotJavaLangOrSerializable(PackageUtils.packageName(tree.packageDeclaration(), "/"))) {
136       runScanner(javaFileScannerContext, new SymbolicExecutionVisitor(executableScanners, behaviorCache), AnalysisError.Kind.SE_ERROR);
137       behaviorCache.cleanup();
138     }
139     executableScanners.forEach(scanner -> runScanner(javaFileScannerContext, scanner, AnalysisError.Kind.CHECK_ERROR));
140     scannerRunner.run(javaFileScannerContext);
141     if (semanticModel != null) {
142       classesNotFound.addAll(semanticModel.classesNotFound());
143     }
144   }
145 
146   private void runScanner(JavaFileScannerContext javaFileScannerContext, JavaFileScanner scanner, AnalysisError.Kind kind) {
147     try {
148       scanner.scanFile(javaFileScannerContext);
149     } catch (IllegalRuleParameterException e) {
150       // bad configuration of a rule parameter, we want to fail analysis fast.
151       throw e;
152     } catch (Exception e) {
153       if (sonarComponents != null && sonarComponents.shouldFailAnalysisOnException()) {
154         throw e;
155       }
156       Throwable rootCause = Throwables.getRootCause(e);
157       if (rootCause instanceof InterruptedIOException || rootCause instanceof InterruptedException) {
158         throw e;
159       }
160       Rule annotation = AnnotationUtils.getAnnotation(scanner.getClass(), Rule.class);
161       String key = "";
162       if (annotation != null) {
163         key = annotation.key();
164       }
165       LOG.error(
166         String.format("Unable to run check %s - %s on file %s, To help improve SonarJava, please report this problem to SonarSource : see https://www.sonarqube.org/community/",
167           scanner.getClass(), key, currentFile.getPath()), e);
168       addAnalysisError(e, currentFile.getPath(), kind);
169     }
170   }
171 
172   private void addAnalysisError(Exception e, String path, AnalysisError.Kind checkError) {
173     if (sonarComponents != null) {
174       sonarComponents.addAnalysisError(new AnalysisError(e, path, checkError));
175     }
176   }

178   private static List<JavaFileScanner> executableScanners(List<JavaFileScanner> scanners, JavaVersion javaVersion) {
179     ImmutableList.Builder<JavaFileScanner> results = ImmutableList.builder();
180     for (JavaFileScanner scanner : scanners) {
181       if (!(scanner instanceof JavaVersionAwareVisitor) || ((JavaVersionAwareVisitor) scanner).isCompatibleWithJavaVersion(javaVersion)) {
182         results.add(scanner);
183       }
184     }
185     return results.build();
186   }

199   private boolean isNotJavaLangOrSerializable(String packageName) {
200     String name = currentFile.getName();
201     return !(inJavaLang(packageName) || isAnnotation(packageName, name) || isSerializable(packageName, name));
202   }
203 
204   private static boolean isSerializable(String packageName, String name) {
205     return "java/io".equals(packageName) && "Serializable.java".equals(name);
206   }
207 
208   private static boolean isAnnotation(String packageName, String name) {
209     return "java/lang/annotation".equals(packageName) && "Annotation.java".equals(name);
210   }
211 
212   private static boolean inJavaLang(String packageName) {
213     return "java/lang".equals(packageName);
214   }

216   private void createSonarSymbolTable(CompilationUnitTree tree) {
217     if (sonarComponents != null && !sonarComponents.isSonarLintContext()) {
218       SonarSymbolTableVisitor symVisitor = new SonarSymbolTableVisitor(sonarComponents.symbolizableFor(currentFile), semanticModel);
219       symVisitor.visitCompilationUnit(tree);
220     }
221   }

223   public void processRecognitionException(RecognitionException e, File file) {
224     addAnalysisError(e, file.getPath(), AnalysisError.Kind.PARSE_ERROR);
225     if(sonarComponents == null || !sonarComponents.reportAnalysisError(e, file)) {
226       this.visitFile(null);
227       executableScanners.stream()
228         .filter(scanner -> scanner instanceof ExceptionHandler)
229         .forEach(scanner -> ((ExceptionHandler) scanner).processRecognitionException(e));
230     }

238   public void endOfAnalysis() {
239     if(!classesNotFound.isEmpty()) {
240       String message = "";
241       if(classesNotFound.size() > 50) {
242         message = ", ...";
243       }
244       LOG.warn("Classes not found during the analysis : [{}{}]", classesNotFound.stream().limit(50).collect(Collectors.joining(", ")), message);
245     }
246     allScanners.stream()
247       .filter(s -> s instanceof EndOfAnalysisCheck)
248       .map(EndOfAnalysisCheck.class::cast)
249       .forEach(EndOfAnalysisCheck::endOfAnalysis);
250     classLoader.close();
251   }

253   private static class ScannerRunner {
254     private EnumMap<Tree.Kind, List<SubscriptionVisitor>> checks;
255     private List<SubscriptionVisitor> subscriptionVisitors;
256 
257     ScannerRunner(List<JavaFileScanner> executableScanners) {
258       checks = new EnumMap<>(Tree.Kind.class);
259       subscriptionVisitors = executableScanners.stream()
260         .filter(isIssuableSubscriptionVisitor)
261         .map(s -> (SubscriptionVisitor) s)
262         .collect(Collectors.toList());
263       subscriptionVisitors.forEach(s -> s.nodesToVisit().forEach(k -> checks.computeIfAbsent(k, key -> new ArrayList<>()).add(s))
264       );
265     }
266 
267     public void run(JavaFileScannerContext javaFileScannerContext) {
268       subscriptionVisitors.forEach(s -> s.setContext(javaFileScannerContext));
269       visit(javaFileScannerContext.getTree());
270       subscriptionVisitors.forEach(s -> s.leaveFile(javaFileScannerContext));
271     }
272 
273     private void visitChildren(Tree tree) {
274       JavaTree javaTree = (JavaTree) tree;
275       if (!javaTree.isLeaf()) {
276         for (Tree next : javaTree.getChildren()) {
277           if (next != null) {
278             visit(next);
279           }
280         }
281       }
282     }
283 
284     private void visit(Tree tree) {
285       Consumer<SubscriptionVisitor> callback;
286       boolean isToken = tree.kind() == Tree.Kind.TOKEN;
287       if (isToken) {
288         callback = s -> {
289           SyntaxToken syntaxToken = (SyntaxToken) tree;
290           s.visitToken(syntaxToken);
291         };
292       } else {
293         callback = s -> s.visitNode(tree);
294       }
295       List<SubscriptionVisitor> subscribed = checks.getOrDefault(tree.kind(), Collections.emptyList());
296       subscribed.forEach(callback);
297       if (isToken) {
298         checks.getOrDefault(Tree.Kind.TRIVIA, Collections.emptyList()).forEach(s -> ((SyntaxToken) tree).trivias().forEach(s::visitTrivia));
299       } else {
300         visitChildren(tree);
301       }
302       if(!isToken) {
303         subscribed.forEach(s -> s.leaveNode(tree));
304       }
305     }
306   }

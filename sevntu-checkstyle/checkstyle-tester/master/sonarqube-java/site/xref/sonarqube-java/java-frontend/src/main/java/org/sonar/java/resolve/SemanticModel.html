44  public class SemanticModel {
45  
46    private final Map<Tree, Symbol> symbolsTree = new HashMap<>();
47  
48    private final Map<Symbol, Resolve.Env> symbolEnvs = Maps.newHashMap();
49    private final BiMap<Tree, Resolve.Env> envs = HashBiMap.create();
50    private final BytecodeCompleter bytecodeCompleter;
51  
52    @VisibleForTesting
53    SemanticModel(BytecodeCompleter bytecodeCompleter) {
54      this.bytecodeCompleter = bytecodeCompleter;
55    }
56  
57    public static SemanticModel createFor(CompilationUnitTree tree, SquidClassLoader classLoader) {
58      ParametrizedTypeCache parametrizedTypeCache = new ParametrizedTypeCache();
59      BytecodeCompleter bytecodeCompleter = new BytecodeCompleter(classLoader, parametrizedTypeCache);
60      Symbols symbols = new Symbols(bytecodeCompleter);
61      SemanticModel semanticModel = new SemanticModel(bytecodeCompleter);
62      try {
63        Resolve resolve = new Resolve(symbols, bytecodeCompleter, parametrizedTypeCache);
64        TypeAndReferenceSolver typeAndReferenceSolver = new TypeAndReferenceSolver(semanticModel, symbols, resolve, parametrizedTypeCache);
65        new FirstPass(semanticModel, symbols, resolve, parametrizedTypeCache, typeAndReferenceSolver).visitCompilationUnit(tree);
66        typeAndReferenceSolver.visitCompilationUnit(tree);
67        new LabelsVisitor(semanticModel).visitCompilationUnit(tree);
68      } finally {
69        handleMissingTypes(tree);
70      }
71      return semanticModel;
72    }
73  
74    /**
75     * Handles missing types in Syntax Tree to prevent NPE in subsequent steps of analysis.
76     */
77    public static void handleMissingTypes(Tree tree) {
78      // (Godin): Another and probably better (safer) way to do the same - is to assign default value during creation of nodes, so that to guarantee that this step won't be skipped.
79      new MissingTypeHandler().handleMissingTypes(tree);
80    }
81  
82  
83    private static class MissingTypeHandler {
84      private Deque<JavaTree> parentList = new LinkedList<>();
85  
86      void handleMissingTypes(Tree tree) {
87        parentList.push((JavaTree) tree);
88        while (!parentList.isEmpty()) {
89          JavaTree parent = parentList.pop();
90          if (parent instanceof AbstractTypedTree) {
91            ((AbstractTypedTree) parent).completeMissingType();
92          }
93          if (!parent.isLeaf()) {
94            for (Tree nextTree : parent.getChildren()) {
95              JavaTree next = (JavaTree) nextTree;
96              if (next != null) {
97                next.setParent(parent);
98                parentList.push(next);
99              }
100           }
101         }
102       }
103     }
104   }

106   public void saveEnv(Symbol symbol, Resolve.Env env) {
107     symbolEnvs.put(symbol, env);
108   }
109 
110   public Resolve.Env getEnv(Symbol symbol) {
111     return symbolEnvs.get(symbol);
112   }
113 
114   public void associateEnv(Tree tree, Resolve.Env env) {
115     envs.put(tree, env);
116   }
117 
118   public Tree getTree(Resolve.Env env) {
119     return envs.inverse().get(env);
120   }
121 
122   public Resolve.Env getEnv(Tree tree) {
123     Resolve.Env result = null;
124     Tree node = tree;
125     while (result == null && node != null) {
126       result = envs.get(node);
127       node = node.parent();
128     }
129     return result;
130   }
131 
132   public Symbol getEnclosingClass(Tree tree) {
133     return getEnv(tree).enclosingClass;
134   }
135 
136   public Type getClassType(String fullyQualifiedName) {
137     return bytecodeCompleter.loadClass(fullyQualifiedName).type();
138   }
139 
140   public void associateSymbol(Tree tree, Symbol symbol) {
141     Preconditions.checkNotNull(symbol);
142     symbolsTree.put(tree, symbol);
143   }
144 
145   @Nullable
146   public Symbol getSymbol(Tree tree) {
147     return symbolsTree.get(tree);
148   }
149 
150   @VisibleForTesting
151   Map<Tree, Symbol> getSymbolsTree() {
152     return Collections.unmodifiableMap(symbolsTree);
153   }
154 
155   public Set<String> classesNotFound() {
156     return bytecodeCompleter.classesNotFound();
157   }
158 
159   @CheckForNull
160   public Object constantValue(Symbol owner, String constantName) {
161     return bytecodeCompleter.constantValue(owner, constantName);
162   }

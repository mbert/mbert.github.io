57    @Test
58    public void buckets() {
59      Object k1 = new Key(42, "k1");
60      Object k2 = new Key(42, "k2");
61      Object k3 = new Key(42, "k3");
62      AVLTree<Object, Object> t = AVLTree.create()
63        .put(k1, "v1")
64        .put(k2, "v2");
65  
66      assertThat(t.toString())
67        .as("should create bucket")
68        .isEqualTo(" k2->v2 k1->v1");
69  
70      AVLTree<Object, Object> t2 = AVLTree.create()
71        .put(k2, "v2")
72        .put(k1, "v1");
73      assertThat(t2.toString())
74        .as("toString depends on order of operations")
75        .isEqualTo(" k1->v1 k2->v2");
76  
77      assertThat(t.equals(t2))
78        .as("should compare buckets")
79        .isTrue();
80      assertThat(t2.equals(t))
81        .as("should compare buckets")
82        .isTrue();
83  
84      assertThat(t.hashCode())
85        .isEqualTo(((31 * k1.hashCode()) ^ "v1".hashCode()) + ((31 * k2.hashCode()) ^ "v2".hashCode()));
86      assertThat(t2.hashCode())
87        .as("hashCode doesn't depend on order of operations")
88        .isEqualTo(t.hashCode());
89  
90      assertThat(t.get(k1))
91        .isEqualTo("v1");
92      assertThat(t.get(k2))
93        .isEqualTo("v2");
94      assertThat(t.get(k3))
95        .as("not such key")
96        .isNull();
97  
98      assertThat(t.put(k2, "new v2").toString())
99        .as("should replace head of bucket")
100       .isEqualTo(" k2->new v2 k1->v1");
101     assertThat(t.put(k1, "new v1").toString())
102       .as("should replace element of bucket")
103       .isEqualTo(" k1->new v1 k2->v2");
104     assertThat(t.put(k1, "v1"))
105       .as("should not change")
106       .isSameAs(t);
107     assertThat(t.put(k2, "v2"))
108       .as("should not change")
109       .isSameAs(t);
110     assertThat(t.put(k3, "v3").toString())
111       .as("should add to bucket")
112       .isEqualTo(" k3->v3 k2->v2 k1->v1");
113 
114     assertThat(t.remove(k2).toString())
115       .as("should remove head of bucket")
116       .isEqualTo(" k1->v1");
117     assertThat(t.remove(k1).toString())
118       .as("should remove element of bucket")
119       .isEqualTo(" k2->v2");
120     assertThat(t.remove(k1).remove(k2).toString())
121       .as("should remove bucket")
122       .isEqualTo("");
123     assertThat(t.remove(k3))
124       .as("should not change")
125       .isSameAs(t);
126 
127     HashMap<Object, Object> biConsumer = new HashMap<>();
128     t.forEach((k, v) -> assertThat(biConsumer.put(k, v)).as("unique key-value").isNull());
129     assertThat(biConsumer)
130       .isEqualTo(ImmutableMap.of(k1, "v1", k2, "v2"));
131 
132     HashSet<Object> consumer = new HashSet<>();
133     t.forEach(k -> assertThat(consumer.add(k)).as("unique key").isTrue());
134     assertThat(consumer)
135       .containsOnly(k1, k2);
136   }

138   @Test
139   public void balancing_should_preserve_buckets() {
140     Object k1 = new Key(1, "k1");
141     Object k2 = new Key(2, "k2");
142     Object k3 = new Key(3, "k3");
143     Object k4 = new Key(4, "k4");
144     AVLTree<Object, Object> t = AVLTree.create()
145       .put(k1, "v1")
146       .put(k2, "v2")
147       .put(k3, "v3");
148 
149     Object k1_1 = new Key(1, "k1_1");
150     t = t.put(k1_1, "v1_1");
151 
152     t = t.put(k4, "v4");
153     assertThat(t.height()).as("height after balancing").isEqualTo(3);
154     assertThat(t.get(k1_1)).isEqualTo("v1_1");
155   }

157   /**
158    * Subtraction must not be used for comparison of keys due to possibility of integer overflow,
159    * this for example will be the case for sequence below, which was generated using random number generator.
160    */
161   @Test
162   public void do_not_use_subtraction_for_comparison_of_keys() {
163     Key[] keys = {
164       new Key(2043979982, ""),
165       new Key(-36348207, ""),
166       new Key(-1864559204, ""),
167       new Key(-2018458363, ""),
168       new Key(-152409201, ""),
169       new Key(-1786252453, ""),
170       new Key(-1853960690, "")
171     };
172     AVLTree<Object, Object> t = AVLTree.create();
173     for (Key key : keys) {
174       t = t.add(key);
175     }
176     for (Key key : keys) {
177       assertThat(t.get(key)).as("found").isNotNull();
178       assertThat(t.remove(key)).as("removed").isNotSameAs(t);
179     }
180   }

182   @Test
183   public void hashCode_and_equals_should_not_depend_on_order_of_construction() {
184     Object o1 = new Key(21, "o1");
185     Object o2 = new Key(45, "o2");
186     AVLTree<Object, Object> t1 = AVLTree.create().add(o1).add(o2);
187     AVLTree<Object, Object> t2 = AVLTree.create().add(o2).add(o1);
188     assertThat(t1.key()).as("shape is different").isNotEqualTo(t2.key());
189 
190     assertThat(t1.hashCode()).isEqualTo(t2.hashCode());
191     assertThat(t1).isEqualTo(t2);
192     assertThat(t2).isEqualTo(t1);
193 
194     Object o3 = new Key(0, "o3");
195     AVLTree<Object, Object> t3 = t1.add(o3);
196     assertThat(t1.hashCode()).isEqualTo(t3.hashCode());
197     assertThat(t1).isNotEqualTo(t3);
198     assertThat(t3).isNotEqualTo(t1);
199   }

255   @Test
256   public void test() {
257     List<Integer> keys = new ArrayList<>();
258     for (int i = 0; i < 100; i++) {
259       keys.add(i);
260     }
261     Collections.shuffle(keys);
262 
263     AVLTree<Integer, Object> t = AVLTree.create();
264     for (Integer key : keys) {
265       t = t.add(key);
266       assertThat(t.add(key)).isSameAs(t);
267     }
268     assertThat(Counter.countSet(t)).isEqualTo(100);
269     assertThat(Counter.countMap(t)).isEqualTo(100);
270     assertThat(t.height())
271       .isGreaterThanOrEqualTo(8)
272       .isLessThanOrEqualTo(10);
273 
274     for (Integer key : keys) {
275       assertThat(t.contains(key)).isTrue();
276       t = t.remove(key);
277       assertThat(t.remove(key)).isSameAs(t);
278     }
279     assertThat(Counter.countSet(t)).isEqualTo(0);
280     assertThat(Counter.countMap(t)).isEqualTo(0);
281   }

283   private static class Counter<K, V> implements BiConsumer<K, V>, Consumer<K> {
284     int count;
285 
286     public static <K> int countSet(PSet<K> set) {
287       Counter<K, K> counter = new Counter<>();
288       set.forEach(counter);
289       return counter.count;
290     }
291 
292     public static <K, V> int countMap(PMap<K, V> map) {
293       Counter<K, V> counter = new Counter<>();
294       map.forEach(counter);
295       return counter.count;
296     }
297 
298     @Override
299     public void accept(K key, V value) {
300       count++;
301     }
302 
303     @Override
304     public void accept(K k) {
305       count++;
306     }
307   }

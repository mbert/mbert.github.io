112   private RelationalSymbolicValue relationalSV(Tree.Kind kind, SymbolicValue... computedFrom) {
113     List<ProgramState.SymbolicValueSymbol> computedFromSymbols = Arrays.stream(computedFrom).map(sv -> new ProgramState.SymbolicValueSymbol(sv, null))
114       .collect(Collectors.toList());
115     return (RelationalSymbolicValue) constraintManager
116       .createBinarySymbolicValue(new BinaryExpressionTreeImpl(kind, mock(ExpressionTree.class), mock(InternalSyntaxToken.class), mock(ExpressionTree.class)),
117         computedFromSymbols);
118   }

148     List<String> actual = new ArrayList<>();
149     for (Tree.Kind operator : operators) {
150       actual.addAll(resolveRelationStateForAllKinds(relationalSV(operator, b, a), () -> relationToString(operator, a, b)));
151       actual.addAll(resolveRelationStateForAllKinds(relationalSV(operator, a, b), () -> relationToString(operator, b, a)));

154     RelationalSymbolicValue eqBA = new RelationalSymbolicValue(RelationalSymbolicValue.Kind.METHOD_EQUALS, b, a);

157     assertThat(actual).isEqualTo(expected);

160   private List<String> resolveRelationStateForAllKinds(RelationalSymbolicValue known, Supplier<String> knownAsString) {
161     List<String> actual = new ArrayList<>();
162     for (Tree.Kind operator : operators) {
163       RelationalSymbolicValue test = relationalSV(operator, b, a);
164       RelationState relationState = test.resolveRelationState(Collections.singleton(known));
165       actual.add(String.format("given %s when %s -> %s", knownAsString.get(), relationToString(operator, a, b), relationState));
166     }
167     RelationalSymbolicValue eq = new RelationalSymbolicValue(RelationalSymbolicValue.Kind.METHOD_EQUALS, a, b);
168     Stream.of(eq, eq.inverse()).forEach(rel -> {
169         RelationState relationState = rel.resolveRelationState(Collections.singleton(known));
170         actual.add(String.format("given %s when %s -> %s", knownAsString.get(), rel, relationState));
171       }
172     );
173     return actual;
174   }

334   private List<String> combineWithAll(RelationalSymbolicValue relation, Supplier<String> relationAsString) {
335     List<String> actual = new ArrayList<>();
336     for (Tree.Kind r : operators) {
337       actual.add(
338         String.format("%s && %s => %s", relationAsString.get(), relationToString(r, b, c), nullableToCollection(relation.deduceTransitiveOrSimplified(relationalSV(r, c, b)))));
339     }
340     RelationalSymbolicValue eq = new RelationalSymbolicValue(RelationalSymbolicValue.Kind.METHOD_EQUALS, b, c);
341     Stream.of(eq, eq.inverse()).forEach(rel ->
342       actual.add(String.format("%s && %s => %s", relationAsString.get(), rel, nullableToCollection(relation.deduceTransitiveOrSimplified(rel))))
343     );
344     return actual;
345   }

412   private String operatorToString(Tree.Kind kind) {
413     switch (kind) {
414       case EQUAL_TO:
415         return "==";
416       case NOT_EQUAL_TO:
417         return "!=";
418       case GREATER_THAN:
419         return ">";
420       case GREATER_THAN_OR_EQUAL_TO:
421         return ">=";
422       case LESS_THAN:
423         return "<";
424       case LESS_THAN_OR_EQUAL_TO:
425         return "<=";
426       default:
427         throw new IllegalStateException();
428     }
429   }

482     ps = Iterables.getOnlyElement(svNotNull.setConstraint(ps, ObjectConstraint.NOT_NULL));
483     SymbolicValue sv = new SymbolicValue();
484     // sv != NOT_NULL
485     RelationalSymbolicValue neq = new RelationalSymbolicValue(NOT_EQUAL, svNotNull, sv);
486     ProgramState result = setTrue(ps, neq);
487     assertNoConstraints(result, sv);
488 
489     // sv != NULL
490     neq = new RelationalSymbolicValue(NOT_EQUAL, NULL_LITERAL, sv);
491     result = setTrue(ps, neq);
492     assertThat(result.getConstraint(sv, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
493 
494     // sv == NULL
495     RelationalSymbolicValue eq = new RelationalSymbolicValue(EQUAL, NULL_LITERAL, sv);
496     result = setTrue(ps, eq);
497     assertNullConstraint(result, sv);
498     // sv == NOT_NULL
499     eq = new RelationalSymbolicValue(METHOD_EQUALS, sv, svNotNull);
500     result = setTrue(ps, eq);
501     assertThat(result.getConstraint(sv, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
502   }

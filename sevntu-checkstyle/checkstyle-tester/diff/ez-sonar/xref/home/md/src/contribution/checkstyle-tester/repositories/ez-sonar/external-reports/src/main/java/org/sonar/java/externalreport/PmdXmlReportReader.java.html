20  package org.sonar.java.externalreport;
21  
22  import java.io.File;
23  import java.io.FileInputStream;
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.util.HashMap;
27  import java.util.Map;
28  import javax.xml.namespace.QName;
29  import javax.xml.stream.XMLEventReader;
30  import javax.xml.stream.XMLInputFactory;
31  import javax.xml.stream.XMLStreamException;
32  import javax.xml.stream.events.Attribute;
33  import javax.xml.stream.events.StartElement;
34  import javax.xml.stream.events.XMLEvent;
35  import org.sonar.api.batch.fs.FilePredicates;
36  import org.sonar.api.batch.fs.InputFile;
37  import org.sonar.api.batch.fs.TextRange;
38  import org.sonar.api.batch.rule.Severity;
39  import org.sonar.api.batch.sensor.SensorContext;
40  import org.sonar.api.batch.sensor.issue.NewExternalIssue;
41  import org.sonar.api.batch.sensor.issue.NewIssueLocation;
42  import org.sonar.api.rule.RuleKey;
43  import org.sonar.api.rules.RuleType;
44  import org.sonar.api.utils.log.Logger;
45  import org.sonar.api.utils.log.Loggers;
46  import org.sonarsource.analyzer.commons.ExternalRuleLoader;
47  
48  public class PmdXmlReportReader {
49  
50    private static final Logger LOG = Loggers.get(PmdXmlReportReader.class);
51  
52    private static final Map<Integer, Severity> SEVERITIES = severities();
53  
54    private final SensorContext context;
55    private final File reportFile;
56    private final ExternalRuleLoader ruleLoader;
57  
58    private InputFile inputFile = null;
59    private NewExternalIssue issue = null;
60    private NewIssueLocation issueLocation = null;
61    private StringBuilder issueMessage = new StringBuilder();
62  
63    public PmdXmlReportReader(SensorContext context, File reportFile, ExternalRuleLoader ruleLoader) {
64      this.context = context;
65      this.reportFile = reportFile;
66      this.ruleLoader = ruleLoader;
67    }
68  
69    public static void read(SensorContext context, File reportFile, ExternalRuleLoader ruleLoader) throws XMLStreamException, IOException {
70      new PmdXmlReportReader(context, reportFile, ruleLoader).parse();
71    }
72  
73    private void parse() throws XMLStreamException, IOException {
74      try (InputStream inputStream = new FileInputStream(reportFile)) {
75        XMLEventReader reader = XMLInputFactory.newInstance().createXMLEventReader(inputStream);
76        while (reader.hasNext()) {
77          onXmlEvent(reader.nextEvent());
78        }
79      }
80    }
81  
82    private void onXmlEvent(XMLEvent event) {
83      if (event.isStartElement()) {
84        StartElement element = event.asStartElement();
85        String elementName = element.getName().getLocalPart();
86        if ("file".equals(elementName)) {
87          String filePath = getAttributeValue(element, "name");
88          FilePredicates predicates = context.fileSystem().predicates();
89          inputFile = context.fileSystem().inputFile(predicates.hasPath(filePath));
90          if (inputFile == null) {
91            LOG.warn("No input file found for {}. No PMD issue will be imported on this file.", filePath);
92          }
93        } else if ("violation".equals(elementName) && inputFile != null) {
94          onViolationStartElement(element);
95        }
96  
97      } else if (event.isCharacters()) {
98        issueMessage.append(event.asCharacters().getData());
99  
100     } else if (event.isEndElement()
101       && "violation".equals(event.asEndElement().getName().getLocalPart())
102       && inputFile != null
103       && issue != null) {
104       issueLocation.message(issueMessage.toString());
105       issue.at(issueLocation).save();
106     }
107   }
108 
109   private void onViolationStartElement(StartElement element) {
110     try {
111       TextRange textRange = textRange(element);
112       RuleKey ruleKey = RuleKey.of(PmdSensor.LINTER_KEY, getAttributeValue(element, "rule"));
113       issue = context.newExternalIssue()
114         .forRule(ruleKey)
115         .type(RuleType.CODE_SMELL)
116         .severity(SEVERITIES.get(getAttributeAsInt(element, "priority")))
117         .remediationEffortMinutes(ruleLoader.ruleConstantDebtMinutes(ruleKey.rule()));
118       issueLocation = issue.newLocation()
119         .on(inputFile)
120         .at(textRange);
121       issueMessage = new StringBuilder();
122     } catch (RuntimeException e) {
123       int lineNumber = element.getLocation().getLineNumber();
124       LOG.warn("Can't import issue at line " + lineNumber + " in " + reportFile + ": " + e.getMessage());
125       issue = null;
126     }
127   }
128 
129   private TextRange textRange(StartElement violationElement) {
130     Integer beginLine = getAttributeAsInt(violationElement, "beginline");
131     try {
132       Integer endLine = getAttributeAsInt(violationElement, "endline");
133       Integer beginColumn = getAttributeAsInt(violationElement, "begincolumn");
134       Integer endColumn = getAttributeAsInt(violationElement, "endcolumn");
135       return inputFile.newRange(beginLine, beginColumn - 1, endLine, endColumn);
136     } catch (RuntimeException e) {
137       // Some PMD rules seem to report invalid line offsets, e.g. TooManyStaticImports
138       return inputFile.selectLine(beginLine);
139     }
140   }
141 
142   private static String getAttributeValue(StartElement startElement, String attributeName) {
143     Attribute attribute = startElement.getAttributeByName(new QName(attributeName));
144     return attribute == null ? "" : attribute.getValue();
145   }
146 
147   private static Integer getAttributeAsInt(StartElement startElement, String attributeName) {
148     return Integer.parseInt(getAttributeValue(startElement, attributeName));
149   }
150 
151   private static Map<Integer, Severity> severities() {
152     Map<Integer, Severity> map = new HashMap<>();
153     map.put(1, Severity.BLOCKER);
154     map.put(2, Severity.CRITICAL);
155     map.put(3, Severity.MAJOR);
156     map.put(4, Severity.MINOR);
157     map.put(5, Severity.INFO);
158     return map;
159   }

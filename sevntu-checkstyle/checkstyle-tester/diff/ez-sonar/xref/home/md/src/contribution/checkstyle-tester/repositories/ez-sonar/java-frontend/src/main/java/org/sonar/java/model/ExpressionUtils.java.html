20  package org.sonar.java.model;
21  
22  import org.sonar.plugins.java.api.tree.AssignmentExpressionTree;
23  import org.sonar.plugins.java.api.tree.BinaryExpressionTree;
24  import org.sonar.plugins.java.api.tree.ExpressionTree;
25  import org.sonar.plugins.java.api.tree.IdentifierTree;
26  import org.sonar.plugins.java.api.tree.MemberSelectExpressionTree;
27  import org.sonar.plugins.java.api.tree.MethodInvocationTree;
28  import org.sonar.plugins.java.api.tree.ParenthesizedTree;
29  import org.sonar.plugins.java.api.tree.Tree;
30  
31  public final class ExpressionUtils {
32  
33    private ExpressionUtils() {
34    }
35  
36    /**
37     * In case of simple assignments, only the expression is evaluated, as we only use the reference to the variable to store the result.
38     * For SE-Based checks, only a single value should be unstacked if its the case. For other cases, two values should be unstacked.
39     * See JLS8-15.26
40     *
41     * @param tree The assignment tree
42     * @return true if the tree is a simple assignment
43     * @see #extractIdentifier(AssignmentExpressionTree)
44     */
45    public static boolean isSimpleAssignment(AssignmentExpressionTree tree) {
46      if (!tree.is(Tree.Kind.ASSIGNMENT)) {
47        // This can't possibly be a simple assignment.
48        return false;
49      }
50  
51      ExpressionTree variable = ExpressionUtils.skipParentheses(tree.variable());
52      return variable.is(Tree.Kind.IDENTIFIER) || isSelectOnThisOrSuper(tree);
53    }
54  
55    /**
56     * Checks of is the given tree is a {@link MemberSelectExpressionTree} which is selecting with <code>this</code> or <code>super</code>
57     * @param tree The tree to check.
58     * @return true when the tree is a select on <code>this</code> or <code>super</code>
59     * @see #isSelectOnThisOrSuper(MemberSelectExpressionTree)
60     */
61    public static boolean isSelectOnThisOrSuper(AssignmentExpressionTree tree) {
62      ExpressionTree variable = ExpressionUtils.skipParentheses(tree.variable());
63      return variable.is(Tree.Kind.MEMBER_SELECT) && isSelectOnThisOrSuper((MemberSelectExpressionTree) variable);
64    }
65  
66    /**
67     * Checks of is the given tree is selecting with <code>this</code> or <code>super</code>
68     * @param tree The tree to check.
69     * @return true when the tree is a select on <code>this</code> or <code>super</code>
70     * @see #isSelectOnThisOrSuper(AssignmentExpressionTree)
71     */
72    public static boolean isSelectOnThisOrSuper(MemberSelectExpressionTree tree) {
73      if (!tree.expression().is(Tree.Kind.IDENTIFIER)) {
74        // This is no longer simple.
75        return false;
76      }
77  
78      String selectSourceName = ((IdentifierTree) tree.expression()).name();
79      return "this".equalsIgnoreCase(selectSourceName) || "super".equalsIgnoreCase(selectSourceName);
80    }
81  
82    public static IdentifierTree extractIdentifier(AssignmentExpressionTree tree) {
83      ExpressionTree variable = skipParentheses(tree.variable());
84      if (variable.is(Tree.Kind.IDENTIFIER)) {
85        return (IdentifierTree) variable;
86      }
87  
88      if (variable.is(Tree.Kind.MEMBER_SELECT)) {
89        MemberSelectExpressionTree selectTree = (MemberSelectExpressionTree) variable;
90        if (isSelectOnThisOrSuper(selectTree)) {
91          return selectTree.identifier();
92        }
93      }
94  
95      // This should not be possible.
96      throw new IllegalArgumentException("Can not extract identifier.");
97    }
98  
99    public static ExpressionTree skipParentheses(ExpressionTree tree) {
100     ExpressionTree result = tree;
101     while (result.is(Tree.Kind.PARENTHESIZED_EXPRESSION)) {
102       result = ((ParenthesizedTree) result).expression();
103     }
104     return result;
105   }
106 
107   public static boolean isNullLiteral(ExpressionTree tree) {
108     return skipParentheses(tree).is(Tree.Kind.NULL_LITERAL);
109   }
110 
111   public static boolean isSecuringByte(ExpressionTree expression) {
112     if (expression.is(Tree.Kind.AND)) {
113       BinaryExpressionTree and = (BinaryExpressionTree) expression;
114       return LiteralUtils.is0xff(and.rightOperand()) || LiteralUtils.is0xff(and.leftOperand());
115     }
116     return false;
117   }
118 
119   /**
120    * Retrieve the identifier corresponding to the method name associated to the method invocation
121    */
122   public static IdentifierTree methodName(MethodInvocationTree mit) {
123     ExpressionTree methodSelect = mit.methodSelect();
124     IdentifierTree id;
125     if (methodSelect.is(Tree.Kind.IDENTIFIER)) {
126       id = (IdentifierTree) methodSelect;
127     } else {
128       id = ((MemberSelectExpressionTree) methodSelect).identifier();
129     }
130     return id;
131   }
132 }

113   @Override
114   public void visitMethod(MethodTree tree) {
115     super.visitMethod(tree);
116     Symbol.MethodSymbol methodSymbol = tree.symbol();
117     List<Type> types = new ArrayList<>(methodSymbol.parameterTypes());
118     if (!tree.is(CONSTRUCTOR)) {
119       types.add(methodSymbol.returnType().type());
120     }
121     if (tree.block() != null && types.stream().noneMatch(Type::isUnknown)) {
122       CFG cfg = CFG.build(tree);
123       serializeUCFG(tree, cfg);
124     }
125   }

127   @VisibleForTesting
128   protected void serializeUCFG(MethodTree tree, CFG cfg) {
129     try {
130       UCFG uCFG = buildUCfg(tree, cfg);
131       UCFGtoProtobuf.toProtobufFile(uCFG, ucfgFilePath());
132     } catch (Exception e) {
133       String msg = String.format("Cannot generate ucfg for file %s for method at line %d", javaFileKey, tree.firstToken().line());
134       LOG.error(msg, e);
135       throw new AnalysisException(msg, e);
136     }
137   }

141     index++;
142     return absolutePath;

145   private UCFG buildUCfg(MethodTree methodTree, CFG cfg) {
146     String signature = signatureFor(methodTree.symbol());
147 
148     IdentifierGenerator idGenerator = new IdentifierGenerator(methodTree);
149     UCFGBuilder builder = UCFGBuilder.createUCFGForMethod(signature);
150 
151     methodTree.parameters().stream()
152       .map(p -> idGenerator.lookupIdFor(p.symbol()))
153       .map(UCFGBuilder::variableWithId)
154       .forEach(builder::addMethodParam);
155 
156     BlockBuilder entryBlockBuilder = buildBasicBlock(cfg.entry(), methodTree, idGenerator);
157 
158     if (hasAnnotatedParameters(methodTree)) {
159       builder.addStartingBlock(buildParameterAnnotationsBlock(methodTree, idGenerator, cfg));
160       builder.addBasicBlock(entryBlockBuilder);
161     } else {
162       builder.addStartingBlock(entryBlockBuilder);
163     }
164 
165     cfg.blocks().stream()
166       .filter(b -> !b.equals(cfg.entry()))
167       .forEach(b -> builder.addBasicBlock(buildBasicBlock(b, methodTree, idGenerator)));
168     return builder.build();
169   }

171   private BlockBuilder buildParameterAnnotationsBlock(MethodTree methodTree, IdentifierGenerator idGenerator, CFG cfg) {
172     LocationInFile parametersLocation = location(methodTree.openParenToken(), methodTree.closeParenToken());
173     UCFGBuilder.BlockBuilder blockBuilder = UCFGBuilder.newBasicBlock("paramAnnotations", parametersLocation);
174 
175     List<AnnotationTree> methodAnnotations = methodTree.modifiers().annotations();
176     getObjectParameters(methodTree).forEach(parameter -> buildBlockForParameter(parameter, methodAnnotations, blockBuilder, idGenerator));
177 
178     Label nextBlockLabel = UCFGBuilder.createLabel(Integer.toString(cfg.entry().id()));
179     blockBuilder.jumpTo(nextBlockLabel);
180     return blockBuilder;
181   }

183   private void buildBlockForParameter(VariableTree parameter, List<AnnotationTree> methodAnnotations, BlockBuilder blockBuilder, IdentifierGenerator idGenerator) {
184     Expression.Variable parameterVariable = variableWithId(idGenerator.lookupIdFor(parameter.symbol()));
185     List<Expression> annotationVariables = new ArrayList<>();
186 
187     List<AnnotationTree> annotations = new ArrayList<>();
188     annotations.addAll(methodAnnotations);
189     annotations.addAll(parameter.modifiers().annotations());
190 
191     if (annotations.isEmpty()) {
192       // method is not annotated && parameter is not annotated
193       return;
194     }
195 
196     // at least one annotation should be applied to the parameter
197     annotations.forEach(annotationTree -> {
198       Expression.Variable var = variableWithId(idGenerator.newId());
199       annotationVariables.add(var);
200       blockBuilder.assignTo(var, annotateCall(annotationTree, parameterVariable), location(annotationTree));
201     });
202 
203     Expression[] args = annotationVariables.toArray(new Expression[annotationVariables.size()]);
204     blockBuilder.assignTo(parameterVariable, call("__annotation").withArgs(args), location(parameter.simpleName()));
205   }

207   private static CallBuilder annotateCall(AnnotationTree annotation, Expression.Variable annotatedVariable) {
208     Expression.Constant fullyQualifiedName = constant(annotation.symbolType().fullyQualifiedName());
209     return call("__annotate").withArgs(fullyQualifiedName, annotatedVariable);
210   }

212   private static boolean hasAnnotatedParameters(MethodTree methodTree) {
213     List<VariableTree> objectParameters = getObjectParameters(methodTree).collect(Collectors.toList());
214     if (objectParameters.isEmpty()) {
215       return false;
216     }
217     return !methodTree.modifiers().annotations().isEmpty() || objectParameters.stream().anyMatch(parameter -> !parameter.modifiers().annotations().isEmpty());
218   }
219 
220   private static Stream<VariableTree> getObjectParameters(MethodTree methodTree) {
221     return methodTree.parameters().stream().filter(parameter -> isObject(parameter.type().symbolType()));
222   }

224   private UCFGBuilder.BlockBuilder buildBasicBlock(CFG.Block javaBlock, MethodTree methodTree, IdentifierGenerator idGenerator) {
225     UCFGBuilder.BlockBuilder blockBuilder = UCFGBuilder.newBasicBlock(String.valueOf(javaBlock.id()), location(javaBlock));
226 
227     javaBlock.elements().forEach(e -> buildCall(e, blockBuilder, idGenerator));
228 
229     Tree terminator = javaBlock.terminator();
230     if (terminator != null && terminator.is(Tree.Kind.RETURN_STATEMENT)) {
231       ExpressionTree returnedExpression = ((ReturnStatementTree) terminator).expression();
232       Expression retExpr = constant(IdentifierGenerator.CONST);
233       if (methodTree.returnType() != null && isObject(methodTree.returnType().symbolType())) {
234         retExpr = idGenerator.lookupExpressionFor(returnedExpression);
235       }
236       blockBuilder.ret(retExpr, location(terminator));
237       return blockBuilder;
238     }
239 
240     Set<CFG.Block> successors = javaBlock.successors();
241     if (!successors.isEmpty()) {
242       blockBuilder.jumpTo(successors.stream().map(b -> UCFGBuilder.createLabel(Integer.toString(b.id()))).toArray(Label[]::new));
243       return blockBuilder;
244     }
245     Preconditions.checkState(javaBlock.id() == 0);
246     blockBuilder.ret(constant("implicit return"), location(methodTree.lastToken()));
247     return blockBuilder;
248   }

250   private void buildCall(Tree element, UCFGBuilder.BlockBuilder blockBuilder, IdentifierGenerator idGenerator) {
251     if (isObjectVarDeclaration(element)) {
252       VariableTree variableTree = (VariableTree) element;
253 
254       String lhs = idGenerator.lookupIdFor(variableTree.simpleName());
255       if (!idGenerator.isConst(lhs)) {
256         Expression source = idGenerator.lookupExpressionFor(variableTree.initializer());
257         blockBuilder.assignTo(variableWithId(lhs), UCFGBuilder.call("__id").withArgs(source), location(element));
258       }
259       return;
260     }
261 
262     if (element.is(METHOD_INVOCATION)) {
263       MethodInvocationTree methodInvocationTree = (MethodInvocationTree) element;
264       buildMethodInvocation(blockBuilder, idGenerator, methodInvocationTree);
265     } else if (element.is(NEW_CLASS)) {
266       NewClassTree newClassTree = (NewClassTree) element;
267       buildConstructorInvocation(blockBuilder, idGenerator, newClassTree);
268     } else if (element.is(NEW_ARRAY)) {
269       NewArrayTree newArrayTree = (NewArrayTree) element;
270       buildNewArrayInvocation(blockBuilder, idGenerator, newArrayTree);
271     } else if (element.is(PLUS)) {
272       BinaryExpressionTree binaryExpressionTree = (BinaryExpressionTree) element;
273       buildConcatenationInvocation(blockBuilder, idGenerator, binaryExpressionTree);
274     } else if (element.is(ASSIGNMENT)) {
275       AssignmentExpressionTree assignmentExpressionTree = (AssignmentExpressionTree) element;
276       buildAssignmentInvocation(blockBuilder, idGenerator, assignmentExpressionTree);
277     } else if (element.is(PLUS_ASSIGNMENT)) {
278       AssignmentExpressionTree assignmentExpressionTree = (AssignmentExpressionTree) element;
279       buildPlusAssignmentInvocation(blockBuilder, idGenerator, assignmentExpressionTree);
280     } else if (element.is(IDENTIFIER, MEMBER_SELECT) && !element.parent().is(PLUS_ASSIGNMENT, ASSIGNMENT)) {
281       buildFieldReadAccessInvocation(blockBuilder, idGenerator, element);
282     } else if (element.is(ARRAY_ACCESS_EXPRESSION) && !element.parent().is(PLUS_ASSIGNMENT, ASSIGNMENT)) {
283       // PLUS_ASSIGNMENT and ASSIGNMENT might imply an array set, otherwise an array access is always a get
284       ArrayAccessExpressionTree arrayAccessExpressionTree = (ArrayAccessExpressionTree) element;
285       if (isObject(arrayAccessExpressionTree.symbolType())) {
286         Expression.Variable getValue = variableWithId(idGenerator.newId());
287         Expression array = idGenerator.lookupExpressionFor(arrayAccessExpressionTree.expression());
288         blockBuilder.assignTo(getValue, arrayGet(array), location(element));
289         idGenerator.varForExpression(element, getValue.id());
290       }
291     }
292   }

294   private void buildNewArrayInvocation(BlockBuilder blockBuilder, IdentifierGenerator idGenerator, NewArrayTree tree) {
295     if (tree.type() != null  && !isObject(tree.type().symbolType())) {
296       return;
297     }
298     Expression.Variable newArray = variableWithId(idGenerator.newIdFor(tree));
299     blockBuilder.newObject(newArray, tree.symbolType().fullyQualifiedName(), location(tree));
300     idGenerator.varForExpression(tree, newArray.id());
301     tree.initializers().stream().filter(i -> isObject(i.symbolType())).forEach(i -> {
302       Expression argument = idGenerator.lookupExpressionFor(i);
303       blockBuilder.assignTo(variableWithId(idGenerator.newId()), arraySet(newArray, argument), location(tree));
304     });
305   }

307   private void buildConstructorInvocation(BlockBuilder blockBuilder, IdentifierGenerator idGenerator, NewClassTree tree) {
308     Symbol constructorSymbol = tree.constructorSymbol();
309     if (!constructorSymbol.isMethodSymbol()) {
310       return;
311     }
312 
313     Expression.Variable newInstance = variableWithId(idGenerator.newIdFor(tree));
314     blockBuilder.newObject(newInstance, tree.symbolType().fullyQualifiedName(), location(tree.identifier()));
315 
316     List<Expression> arguments = new ArrayList<>();
317     arguments.add(newInstance);
318     arguments.addAll(argumentIds(idGenerator, tree.arguments()));
319 
320     blockBuilder.assignTo(variableWithId(idGenerator.newId()),
321         UCFGBuilder.call(signatureFor((Symbol.MethodSymbol) constructorSymbol)).withArgs(arguments.toArray(new Expression[0])),
322         location(tree));
323   }

325   private void buildMethodInvocation(UCFGBuilder.BlockBuilder blockBuilder, IdentifierGenerator idGenerator, MethodInvocationTree tree) {
326     if (tree.symbol().isUnknown()) {
327       return;
328     }
329 
330     List<Expression> arguments = new ArrayList<>();
331     if (tree.symbol().isStatic()) {
332       arguments.add(new Expression.ClassName(tree.symbol().type().fullyQualifiedName()));
333     } else if (tree.methodSelect().is(MEMBER_SELECT)) {
334       arguments.add(idGenerator.lookupExpressionFor(((MemberSelectExpressionTree) tree.methodSelect()).expression()));
335     } else if (tree.methodSelect().is(IDENTIFIER)) {
336       arguments.add(Expression.THIS);
337     }
338     arguments.addAll(argumentIds(idGenerator, tree.arguments()));
339 
340     buildAssignCall(blockBuilder, idGenerator, arguments, tree, (Symbol.MethodSymbol) tree.symbol());
341   }

343   private void buildConcatenationInvocation(BlockBuilder blockBuilder, IdentifierGenerator idGenerator, BinaryExpressionTree tree) {
344     if (!isObject(tree.symbolType())) {
345       return;
346     }
347     Expression leftOperand = idGenerator.lookupExpressionFor(tree.leftOperand());
348     Expression rightOperand = idGenerator.lookupExpressionFor(tree.rightOperand());
349     Expression.Variable var = variableWithId(idGenerator.newIdFor(tree));
350     blockBuilder.assignTo(var, concat(leftOperand, rightOperand), location(tree));
351   }

353   private void buildAssignmentInvocation(BlockBuilder blockBuilder, IdentifierGenerator idGenerator, AssignmentExpressionTree tree) {
354     if (!isObject(tree.symbolType())) {
355       return;
356     }
357     ExpressionTree lhsTree = tree.variable();
358     ExpressionTree rhsTree = tree.expression();
359     Expression rightSide = lookupExpression(blockBuilder, idGenerator, rhsTree);
360     Optional<Expression.FieldAccess> leftFieldAccess = buildFieldAccess(idGenerator, lhsTree);
361     if (leftFieldAccess.isPresent()) {
362       blockBuilder.assignTo(leftFieldAccess.get(), call("__id").withArgs(rightSide), location(tree));
363     } else if (lhsTree.is(ARRAY_ACCESS_EXPRESSION)) {
364       Expression leftSide = idGenerator.lookupExpressionFor(((ArrayAccessExpressionTree)lhsTree).expression());
365       // when an assignment implies both get and set on arrays, the get must be stored in an auxiliary local variable
366       blockBuilder.assignTo(variableWithId(idGenerator.newId()), arraySet(leftSide, rightSide), location(tree));
367     } else {
368       Expression leftSide = idGenerator.lookupExpressionFor(lhsTree);
369       if (leftSide.isVariable()) {
370         blockBuilder.assignTo((Expression.Variable) leftSide, call("__id").withArgs(rightSide), location(tree));
371       }
372     }
373   }

375   private void buildPlusAssignmentInvocation(BlockBuilder blockBuilder, IdentifierGenerator idGenerator, AssignmentExpressionTree tree) {
376     if (!isObject(tree.symbolType())) {
377       return;
378     }
379     ExpressionTree lhsTree = tree.variable();
380     ExpressionTree rhsTree = tree.expression();
381     // '+=' is the only expression which can imply two reads and one write on an array / field
382     Expression leftSide = lookupExpression(blockBuilder, idGenerator, lhsTree);
383     Expression rightSide = lookupExpression(blockBuilder, idGenerator, rhsTree);
384     Optional<Expression.FieldAccess> leftFieldAccess = buildFieldAccess(idGenerator, lhsTree);
385     if (leftFieldAccess.isPresent()) {
386       Expression.Variable concatAux = variableWithId(idGenerator.newId());
387       blockBuilder.assignTo(concatAux, concat(leftSide, rightSide), location(tree));
388       blockBuilder.assignTo(leftFieldAccess.get(), call("__id").withArgs(concatAux), location(tree));
389     } else if (lhsTree.is(ARRAY_ACCESS_EXPRESSION)) {
390       Expression.Variable concatAux = variableWithId(idGenerator.newId());
391       blockBuilder.assignTo(concatAux, concat(leftSide, rightSide), location(tree));
392       blockBuilder.assignTo(variableWithId(idGenerator.newId()),
393           arraySet(idGenerator.lookupExpressionFor(((ArrayAccessExpressionTree) lhsTree).expression()), concatAux),
394           location(tree));
395     } else if (leftSide.isVariable()) {
396       idGenerator.varForExpression(tree, ((Expression.Variable) leftSide).id());
397       blockBuilder.assignTo((Expression.Variable) leftSide, concat(leftSide, rightSide), location(tree));
398     }
399   }

401   private void buildAssignCall(BlockBuilder blockBuilder, IdentifierGenerator idGenerator,  List<Expression> arguments, Tree tree, Symbol.MethodSymbol symbol) {
402     String destination = idGenerator.newIdFor(tree);
403     blockBuilder.assignTo(variableWithId(destination), UCFGBuilder.call(signatureFor(symbol)).withArgs(arguments.toArray(new Expression[0])), location(tree));
404   }

406   /**
407    * Field access can be a read or a write access, depending on the context.
408    * This method should be called for field read access.
409    */
410   private void buildFieldReadAccessInvocation(BlockBuilder blockBuilder, IdentifierGenerator idGenerator, Tree tree) {
411     buildFieldAccess(idGenerator, tree).ifPresent(fieldAccess -> {
412       Expression.Variable aux = variableWithId(idGenerator.newId());
413       blockBuilder.assignTo(aux, call("__id").withArgs(fieldAccess), location(tree));
414       idGenerator.varForExpression(tree, aux.id());
415     });
416   }

418   private static Optional<Expression.FieldAccess> buildFieldAccess(IdentifierGenerator identifierGenerator, Tree tree) {
419     if (tree.is(IDENTIFIER)) {
420       return buildFieldAccess((IdentifierTree) tree);
421     }
422     if (tree.is(MEMBER_SELECT)) {
423       return buildFieldAccess(identifierGenerator, (MemberSelectExpressionTree) tree);
424     }
425     return Optional.empty();
426   }
427 
428   private static Optional<Expression.FieldAccess> buildFieldAccess(IdentifierGenerator idGenerator, MemberSelectExpressionTree memberSelectTree) {
429     ExpressionTree lhsTree = memberSelectTree.expression();
430     Symbol rhsTreeSymbol = memberSelectTree.identifier().symbol();
431     if (!rhsTreeSymbol.isVariableSymbol()) {
432       return Optional.empty();
433     }
434     Expression.Variable rightSide = variableWithId(rhsTreeSymbol.name());
435     if (!rhsTreeSymbol.isStatic()) {
436       Expression leftSide = idGenerator.lookupExpressionFor(lhsTree);
437       if (leftSide.equals(Expression.THIS)) {
438         return Optional.of(new Expression.FieldAccess(rightSide));
439       } else if (leftSide.isVariable()) {
440         return  Optional.of(new Expression.FieldAccess((Expression.Variable) leftSide, rightSide));
441       }
442     }
443     if (lhsTree.is(IDENTIFIER)) {
444       IdentifierTree lhsIdentifierTree = (IdentifierTree) lhsTree;
445       return Optional.of(new Expression.FieldAccess(
446         new Expression.ClassName(lhsIdentifierTree.symbol().type().fullyQualifiedName()),
447         rightSide));
448     }
449     return Optional.empty();
450   }
451 
452   private static Optional<Expression.FieldAccess> buildFieldAccess(IdentifierTree identifierTree) {
453     Symbol identifierTreeSymbol = identifierTree.symbol();
454     if (!identifierTreeSymbol.isVariableSymbol()
455         || identifierTreeSymbol.owner().isMethodSymbol()
456         || identifierTree.name().equals("this")
457         || identifierTree.name().equals("super")) {
458       return Optional.empty();
459     }
460     Expression.Variable rightSide = variableWithId(identifierTree.name());
461     if (identifierTreeSymbol.isStatic()) {
462       return Optional.of(new Expression.FieldAccess(
463         new Expression.ClassName(identifierTreeSymbol.owner().type().fullyQualifiedName()),
464         rightSide));
465     }
466     return Optional.of(new Expression.FieldAccess(rightSide));
467   }

469   private static List<Expression> argumentIds(IdentifierGenerator idGenerator, Arguments arguments) {
470     return arguments.stream().map(idGenerator::lookupExpressionFor).collect(Collectors.toList());
471   }

473   /**
474    * This method should be used for the contexts where
475    * - the array / field access is not already in the cache of the idGenerator
476    * - it is clear from the context that the array / field access (if present) is a read access
477    */
478   private Expression lookupExpression(BlockBuilder blockBuilder, IdentifierGenerator idGenerator, ExpressionTree expressionTree) {
479     if (expressionTree.is(ARRAY_ACCESS_EXPRESSION)) {
480       Expression array = idGenerator.lookupExpressionFor(((ArrayAccessExpressionTree)expressionTree).expression());
481       Expression.Variable aux = variableWithId(idGenerator.newId());
482       blockBuilder.assignTo(aux, arrayGet(array), location(expressionTree));
483       return aux;
484     }
485     buildFieldReadAccessInvocation(blockBuilder, idGenerator, expressionTree);
486     return idGenerator.lookupExpressionFor(expressionTree);
487   }

505   @Nullable
506   private LocationInFile location(CFG.Block javaBlock) {
507     Tree firstTree = null;
508     List<Tree> elements = javaBlock.elements();
509     if (!elements.isEmpty()) {
510       firstTree = elements.get(0);
511     } else if (javaBlock.terminator() != null) {
512       firstTree = javaBlock.terminator();
513     }
514     if (firstTree == null) {
515       return null;
516     }
517     return location(firstTree);
518   }

520   private LocationInFile location(Tree tree) {
521     return location(tree.firstToken(), tree.lastToken());
522   }

524   private LocationInFile location(SyntaxToken firstToken, SyntaxToken lastToken) {
525     return new LocationInFile(
526       javaFileKey,
527       firstToken.line(), firstToken.column(),
528       lastToken.line(), lastToken.column() + lastToken.text().length()
529     );
530   }

532   private static boolean isObjectVarDeclaration(Tree tree) {
533     if (!tree.is(Tree.Kind.VARIABLE)) {
534       return false;
535     }
536 
537     VariableTree var = (VariableTree) tree;
538     return isObject(var.type().symbolType());
539   }

541   private static boolean isObject(Type type) {
542     if (type.isArray()) {
543       return isObject(((Type.ArrayType)type).elementType());
544     }
545     return !type.isPrimitive() && !type.isUnknown();
546   }

552     private final Map<Symbol, String> objectVars;
553     private final Map<Tree, String> temps;
554     private int counter;
555 
556     public IdentifierGenerator(MethodTree methodTree) {
557       Set<Symbol> objectParameters = methodTree.parameters().stream().filter(p -> isObject(p.symbol().type())).map(VariableTree::symbol).collect(Collectors.toSet());
558       VariableReadExtractor variableReadExtractor = new VariableReadExtractor(methodTree.symbol(), false);
559       methodTree.accept(variableReadExtractor);
560       Set<Symbol> objectLocals = variableReadExtractor.usedVariables().stream().filter(s -> isObject(s.type())).collect(Collectors.toSet());
561       objectVars = Sets.union(objectParameters, objectLocals).stream().collect(Collectors.toMap(s -> s, Symbol::name));
562       temps = new HashMap<>();
563       counter = 0;
564     }
565 
566     public boolean isConst(String id) {
567       return CONST.equals(id);
568     }
569 
570     public String newIdFor(@Nullable Tree tree) {
571       return temps.computeIfAbsent(tree, t -> newId());
572     }
573 
574     private String newId() {
575       String result = "%" + counter;
576       counter++;
577       return result;
578     }
579 
580     public Expression lookupExpressionFor(@Nullable Tree tree) {
581       String id = lookupIdFor(tree);
582       if (isConst(id)) {
583         if (tree != null) {
584           if (tree.is(Tree.Kind.STRING_LITERAL)) {
585             return constant(LiteralUtils.trimQuotes(((LiteralTree) tree).value()));
586           } else if (tree.is(IDENTIFIER) && ((IdentifierTree) tree).name().equals("this")) {
587             return Expression.THIS;
588           }
589         }
590         return constant(id);
591       }
592       return variableWithId(id);
593     }
594 
595     public String lookupIdFor(@Nullable Tree tree) {
596       if (tree == null) {
597         return CONST;
598       }
599       Tree noParentheses = skipParentheses(tree);
600       if (noParentheses.is(IDENTIFIER) && !temps.containsKey(noParentheses)) {
601         return lookupIdFor(((IdentifierTree) noParentheses).symbol());
602       } else {
603         return temps.getOrDefault(noParentheses, CONST);
604       }
605     }
606 
607     public String lookupIdFor(Symbol symbol) {
608       return objectVars.getOrDefault(symbol, CONST);
609     }
610 
611     public void varForExpression(Tree element, String id) {
612       temps.put(element, id);
613     }
614 
615     private static Tree skipParentheses(Tree tree) {
616       if (tree.is(Tree.Kind.PARENTHESIZED_EXPRESSION)) {
617         return ExpressionUtils.skipParentheses((ParenthesizedTree) tree);
618       }
619       return tree;
620     }

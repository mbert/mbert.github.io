20  package org.sonar.java.it;
21  
22  import com.google.common.base.Throwables;
23  import com.google.common.collect.ImmutableMap;
24  import com.google.common.io.Files;
25  import com.sonar.orchestrator.Orchestrator;
26  import com.sonar.orchestrator.locator.FileLocation;
27  import java.io.File;
28  import java.io.IOException;
29  import java.nio.charset.StandardCharsets;
30  import java.util.ArrayList;
31  import java.util.List;
32  import java.util.Map;
33  import java.util.Set;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  import org.sonar.wsclient.internal.HttpRequestFactory;
37  import org.sonar.wsclient.jsonsimple.JSONValue;
38  
39  
40  public class ProfileGenerator {
41    private static final String LANGUAGE = "java";
42    private static final String REPOSITORY_KEY = "squid";
43    // note that 500 is the maximum allowed by SQ rules/api/search
44    private static final int NUMBER_RULES_BY_PAGE = 500;
45  
46    private static final Logger LOG = LoggerFactory.getLogger(ProfileGenerator.class);
47  
48    static void generate(Orchestrator orchestrator, ImmutableMap<String, ImmutableMap<String, String>> rulesParameters,
49      Set<String> excluded, Set<String> subsetOfEnabledRules, Set<String> activatedRuleKeys) {
50      try {
51        LOG.info("Generating profile containing all the rules");
52        StringBuilder sb = new StringBuilder()
53          .append("<profile>")
54          .append("<name>rules</name>")
55          .append("<language>").append(LANGUAGE).append("</language>")
56          .append("<rules>");
57  
58        for (String key : getRuleKeys(orchestrator)) {
59          if (excluded.contains(key) || (!subsetOfEnabledRules.isEmpty() && !subsetOfEnabledRules.contains(key))) {
60            continue;
61          }
62          activatedRuleKeys.add(key);
63          sb.append("<rule>")
64            .append("<repositoryKey>").append(REPOSITORY_KEY).append("</repositoryKey>")
65            .append("<key>").append(key).append("</key>")
66            .append("<priority>INFO</priority>");
67          if (rulesParameters.containsKey(key)) {
68            sb.append("<parameters>");
69            for (Map.Entry<String, String> parameter : rulesParameters.get(key).entrySet()) {
70              sb.append("<parameter>")
71                .append("<key>").append(parameter.getKey()).append("</key>")
72                .append("<value>").append(parameter.getValue()).append("</value>")
73                .append("</parameter>");
74            }
75            sb.append("</parameters>");
76          }
77          sb.append("</rule>");
78        }
79  
80        sb.append("</rules>")
81          .append("</profile>");
82  
83        File file = File.createTempFile("profile", ".xml");
84        Files.write(sb, file, StandardCharsets.UTF_8);
85        LOG.info("Restoring profile to SonarQube");
86        orchestrator.getServer().restoreProfile(FileLocation.of(file));
87        file.delete();
88      } catch (IOException e) {
89        throw Throwables.propagate(e);
90      }
91    }
92  
93    private static List<String> getRuleKeys(Orchestrator orchestrator) {
94      List<String> ruleKeys = new ArrayList<>();
95      // pages are 1-based
96      int currentPage = 1;
97  
98      Long totalNumberRules;
99      int collectedRulesNumber;
100     do {
101       String json = new HttpRequestFactory(orchestrator.getServer().getUrl())
102         .get("/api/rules/search", ImmutableMap.<String, Object>of("languages", LANGUAGE, "repositories", REPOSITORY_KEY, "p", currentPage, "ps", NUMBER_RULES_BY_PAGE));
103 
104       @SuppressWarnings("unchecked")
105       Map<Object, Object> jsonObject = (Map<Object, Object>) JSONValue.parse(json);
106 
107       @SuppressWarnings("unchecked")
108       List<Map<Object, Object>> jsonRules = (List<Map<Object, Object>>) jsonObject.get("rules");
109 
110       jsonRules.stream()
111         .map(jsonRule -> (String) jsonRule.get("key"))
112         .map(key -> key.split(":")[1])
113         .forEach(ruleKeys::add);
114 
115       // update number of rules
116       collectedRulesNumber = ruleKeys.size();
117       totalNumberRules = (Long) jsonObject.get("total");
118       LOG.info("Collected rule keys: {} / {}", collectedRulesNumber, totalNumberRules);
119       // prepare for next page
120       currentPage++;
121     } while (collectedRulesNumber != totalNumberRules);
122 
123     return ruleKeys;
124   }

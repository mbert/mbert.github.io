20  package org.sonar.java.checks;
21  
22  import org.sonar.check.Rule;
23  import org.sonar.plugins.java.api.JavaFileScanner;
24  import org.sonar.plugins.java.api.JavaFileScannerContext;
25  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
26  import org.sonar.plugins.java.api.tree.ClassTree;
27  import org.sonar.plugins.java.api.tree.ExpressionTree;
28  import org.sonar.plugins.java.api.tree.IdentifierTree;
29  import org.sonar.plugins.java.api.tree.MemberSelectExpressionTree;
30  import org.sonar.plugins.java.api.tree.Tree;
31  import org.sonar.plugins.java.api.tree.TypeTree;
32  
33  @Rule(key = "S1194")
34  public class ErrorClassExtendedCheck extends BaseTreeVisitor implements JavaFileScanner {
35  
36    private JavaFileScannerContext context;
37  
38    @Override
39    public void scanFile(JavaFileScannerContext context) {
40      this.context = context;
41      scan(context.getTree());
42    }
43  
44    @Override
45    public void visitClass(ClassTree tree) {
46      TypeTree superClass = tree.superClass();
47      if (tree.is(Tree.Kind.CLASS) && superClass != null) {
48        if (superClass.is(Tree.Kind.IDENTIFIER)) {
49          IdentifierTree idt = (IdentifierTree) superClass;
50          if ("Error".equals(idt.name())) {
51            context.reportIssue(this, superClass, "Extend \"java.lang.Exception\" or one of its subclasses.");
52          }
53        } else if (superClass.is(Tree.Kind.MEMBER_SELECT)) {
54          MemberSelectExpressionTree mse = (MemberSelectExpressionTree) superClass;
55          if ("Error".equals(mse.identifier().name()) && isJavaLang(mse.expression())) {
56            context.reportIssue(this, superClass, "Extend \"java.lang.Exception\" or one of its subclasses.");
57          }
58        }
59      }
60      super.visitClass(tree);
61    }
62  
63    private static boolean isJavaLang(ExpressionTree tree) {
64      if (tree.is(Tree.Kind.MEMBER_SELECT)) {
65        MemberSelectExpressionTree mse = (MemberSelectExpressionTree) tree;
66        if (!"lang".equals(mse.identifier().name())) {
67          return false;
68        }
69        if (mse.expression().is(Tree.Kind.IDENTIFIER)) {
70          IdentifierTree idt = (IdentifierTree) mse.expression();
71          return "java".equals(idt.name());
72        }
73      }
74      return false;
75    }
76  
77  }

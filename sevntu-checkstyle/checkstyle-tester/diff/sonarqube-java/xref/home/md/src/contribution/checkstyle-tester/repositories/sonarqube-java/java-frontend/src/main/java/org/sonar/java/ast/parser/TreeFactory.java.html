225     InternalSyntaxToken semicolonToken) {
226     ModifiersTreeImpl newModifiers = ModifiersTreeImpl.emptyModifiers();
227     if (modifiers.isPresent()) {
228       List<ModifierTree> modifierKeywords = new ArrayList<>();
229       // JLS9 - ยง7.7.1 'requires' only 'static' and 'transitive' modifiers are allowed
230       for (InternalSyntaxToken modifierAsSyntaxToken : modifiers.get()) {
231         if (JavaRestrictedKeyword.TRANSITIVE.getValue().equals(modifierAsSyntaxToken.text())) {
232           modifierKeywords.add(new ModifierKeywordTreeImpl(Modifier.TRANSITIVE, modifierAsSyntaxToken));
233         } else {
234           modifierKeywords.add(new ModifierKeywordTreeImpl(Modifier.STATIC, modifierAsSyntaxToken));
235         }
236       }
237       newModifiers = new ModifiersTreeImpl(modifierKeywords);
238     }
239     return new RequiresDirectiveTreeImpl(requiresToken, newModifiers, moduleName, semicolonToken);
240   }

489   public ClassTreeImpl newEnumDeclaration(
490     InternalSyntaxToken enumToken,
491     InternalSyntaxToken identifierToken,
492     Optional<Tuple<InternalSyntaxToken, QualifiedIdentifierListTreeImpl>> implementsClause,
493     InternalSyntaxToken openBraceToken,
494     Optional<List<EnumConstantTreeImpl>> enumConstants,
495     Optional<InternalSyntaxToken> semicolonToken,
496     Optional<List<JavaTree>> enumDeclarations,
497     InternalSyntaxToken closeBraceToken) {
498 
499     List<JavaTree> members = Lists.newLinkedList();
500     EnumConstantTreeImpl lastEnumConstant = null;
501     if (enumConstants.isPresent()) {
502       for (EnumConstantTreeImpl enumConstant : enumConstants.get()) {
503         members.add(enumConstant);
504         lastEnumConstant = enumConstant;
505       }
506     }
507     if (semicolonToken.isPresent()) {
508       InternalSyntaxToken semicolon = semicolonToken.get();
509       // add the semicolon as endToken of the last enumConstant, or as empty statement in the enum members
510       if (lastEnumConstant != null) {
511         lastEnumConstant.setEndToken(semicolon);
512       } else {
513         members.add(newEmptyMember(semicolon));
514       }
515     }
516     if (enumDeclarations.isPresent()) {
517       for (JavaTree enumDeclaration : enumDeclarations.get()) {
518         members.add(enumDeclaration);
519       }
520     }
521 
522     ClassTreeImpl result = newClassBody(Kind.ENUM, openBraceToken, Optional.of((List<JavaTree>) ImmutableList.<JavaTree>builder().addAll(members).build()), closeBraceToken);
523 
524     result.completeDeclarationKeyword(enumToken);
525 
526     IdentifierTreeImpl identifier = new IdentifierTreeImpl(identifierToken);
527     result.completeIdentifier(identifier);
528 
529     if (implementsClause.isPresent()) {
530       InternalSyntaxToken implementsKeyword = implementsClause.get().first();
531       QualifiedIdentifierListTreeImpl interfaces = implementsClause.get().second();
532       result.completeInterfaces(implementsKeyword, interfaces);
533     }
534 
535     return result;
536   }

1764     ExpressionTree result = primary;
1765 
1766     if (selectors.isPresent()) {
1767       for (Tuple<Optional<InternalSyntaxToken>, ExpressionTree> tuple : selectors.get()) {
1768         Optional<InternalSyntaxToken> dotTokenOptional = tuple.first();
1769         ExpressionTree selector = tuple.second();
1770 
1771         if (dotTokenOptional.isPresent()) {
1772           InternalSyntaxToken dotToken = dotTokenOptional.get();
1773 
1774           if (selector.is(Kind.IDENTIFIER)) {
1775             IdentifierTreeImpl identifier = (IdentifierTreeImpl) selector;
1776             result = new MemberSelectExpressionTreeImpl(result, dotToken, identifier);
1777           } else {
1778             MethodInvocationTreeImpl methodInvocation = (MethodInvocationTreeImpl) selector;
1779             IdentifierTreeImpl identifier = (IdentifierTreeImpl) methodInvocation.methodSelect();
1780             MemberSelectExpressionTreeImpl memberSelect = new MemberSelectExpressionTreeImpl(result, dotToken, identifier);
1781 
1782             result = new MethodInvocationTreeImpl(memberSelect, methodInvocation.typeArguments(), (ArgumentListTreeImpl) methodInvocation.arguments());
1783           }
1784         } else if (selector.is(Kind.NEW_CLASS)) {
1785           NewClassTreeImpl newClass = (NewClassTreeImpl) selector;
1786           result = newClass.completeWithEnclosingExpression(result);
1787         } else if (selector.is(Kind.ARRAY_ACCESS_EXPRESSION)) {
1788           ArrayAccessExpressionTreeImpl arrayAccess = (ArrayAccessExpressionTreeImpl) selector;
1789           result = arrayAccess.complete(result);
1790         } else if (selector.is(Kind.MEMBER_SELECT)) {
1791           MemberSelectExpressionTreeImpl memberSelect = (MemberSelectExpressionTreeImpl) selector;
1792           result = memberSelect.completeWithExpression(result);
1793         } else {
1794           throw new IllegalStateException();
1795         }
1796       }
1797     }
1798 
1799     return result;

247   @Test
248   public void add_issue_or_parse_error() throws Exception {
249     JavaCheck expectedCheck = new CustomCheck();
250     CheckRegistrar expectedRegistrar = getRegistrar(expectedCheck);
251     SensorContextTester context = SensorContextTester.create(new File(""));
252 
253     DefaultFileSystem fileSystem = context.fileSystem();
254     File file = new File("file.java");
255     TestInputFileBuilder inputFile = new TestInputFileBuilder("", "file.java");
256     inputFile.setLines(45);
257     int[] lineStartOffsets = new int[45];
258     lineStartOffsets[35] = 12;
259     lineStartOffsets[42] = 1;
260     int lastValidOffset = 420;
261     inputFile.setOriginalLineStartOffsets(lineStartOffsets);
262     inputFile.setOriginalLineEndOffsets(computeLineEndOffsets(lineStartOffsets, lastValidOffset));
263     inputFile.setLastValidOffset(lastValidOffset);
264     fileSystem.add(inputFile.build());
265 
266     when(this.checks.ruleKey(any(JavaCheck.class))).thenReturn(mock(RuleKey.class));
267 
268     SonarComponents sonarComponents = new SonarComponents(fileLinesContextFactory, fileSystem, null, null, checkFactory, new CheckRegistrar[] {
269       expectedRegistrar
270     });
271     sonarComponents.setSensorContext(context);
272 
273     sonarComponents.addIssue(file, expectedCheck, -5, "message on wrong line", null);
274     sonarComponents.addIssue(file, expectedCheck, 42, "message on line", 1);
275     sonarComponents.addIssue(new File("."), expectedCheck, 42, "message on line", 1);
276     sonarComponents.addIssue(new File("unknown_file"), expectedCheck, 42, "message on line", 1);
277     sonarComponents.reportIssue(new AnalyzerMessage(expectedCheck, file, 35, "other message", 0));
278 
279     assertThat(context.allIssues()).hasSize(3);
280 
281     RecognitionException parseError = new RecognitionException(new LexerException("parse error"));
282 
283     context.setRuntime(SonarRuntimeImpl.forSonarLint(V6_7));
284     assertThat(sonarComponents.reportAnalysisError(parseError, file)).isTrue();
285 
286     context.setRuntime(SonarRuntimeImpl.forSonarQube(V6_7, SonarQubeSide.SCANNER));
287     assertThat(sonarComponents.reportAnalysisError(parseError, file)).isFalse();
288 
289   }

291   @Test
292   public void fail_on_empty_location() {
293     JavaCheck expectedCheck = new CustomCheck();
294     CheckRegistrar expectedRegistrar = getRegistrar(expectedCheck);
295     RuleKey ruleKey = RuleKey.of("MyRepo", "CustomCheck");
296 
297     File file = new File("file.java");
298     DefaultInputFile inputFile = new TestInputFileBuilder("", file.getPath())
299     .initMetadata("class A {\n"
300       + "  void foo() {\n"
301       + "    System.out.println();\n"
302       + "  }\n"
303       + "}\n").build();
304 
305     SensorContextTester context = SensorContextTester.create(new File(""));
306     DefaultFileSystem fileSystem = context.fileSystem();
307     SonarComponents sonarComponents = new SonarComponents(fileLinesContextFactory, fileSystem, null, null, checkFactory, new CheckRegistrar[] {expectedRegistrar});
308     sonarComponents.setSensorContext(context);
309 
310     AnalyzerMessage.TextSpan emptyTextSpan = new AnalyzerMessage.TextSpan(3, 10, 3, 10);
311     AnalyzerMessage analyzerMessageEmptyLocation = new AnalyzerMessage(expectedCheck, file, emptyTextSpan, "message", 0);
312 
313     assertThatThrownBy(() -> sonarComponents.reportIssue(analyzerMessageEmptyLocation, ruleKey, inputFile, 0.0))
314       .isInstanceOf(IllegalStateException.class).hasMessageContaining("Issue location should not be empty");
315     assertThat(context.allIssues()).isEmpty();
316 
317     AnalyzerMessage.TextSpan nonEmptyTextSpan = new AnalyzerMessage.TextSpan(3, 10, 3, 15);
318     AnalyzerMessage analyzerMessageValidLocation = new AnalyzerMessage(expectedCheck, file, nonEmptyTextSpan, "message", 0);
319     sonarComponents.reportIssue(analyzerMessageValidLocation, ruleKey, inputFile, 0.0);
320     assertThat(context.allIssues()).isNotEmpty();
321   }

382       fail("reading file content should have failed", e);

385       sonarComponents.fileLines(unknownFile);

387     } catch (AnalysisException e) {
388       assertThat(e).hasMessage("Unable to read file unknown_file.java").hasCauseInstanceOf(NoSuchFileException.class);
389     } catch (Exception e) {
390       fail("reading file content should have failed");
391     }

420 
421     AnalysisError analysisError;

425       analysisError = new AnalysisError(iae, "/abcde/abcde/abcde/abcde/abcde/abcde/abcde/abcde/abcde/abcde/abcde/abcde/abcde/some_very/long/path/FileInError.java", AnalysisError.Kind.PARSE_ERROR);

427 
428     for (int i = 0; i < 200_000; i++) {
429       sonarComponents.addAnalysisError(analysisError);
430     }
431 
432     sonarComponents.saveAnalysisErrors();
433 
434     return sensorContext.measure("projectKey", "sonarjava_feedback");
435   }

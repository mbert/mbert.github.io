20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ArrayListMultimap;
23  import com.google.common.collect.Multimap;
24  import java.util.Collection;
25  import java.util.HashMap;
26  import java.util.List;
27  import java.util.Map;
28  import java.util.stream.Collectors;
29  import org.sonar.check.Rule;
30  import org.sonar.check.RuleProperty;
31  import org.sonar.java.model.ModifiersUtils;
32  import org.sonar.plugins.java.api.JavaFileScanner;
33  import org.sonar.plugins.java.api.JavaFileScannerContext;
34  import org.sonar.plugins.java.api.tree.AnnotationTree;
35  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
36  import org.sonar.plugins.java.api.tree.ExpressionTree;
37  import org.sonar.plugins.java.api.tree.LiteralTree;
38  import org.sonar.plugins.java.api.tree.MethodTree;
39  import org.sonar.plugins.java.api.tree.Modifier;
40  import org.sonar.plugins.java.api.tree.Tree;
41  import org.sonar.plugins.java.api.tree.VariableTree;
42  
43  @Rule(key = "S1192")
44  public class StringLiteralDuplicatedCheck extends BaseTreeVisitor implements JavaFileScanner {
45  
46    private static final int DEFAULT_THRESHOLD = 3;
47  
48    // String literals include quotes, so this means length 5 as defined in RSPEC
49    private static final int MINIMAL_LITERAL_LENGTH = 7;
50  
51    @RuleProperty(
52      key = "threshold",
53      description = "Number of times a literal must be duplicated to trigger an issue",
54      defaultValue = "" + DEFAULT_THRESHOLD)
55    public int threshold = DEFAULT_THRESHOLD;
56  
57    private final Multimap<String, LiteralTree> occurrences = ArrayListMultimap.create();
58    private final Map<String, VariableTree> constants = new HashMap<>();
59  
60    @Override
61    public void scanFile(JavaFileScannerContext context) {
62      occurrences.clear();
63      constants.clear();
64      scan(context.getTree());
65      for (String entry : occurrences.keySet()) {
66        Collection<LiteralTree> literalTrees = occurrences.get(entry);
67        int literalOccurrence = literalTrees.size();
68        if (constants.containsKey(entry)) {
69          VariableTree constant = constants.get(entry);
70          List<LiteralTree> duplications = literalTrees.stream().filter(literal -> literal.parent() != constant).collect(Collectors.toList());
71          context.reportIssue(this, duplications.iterator().next(),
72            "Use already-defined constant '" + constant.simpleName() + "' instead of duplicating its value here.",
73            secondaryLocations(duplications.subList(1, duplications.size())), literalOccurrence);
74        } else if (literalOccurrence >= threshold) {
75          context.reportIssue(
76            this,
77            literalTrees.iterator().next(),
78            "Define a constant instead of duplicating this literal " + entry + " " + literalOccurrence + " times.",
79            secondaryLocations(literalTrees), literalOccurrence);
80        }
81      }
82    }
83  
84    private static List<JavaFileScannerContext.Location> secondaryLocations(Collection<LiteralTree> literalTrees) {
85      return literalTrees.stream().map(element -> new JavaFileScannerContext.Location("Duplication", element)).collect(Collectors.toList());
86    }
87  
88    @Override
89    public void visitLiteral(LiteralTree tree) {
90      if (tree.is(Tree.Kind.STRING_LITERAL)) {
91        String literal = tree.value();
92        if (literal.length() >= MINIMAL_LITERAL_LENGTH) {
93          occurrences.put(literal, tree);
94        }
95      }
96    }
97  
98    @Override
99    public void visitVariable(VariableTree tree) {
100     ExpressionTree initializer = tree.initializer();
101     if (initializer != null && initializer.is(Tree.Kind.STRING_LITERAL)
102       && ModifiersUtils.hasModifier(tree.modifiers(), Modifier.STATIC)
103       && ModifiersUtils.hasModifier(tree.modifiers(), Modifier.FINAL)) {
104       constants.putIfAbsent(((LiteralTree) initializer).value(), tree);
105       return;
106     }
107     super.visitVariable(tree);
108   }
109 
110   @Override
111   public void visitMethod(MethodTree tree) {
112     if(ModifiersUtils.hasModifier(tree.modifiers(), Modifier.DEFAULT)) {
113       //Ignore default methods to avoid catch-22 with S1214
114       return;
115     }
116     super.visitMethod(tree);
117   }
118 
119   @Override
120   public void visitAnnotation(AnnotationTree annotationTree) {
121     //Ignore literals within annotation
122   }
123 }

269       ProgramState programState = execute(new Instruction(opcode), initState);

274       assertThat(result.values).hasSize(1);
275       assertThat(result.values.get(0)).isNotEqualTo(array);
276       assertThat(result.values.get(0)).isNotEqualTo(index);
277       if (opcode == Opcodes.DALOAD || opcode == Opcodes.LALOAD) {
278         assertThat(isDoubleOrLong(programState, result.values.get(0))).isTrue();
279       }
280     }

1065     ProgramState.Pop pop = ps.unstackValue(constraints.length);
1066     assertEmptyStack(pop.state);
1067     assertThat(pop.valuesAndSymbols).hasSize(constraints.length);
1068     List<SymbolicValue> symbolicValues = pop.values;
1069     int idx = 0;
1070     for (SymbolicValue sv : symbolicValues) {
1071       ConstraintsByDomain constraintsByDomain = ps.getConstraints(sv);
1072       if (constraintsByDomain != null) {
1073         constraintsByDomain = constraintsByDomain.remove(BytecodeEGWalker.StackValueCategoryConstraint.class);
1074       }
1075       for (Constraint expectedConstraint : constraints[idx]) {
1076         if (expectedConstraint != null) {
1077           Class<? extends Constraint> expectedConstraintDomain = expectedConstraint.getClass();
1078           Constraint constraint = constraintsByDomain.get(expectedConstraintDomain);
1079           assertThat(constraint).isEqualTo(expectedConstraint);
1080           constraintsByDomain = constraintsByDomain.remove(expectedConstraintDomain);
1081         } else {
1082           assertThat(constraintsByDomain).isNull();
1083         }
1084       }
1085       if (constraintsByDomain != null) {
1086         assertThat(constraintsByDomain.isEmpty()).isTrue();
1087       }
1088       idx++;
1089     }

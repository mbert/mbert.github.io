110     // Each list of FunctionalInterface has to be sorted ascending by number of parametrized types so that smallest number
111     // of parametrized types take precedence. For example UnaryOperator<String> and Function<String,String> are equivalent,
112     // but UnaryOperator is preferred.
113     STD_INTERFACE_BY_PARAMETER_COUNT.values().forEach(list -> list.sort((a, b) -> Integer.compare(a.getGenericTypeCount(), b.getGenericTypeCount())));

116   private static void registerInterface(String name, String returnType, String... parameters) {
117     FunctionalInterface functionalInterface = new FunctionalInterface(name, returnType, parameters);
118     STD_INTERFACE_NAMES.add(functionalInterface.getName());
119     STD_INTERFACE_BY_PARAMETER_COUNT.computeIfAbsent(functionalInterface.getParameterCount(), key -> new ArrayList<>()).add(functionalInterface);
120   }
121 
122   private static MethodMatcher methodMatcherWithName(String name, String... parameters) {
123     MethodMatcher methodMatcher = MethodMatcher.create().typeDefinition(TypeCriteria.anyType()).name(name);
124     if(parameters.length == 0) {
125       methodMatcher.withoutParameter();
126     }
127     for (String parameter : parameters) {
128       methodMatcher.addParameter(parameter);
129     }
130     return methodMatcher;
131   }

133   @Override
134   public List<Tree.Kind> nodesToVisit() {
135     return Collections.singletonList(Tree.Kind.INTERFACE);
136   }
137 
138   @Override
139   public void visitNode(Tree tree) {
140     if (!hasSemantic()) {
141       return;
142     }
143     ClassTree classTree = (ClassTree) tree;
144     // classTree.simpleName() never null for Tree.Kind.INTERFACE
145     IdentifierTree issueLocation = classTree.simpleName();
146     // The question "Why we raise issue only for interface annotated with @FunctionalInterface?"
147     // is discussed in comments of https://jira.sonarsource.com/browse/SONARJAVA-504
148     Optional.of(classTree)
149       .filter(StandardFunctionalInterfaceCheck::isFunctionalInterface)
150       .filter(StandardFunctionalInterfaceCheck::isNonStandardFunctionalInterface)
151       .filter(StandardFunctionalInterfaceCheck::hasNoExtension)
152       .flatMap(StandardFunctionalInterfaceCheck::lookupFunctionalMethod)
153       .flatMap(StandardFunctionalInterfaceCheck::lookupMatchingStandardInterface)
154       .ifPresent(standardInterface -> reportIssue(issueLocation, buildIssueMessage(classTree, standardInterface)));
155   }
156 
157   private static boolean isFunctionalInterface(ClassTree tree) {
158     return tree.symbol().metadata().isAnnotatedWith("java.lang.FunctionalInterface");
159   }
160 
161   private static boolean isNonStandardFunctionalInterface(ClassTree tree) {
162     return !STD_INTERFACE_NAMES.contains(tree.symbol().type().fullyQualifiedName());
163   }
164 
165   private static boolean hasNoExtension(ClassTree tree) {
166     return tree.superInterfaces().isEmpty();
167   }
168 
169   private static Optional<MethodSymbol> lookupFunctionalMethod(ClassTree interfaceTree) {
170     return interfaceTree.symbol().memberSymbols().stream()
171         .filter(Symbol::isMethodSymbol)
172         .map(MethodSymbol.class::cast)
173         .filter(MethodSymbol::isAbstract)
174         .filter(StandardFunctionalInterfaceCheck::isNotObjectMethod)
175         .findFirst();
176   }
177 
178   private static Optional<String> lookupMatchingStandardInterface(MethodSymbol functionalMethod) {
179     MethodTree declaration = functionalMethod.declaration();
180     if (!functionalMethod.thrownTypes().isEmpty() || (declaration != null && !declaration.typeParameters().isEmpty())) {
181       return Optional.empty();
182     }
183     Type returnType = declaration != null ? declaration.returnType().symbolType() : functionalMethod.returnType().type();
184     return STD_INTERFACE_BY_PARAMETER_COUNT.getOrDefault(functionalMethod.parameterTypes().size(), Collections.emptyList()).stream()
185         .map(standardInterface -> standardInterface.matchingSpecialization(functionalMethod, returnType))
186         .filter(Objects::nonNull)
187         .findFirst();
188   }
189 
190   private static String buildIssueMessage(ClassTree interfaceTree, String standardInterface) {
191     if (interfaceTree.members().size() <= 1) {
192       return "Drop this interface in favor of \"" + standardInterface + "\".";
193     } else {
194       return "Make this interface extend \"" + standardInterface + "\" and remove the functional method declaration.";
195     }
196   }
197 
198   private static boolean isNotObjectMethod(MethodSymbol method) {
199     MethodTree declaration = method.declaration();
200     return declaration == null || !OBJECT_METHODS.anyMatch(declaration);
201   }
202 
203   private static class FunctionalInterface {
204 
205     private final String name;
206     private final List<String> genericTypes;
207     private final String returnType;
208     private final List<String> parameters;
209 
210     private FunctionalInterface(String name, String returnType, String... parameters) {
211       int genericStart = name.indexOf('<');
212       if (genericStart != -1) {
213         this.name = name.substring(0, genericStart);
214         this.genericTypes = Arrays.asList(name.substring(genericStart + 1, name.length() - 1).split(","));
215       } else {
216         this.name = name;
217         this.genericTypes = Collections.emptyList();
218       }
219       this.returnType = returnType;
220       this.parameters = Arrays.asList(parameters);
221     }
222 
223     private String getName() {
224       return name;
225     }
226 
227     private int getGenericTypeCount() {
228       return genericTypes.size();
229     }
230 
231     private int getParameterCount() {
232       return parameters.size();
233     }
234 
235     @CheckForNull
236     private String matchingSpecialization(MethodSymbol method, Type actualReturnType) {
237       Map<String, String> genericTypeMapping = genericTypes.isEmpty() ? Collections.emptyMap() : new HashMap<>();
238       String expectedReturnType = convertGenericType(returnType, actualReturnType, genericTypeMapping);
239       if (!expectedReturnType.equals(actualReturnType.fullyQualifiedName())) {
240         return null;
241       }
242       List<Type> methodParameters = method.parameterTypes();
243       for (int i = 0; i < parameters.size(); i++) {
244         Type actualType = methodParameters.get(i);
245         String expectedType = convertGenericType(parameters.get(i), actualType, genericTypeMapping);
246         if (!expectedType.equals(actualType.fullyQualifiedName())) {
247           return null;
248         }
249       }
250       return buildSpecializationName(genericTypeMapping);
251     }
252 
253     private String convertGenericType(String expectedType, Type actualType, Map<String, String> genericTypeMapping) {
254       if (genericTypes.isEmpty() || !genericTypes.contains(expectedType)) {
255         return expectedType;
256       }
257       String convertedType = genericTypeMapping.get(expectedType);
258       if (convertedType == null) {
259         if (actualType.isPrimitive() || actualType.isVoid() || actualType.isArray() || actualType.isUnknown()) {
260           return "!unknown!";
261         }
262         convertedType = actualType.fullyQualifiedName();
263         genericTypeMapping.put(expectedType, convertedType);
264       }
265       return convertedType;
266     }
267 
268     private String buildSpecializationName(Map<String, String> genericTypeMapping) {
269       if (genericTypes.isEmpty()) {
270         return name;
271       }
272       StringBuilder genericName = new StringBuilder();
273       genericName.append(name);
274       genericName.append('<');
275       boolean addComma = false;
276       for (String genericType : genericTypes) {
277         if (addComma) {
278           genericName.append(',');
279         } else {
280           addComma = true;
281         }
282         String typeName = genericTypeMapping.getOrDefault(genericType, genericType);
283         int packageEnd = typeName.lastIndexOf('.');
284         if (packageEnd != -1) {
285           typeName = typeName.substring(packageEnd + 1);
286         }
287         genericName.append(typeName);
288       }
289       genericName.append('>');
290       return genericName.toString();
291     }

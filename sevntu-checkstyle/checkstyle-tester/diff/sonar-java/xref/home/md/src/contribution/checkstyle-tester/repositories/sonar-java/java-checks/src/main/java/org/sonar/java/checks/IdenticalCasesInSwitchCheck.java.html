20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.HashMultimap;
23  import com.google.common.collect.Multimap;
24  import com.google.common.collect.SetMultimap;
25  
26  import org.sonar.check.Rule;
27  import org.sonar.java.model.SyntacticEquivalence;
28  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
29  import org.sonar.plugins.java.api.JavaFileScannerContext;
30  import org.sonar.plugins.java.api.tree.BlockTree;
31  import org.sonar.plugins.java.api.tree.CaseGroupTree;
32  import org.sonar.plugins.java.api.tree.IfStatementTree;
33  import org.sonar.plugins.java.api.tree.StatementTree;
34  import org.sonar.plugins.java.api.tree.SwitchStatementTree;
35  import org.sonar.plugins.java.api.tree.Tree;
36  
37  import java.util.ArrayList;
38  import java.util.Collections;
39  import java.util.Arrays;
40  import java.util.List;
41  
42  @Rule(key = "S1871")
43  public class IdenticalCasesInSwitchCheck extends IssuableSubscriptionVisitor {
44  
45    @Override
46    public List<Tree.Kind> nodesToVisit() {
47      return Arrays.asList(Tree.Kind.SWITCH_STATEMENT, Tree.Kind.IF_STATEMENT);
48    }
49  
50    @Override
51    public void visitNode(Tree node) {
52      if (node.is(Tree.Kind.SWITCH_STATEMENT)) {
53        SwitchStatementTree switchStatement = (SwitchStatementTree) node;
54        Multimap<CaseGroupTree, CaseGroupTree> identicalBranches = checkSwitchStatement(switchStatement);
55        boolean allBranchesSame = allBranchesSame(identicalBranches, switchStatement.cases().size());
56        boolean allBranchesSameWithoutDefault = allBranchesSame && !hasDefaultClause(switchStatement);
57        if (!allBranchesSame || allBranchesSameWithoutDefault) {
58          identicalBranches.asMap().forEach((first, others) -> {
59            if (!isTrivialCase(first.body()) || allBranchesSameWithoutDefault) {
60              others.forEach(other -> createIssue(other, issueMessage("case", first), first));
61            }
62          });
63        }
64      } else if (node.is(Tree.Kind.IF_STATEMENT) && !node.parent().is(Tree.Kind.IF_STATEMENT)) {
65        IfStatementTree ifStatement = (IfStatementTree) node;
66        IfElseChain ifElseChain = checkIfStatement(ifStatement);
67        reportIdenticalIfChainBranches(ifElseChain.branches, ifElseChain.totalBranchCount, hasElseClause(ifStatement));
68      }
69    }
70  
71    protected static boolean allBranchesSame(Multimap<? extends Tree, ? extends Tree> identicalBranches, int size) {
72      return identicalBranches.keySet().size() == 1 && identicalBranches.size() == size - 1;
73    }
74  
75    private static boolean isTrivialCase(List<StatementTree> body) {
76      return body.size() == 1 || (body.size() == 2 && body.get(1).is(Tree.Kind.BREAK_STATEMENT));
77    }
78  
79    protected Multimap<CaseGroupTree, CaseGroupTree> checkSwitchStatement(SwitchStatementTree node) {
80      SetMultimap<CaseGroupTree, CaseGroupTree> identicalBranches = HashMultimap.create();
81      int index = 0;
82      List<CaseGroupTree> cases = node.cases();
83      for (CaseGroupTree caseGroupTree : cases) {
84        index++;
85        if (identicalBranches.containsValue(caseGroupTree)) {
86          continue;
87        }
88        for (int i = index; i < cases.size(); i++) {
89          if (SyntacticEquivalence.areEquivalent(caseGroupTree.body(), cases.get(i).body())) {
90            identicalBranches.put(caseGroupTree, cases.get(i));
91          }
92        }
93      }
94      return identicalBranches;
95    }
96  
97    protected static class IfElseChain {
98      Multimap<StatementTree, StatementTree> branches = HashMultimap.create();
99      int totalBranchCount;
100   }
101 
102   protected static IfElseChain checkIfStatement(IfStatementTree node) {
103     IfElseChain ifElseChain = new IfElseChain();
104     ifElseChain.totalBranchCount = 1;
105     List<StatementTree> allBranches = new ArrayList<>();
106     allBranches.add(node.thenStatement());
107     StatementTree elseStatement = node.elseStatement();
108     while (elseStatement != null && elseStatement.is(Tree.Kind.IF_STATEMENT)) {
109       IfStatementTree ifStatement = (IfStatementTree) elseStatement;
110       allBranches.add(ifStatement.thenStatement());
111       elseStatement = ifStatement.elseStatement();
112     }
113     if (elseStatement != null) {
114       allBranches.add(elseStatement);
115     }
116     return collectIdenticalBranches(allBranches);
117   }
118 
119   private static IfElseChain collectIdenticalBranches(List<StatementTree> allBranches) {
120     IfElseChain ifElseChain = new IfElseChain();
121     for (int i = 0; i < allBranches.size(); i++) {
122       if (ifElseChain.branches.containsValue(allBranches.get(i))) {
123         continue;
124       }
125       for (int j = i + 1; j < allBranches.size(); j++) {
126         if (SyntacticEquivalence.areEquivalent(allBranches.get(i), allBranches.get(j))) {
127           ifElseChain.branches.put(allBranches.get(i), allBranches.get(j));
128         }
129       }
130     }
131     ifElseChain.totalBranchCount = allBranches.size();
132     return ifElseChain;
133   }
134 
135   private void reportIdenticalIfChainBranches(Multimap<StatementTree, StatementTree> identicalBranches, int totalBranchCount, boolean withElseClause) {
136     boolean allBranchesSame = allBranchesSame(identicalBranches, totalBranchCount);
137     boolean allBranchesSameWithoutElse = allBranchesSame && !withElseClause;
138     if (!allBranchesSame || allBranchesSameWithoutElse) {
139       identicalBranches.asMap().forEach((first, others) -> {
140         if (!isTrivialIfStatement(first) || allBranchesSameWithoutElse) {
141           others.forEach(other -> createIssue(other, issueMessage("branch", first), first));
142         }
143       });
144     }
145   }
146 
147   private static boolean isTrivialIfStatement(StatementTree node) {
148     return !node.is(Tree.Kind.BLOCK) || ((BlockTree) node).body().size() <= 1;
149   }
150 
151   protected static boolean hasDefaultClause(SwitchStatementTree switchStatement) {
152     return switchStatement.cases().stream()
153       .flatMap(caseGroupTree -> caseGroupTree.labels().stream())
154       .anyMatch(caseLabelTree -> caseLabelTree.caseOrDefaultKeyword().text().equals("default"));
155   }
156 
157   protected static boolean hasElseClause(IfStatementTree ifStatement) {
158     StatementTree elseStatement = ifStatement.elseStatement();
159     while (elseStatement != null && elseStatement.is(Tree.Kind.IF_STATEMENT)) {
160       elseStatement = ((IfStatementTree) elseStatement).elseStatement();
161     }
162     return elseStatement != null;
163   }
164 
165   private void createIssue(Tree node, String message, Tree secondary) {
166     reportIssue(node, message, Collections.singletonList(new JavaFileScannerContext.Location("Original", secondary)), null);
167   }
168 
169   private static String issueMessage(String type, Tree node) {
170     return "This " + type + "'s code block is the same as the block for the " + type + " on line " + node.firstToken().line() + ".";
171   }

57    @Test
58    public void spotbugs_rules_definition() {
59      RulesDefinition.Context context = new RulesDefinition.Context();
60      new ExternalRulesDefinition(SpotBugsSensor.RULE_LOADER, SpotBugsSensor.SPOTBUGS_KEY).define(context);
61      new ExternalRulesDefinition(SpotBugsSensor.FINDSECBUGS_LOADER, SpotBugsSensor.FINDSECBUGS_KEY).define(context);
62      assertThat(context.repositories()).hasSize(2);
63  
64      RulesDefinition.Repository repository = context.repository("external_spotbugs");
65      assertThat(repository.name()).isEqualTo("SpotBugs");
66      assertThat(repository.language()).isEqualTo("java");
67      assertThat(repository.isExternal()).isEqualTo(true);
68      assertThat(repository.rules().size()).isEqualTo(468);
69  
70      RulesDefinition.Rule rule = repository.rule("AM_CREATES_EMPTY_JAR_FILE_ENTRY");
71      assertThat(rule).isNotNull();
72      assertThat(rule.name()).isEqualTo("Bad practice - Creates an empty jar file entry");
73      assertThat(rule.type()).isEqualTo(RuleType.CODE_SMELL);
74      assertThat(rule.severity()).isEqualTo("MAJOR");
75      assertThat(rule.htmlDescription()).isEqualTo("See description of SpotBugs rule <code>AM_CREATES_EMPTY_JAR_FILE_ENTRY</code> at the " +
76        "<a href=\"https://spotbugs.readthedocs.io/en/latest/bugDescriptions.html#AM_CREATES_EMPTY_JAR_FILE_ENTRY\">SpotBugs website</a>.");
77      assertThat(rule.debtRemediationFunction().baseEffort()).isEqualTo("5min");
78  
79      RulesDefinition.Repository findsecbugsRepo = context.repository("external_findsecbugs");
80      assertThat(findsecbugsRepo.name()).isEqualTo("FindSecBugs");
81      assertThat(findsecbugsRepo.language()).isEqualTo("java");
82      assertThat(findsecbugsRepo.isExternal()).isEqualTo(true);
83      repository = context.repository("external_findsecbugs");
84      assertThat(repository.rules().size()).isEqualTo(128);
85    }

103   @Test
104   public void issues_with_sonarqube_72() throws IOException {
105     List<ExternalIssue> externalIssues = executeSensorImporting(7, 2, "spotbugsXml.xml");
106     assertThat(externalIssues).hasSize(1);
107 
108     ExternalIssue first = externalIssues.get(0);
109     assertThat(first.primaryLocation().inputComponent().key()).isEqualTo("spotbugs-project:src/main/java/org/myapp/Main.java");
110     assertThat(first.engineId()).isEqualTo("spotbugs");
111     assertThat(first.ruleId()).isEqualTo("HE_EQUALS_USE_HASHCODE");
112     assertThat(first.ruleKey().rule()).isEqualTo("HE_EQUALS_USE_HASHCODE");
113     assertThat(first.ruleKey().repository()).isEqualTo("external_spotbugs");
114     assertThat(first.type()).isEqualTo(RuleType.CODE_SMELL);
115     assertThat(first.severity()).isEqualTo(Severity.MAJOR);
116     assertThat(first.remediationEffort().longValue()).isEqualTo(5L);
117     assertThat(first.primaryLocation().message()).isEqualTo("org.myapp.Main defines equals and uses Object.hashCode()");
118     assertThat(first.primaryLocation().textRange().start().line()).isEqualTo(6);
119 
120     assertThat(logTester.logs(LoggerLevel.ERROR)).isEmpty();
121   }

186   @Test
187   public void issues_when_xml_file_has_errors() throws IOException {
188     List<ExternalIssue> externalIssues = executeSensorImporting(7, 2, "spotbugsXml-with-errors.xml");
189     assertThat(externalIssues).hasSize(1);
190 
191     ExternalIssue first = externalIssues.get(0);
192     assertThat(first.primaryLocation().inputComponent().key()).isEqualTo("spotbugs-project:src/main/java/org/myapp/Main.java");
193     assertThat(first.ruleKey().rule()).isEqualTo("UNKNOWN_RULE");
194     assertThat(first.type()).isEqualTo(RuleType.CODE_SMELL);
195     assertThat(first.severity()).isEqualTo(Severity.MAJOR);
196     assertThat(first.primaryLocation().message()).isEqualTo("Message for unknown rule.");
197     assertThat(first.primaryLocation().textRange()).isNull();
198 
199     assertThat(logTester.logs(LoggerLevel.ERROR)).isEmpty();
200     assertThat(onlyOneLogElement(logTester.logs(LoggerLevel.WARN)))
201       .startsWith("No input file found for '")
202       .endsWith("not-existing-file.java'. No SpotBugs issues will be imported on this file.");
203     assertThat(logTester.logs(LoggerLevel.DEBUG)).containsExactlyInAnyOrder(
204       "Unexpected empty 'BugCollection/BugInstance/@type'.",
205       "Unexpected empty 'BugCollection/BugInstance/SourceLine/@sourcepath' for bug 'HE_EQUALS_USE_HASHCODE'.",
206       "Unexpected empty 'BugCollection/BugInstance/LongMessage/text()' for bug 'NO_MESSAGE'");
207   }

217   private List<ExternalIssue> executeSensorImporting(int majorVersion, int minorVersion, @Nullable String fileName) throws IOException {
218     SensorContextTester context = ExternalReportTestUtils.createContext(PROJECT_DIR, majorVersion, minorVersion);
219     if (fileName != null) {
220       File reportFile = ExternalReportTestUtils.generateReport(PROJECT_DIR, tmp, fileName);
221       context.settings().setProperty("sonar.java.spotbugs.reportPaths", reportFile.getPath());
222     }
223     spotBugsSensor.execute(context);
224     return new ArrayList<>(context.allExternalIssues());
225   }

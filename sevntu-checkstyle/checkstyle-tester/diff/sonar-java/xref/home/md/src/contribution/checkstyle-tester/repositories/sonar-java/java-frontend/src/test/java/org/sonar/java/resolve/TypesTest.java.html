32  public class TypesTest {

34    private Symbols symbols = new Symbols(new BytecodeCompleter(new SquidClassLoader(Collections.emptyList()), new ParametrizedTypeCache()));
35    private Types types = new Types();
36  
37    @Test
38    public void isSubtype() {
39      // byte is direct subtype of short
40      shouldNotBeSubtype(symbols.byteType, Arrays.asList(symbols.booleanType, symbols.charType));
41      shouldBeSubtype(symbols.byteType, Arrays.asList(symbols.byteType, symbols.shortType, symbols.intType, symbols.longType, symbols.floatType, symbols.doubleType));
42      // char is direct subtype of int
43      shouldNotBeSubtype(symbols.charType, Arrays.asList(symbols.booleanType, symbols.byteType, symbols.shortType));
44      shouldBeSubtype(symbols.charType, Arrays.asList(symbols.charType, symbols.intType, symbols.longType, symbols.floatType, symbols.doubleType));
45      // short is direct subtype of int
46      shouldNotBeSubtype(symbols.shortType, Arrays.asList(symbols.booleanType, symbols.byteType, symbols.charType));
47      shouldBeSubtype(symbols.shortType, Arrays.asList(symbols.shortType, symbols.intType, symbols.longType, symbols.floatType, symbols.doubleType));
48      // int is direct subtype of long
49      shouldNotBeSubtype(symbols.intType, Arrays.asList(symbols.booleanType, symbols.byteType, symbols.charType, symbols.shortType));
50      shouldBeSubtype(symbols.intType, Arrays.asList(symbols.intType, symbols.longType, symbols.floatType, symbols.doubleType));
51      // long is direct subtype of float
52      shouldNotBeSubtype(symbols.longType, Arrays.asList(symbols.booleanType, symbols.byteType, symbols.charType, symbols.shortType, symbols.intType));
53      shouldBeSubtype(symbols.longType, Arrays.asList(symbols.longType, symbols.floatType, symbols.doubleType));
54      // float is direct subtype of double
55      shouldNotBeSubtype(symbols.floatType, Arrays.asList(symbols.booleanType, symbols.byteType, symbols.charType, symbols.shortType, symbols.intType, symbols.longType));
56      shouldBeSubtype(symbols.floatType, Arrays.asList(symbols.floatType, symbols.doubleType));
57      // double
58      shouldNotBeSubtype(symbols.doubleType, Arrays.asList(symbols.booleanType, symbols.byteType, symbols.charType, symbols.shortType, symbols.intType, symbols.longType, symbols.floatType));
59      shouldBeSubtype(symbols.doubleType, Arrays.asList(symbols.doubleType));
60      // boolean
61      shouldNotBeSubtype(symbols.booleanType, Arrays.asList(symbols.byteType, symbols.charType, symbols.shortType, symbols.intType, symbols.longType, symbols.floatType, symbols.doubleType));
62      shouldBeSubtype(symbols.booleanType, Arrays.asList(symbols.booleanType));
63  
64      // TODO test void
65  
66      // null
67      ArrayJavaType arrayTypeInt = new ArrayJavaType(symbols.intType, symbols.arrayClass);
68      ArrayJavaType arrayTypeShort = new ArrayJavaType(symbols.shortType, symbols.arrayClass);
69      shouldBeSubtype(arrayTypeShort, Arrays.<JavaType>asList(arrayTypeShort, arrayTypeInt));
70      shouldNotBeSubtype(symbols.nullType, Arrays.asList(symbols.booleanType, symbols.byteType, symbols.charType, symbols.shortType, symbols.intType, symbols.longType, symbols.floatType, symbols.doubleType));
71      shouldBeSubtype(symbols.nullType, Arrays.asList(symbols.nullType, arrayTypeInt, symbols.objectType));
72      shouldBeSubtype(arrayTypeInt, Arrays.asList(symbols.objectType));
73      JavaSymbol.TypeJavaSymbol typeSymbol = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, "MyType", symbols.defaultPackage);
74      ClassJavaType classType = (ClassJavaType) typeSymbol.type;
75      classType.interfaces = new ArrayList<>();
76      JavaSymbol.TypeJavaSymbol subtypeSymbol = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, "MySubtype", symbols.defaultPackage);
77      ClassJavaType subClassType = (ClassJavaType) subtypeSymbol.type;
78      subClassType.supertype = classType;
79      subClassType.interfaces = new ArrayList<>();
80      shouldBeSubtype(subClassType, Arrays.<JavaType>asList(classType, subClassType));
81  
82    }
83  
84    @Test
85    public void array_types_equality() throws Exception {
86      ArrayJavaType arrayInt= new ArrayJavaType(symbols.intType, symbols.arrayClass);
87      ArrayJavaType arrayInt2= new ArrayJavaType(symbols.intType, symbols.arrayClass);
88      ArrayJavaType arrayBoolean = new ArrayJavaType(symbols.booleanType, symbols.arrayClass);
89      assertThat(arrayInt.equals(arrayInt2)).isTrue();
90      assertThat(arrayInt2.equals(arrayInt)).isTrue();
91      assertThat(arrayInt2.equals(arrayBoolean)).isFalse();
92      assertThat(arrayInt2.equals(arrayInt2)).isTrue();
93      assertThat(arrayInt2.equals(null)).isFalse();
94      assertThat(arrayInt2.equals(symbols.charType)).isFalse();
95  
96    }
97  
98    private void shouldNotBeSubtype(JavaType t, List<JavaType> s) {
99      for (JavaType type : s) {
100       assertThat(types.isSubtype(t, type)).as(t + " is subtype of " + type).isFalse();
101     }
102   }
103 
104   private void shouldBeSubtype(JavaType t, List<JavaType> s) {
105     for (JavaType type : s) {
106       assertThat(types.isSubtype(t, type)).as(t + " is subtype of " + type).isTrue();
107     }
108   }

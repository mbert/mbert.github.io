20  package org.sonar.java.resolve;
21  
22  import org.sonar.plugins.java.api.semantic.Type;
23  
24  public class WildCardType extends JavaType {
25  
26    public enum BoundType {
27      UNBOUNDED("?"),
28      SUPER("? super "),
29      EXTENDS("? extends ");
30  
31      private final String name;
32  
33      BoundType(String name) {
34        this.name = name;
35      }
36  
37      @Override
38      public String toString() {
39        return name;
40      }
41    }
42  
43    final JavaType bound;
44    final BoundType boundType;
45  
46    public WildCardType(JavaType bound, BoundType boundType) {
47      super(WILDCARD, new JavaSymbol.WildcardSymbol(boundType == BoundType.UNBOUNDED ? boundType.toString() : (boundType + bound.symbol.name())));
48      this.bound = bound;
49      this.boundType = boundType;
50      this.symbol.type = this;
51    }
52  
53    @Override
54    public boolean isSubtypeOf(String fullyQualifiedName) {
55      return "java.lang.Object".equals(fullyQualifiedName) || (boundType == BoundType.EXTENDS && bound.isSubtypeOf(fullyQualifiedName));
56    }
57  
58    @Override
59    public boolean isSubtypeOf(Type superType) {
60      if (((JavaType) superType).isTagged(WILDCARD)) {
61        WildCardType superTypeWildcard = (WildCardType) superType;
62        JavaType superTypeBound = superTypeWildcard.bound;
63        switch (superTypeWildcard.boundType) {
64          case UNBOUNDED:
65            return true;
66          case SUPER:
67            return boundType == BoundType.SUPER && superTypeBound.isSubtypeOf(bound);
68          case EXTENDS:
69            return boundType != BoundType.SUPER && bound.isSubtypeOf(superTypeBound);
70        }
71      }
72      return "java.lang.Object".equals(superType.fullyQualifiedName()) || (boundType == BoundType.EXTENDS && bound.isSubtypeOf(superType));
73    }
74  
75    public boolean isSubtypeOfBound(JavaType type) {
76      switch (boundType) {
77        case SUPER:
78          return bound.isSubtypeOf(type);
79        case EXTENDS:
80          return !boundIsTypeVarAndNotType(type) && type.isSubtypeOf(bound);
81        case UNBOUNDED:
82        default:
83          return true;
84      }
85    }
86  
87    private boolean boundIsTypeVarAndNotType(JavaType type) {
88      return bound.isTagged(TYPEVAR) && !type.isTagged(TYPEVAR);
89    }
90  }

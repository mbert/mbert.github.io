144   @Test
145   public void test_yield_equality() {
146     MethodBehavior methodBehavior = mockMethodBehavior(1, false);
147     MethodYield yield = newMethodYield(methodBehavior);
148     MethodYield otherYield;
149 
150     assertThat(yield).isNotEqualTo(null);
151     assertThat(yield).isNotEqualTo(new Object());
152 
153     // same instance
154     assertThat(yield).isEqualTo(yield);
155 
156     // method behavior not taken into account
157     MethodYield myYield = newMethodYield(null);
158     assertThat(yield).isEqualTo(myYield);
159 
160     // node not taken into account
161     otherYield = newMethodYield(mockNode(), methodBehavior);
162     assertThat(yield).isEqualTo(otherYield);
163 
164     // same arity and constraints on parameters but exceptional path
165     otherYield = new ExceptionalYield(methodBehavior);
166     assertThat(yield).isNotEqualTo(otherYield);
167 
168     // same arity and constraints on parameters but happy path path
169     otherYield = new HappyPathYield(methodBehavior);
170     assertThat(yield).isNotEqualTo(otherYield);
171 
172     ConstraintsByDomain nullConstraint = ConstraintsByDomain.empty().put(ObjectConstraint.NULL);
173     MethodYield yield1 = newMethodYield(methodBehavior);
174     yield1.parametersConstraints.add(nullConstraint);
175     yield1.parametersConstraints.add(ConstraintsByDomain.empty());
176     yield1.parametersConstraints.add(nullConstraint);
177     MethodYield yield2 = newMethodYield(methodBehavior);
178     yield2.parametersConstraints.add(nullConstraint);
179     yield2.parametersConstraints.add(nullConstraint);
180     yield2.parametersConstraints.add(ConstraintsByDomain.empty());
181 
182     assertThat(yield1).isNotEqualTo(yield2);
183   }

225   @Test
226   public void constraints_on_varargs() throws Exception {
227     ActionParser<Tree> p = JavaParser.createParser();
228     CompilationUnitTree cut = (CompilationUnitTree) p.parse(new File("src/test/files/se/VarArgsYields.java"));
229     SemanticModel semanticModel = SemanticModel.createFor(cut, new SquidClassLoader(new ArrayList<>()));
230     SymbolicExecutionVisitor sev = new SymbolicExecutionVisitor(Lists.newArrayList(new SECheck[]{}), new BehaviorCache(new SquidClassLoader(new ArrayList<>())));
231     JavaFileScannerContext context = mock(JavaFileScannerContext.class);
232     when(context.getTree()).thenReturn(cut);
233     when(context.getSemanticModel()).thenReturn(semanticModel);
234     sev.scanFile(context);
235 
236     MethodSymbol methodSymbol = ((MethodTree) ((ClassTree) cut.types().get(0)).members().get(0)).symbol();
237     MethodBehavior mb = getMethodBehavior(sev, "varArgMethod");
238     List<MethodYield> yields = mb.yields();
239     assertThat(yields).hasSize(5);
240     assertThat(mb.exceptionalPathYields()).hasSize(4);
241 
242     MethodYield yield = mb.happyPathYields().findFirst().get();
243 
244     // check that we have NOT_NULL constraint on the first argument
245     assertThat(yield.parametersConstraints.get(0).get(ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
246     // check that we have NOT_NULL constraint on the variadic argument
247     assertThat(yield.parametersConstraints.get(1).get(ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
248 
249     List<IdentifierTree> usages = methodSymbol.usages();
250     assertThat(usages).hasSize(6);
251 
252     List<List<Type>> arguments = usages.stream()
253       .map(MethodYieldTest::getMethodIncoationArgumentsTypes)
254       .collect(Collectors.toList());
255 
256     ProgramState ps = ProgramState.EMPTY_STATE;
257     ProgramState psResult;
258     SymbolicValue svFirstArg = new SymbolicValue();
259     SymbolicValue svVarArg1 = new SymbolicValue();
260     SymbolicValue svVarArg2 = new SymbolicValue();
261     SymbolicValue svResult = new SymbolicValue();
262 
263     // apply constraint NotNull to parameter
264     Collection<ProgramState> arrayOfA = yield.statesAfterInvocation(Lists.newArrayList(svFirstArg, svVarArg1), arguments.get(0), ps, () -> svResult).collect(Collectors.toList());
265     assertThat(arrayOfA).hasSize(1);
266     psResult = arrayOfA.iterator().next();
267     assertThat(psResult.getConstraint(svFirstArg, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
268     assertThat(psResult.getConstraint(svVarArg1, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
269 
270     // apply constraint NotNull to parameter (B[] is a subtype of A[])
271     Collection<ProgramState> arrayOfB = yield.statesAfterInvocation(Lists.newArrayList(svFirstArg, svVarArg1), arguments.get(1), ps, () -> svResult).collect(Collectors.toList());
272     assertThat(arrayOfB).hasSize(1);
273     psResult = arrayOfB.iterator().next();
274     assertThat(psResult.getConstraint(svFirstArg, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
275     assertThat(psResult.getConstraint(svVarArg1, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
276     // no constraint, as 'a' is not an array
277     Collection<ProgramState> objectA = yield.statesAfterInvocation(Lists.newArrayList(svFirstArg, svVarArg1), arguments.get(2), ps, () -> svResult).collect(Collectors.toList());
278     assertThat(objectA).hasSize(1);
279     psResult = objectA.iterator().next();
280     assertThat(psResult.getConstraint(svFirstArg, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
281     assertThat(psResult.getConstraint(svVarArg1, ObjectConstraint.class)).isNull();
282 
283     // no constraint, as 'a' and 'b' can not receive the constraint of the array
284     Collection<ProgramState> objectsAandB = yield.statesAfterInvocation(Lists.newArrayList(svFirstArg, svVarArg1, svVarArg2), arguments.get(3), ps, () -> svResult).collect(Collectors.toList());
285     assertThat(objectsAandB).hasSize(1);
286     psResult = objectsAandB.iterator().next();
287     assertThat(psResult.getConstraint(svFirstArg, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
288     assertThat(psResult.getConstraint(svVarArg1, ObjectConstraint.class)).isNull();
289     assertThat(psResult.getConstraint(svVarArg2, ObjectConstraint.class)).isNull();
290 
291     // no param, we only learn something about the argument which is not variadic
292     Collection<ProgramState> noParam = yield.statesAfterInvocation(Lists.newArrayList(svFirstArg), arguments.get(4), ps, () -> svResult).collect(Collectors.toList());
293     assertThat(noParam).hasSize(1);
294     psResult = noParam.iterator().next();
295     assertThat(psResult.getConstraint(svFirstArg, ObjectConstraint.class)).isEqualTo(ObjectConstraint.NOT_NULL);
296     // null param, contradiction, no resulting program state
297     ps = ProgramState.EMPTY_STATE.addConstraint(svFirstArg, ObjectConstraint.NULL);
298     Collection<ProgramState> nullParam = yield.statesAfterInvocation(Lists.newArrayList(svFirstArg, svVarArg1), arguments.get(5), ps, () -> svResult).collect(Collectors.toList());
299     assertThat(nullParam).isEmpty();
300   }

302   private static List<Type> getMethodIncoationArgumentsTypes(IdentifierTree identifier) {
303     Tree tree = identifier;
304     while(!tree.is(Tree.Kind.METHOD_INVOCATION)) {
305       tree = tree.parent();
306     }
307     return ((MethodInvocationTree) tree).arguments().stream().map(ExpressionTree::symbolType).collect(Collectors.toList());
308   }

328   private static MethodYield buildMethodYield(int resultIndex, @Nullable ConstraintsByDomain resultConstraint) {
329     HappyPathYield methodYield = new HappyPathYield(mockMethodBehavior(1, false));
330     methodYield.parametersConstraints = new ArrayList<>();
331     methodYield.parametersConstraints.add(ConstraintsByDomain.empty());
332     methodYield.setResult(resultIndex, resultConstraint);

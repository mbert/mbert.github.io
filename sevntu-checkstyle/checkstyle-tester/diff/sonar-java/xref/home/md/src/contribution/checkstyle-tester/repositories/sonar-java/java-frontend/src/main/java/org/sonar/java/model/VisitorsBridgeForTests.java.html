20  package org.sonar.java.model;
21  
22  import com.google.common.annotations.VisibleForTesting;
23  import com.google.common.collect.Iterables;
24  import java.io.File;
25  import java.util.ArrayList;
26  import java.util.Arrays;
27  import java.util.Collections;
28  import java.util.HashSet;
29  import java.util.List;
30  import java.util.Set;
31  import java.util.stream.Collectors;
32  import javax.annotation.Nullable;
33  import org.sonar.api.batch.fs.InputFile;
34  import org.sonar.java.AnalyzerMessage;
35  import org.sonar.java.SonarComponents;
36  import org.sonar.java.resolve.SemanticModel;
37  import org.sonar.java.se.SymbolicExecutionMode;
38  import org.sonar.plugins.java.api.JavaCheck;
39  import org.sonar.plugins.java.api.JavaFileScanner;
40  import org.sonar.plugins.java.api.JavaFileScannerContext;
41  import org.sonar.plugins.java.api.JavaVersion;
42  import org.sonar.plugins.java.api.tree.CompilationUnitTree;
43  import org.sonar.plugins.java.api.tree.Tree;
44  
45  public class VisitorsBridgeForTests extends VisitorsBridge {
46  
47    private TestJavaFileScannerContext testContext;
48    private boolean enableSemantic = true;
49  
50  
51    @VisibleForTesting
52    public VisitorsBridgeForTests(JavaFileScanner visitor, SonarComponents sonarComponents) {
53      this(Collections.singletonList(visitor), new ArrayList<>(), sonarComponents);
54    }
55  
56    public VisitorsBridgeForTests(Iterable visitors, @Nullable SonarComponents sonarComponents) {
57      super(visitors, new ArrayList<>(), sonarComponents, SymbolicExecutionMode.DISABLED);
58      enableSemantic = false;
59    }
60  
61    public VisitorsBridgeForTests(Iterable visitors, List<File> projectClasspath, @Nullable SonarComponents sonarComponents) {
62      super(visitors, projectClasspath, sonarComponents, SymbolicExecutionMode.getMode(Iterables.<JavaCheck>toArray(visitors, JavaCheck.class), true));
63    }
64  
65    @Override
66    protected JavaFileScannerContext createScannerContext(CompilationUnitTree tree, SemanticModel semanticModel,
67                                                          SonarComponents sonarComponents, boolean failedParsing) {
68      SemanticModel model = enableSemantic ? semanticModel : null;
69      testContext = new TestJavaFileScannerContext(tree, currentFile, model, sonarComponents, javaVersion, failedParsing);
70      return testContext;
71    }
72  
73    public TestJavaFileScannerContext lastCreatedTestContext() {
74      return testContext;
75    }
76  
77    public static class TestJavaFileScannerContext extends DefaultJavaFileScannerContext {
78  
79      private final Set<AnalyzerMessage> issues = new HashSet<>();
80      private final SonarComponents sonarComponents;
81  
82      public TestJavaFileScannerContext(CompilationUnitTree tree, InputFile inputFile, SemanticModel semanticModel,
83                                        @Nullable SonarComponents sonarComponents, JavaVersion javaVersion, boolean failedParsing) {
84        super(tree, inputFile, semanticModel, sonarComponents, javaVersion, failedParsing);
85        this.sonarComponents = sonarComponents;
86      }
87  
88      public Set<AnalyzerMessage> getIssues() {
89        return issues;
90      }
91  
92      /**
93       * @deprecated since SonarJava 5.12 - Should only report on InputComponent
94       */
95      @Deprecated
96      @Override
97      public void addIssue(File file, JavaCheck javaCheck, int line, String message) {
98        issues.add(new AnalyzerMessage(javaCheck, sonarComponents.inputFromIOFileOrDirectory(file), line, message, 0));
99      }
100 
101     @Override
102     public void addIssueOnProject(JavaCheck javaCheck, String message) {
103       issues.add(new AnalyzerMessage(javaCheck, sonarComponents.project(), null, message, 0));
104     }
105 
106     @Override
107     public void addIssue(int line, JavaCheck javaCheck, String message, @Nullable Integer cost) {
108       issues.add(new AnalyzerMessage(javaCheck, getInputFile(), line, message, cost != null ? cost.intValue() : 0));
109     }
110 
111     @Override
112     public void reportIssue(JavaCheck javaCheck, Tree syntaxNode, String message, List<Location> secondary, @Nullable Integer cost) {
113       List<List<Location>> flows = secondary.stream().map(Collections::singletonList).collect(Collectors.toList());
114       issues.add(createAnalyzerMessage(getInputFile(), javaCheck, syntaxNode, null, message, flows, cost));
115     }
116 
117     @Override
118     public void reportIssue(JavaCheck javaCheck, Tree startTree, Tree endTree, String message) {
119       issues.add(createAnalyzerMessage(javaCheck, startTree, endTree, message, Collections.emptyList(), null));
120     }
121 
122     @Override
123     public void reportIssue(JavaCheck javaCheck, Tree startTree, Tree endTree, String message, List<Location> secondary, @Nullable Integer cost) {
124       issues.add(createAnalyzerMessage(javaCheck, startTree, endTree, message, secondary, cost));
125     }
126 
127     @Override
128     public void reportIssueWithFlow(JavaCheck javaCheck, Tree syntaxNode, String message, Iterable<List<Location>> flows, @Nullable Integer cost) {
129       issues.add(createAnalyzerMessage(getInputFile(), javaCheck, syntaxNode, null, message, flows, cost));
130     }
131 
132     @Override
133     public void reportIssue(AnalyzerMessage message) {
134       issues.add(message);
135     }
136 
137     @Override
138     public AnalyzerMessage createAnalyzerMessage(JavaCheck javaCheck, Tree startTree, String message) {
139       return createAnalyzerMessage(getInputFile(), javaCheck, startTree, null, message, Arrays.asList(), null);
140     }
141 
142     private AnalyzerMessage createAnalyzerMessage(JavaCheck javaCheck, Tree startTree, @Nullable Tree endTree, String message, List<Location> secondary, @Nullable Integer cost) {
143       List<List<Location>> flows = secondary.stream().map(Collections::singletonList).collect(Collectors.toList());  
144       return createAnalyzerMessage(getInputFile(), javaCheck, startTree, endTree, message, flows, cost);
145     }
146   }
147 }

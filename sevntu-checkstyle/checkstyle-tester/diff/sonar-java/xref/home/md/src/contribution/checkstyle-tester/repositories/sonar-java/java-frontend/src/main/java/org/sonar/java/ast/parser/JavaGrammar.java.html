201     return b.<ModuleDirectiveTree>nonterminal(JavaLexer.REQUIRES_DIRECTIVE)
202       .is(b.firstOf(
203         // JLS9 - §3.9 : 'transitive' restricted keyword can be used as module name instead of modifier
204         f.newRequiresModuleDirective(
205           b.token(JavaRestrictedKeyword.REQUIRES),
206           b.token(JavaRestrictedKeyword.TRANSITIVE),
207           b.token(JavaPunctuator.SEMI)),
208         f.newRequiresModuleDirective(
209           b.token(JavaRestrictedKeyword.REQUIRES),
210           b.token(JavaKeyword.STATIC),
211           b.token(JavaRestrictedKeyword.TRANSITIVE),
212           b.token(JavaPunctuator.SEMI)),
213         // ordinary requires directives
214         f.newRequiresModuleDirective(
215           b.token(JavaRestrictedKeyword.REQUIRES),
216           b.zeroOrMore(REQUIRES_MODIFIER()),
217           MODULE_NAME(),
218           b.token(JavaPunctuator.SEMI))));
219   }

20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.Lists;
23  import org.sonar.check.Rule;
24  import org.sonar.java.JavaVersionAwareVisitor;
25  import org.sonar.java.model.SyntacticEquivalence;
26  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
27  import org.sonar.plugins.java.api.JavaFileScannerContext;
28  import org.sonar.plugins.java.api.JavaVersion;
29  import org.sonar.plugins.java.api.tree.CatchTree;
30  import org.sonar.plugins.java.api.tree.Tree;
31  import org.sonar.plugins.java.api.tree.TryStatementTree;
32  
33  import java.util.ArrayList;
34  import java.util.Collections;
35  import java.util.List;
36  
37  @Rule(key = "S2147")
38  public class CombineCatchCheck extends IssuableSubscriptionVisitor implements JavaVersionAwareVisitor {
39  
40    @Override
41    public List<Tree.Kind> nodesToVisit() {
42      return Collections.singletonList(Tree.Kind.TRY_STATEMENT);
43    }
44  
45    @Override
46    public void visitNode(Tree tree) {
47      List<CatchTree> catches = new ArrayList<>();
48      for (CatchTree catchTree : ((TryStatementTree) tree).catches()) {
49        for (CatchTree catchTreeToBeCompared : catches) {
50          if (SyntacticEquivalence.areEquivalent(catchTree.block(), catchTreeToBeCompared.block())) {
51            reportIssue(catchTree, catchTreeToBeCompared);
52            break;
53          }
54        }
55        catches.add(catchTree);
56      }
57    }
58  
59    private void reportIssue(CatchTree catchTree, CatchTree catchTreeToBeCompared) {
60      String message = "Combine this catch with the one at line " + catchTreeToBeCompared.catchKeyword().line()
61        + ", which has the same body." + context.getJavaVersion().java7CompatibilityMessage();
62      List<JavaFileScannerContext.Location> flow = Lists.newArrayList(new JavaFileScannerContext.Location("Combine with this catch", catchTreeToBeCompared));
63      reportIssue(catchTree.parameter(), message, flow, null);
64    }
65  
66    @Override
67    public boolean isCompatibleWithJavaVersion(JavaVersion version) {
68      return version.isNotSet() || version.asInt() >= 7;
69    }
70  }

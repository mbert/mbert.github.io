20  package org.sonar.java.model;
21  
22  import javax.annotation.Nullable;
23  import org.sonar.api.utils.log.Logger;
24  import org.sonar.api.utils.log.Loggers;
25  import org.sonar.plugins.java.api.JavaVersion;
26  
27  public class JavaVersionImpl implements JavaVersion {
28  
29    private static final Logger LOG = Loggers.get(JavaVersionImpl.class);
30  
31    private static final int JAVA_6 = 6;
32    private static final int JAVA_7 = 7;
33    private static final int JAVA_8 = 8;
34    private static final int JAVA_12 = 12;
35    private final int javaVersion;
36  
37    public JavaVersionImpl() {
38      this.javaVersion = -1;
39    }
40  
41    public JavaVersionImpl(int javaVersion) {
42      this.javaVersion = javaVersion;
43    }
44  
45    public static JavaVersion fromString(@Nullable String javaVersion) {
46      if (javaVersion == null) {
47        return new JavaVersionImpl();
48      }
49      try {
50        String cleanedVersion = javaVersion.startsWith("1.") ? javaVersion.substring(2) : javaVersion;
51        int versionAsInt = Integer.parseInt(cleanedVersion);
52        return new JavaVersionImpl(versionAsInt);
53      } catch (NumberFormatException e) {
54        LOG.warn("Invalid java version (got \"" + javaVersion + "\"). "
55          + "The version will be ignored. Accepted formats are \"1.X\", or simply \"X\" "
56          + "(for instance: \"1.5\" or \"5\", \"1.6\" or \"6\", \"1.7\" or \"7\", etc.)");
57        return new JavaVersionImpl();
58      }
59    }
60  
61    @Override
62    public boolean isJava6Compatible() {
63      return notSetOrAtLeast(JAVA_6);
64    }
65  
66    @Override
67    public boolean isJava7Compatible() {
68      return notSetOrAtLeast(JAVA_7);
69    }
70  
71    @Override
72    public boolean isJava8Compatible() {
73      return notSetOrAtLeast(JAVA_8);
74    }
75  
76    @Override
77    public boolean isJava12Compatible() {
78      return JAVA_12 <= javaVersion;
79    }
80  
81    private boolean notSetOrAtLeast(int requiredJavaVersion) {
82      return isNotSet() || requiredJavaVersion <= javaVersion;
83    }
84  
85    @Override
86    public String java6CompatibilityMessage() {
87      return compatibilityMessage(JAVA_6);
88    }
89  
90    @Override
91    public String java7CompatibilityMessage() {
92      return compatibilityMessage(JAVA_7);
93    }
94  
95    @Override
96    public String java8CompatibilityMessage() {
97      return compatibilityMessage(JAVA_8);
98    }
99  
100   private String compatibilityMessage(int expected) {
101     return isNotSet() ? (" (sonar.java.source not set. Assuming " + expected + " or greater.)") : "";
102   }
103 
104   @Override
105   public int asInt() {
106     return javaVersion;
107   }
108 
109   @Override
110   public boolean isNotSet() {
111     return javaVersion == -1;
112   }
113 
114   @Override
115   public String toString() {
116     return isNotSet() ? "none" : Integer.toString(javaVersion);
117   }
118 }

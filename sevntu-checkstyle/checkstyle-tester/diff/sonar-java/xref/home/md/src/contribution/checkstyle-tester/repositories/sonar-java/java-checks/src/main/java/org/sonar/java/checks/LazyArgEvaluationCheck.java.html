59    private static final TypeCriteria STRING = TypeCriteria.is("java.lang.String");
60    private static final TypeCriteria OBJECT_ARR = TypeCriteria.is("java.lang.Object[]");

62    interface LogLevels {
63      /**
64       * The methods used by the logger to log messages.
65       */
66      List<MethodMatcher> log();
67  
68      /**
69       * The method matcher corresponding to the test method used to check if a logger level is enabled.
70       * For instance: 'isTraceEnabled()'.
71       */
72      MethodMatcher test();
73  
74      static Stream<LogLevels> logLevels() {
75        return Stream.of(SLF4J.values(), JUL.values(), LOG4J.values())
76          .flatMap(Arrays::stream);
77      }
78  
79      static MethodMatcher levelTestMatcher(TypeCriteria typeDefinition, String level) {
80        return MethodMatcher.create()
81          .typeDefinition(typeDefinition)
82          .name(String.format("is%c%sEnabled", level.charAt(0), level.toLowerCase(Locale.ROOT).substring(1)));
83      }
84  
85      enum SLF4J implements LogLevels {
86        TRACE,
87        DEBUG,
88        INFO,
89        WARN,
90        ERROR
91        ;
92  
93        private static final TypeCriteria LOGGER = TypeCriteria.subtypeOf("org.slf4j.Logger");
94        private static final TypeCriteria MARKER = TypeCriteria.is("org.slf4j.Marker");
95  
96        @Override
97        public List<MethodMatcher> log() {
98          return slf4jVariants(() -> MethodMatcher.create().typeDefinition(LOGGER).name(toString().toLowerCase(Locale.ROOT)));
99        }
100 
101       @Override
102       public MethodMatcher test() {
103         return LogLevels.levelTestMatcher(LOGGER, toString()).withoutParameter();
104       }
105 
106       private static List<MethodMatcher> slf4jVariants(Supplier<MethodMatcher> prototype) {
107         return Arrays.asList(
108           prototype.get().parameters(STRING),
109           prototype.get().parameters(STRING, anyType()),
110           prototype.get().parameters(STRING, anyType(), anyType()),
111           prototype.get().parameters(STRING, OBJECT_ARR),
112           prototype.get().parameters(MARKER, STRING),
113           prototype.get().parameters(MARKER, STRING, anyType()),
114           prototype.get().parameters(MARKER, STRING, anyType(), anyType()),
115           prototype.get().parameters(MARKER, STRING, OBJECT_ARR)
116         );
117       }
118     }

122       WARNING,

124       CONFIG,
125       FINE,
126       FINER,

129       private static final String LOGGER = "java.util.logging.Logger";
130       private static final MethodMatcher LOG = MethodMatcher.create()
131         .typeDefinition(LOGGER)
132         .name("log")
133         .addParameter("java.util.logging.Level")
134         .addParameter(STRING);

138         return Collections.singletonList(
139           MethodMatcher.create()
140           .typeDefinition(LOGGER)
141           .name(toString().toLowerCase(Locale.ROOT))

143       }

145       @Override
146       public MethodMatcher test() {
147         return MethodMatcher.create()
148           .typeDefinition(LOGGER)
149           .name("isLoggable")
150           .addParameter("java.util.logging.Level");
151       }

154     enum LOG4J implements LogLevels {
155       DEBUG,
156       ERROR,
157       FATAL,
158       INFO,
159       TRACE,
160       WARN;
161 
162       private static final String LEVEL = "org.apache.logging.log4j.Level";
163 
164       private static final TypeCriteria LOGGER = TypeCriteria.subtypeOf("org.apache.logging.log4j.Logger");
165       private static final TypeCriteria MARKER = TypeCriteria.is("org.apache.logging.log4j.Marker");
166       private static final Predicate<Type> SUPPLIER = TypeCriteria.subtypeOf("org.apache.logging.log4j.util.Supplier")
167         .or(TypeCriteria.subtypeOf("org.apache.logging.log4j.util.MessageSupplier"));
168 
169       private static final List<MethodMatcher> TESTS = Arrays.asList(
170         MethodMatcher.create().typeDefinition(LOGGER).name("isEnabled").addParameter(LEVEL),
171         MethodMatcher.create().typeDefinition(LOGGER).name("isEnabled").addParameter(LEVEL).addParameter(MARKER));

173       private static final MethodMatcher LOG = MethodMatcher.create().typeDefinition(LOGGER).name("log").withAnyParameters();
174 
175       @Override
176       public List<MethodMatcher> log() {
177         return Collections.singletonList(MethodMatcher.create()
178           .typeDefinition(LOGGER)
179           .name(toString().toLowerCase(Locale.ROOT))
180           .withAnyParameters());
181       }
182 
183       @Override
184       public MethodMatcher test() {
185         return LogLevels.levelTestMatcher(LOGGER, toString()).withAnyParameters();
186       }
187     }

190   private static final MethodMatcher PRECONDITIONS = MethodMatcher.create()
191     .typeDefinition("com.google.common.base.Preconditions")
192     .name("checkState")
193     .withAnyParameters();
194 
195   private static final MethodMatcherCollection LAZY_ARG_METHODS = MethodMatcherCollection.create(PRECONDITIONS, LogLevels.JUL.LOG, LogLevels.LOG4J.LOG);
196   static {
197     LogLevels.logLevels().map(LogLevels::log).forEach(LAZY_ARG_METHODS::addAll);
198   }
199 
200   private static final MethodMatcherCollection LOG_LEVEL_TESTS = MethodMatcherCollection.create().addAll(LogLevels.LOG4J.TESTS);
201   static {
202     LogLevels.logLevels().map(LogLevels::test).forEach(LOG_LEVEL_TESTS::add);
203   }

205   private JavaFileScannerContext context;
206   private Deque<Tree> treeStack = new ArrayDeque<>();
207 
208   @Override
209   public void scanFile(JavaFileScannerContext context) {
210     this.context = context;
211     if (context.getSemanticModel() == null) {
212       return;
213     }
214     scan(context.getTree());
215   }
216 
217   @Override
218   public void visitMethodInvocation(MethodInvocationTree tree) {
219     if (LAZY_ARG_METHODS.anyMatch(tree) && !insideCatchStatement() && !insideLevelTest() && !argsUsingSuppliers(tree)) {
220       onMethodInvocationFound(tree);
221     }
222   }
223 
224   private static boolean argsUsingSuppliers(MethodInvocationTree tree) {
225     return tree.arguments().stream().map(ExpressionTree::symbolType).anyMatch(LogLevels.LOG4J.SUPPLIER::test);
226   }
227 
228   @Override
229   public void visitIfStatement(IfStatementTree ifTree) {
230     LevelTestVisitor levelTestVisitor = new LevelTestVisitor();
231     ifTree.condition().accept(levelTestVisitor);
232     if (levelTestVisitor.match) {
233       stackAndContinue(ifTree, super::visitIfStatement);
234     } else {
235       super.visitIfStatement(ifTree);
236     }
237   }
238 
239   @Override
240   public void visitCatch(CatchTree tree) {
241     stackAndContinue(tree, super::visitCatch);
242   }
243 
244   @Override
245   public void visitMethod(MethodTree tree) {
246     // we put method trees on stack to be able to detect log statements in anonymous classes
247     stackAndContinue(tree, super::visitMethod);
248   }
249 
250   private boolean insideLevelTest() {
251     return treeStack.stream().anyMatch(t -> t.is(Tree.Kind.IF_STATEMENT));
252   }
253 
254   private boolean insideCatchStatement() {
255     return treeStack.peek() != null && treeStack.peek().is(Tree.Kind.CATCH);
256   }
257 
258   private <T extends Tree> void stackAndContinue(T tree, Consumer<T> visit) {
259     treeStack.push(tree);
260     visit.accept(tree);
261     treeStack.pop();
262   }
263 
264   private void onMethodInvocationFound(MethodInvocationTree mit) {
265     List<JavaFileScannerContext.Location> flow = findStringArg(mit)
266       .flatMap(LazyArgEvaluationCheck::checkArgument)
267       .collect(Collectors.toList());
268     if (!flow.isEmpty()) {
269       context.reportIssue(this, flow.get(0).syntaxNode, flow.get(0).msg, flow.subList(1, flow.size()), null);
270     }
271   }
272 
273   private static Stream<JavaFileScannerContext.Location> checkArgument(ExpressionTree stringArgument) {
274     StringExpressionVisitor visitor = new StringExpressionVisitor();
275     stringArgument.accept(visitor);
276     if (visitor.shouldReport) {
277       return Stream.of(locationFromArg(stringArgument, visitor));
278     } else {
279       return Stream.empty();
280     }
281   }
282 
283   private static JavaFileScannerContext.Location locationFromArg(ExpressionTree stringArgument, StringExpressionVisitor visitor) {
284     StringBuilder msg = new StringBuilder();
285     if (visitor.hasMethodInvocation) {
286       msg.append("Invoke method(s) only conditionally. ");
287     }
288     if (visitor.hasBinaryExpression) {
289       msg.append("Use the built-in formatting to construct this argument.");
290     }
291     return new JavaFileScannerContext.Location(msg.toString(), stringArgument);
292   }
293 
294   private static Stream<ExpressionTree> findStringArg(MethodInvocationTree mit) {
295     return mit.arguments().stream()
296       .filter(arg -> arg.symbolType().is("java.lang.String"));
297   }
298 
299   private static class StringExpressionVisitor extends BaseTreeVisitor {
300 
301     private boolean hasBinaryExpression;
302     private boolean shouldReport;
303     private boolean hasMethodInvocation;
304 
305     @Override
306     public void visitMethodInvocation(MethodInvocationTree tree) {
307       if (!isGetter(tree) && !isAnnotationMethod(tree)) {
308         shouldReport = true;
309         hasMethodInvocation = true;
310       }
311     }
312 
313     private static boolean isGetter(MethodInvocationTree tree) {
314       String methodName = tree.symbol().name();
315       return methodName != null && (methodName.startsWith("get") || methodName.startsWith("is"));
316     }
317 
318     private static boolean isAnnotationMethod(MethodInvocationTree tree) {
319       Symbol owner = tree.symbol().owner();
320       return owner.isTypeSymbol() && ((JavaSymbol.TypeJavaSymbol) owner).isAnnotation();
321     }
322 
323     @Override
324     public void visitIdentifier(IdentifierTree tree) {
325       if (hasBinaryExpression) {
326         shouldReport = true;
327       }
328     }
329 
330     @Override
331     public void visitNewClass(NewClassTree tree) {
332       hasMethodInvocation = true;
333       shouldReport = true;
334     }
335 
336     @Override
337     public void visitBinaryExpression(BinaryExpressionTree tree) {
338       hasBinaryExpression = true;
339       if (!isConstant(tree.rightOperand())) {
340         tree.rightOperand().accept(this);
341       }
342       if (!isConstant(tree.leftOperand())) {
343         tree.leftOperand().accept(this);
344       }
345     }
346 
347     private static boolean isConstant(ExpressionTree operand) {
348       switch (operand.kind()) {
349         case BOOLEAN_LITERAL:
350         case CHAR_LITERAL:
351         case DOUBLE_LITERAL:
352         case FLOAT_LITERAL:
353         case INT_LITERAL:
354         case LONG_LITERAL:
355         case STRING_LITERAL:
356         case NULL_LITERAL:
357           return true;
358         case IDENTIFIER:
359           return isConstant(((IdentifierTree) operand).symbol());
360         case MEMBER_SELECT:
361           MemberSelectExpressionTree mset = (MemberSelectExpressionTree) operand;
362           return isConstant(mset.identifier().symbol());
363         default:
364           return false;
365       }
366     }
367 
368     private static boolean isConstant(Symbol symbol) {
369       return symbol.isStatic() && symbol.isFinal();
370     }
371   }
372 
373   private static class LevelTestVisitor extends BaseTreeVisitor {
374     boolean match = false;
375 
376     @Override
377     public void visitMethodInvocation(MethodInvocationTree mit) {
378       if (LOG_LEVEL_TESTS.anyMatch(mit)) {
379         match = true;
380       }
381     }

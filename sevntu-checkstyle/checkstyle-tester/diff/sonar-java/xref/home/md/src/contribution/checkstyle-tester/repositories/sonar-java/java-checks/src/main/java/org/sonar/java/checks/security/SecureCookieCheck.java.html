47  @Rule(key = "S2092")
48  public class SecureCookieCheck extends IssuableSubscriptionVisitor {
49  
50    private static final String MESSAGE = "Make sure creating this cookie without the \"secure\" flag is safe here.";
51  
52    private static final String JAX_RS_COOKIE = "javax.ws.rs.core.Cookie";
53    private static final String JAX_RS_NEW_COOKIE = "javax.ws.rs.core.NewCookie";
54    private static final String SPRING_SAVED_COOKIE = "org.springframework.security.web.savedrequest.SavedCookie";
55    private static final String PLAY_COOKIE = "play.mvc.Http$Cookie";
56    private static final List<String> COOKIES = Arrays.asList(
57      "javax.servlet.http.Cookie",
58      "java.net.HttpCookie",
59      JAX_RS_COOKIE,
60      JAX_RS_NEW_COOKIE,
61      "org.apache.shiro.web.servlet.SimpleCookie",
62      SPRING_SAVED_COOKIE,
63      PLAY_COOKIE,
64      "play.mvc.Http$CookieBuilder");
65  
66    private static final List<String> SETTER_NAMES = Arrays.asList("setSecure", "withSecure");
67  
68    /**
69     * Some constructors have the 'secure' parameter and do not need a 'setSecure' call afterwards.
70     */
71    private static final String JAVA_LANG_STRING = "java.lang.String";
72    private static final String INT = "int";
73    private static final String BOOLEAN = "boolean";
74  
75    private static final List<MethodMatcher> CONSTRUCTORS_WITH_SECURE_PARAM_LAST = Arrays.asList(
76      constructorMatcher(JAX_RS_NEW_COOKIE).parameters(JAX_RS_COOKIE, JAVA_LANG_STRING, INT, BOOLEAN),
77      constructorMatcher(JAX_RS_NEW_COOKIE).parameters(JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, INT, JAVA_LANG_STRING, INT, BOOLEAN),
78      constructorMatcher(JAX_RS_NEW_COOKIE).parameters(JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, INT, BOOLEAN));
79  
80    private static final List<MethodMatcher> CONSTRUCTORS_WITH_SECURE_PARAM_BEFORE_LAST = Arrays.asList(
81      constructorMatcher(JAX_RS_NEW_COOKIE)
82        .parameters(JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, INT, JAVA_LANG_STRING, INT, "java.util.Date", BOOLEAN, BOOLEAN),
83      constructorMatcher(JAX_RS_NEW_COOKIE).parameters(JAX_RS_COOKIE, JAVA_LANG_STRING, INT, "java.util.Date", BOOLEAN, BOOLEAN),
84      constructorMatcher(JAX_RS_NEW_COOKIE).parameters(JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, INT, BOOLEAN, BOOLEAN),
85      constructorMatcher(SPRING_SAVED_COOKIE).parameters(JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, JAVA_LANG_STRING, INT, JAVA_LANG_STRING, BOOLEAN, INT),
86      constructorMatcher(PLAY_COOKIE).parameters(JAVA_LANG_STRING, JAVA_LANG_STRING, "java.lang.Integer", JAVA_LANG_STRING, JAVA_LANG_STRING, BOOLEAN, BOOLEAN));
87  
88    private static final List<MethodMatcher> CONSTRUCTORS_WITH_SECURE_PARAM_BEFORE_BEFORE_LAST = Collections.singletonList(
89      constructorMatcher(PLAY_COOKIE)
90        .parameters(JAVA_LANG_STRING, JAVA_LANG_STRING, "java.lang.Integer", JAVA_LANG_STRING, JAVA_LANG_STRING, BOOLEAN, BOOLEAN, "play.mvc.Http$Cookie$SameSite"));
91  
92    private final Map<Symbol.VariableSymbol, NewClassTree> unsecuredCookies = new HashMap<>();
93    private final Set<NewClassTree> cookieConstructors = new HashSet<>();
94  
95    @Override
96    public List<Tree.Kind> nodesToVisit() {
97      return Arrays.asList(
98        Tree.Kind.VARIABLE,
99        Tree.Kind.ASSIGNMENT,
100       Tree.Kind.METHOD_INVOCATION,
101       Tree.Kind.NEW_CLASS);
102   }
103 
104   @Override
105   public void setContext(JavaFileScannerContext context) {
106     unsecuredCookies.clear();
107     cookieConstructors.clear();
108     super.setContext(context);
109   }
110 
111   @Override
112   public void leaveFile(JavaFileScannerContext context) {
113     cookieConstructors.forEach(r -> reportIssue(r.identifier(), MESSAGE));
114   }
115 
116   @Override
117   public void visitNode(Tree tree) {
118     if (!hasSemantic()) {
119       return;
120     }
121     if (tree.is(Tree.Kind.VARIABLE)) {
122       addToUnsecuredCookies((VariableTree) tree);
123     } else if (tree.is(Tree.Kind.ASSIGNMENT)) {
124       addToUnsecuredCookies((AssignmentExpressionTree) tree);
125     } else if (tree.is(Tree.Kind.METHOD_INVOCATION)) {
126       checkSecureCall((MethodInvocationTree) tree);
127     } else {
128       checkConstructor((NewClassTree) tree);
129     }
130   }
131 
132   private void addToUnsecuredCookies(VariableTree variableTree) {
133     ExpressionTree initializer = variableTree.initializer();
134     Symbol variableTreeSymbol = variableTree.symbol();
135 
136     if (initializer != null && variableTreeSymbol.isVariableSymbol()) {
137       boolean isInitializedWithConstructor = initializer.is(Tree.Kind.NEW_CLASS);
138       boolean isMatchedType = isCookieClass(variableTreeSymbol.type()) || isCookieClass(initializer.symbolType());
139       if (isInitializedWithConstructor && isMatchedType && isSecureParamFalse((NewClassTree) initializer)) {
140         unsecuredCookies.put((Symbol.VariableSymbol) variableTreeSymbol, (NewClassTree) initializer);
141       }
142     }
143   }
144 
145   private void addToUnsecuredCookies(AssignmentExpressionTree assignment) {
146     if (assignment.expression().is(Tree.Kind.NEW_CLASS) && assignment.variable().is(Tree.Kind.IDENTIFIER)) {
147       IdentifierTree assignmentVariable = (IdentifierTree) assignment.variable();
148       Symbol assignmentVariableSymbol = assignmentVariable.symbol();
149       boolean isMatchedType = isCookieClass(assignmentVariable.symbolType()) || isCookieClass(assignment.expression().symbolType());
150       if (isMatchedType && isSecureParamFalse((NewClassTree) assignment.expression())) {
151         unsecuredCookies.put((Symbol.VariableSymbol) assignmentVariableSymbol, (NewClassTree) assignment.expression());
152       }
153     }
154   }
155 
156   private void checkSecureCall(MethodInvocationTree mit) {
157     if (isSetSecureCall(mit) && mit.methodSelect().is(Tree.Kind.MEMBER_SELECT)) {
158       Boolean secureArgument = ConstantUtils.resolveAsBooleanConstant(mit.arguments().get(0));
159       boolean isFalse = secureArgument != null && !secureArgument;
160       if (isFalse) {
161         reportIssue(mit.arguments(), MESSAGE);
162       }
163       ExpressionTree methodObject = ((MemberSelectExpressionTree) mit.methodSelect()).expression();
164       if (methodObject.is(Tree.Kind.IDENTIFIER)) {
165         IdentifierTree identifierTree = (IdentifierTree) methodObject;
166         NewClassTree newClassTree = unsecuredCookies.remove(identifierTree.symbol());
167         cookieConstructors.remove(newClassTree);
168       }
169     }
170   }
171 
172   private void checkConstructor(NewClassTree tree) {
173     if (isCookieClass(tree.symbolType()) && isSecureParamFalse(tree)) {
174       cookieConstructors.add(tree);
175     }
176   }
177 
178   private static boolean isSecureParamFalse(NewClassTree newClassTree) {
179     ExpressionTree secureArgument = null;
180     Arguments arguments = newClassTree.arguments();
181     if (CONSTRUCTORS_WITH_SECURE_PARAM_LAST.stream().anyMatch(m -> m.matches(newClassTree))) {
182       secureArgument = arguments.get(arguments.size() - 1);
183     } else if (CONSTRUCTORS_WITH_SECURE_PARAM_BEFORE_LAST.stream().anyMatch(m -> m.matches(newClassTree))) {
184       secureArgument = arguments.get(arguments.size() - 2);
185     } else if (CONSTRUCTORS_WITH_SECURE_PARAM_BEFORE_BEFORE_LAST.stream().anyMatch(m -> m.matches(newClassTree))) {
186       secureArgument = arguments.get(arguments.size() - 3);
187     }
188     if (secureArgument != null) {
189       return LiteralUtils.isFalse(secureArgument);
190     }
191     return true;
192   }
193 
194   private static boolean isSetSecureCall(MethodInvocationTree mit) {
195     return mit.arguments().size() == 1
196       && !mit.symbol().isUnknown()
197       && !mit.symbol().owner().isUnknown()
198       && isCookieClass(mit.symbol().owner().type())
199       && SETTER_NAMES.stream().anyMatch(getIdentifier(mit).name()::equals);
200   }
201 
202   private static boolean isCookieClass(Type type) {
203     return COOKIES.stream().anyMatch(type::isSubtypeOf);
204   }
205 
206   private static IdentifierTree getIdentifier(MethodInvocationTree mit) {
207     IdentifierTree id;
208     if (mit.methodSelect().is(Tree.Kind.IDENTIFIER)) {
209       id = (IdentifierTree) mit.methodSelect();
210     } else {
211       id = ((MemberSelectExpressionTree) mit.methodSelect()).identifier();
212     }
213     return id;
214   }
215 
216   private static MethodMatcher constructorMatcher(String fullyQualifiedName) {
217     return MethodMatcher.create().typeDefinition(fullyQualifiedName).name("<init>");
218   }

294   private void assertAnnotationValue(Object value) {
295     if (value instanceof JavaSymbol.VariableJavaSymbol) {
296       JavaSymbol.VariableJavaSymbol var = (JavaSymbol.VariableJavaSymbol) value;
297       assertThat(var.getName()).isEqualTo("ONE");
298       assertThat(var.type.is("org.sonar.java.resolve.targets.MyEnum")).isTrue();

304       return;
305     }
306     fail("value is not array nor variableSymbol");

311   public void type_parameters_should_be_read_from_bytecode() {
312     JavaSymbol.TypeJavaSymbol typeParametersSymbol = bytecodeCompleter.getClassSymbol("org.sonar.java.resolve.targets.TypeParameters");
313     typeParametersSymbol.complete();
314     assertThat(typeParametersSymbol.typeParameters).isNotNull();
315     assertThat(typeParametersSymbol.typeParameters.scopeSymbols()).hasSize(2);
316     assertThat(typeParametersSymbol.typeVariableTypes).hasSize(2);
317 
318     TypeVariableJavaType TtypeVariableType = typeParametersSymbol.typeVariableTypes.get(0);
319     assertThat(TtypeVariableType.erasure().getSymbol().getName()).isEqualTo("Object");
320     assertThat(typeParametersSymbol.typeVariableTypes.get(1).erasure().getSymbol().getName()).isEqualTo("CharSequence");
321 
322     assertThat(typeParametersSymbol.getSuperclass()).isInstanceOf(ParametrizedTypeJavaType.class);
323     assertThat(((ParametrizedTypeJavaType) typeParametersSymbol.getSuperclass()).typeSubstitution.typeVariables()).hasSize(1);
324     TypeVariableJavaType keyTypeVariable = ((ParametrizedTypeJavaType) typeParametersSymbol.getSuperclass()).typeSubstitution.typeVariables().iterator().next();
325     assertThat(keyTypeVariable.symbol.getName()).isEqualTo("S");
326     JavaType actual = ((ParametrizedTypeJavaType) typeParametersSymbol.getSuperclass()).typeSubstitution.substitutedType(keyTypeVariable);
327     assertThat(actual).isInstanceOf(ParametrizedTypeJavaType.class);
328     assertThat(((ParametrizedTypeJavaType) actual).typeSubstitution.typeVariables()).hasSize(1);
329 
330     assertThat(typeParametersSymbol.getInterfaces()).hasSize(2);
331     assertThat(typeParametersSymbol.getInterfaces().get(0)).isInstanceOf(ParametrizedTypeJavaType.class);
332 
333     JavaSymbol.MethodJavaSymbol funMethod = (JavaSymbol.MethodJavaSymbol) typeParametersSymbol.members().lookup("fun").get(0);
334     assertThat(funMethod.getReturnType().type).isSameAs(TtypeVariableType);
335     assertThat(funMethod.parameterTypes().get(0)).isSameAs(TtypeVariableType);
336 
337     JavaSymbol.MethodJavaSymbol fooMethod = (JavaSymbol.MethodJavaSymbol) typeParametersSymbol.members().lookup("foo").get(0);
338     TypeVariableJavaType WtypeVariableType = fooMethod.typeVariableTypes.get(0);
339     assertThat(fooMethod.parameterTypes().get(0).isArray()).isTrue();
340     assertThat(((ArrayJavaType)fooMethod.parameterTypes().get(0)).elementType()).isSameAs(WtypeVariableType);
341     JavaType resultType = ((MethodJavaType) fooMethod.type).resultType;
342     assertThat(resultType).isInstanceOf(ParametrizedTypeJavaType.class);
343     ParametrizedTypeJavaType actualResultType = (ParametrizedTypeJavaType) resultType;
344     assertThat(actualResultType.typeSubstitution.typeVariables()).hasSize(1);
345     assertThat(actualResultType.typeSubstitution.substitutedTypes().iterator().next()).isSameAs(WtypeVariableType);
346 
347     //primitive types
348     assertThat(fooMethod.parameterTypes().get(1).isPrimitive(org.sonar.plugins.java.api.semantic.Type.Primitives.INT)).isTrue();
349     assertThat(fooMethod.parameterTypes().get(2).isPrimitive(org.sonar.plugins.java.api.semantic.Type.Primitives.LONG)).isTrue();
350 
351     //read field.
352     JavaSymbol.VariableJavaSymbol field = (JavaSymbol.VariableJavaSymbol) typeParametersSymbol.members().lookup("field").get(0);
353     assertThat(field.type).isInstanceOf(TypeVariableJavaType.class);
354     assertThat(field.type).isSameAs(TtypeVariableType);
355   }

396     Symbol typeParameterTSymbol = ((MethodJavaType) method.getType()).argTypes().get(1).symbol();
397     assertThat(typeParameterTSymbol.name()).isEqualTo("T");
398     assertThat(typeParameterTSymbol.metadata().isAnnotatedWith("org.sonar.java.resolve.targets.TypeAnnotation")).isFalse();

51  public class SonarSymbolTableVisitorTest {
52  
53    @Rule
54    public TemporaryFolder temp = new TemporaryFolder();
55    private static final String EOL = "\n";
56    private List<String> lines;
57    private SensorContextTester context;
58    private DefaultFileSystem fs;
59    private SonarComponents sonarComponents;
60  
61    @Before
62    public void setUp() {
63      context = SensorContextTester.create(temp.getRoot());
64      fs = context.fileSystem();
65      sonarComponents = new SonarComponents(mock(FileLinesContextFactory.class), fs,
66        mock(JavaClasspath.class), mock(JavaTestClasspath.class), mock(CheckFactory.class));
67      sonarComponents.setSensorContext(context);
68    }
69  
70    @Test
71    public void sonar_symbol_table() throws Exception {
72      File file = temp.newFile().getAbsoluteFile();
73      Files.write(Files.toString(new File("src/test/files/highlighter/SonarSymTable.java"), StandardCharsets.UTF_8).replaceAll("\\r\\n", "\n").replaceAll("\\n", EOL), file, StandardCharsets.UTF_8);
74      lines = Files.readLines(file, StandardCharsets.UTF_8);
75      String content  = Joiner.on(EOL).join(lines);
76      fs.add(new TestInputFileBuilder("", file.getName()).initMetadata(content).build());
77      JavaAstScanner.scanSingleFileForTests(file, new VisitorsBridge(ImmutableList.of(), sonarComponents.getJavaClasspath(), sonarComponents));
78      String componentKey = ":" + file.getName();
79      verifyUsages(componentKey, 1, 17, reference(5,2), reference(9,10));
80      // Example class declaration
81      verifyUsages(componentKey, 4, 6);
82      verifyUsages(componentKey, 4, 14);
83      // list field
84      verifyUsages(componentKey, 5, 15, reference(10, 9));
85  
86      // Example empty constructor
87      verifyUsages(componentKey, 6, 2);
88      // Do not reference constructor of class using this() and super() as long as SONAR-5894 is not fixed
89      //verify(symboltableBuilder).newReference(any(Symbol.class), eq(offset(7, 5)));
90  
91      // Example list constructor
92      verifyUsages(componentKey, 9, 2, reference(7, 4));
93  
94      // list local var
95      verifyUsages(componentKey, 9, 23, reference(10, 16));
96      // method
97      verifyUsages(componentKey, 12, 6);
98      //label
99      verifyUsages(componentKey, 13, 4);
100     //Enum
101     verifyUsages(componentKey, 16, 7);
102     verifyUsages(componentKey, 17, 5);
103     // Do not reference constructor of enum as it can leads to failure in analysis as long as SONAR-5894 is not fixed
104     //verify(symboltableBuilder).newReference(any(Symbol.class), eq(offset(14, 5)));
105 
106     verifyUsages(componentKey, 18, 4, reference(17, 4));
107     verifyUsages(componentKey, 21, 3, reference(21, 19));
108     verifyUsages(componentKey, 21, 11);
109     verifyUsages(componentKey, 21, 21);
110   }
111 
112   private void verifyUsages(String componentKey, int line, int offset, TextPointer... tps) {
113     Collection<TextRange> textRanges = context.referencesForSymbolAt(componentKey, line, offset);
114     if(tps.length == 0) {
115       // TODO assert correctly that symbol is effectevly created see : SONAR-7850
116       assertThat(textRanges).isEmpty();
117     } else {
118       assertThat(textRanges.stream().map(TextRange::start).collect(Collectors.toList())).isNotEmpty().containsOnly(tps);
119     }
120   }
121 
122   private static TextPointer reference(int line, int column) {
123     return new DefaultTextPointer(line, column);
124   }

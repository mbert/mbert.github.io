99    public BehaviorCache(SquidClassLoader classLoader) {
100     this(classLoader, true);
101   }

103   public BehaviorCache(SquidClassLoader classLoader, boolean crossFileEnabled) {
104     this.classLoader = classLoader;
105     this.crossFileEnabled = crossFileEnabled;
106   }
107 
108   public void setFileContext(@Nullable SymbolicExecutionVisitor sev,@Nullable SemanticModel semanticModel) {
109     this.sev = sev;
110     this.semanticModel = semanticModel;
111   }
112 
113   public void cleanup() {
114     behaviors.clear();
115   }
116 
117   public MethodBehavior methodBehaviorForSymbol(Symbol.MethodSymbol symbol) {
118     String signature = symbol.signature();
119     boolean varArgs = ((JavaSymbol.MethodJavaSymbol) symbol).isVarArgs();
120     return behaviors.computeIfAbsent(signature, k -> new MethodBehavior(signature, varArgs));
121   }
122 
123   public MethodBehavior methodBehaviorForSymbol(String signature) {
124     return bytecodeBehaviors.computeIfAbsent(signature, k -> new MethodBehavior(signature));
125   }
126 
127   @CheckForNull
128   public MethodBehavior get(Symbol.MethodSymbol symbol) {
129     return get(symbol.signature(), symbol);
130   }
131 
132   @CheckForNull
133   public MethodBehavior get(String signature) {
134     return get(signature, null);
135   }
136 
137   @CheckForNull
138   private MethodBehavior get(String signature, @Nullable Symbol.MethodSymbol symbol) {
139     MethodBehavior mb = behaviors.get(signature);
140     if(mb != null) {
141       return mb;
142     }
143     if (symbol != null) {
144       MethodTree declaration = symbol.declaration();
145       if (SymbolicExecutionVisitor.methodCanNotBeOverriden(symbol) && declaration != null) {
146         sev.execute(declaration);
147         return behaviors.get(signature);
148       }
149     }
150 
151     // disabled x-file analysis, behavior based on source code can still be used
152     if (!crossFileEnabled && !isKnownSignature(signature)) {
153       return null;
154     }
155 
156     if (!bytecodeBehaviors.containsKey(signature)) {
157       new BytecodeEGWalker(this, semanticModel).getMethodBehavior(signature, classLoader);
158     }
159     return bytecodeBehaviors.get(signature);
160   }
161 
162   /**
163    * Do not trigger any new computation of method behavior, just check if there is a known method behavior for the symbol.
164    *
165    * @param symbol The targeted method.
166    * @return null for methods having no computed method behavior yet, or its method behavior, based on bytecode or source
167    */
168   @CheckForNull
169   public MethodBehavior peek(String signature) {
170     // directly query the cache, to not trigger computation of new method behaviors
171     MethodBehavior mb = behaviors.get(signature);
172     if (mb != null) {
173       return mb;
174     }
175     // check for bytecode signatures
176     return bytecodeBehaviors.get(signature);
177   }
178 
179   private static boolean isKnownSignature(String signature) {
180     return WHITELIST.stream().anyMatch(signature::startsWith);
181   }

51  @Rule(key = "S1994")
52  public class ForLoopIncrementAndUpdateCheck extends IssuableSubscriptionVisitor {
53  
54    @Override
55    public List<Tree.Kind> nodesToVisit() {
56      return ImmutableList.of(Tree.Kind.FOR_STATEMENT);
57    }
58  
59    @Override
60    public void visitNode(Tree tree) {
61      if (!hasSemantic()) {
62        return;
63      }
64      ForStatementTree forStatementTree = (ForStatementTree) tree;
65      if (forStatementTree.update().isEmpty() || forStatementTree.condition() == null) {
66        return;
67      }
68      Collection<Symbol> symbolsFromConditionsNotUpdated = symbolsFromConditionsNotUpdated(forStatementTree);
69      if (!symbolsFromConditionsNotUpdated.isEmpty()) {
70        Map<Symbol, Tree> updatesInBody = singleUpdatesInBody(forStatementTree.statement(), symbolsFromConditionsNotUpdated);
71        if (!updatesInBody.isEmpty()) {
72          reportIssue(forStatementTree.forKeyword(), getMessage(updatesInBody.keySet()), getSecondaries(updatesInBody.values()), null);
73        }
74      }
75    }
76  
77    private static String getMessage(Set<Symbol> updates) {
78      return String.format("Move the update of %s into this loop's update clause.", symbolNames(updates));
79    }
80  
81    private static Collection<Symbol> symbolsFromConditionsNotUpdated(ForStatementTree forStatementTree) {
82      Collection<Symbol> symbols = getConditionSymbols(forStatementTree.condition());
83      symbols.removeAll(getUpdatedSymbols(forStatementTree.update()));
84      return symbols;
85    }
86  
87    private static Collection<Symbol> getConditionSymbols(ExpressionTree condition) {
88      ConditionVisitor conditionVisitor = new ConditionVisitor();
89      condition.accept(conditionVisitor);
90      return conditionVisitor.symbols;
91    }
92  
93    private static Collection<Symbol> getUpdatedSymbols(ListTree<StatementTree> updates) {
94      UpdateVisitor updateVisitor = new UpdateVisitor();
95      updates.accept(updateVisitor);
96      return updateVisitor.symbols;
97    }
98  
99    private static Map<Symbol, Tree> singleUpdatesInBody(StatementTree statement, Collection<Symbol> conditionsNotUpdated) {
100     UpdatesInBodyVisitor updatedInBodyVisitor = new UpdatesInBodyVisitor(conditionsNotUpdated);
101     statement.accept(updatedInBodyVisitor);
102     // only report if there is symbols which are updated only once, as multiple update cannot easily be moved to updates
103     return updatedOnlyOnceWithUnaryExpression(updatedInBodyVisitor.updates, statement);
104   }
105 
106   private static String symbolNames(Set<Symbol> symbols) {
107     return symbols.stream().map(s -> "\"" + s.name() + "\"").sorted().collect(Collectors.joining(","));
108   }
109 
110   private static List<JavaFileScannerContext.Location> getSecondaries(Collection<Tree> updateTrees) {
111     return updateTrees.stream().map(t -> new JavaFileScannerContext.Location("", t)).collect(Collectors.toList());
112   }
113 
114   private static Map<Symbol, Tree> updatedOnlyOnceWithUnaryExpression(Multimap<Symbol, Tree> updatesInBody, StatementTree forStatementBody) {
115     Map<Symbol, Tree> result = new HashMap<>();
116     for (Symbol updatedSymbol : updatesInBody.keySet()) {
117       Collection<Tree> updateTrees = updatesInBody.get(updatedSymbol);
118       if (updateTrees.size() == 1) {
119         Tree updateTree = updateTrees.iterator().next();
120         if (updateTree.is(Tree.Kind.POSTFIX_INCREMENT, Tree.Kind.POSTFIX_DECREMENT, Tree.Kind.PREFIX_INCREMENT, Tree.Kind.PREFIX_DECREMENT)
121           // only consider unary expression as single statement, and not nested
122           && updateTree.parent().is(Tree.Kind.EXPRESSION_STATEMENT) && updateTree.parent().parent() == forStatementBody) {
123           result.put(updatedSymbol, updateTree);
124         }
125       }
126     }
127     return result;
128   }
129 
130   private static class ConditionVisitor extends BaseTreeVisitor {
131     Collection<Symbol> symbols = new HashSet<>();
132 
133     @Override
134     public void visitIdentifier(IdentifierTree tree) {
135       Symbol symbol = tree.symbol();
136       if (symbol.isVariableSymbol()) {
137         symbols.add(symbol);
138       }
139       super.visitIdentifier(tree);
140     }
141   }
142 
143   private static class UpdateVisitor extends BaseTreeVisitor {
144     Collection<Symbol> symbols = new ArrayList<>();
145 
146     @Override
147     public void visitUnaryExpression(UnaryExpressionTree tree) {
148       checkIdentifier(tree.expression());
149       super.visitUnaryExpression(tree);
150     }
151 
152     @Override
153     public void visitAssignmentExpression(AssignmentExpressionTree tree) {
154       checkIdentifier(tree.variable());
155       super.visitAssignmentExpression(tree);
156     }
157 
158     @Override
159     public void visitMemberSelectExpression(MemberSelectExpressionTree tree) {
160       checkIdentifier(tree.identifier());
161       // skip expression
162     }
163 
164     private void checkIdentifier(ExpressionTree expression) {
165       ExpressionTree expr = ExpressionUtils.skipParentheses(expression);
166       if (expr.is(Tree.Kind.IDENTIFIER)) {
167         addSymbol((IdentifierTree) expr);
168       } else if (expr.is(Tree.Kind.MEMBER_SELECT)) {
169         addSymbol(((MemberSelectExpressionTree) expr).identifier());
170       }
171     }
172 
173     private void addSymbol(IdentifierTree identifierTree) {
174       Symbol symbol = identifierTree.symbol();
175       if (!symbol.isUnknown()) {
176         symbols.add(symbol);
177       }
178     }
179   }
180 
181   private static class UpdatesInBodyVisitor extends BaseTreeVisitor {
182     private final Collection<Symbol> targets;
183     private final Multimap<Symbol, Tree> updates = ArrayListMultimap.create();
184 
185     private UpdatesInBodyVisitor(Collection<Symbol> targets) {
186       this.targets = targets;
187     }
188 
189     @Override
190     public void visitAssignmentExpression(AssignmentExpressionTree tree) {
191       checkIdentifier(tree.variable(), tree);
192       super.visitAssignmentExpression(tree);
193     }
194 
195     @Override
196     public void visitUnaryExpression(UnaryExpressionTree tree) {
197       checkIdentifier(tree.expression(), tree);
198       super.visitUnaryExpression(tree);
199     }
200 
201     private void checkIdentifier(ExpressionTree expression, Tree root) {
202       ExpressionTree expr = ExpressionUtils.skipParentheses(expression);
203       if (expr.is(Tree.Kind.IDENTIFIER)) {
204         addSymbol((IdentifierTree) expr, root);
205       } else if (expr.is(Tree.Kind.MEMBER_SELECT)) {
206         addSymbol(((MemberSelectExpressionTree) expr).identifier(), root);
207       }
208     }
209 
210     private void addSymbol(IdentifierTree identifierTree, Tree root) {
211       Symbol symbol = identifierTree.symbol();
212       if (targets.contains(symbol)) {
213         updates.put(symbol, root);
214       }
215     }
216   }

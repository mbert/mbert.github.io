20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableSet;
23  import com.google.common.collect.Lists;
24  import java.util.List;
25  import java.util.Set;
26  import org.sonar.check.Rule;
27  import org.sonar.java.checks.methods.AbstractMethodDetection;
28  import org.sonar.java.matcher.MethodMatcher;
29  import org.sonar.java.matcher.NameCriteria;
30  import org.sonar.java.matcher.TypeCriteria;
31  import org.sonar.plugins.java.api.semantic.Symbol;
32  import org.sonar.plugins.java.api.semantic.Type;
33  import org.sonar.plugins.java.api.tree.BaseTreeVisitor;
34  import org.sonar.plugins.java.api.tree.ExpressionTree;
35  import org.sonar.plugins.java.api.tree.MethodInvocationTree;
36  
37  import static org.sonar.plugins.java.api.semantic.Type.Primitives.DOUBLE;
38  import static org.sonar.plugins.java.api.semantic.Type.Primitives.FLOAT;
39  
40  @Rule(key = "S2698")
41  public class AssertionsWithoutMessageCheck extends AbstractMethodDetection {
42  
43    private static final String MESSAGE = "Add a message to this assertion.";
44    private static final String ASSERT = "assert";
45  
46    private static final String GENERIC_ASSERT = "org.fest.assertions.GenericAssert";
47    private static final MethodMatcher FEST_AS_METHOD = MethodMatcher.create()
48      .typeDefinition(GENERIC_ASSERT).name("as").addParameter("java.lang.String");
49    private static final Set<String> ASSERT_METHODS_WITH_ONE_PARAM = ImmutableSet.of("assertNull", "assertNotNull");
50    private static final Set<String> ASSERT_METHODS_WITH_TWO_PARAMS = ImmutableSet.of("assertEquals", "assertSame", "assertNotSame", "assertThat");
51    private static final Set<String> JUNIT5_ASSERT_METHODS_IGNORED = ImmutableSet.of("assertAll", "assertLinesMatch");
52    private static final Set<String> JUNIT5_ASSERT_METHODS_WITH_ONE_PARAM = ImmutableSet.of("assertTrue", "assertFalse", "assertNull", "assertNotNull");
53    private static final Set<String> JUNIT5_ASSERT_METHODS_WITH_DELTA = ImmutableSet.of("assertArrayEquals", "assertEquals");
54  
55    @Override
56    protected List<MethodMatcher> getMethodInvocationMatchers() {
57      return Lists.newArrayList(
58        MethodMatcher.create().typeDefinition("org.junit.jupiter.api.Assertions").name(NameCriteria.startsWith(ASSERT)).withAnyParameters(),
59        MethodMatcher.create().typeDefinition("org.junit.Assert").name(NameCriteria.startsWith(ASSERT)).withAnyParameters(),
60        MethodMatcher.create().typeDefinition("org.junit.Assert").name("fail").withAnyParameters(),
61        MethodMatcher.create().typeDefinition("junit.framework.Assert").name(NameCriteria.startsWith(ASSERT)).withAnyParameters(),
62        MethodMatcher.create().typeDefinition("junit.framework.Assert").name(NameCriteria.startsWith("fail")).withAnyParameters(),
63        MethodMatcher.create().typeDefinition("org.fest.assertions.Fail").name(NameCriteria.startsWith("fail")).withAnyParameters(),
64        MethodMatcher.create().typeDefinition(TypeCriteria.subtypeOf(GENERIC_ASSERT)).name(NameCriteria.any()).withAnyParameters()
65      );
66    }
67  
68    @Override
69    protected void onMethodInvocationFound(MethodInvocationTree mit) {
70      Symbol symbol = mit.symbol();
71      if (symbol.owner().type().isSubtypeOf(GENERIC_ASSERT) && !FEST_AS_METHOD.matches(mit)) {
72        if (isConstructor(symbol)) {
73          return;
74        }
75        FestVisitor visitor = new FestVisitor();
76        mit.methodSelect().accept(visitor);
77        if (!visitor.useDescription) {
78          reportIssue(mit, MESSAGE);
79        }
80      } else if (symbol.owner().type().is("org.junit.jupiter.api.Assertions")) {
81        checkJUnit5(mit);
82      } else if (mit.arguments().isEmpty() || !isString(mit.arguments().get(0)) || isAssertingOnStringWithNoMessage(mit)) {
83        reportIssue(mit, MESSAGE);
84      }
85    }
86  
87    private void checkJUnit5(MethodInvocationTree mit) {
88      String methodName = mit.symbol().name();
89      if (JUNIT5_ASSERT_METHODS_IGNORED.contains(methodName)) {
90        return;
91      }
92  
93      if (JUNIT5_ASSERT_METHODS_WITH_ONE_PARAM.contains(methodName)) {
94        if (mit.arguments().size() == 1) {
95          reportIssue(mit, MESSAGE);
96        }
97      } else if (mit.arguments().size() == 2) {
98        reportIssue(mit, MESSAGE);
99      } else if (JUNIT5_ASSERT_METHODS_WITH_DELTA.contains(methodName) && mit.arguments().size() == 3) {
100       Type thirdArgumentType = mit.arguments().get(2).symbolType();
101       if (thirdArgumentType.isPrimitive(DOUBLE) || thirdArgumentType.isPrimitive(FLOAT)) {
102         reportIssue(mit, MESSAGE);
103       }
104     }
105   }
106 
107   private static boolean isConstructor(Symbol symbol) {
108     return "<init>".equals(symbol.name());
109   }
110 
111   private static boolean isAssertingOnStringWithNoMessage(MethodInvocationTree mit) {
112     return isAssertWithTwoParams(mit) || isAssertWithOneParam(mit);
113   }
114 
115   private static boolean isAssertWithOneParam(MethodInvocationTree mit) {
116     return ASSERT_METHODS_WITH_ONE_PARAM.contains(mit.symbol().name()) && mit.arguments().size() == 1;
117   }
118 
119   private static boolean isAssertWithTwoParams(MethodInvocationTree mit) {
120     return ASSERT_METHODS_WITH_TWO_PARAMS.contains(mit.symbol().name()) && mit.arguments().size() == 2;
121   }
122 
123   private static boolean isString(ExpressionTree expressionTree) {
124     return expressionTree.symbolType().is("java.lang.String");
125   }
126 
127   private static class FestVisitor extends BaseTreeVisitor {
128     boolean useDescription = false;
129 
130     @Override
131     public void visitMethodInvocation(MethodInvocationTree tree) {
132       useDescription |= FEST_AS_METHOD.matches(tree);
133       super.visitMethodInvocation(tree);
134     }
135 
136   }

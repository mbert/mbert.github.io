43  public class JaCoCoSensor implements Sensor {
44  
45    private static final String JACOCO_MERGED_FILENAME = "jacoco-merged.exec";
46    private static final Set<String> DEPRECATED_PROPERTIES = ImmutableSet.of(REPORT_PATH_PROPERTY, IT_REPORT_PATH_PROPERTY);
47    private final ResourcePerspectives perspectives;
48    private final JavaResourceLocator javaResourceLocator;
49    private final JavaClasspath javaClasspath;
50    static final String JACOCO_XML_PROPERTY = "sonar.coverage.jacoco.xmlReportPaths";
51    private AnalysisWarningsWrapper analysisWarnings;
52  
53    /**
54     * Used by SQ < 7.4 where AnalysisWarning is not yet available. Should be dropped once SQ > 7.4 is the minimal version
55     */
56    public JaCoCoSensor(ResourcePerspectives perspectives, JavaResourceLocator javaResourceLocator, JavaClasspath javaClasspath) {
57      this(perspectives, javaResourceLocator, javaClasspath, AnalysisWarningsWrapper.NOOP_ANALYSIS_WARNINGS);
58    }
59  
60    public JaCoCoSensor(ResourcePerspectives perspectives, JavaResourceLocator javaResourceLocator, JavaClasspath javaClasspath, AnalysisWarningsWrapper analysisWarnings) {
61      this.perspectives = perspectives;
62      this.javaResourceLocator = javaResourceLocator;
63      this.javaClasspath = javaClasspath;
64      this.analysisWarnings = analysisWarnings;
65    }
66  
67    @Override
68    public void describe(SensorDescriptor descriptor) {
69      descriptor
70        .onlyOnLanguage("java")
71        .name("JaCoCoSensor");
72    }
73  
74    @Override
75    public void execute(SensorContext context) {
76      warnAboutDeprecatedProperties(context.config());
77      if (context.config().hasKey(JACOCO_XML_PROPERTY)) {
78        LOG.debug("Property '{}' was specified, skipping processing of binary JaCoCo exec report.", JACOCO_XML_PROPERTY);
79        return;
80      }
81      Set<File> reportPaths = getReportPaths(context);
82      if (reportPaths.isEmpty()) {
83        return;
84      }
85      // Merge JaCoCo reports
86      File reportMerged;
87      if (reportPaths.size() == 1) {
88        reportMerged = reportPaths.iterator().next();
89      } else {
90        reportMerged = new File(context.fileSystem().workDir(), JACOCO_MERGED_FILENAME);
91        reportMerged.getParentFile().mkdirs();
92        JaCoCoReportMerger.mergeReports(reportMerged, reportPaths.toArray(new File[0]));
93      }
94      new UnitTestAnalyzer(reportMerged, perspectives, javaResourceLocator, javaClasspath, analysisWarnings).analyse(context);
95    }
96  
97    private void warnAboutDeprecatedProperties(Configuration config) {
98      if (config.hasKey(REPORT_MISSING_FORCE_ZERO)) {
99        addAnalysisWarning("Property '%s' is deprecated and its value will be ignored.", REPORT_MISSING_FORCE_ZERO);
100     }
101     DEPRECATED_PROPERTIES.forEach(deprecatedProperty -> {
102       if (config.hasKey(deprecatedProperty) && !config.hasKey(REPORT_PATHS_PROPERTY) && !config.hasKey(JACOCO_XML_PROPERTY)) {
103         addAnalysisWarning("Property '%s' is deprecated. '%s' should be used instead (JaCoCo XML format).", deprecatedProperty, JACOCO_XML_PROPERTY);
104       }
105     });
106     if (config.hasKey(REPORT_PATHS_PROPERTY)) {
107       if (config.hasKey(JACOCO_XML_PROPERTY)) {
108         LOG.info("Both '{}' and '{}' were set. '{}' is deprecated therefore, only '{}' will be taken into account.",
109           REPORT_PATHS_PROPERTY, JACOCO_XML_PROPERTY, REPORT_PATHS_PROPERTY, JACOCO_XML_PROPERTY);
110       } else {
111         addAnalysisWarning("'%s' is deprecated (JaCoCo binary format). '%s' should be used instead (JaCoCo XML format).", REPORT_PATHS_PROPERTY, JACOCO_XML_PROPERTY);
112       }
113     }
114   }
115 
116   private void addAnalysisWarning(String format, Object... args) {
117     String msg = String.format(format, args);
118     LOG.warn(msg);
119     analysisWarnings.addUnique(msg);
120   }
121 
122   private static Set<File> getReportPaths(SensorContext context) {
123     Set<File> reportPaths = new HashSet<>();
124     Configuration settings = context.config();
125     FileSystem fs = context.fileSystem();
126     for (String reportPath : settings.getStringArray(REPORT_PATHS_PROPERTY)) {
127       File report = fs.resolvePath(reportPath);
128       if (!report.isFile()) {
129         if (settings.hasKey(REPORT_PATHS_PROPERTY)) {
130           LOG.info("JaCoCo report not found: '{}'", reportPath);
131         }
132       } else {
133         reportPaths.add(report);
134       }
135     }
136     getReport(settings, fs, REPORT_PATH_PROPERTY, "JaCoCo UT report not found: '{}'").ifPresent(reportPaths::add);
137     getReport(settings, fs, IT_REPORT_PATH_PROPERTY, "JaCoCo IT report not found: '{}'").ifPresent(reportPaths::add);
138     return reportPaths;
139   }
140 
141   private static Optional<File> getReport(Configuration settings, FileSystem fs, String reportPathPropertyKey, String msg) {
142     Optional<String> reportPathProp = settings.get(reportPathPropertyKey);
143     if (reportPathProp.isPresent()) {
144       String reportPathProperty = reportPathProp.get();
145       File report = fs.resolvePath(reportPathProperty);
146       if (!report.isFile()) {
147         LOG.info(msg, reportPathProperty);
148       } else {
149         return Optional.of(report);
150       }
151     }
152     return Optional.empty();
153   }
154

157     return getClass().getSimpleName();

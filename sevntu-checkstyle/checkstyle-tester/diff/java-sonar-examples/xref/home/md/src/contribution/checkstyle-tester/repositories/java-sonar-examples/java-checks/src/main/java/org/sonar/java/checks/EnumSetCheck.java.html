20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  import java.util.List;
24  import org.sonar.check.Rule;
25  import org.sonar.java.matcher.MethodMatcher;
26  import org.sonar.java.matcher.MethodMatcherCollection;
27  import org.sonar.java.matcher.NameCriteria;
28  import org.sonar.java.resolve.ParametrizedTypeJavaType;
29  import org.sonar.java.resolve.TypeVariableJavaType;
30  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
31  import org.sonar.plugins.java.api.semantic.Type;
32  import org.sonar.plugins.java.api.tree.ExpressionTree;
33  import org.sonar.plugins.java.api.tree.MethodInvocationTree;
34  import org.sonar.plugins.java.api.tree.Tree;
35  import org.sonar.plugins.java.api.tree.Tree.Kind;
36  import org.sonar.plugins.java.api.tree.TypeTree;
37  import org.sonar.plugins.java.api.tree.VariableTree;
38  
39  @Rule(key = "S1641")
40  public class EnumSetCheck extends IssuableSubscriptionVisitor {
41  
42    private static final MethodMatcher COLLECTIONS_UNMODIFIABLE = MethodMatcher.create().typeDefinition("java.util.Collections").name("unmodifiableSet").withAnyParameters();
43    private static final MethodMatcherCollection SET_CREATION_METHODS = MethodMatcherCollection.create(
44      // Java 9 factory methods
45      MethodMatcher.create().typeDefinition("java.util.Set").name("of").withAnyParameters(),
46      // guava
47      MethodMatcher.create().typeDefinition("com.google.common.collect.ImmutableSet").name("of").withAnyParameters(),
48      MethodMatcher.create().typeDefinition("com.google.common.collect.Sets").name(NameCriteria.any()).withAnyParameters());
49  
50    @Override
51    public List<Kind> nodesToVisit() {
52      return ImmutableList.of(Kind.VARIABLE);
53    }
54  
55    @Override
56    public void visitNode(Tree tree) {
57      if (!hasSemantic()) {
58        return;
59      }
60      VariableTree variableTree = (VariableTree) tree;
61      ExpressionTree initializer = variableTree.initializer();
62      if (initializer == null) {
63        return;
64      }
65      if (initializer.is(Kind.METHOD_INVOCATION)) {
66        MethodInvocationTree mit = (MethodInvocationTree) initializer;
67        if (COLLECTIONS_UNMODIFIABLE.matches(mit)) {
68          // check the collection used as parameter
69          initializer = mit.arguments().get(0);
70        } else if (!SET_CREATION_METHODS.anyMatch(mit) || "immutableEnumSet".equals(mit.symbol().name())) {
71          // Methods from Guava 'Sets' except 'immutableEnumSet' should be checked,
72          // but discard any other method invocations (killing the noise)
73          return;
74        }
75      }
76      checkIssue(initializer.symbolType(), initializer, variableTree.type());
77    }
78  
79    private void checkIssue(Type type, Tree reportTree, TypeTree typeTree) {
80      if (type.isSubtypeOf("java.util.Set") && !type.isSubtypeOf("java.util.EnumSet") && type instanceof ParametrizedTypeJavaType) {
81        ParametrizedTypeJavaType parametrizedType = (ParametrizedTypeJavaType) type;
82        List<TypeVariableJavaType> typeParameters = parametrizedType.typeParameters();
83        Type variableType = typeTree.symbolType();
84        if(typeParameters.isEmpty() && variableType instanceof ParametrizedTypeJavaType) {
85          // for java 7 diamond operator lookup declaration.
86          parametrizedType = (ParametrizedTypeJavaType) variableType;
87          typeParameters = parametrizedType.typeParameters();
88        }
89        if(!typeParameters.isEmpty()) {
90          Type typeParameter = parametrizedType.substitution(typeParameters.get(0));
91          if (typeParameter != null && typeParameter.symbol().isEnum()) {
92            reportIssue(reportTree, "Convert this Set to an EnumSet.");
93          }
94        }
95      }
96    }
97  
98  }

97    @Test
98    public void applySubstitution_on_array_with_generic() {
99      TypeVariableJavaType X = getTypeVariable("X");
100     JavaSymbol.TypeJavaSymbol aSymbol = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, "A", symbols.defaultPackage);
101 
102     // A<{X=T}>
103     JavaType aXT = parametrizedTypeCache.getParametrizedTypeType(aSymbol, new TypeSubstitution().add(X, T));
104     // A<T>[]
105     JavaType formal1 = new ArrayJavaType(aXT, symbols.arrayClass);
106     // A<T>[][]
107     JavaType formal2 = new ArrayJavaType(new ArrayJavaType(aXT, symbols.arrayClass), symbols.arrayClass);
108 
109     List<JavaType> substitutedTypes = typeSubstitutionSolver.applySubstitutionToFormalParameters(ImmutableList.of(formal1, formal2),
110       new TypeSubstitution().add(T, symbols.stringType));
111 
112     JavaType substituted1 = substitutedTypes.get(0);
113     assertThat(substituted1).isInstanceOf(ArrayJavaType.class);
114     JavaType elementType = ((ArrayJavaType) substituted1).elementType;
115     assertThat(elementType).isInstanceOf(ParametrizedTypeJavaType.class);
116     ParametrizedTypeJavaType ptt = (ParametrizedTypeJavaType) elementType;
117     assertThat(ptt.substitution(X)).isSameAs(symbols.stringType);
118 
119     JavaType substituted2 = substitutedTypes.get(1);
120     assertThat(substituted2).isInstanceOf(ArrayJavaType.class);
121     elementType = ((ArrayJavaType) ((ArrayJavaType) substituted2).elementType).elementType;
122     assertThat(elementType).isInstanceOf(ParametrizedTypeJavaType.class);
123     ptt = (ParametrizedTypeJavaType) elementType;
124     assertThat(ptt.substitution(X)).isSameAs(symbols.stringType);
125   }

127   @Test
128   public void applySubstitution_on_nested_parametrized_types() {
129     TypeVariableJavaType k = getTypeVariable("K");
130 
131     TypeSubstitution aSubs = new TypeSubstitution();
132     aSubs.add(k, T);
133     JavaSymbol.TypeJavaSymbol aSymbol = new JavaSymbol.TypeJavaSymbol(Flags.PUBLIC, "A", Symbols.rootPackage);
134     // A<{K=T}>
135     ParametrizedTypeJavaType aRoot = new ParametrizedTypeJavaType(aSymbol, aSubs, typeSubstitutionSolver);
136 
137     // A<...n-1...<A<T>>...>
138     JavaType last = aRoot;
139     int n = 10;
140     for (int i = 0; i < n; i++) {
141       TypeSubstitution newSubs = new TypeSubstitution();
142       newSubs.add(k, last);
143       last = new ParametrizedTypeJavaType(aSymbol, newSubs, typeSubstitutionSolver);
144     }
145 
146     List<JavaType> formals = Lists.newArrayList(last);
147     TypeSubstitution substitution = new TypeSubstitution();
148     substitution.add(T, symbols.stringType);
149     List<JavaType> substitutedFormals = typeSubstitutionSolver.applySubstitutionToFormalParameters(formals, substitution);
150 
151     JavaType type = substitutedFormals.get(0);
152     int nbNestedGenerics = 0;
153     while (type instanceof ParametrizedTypeJavaType) {
154       type = ((ParametrizedTypeJavaType) type).substitution(k);
155       nbNestedGenerics++;
156     }
157     assertThat(nbNestedGenerics).isEqualTo(n + 1);
158     assertThat(type).isEqualTo(symbols.stringType);
159   }

364   @Test
365   public void type_hierarchy_visit_should_be_limited() {
366     ParametrizedTypeCache parametrizedTypeCache = new ParametrizedTypeCache();
367     BytecodeCompleter bytecodeCompleter = new BytecodeCompleter(new SquidClassLoader(new ArrayList<>()), parametrizedTypeCache);
368     Symbols symbols = new Symbols(bytecodeCompleter);
369 
370     ActionParser<Tree> parser = JavaParser.createParser();
371     SemanticModel semanticModel = new SemanticModel(bytecodeCompleter);
372     Resolve resolve = new Resolve(symbols, bytecodeCompleter, parametrizedTypeCache);
373     TypeAndReferenceSolver typeAndReferenceSolver = new TypeAndReferenceSolver(semanticModel, symbols, resolve, parametrizedTypeCache);
374     CompilationUnitTree tree = (CompilationUnitTree) parser.parse(new File("src/test/files/sym/ComplexHierarchy.java"));
375     new FirstPass(semanticModel, symbols, resolve, parametrizedTypeCache, typeAndReferenceSolver).visitCompilationUnit(tree);
376     typeAndReferenceSolver.visitCompilationUnit(tree);
377 
378     ClassTree classTree = (ClassTree) tree.types().get(tree.types().size() - 1);
379     JavaType site = (JavaType) classTree.symbol().type();
380     MethodInvocationTree mit = (MethodInvocationTree) ((ExpressionStatementTree) ((MethodTree) classTree.members().get(0)).block().body().get(0)).expression();
381 
382     TypeSubstitutionSolver typeSubstitutionSolver = Mockito.spy(new TypeSubstitutionSolver(parametrizedTypeCache, symbols));
383     // call with empty formals should return.
384     typeSubstitutionSolver.applySiteSubstitutionToFormalParameters(new ArrayList<>(), site);
385     verify(typeSubstitutionSolver, times(0)).applySiteSubstitutionToFormalParameters(anyList(), any(JavaType.class), anySet());
386 
387     JavaSymbol.MethodJavaSymbol methodJavaSymbol = (JavaSymbol.MethodJavaSymbol) mit.symbol();
388     typeSubstitutionSolver.applySiteSubstitutionToFormalParameters(((MethodJavaType) methodJavaSymbol.type).argTypes, site);
389     verify(typeSubstitutionSolver, times(11)).applySiteSubstitutionToFormalParameters(anyList(), any(JavaType.class), anySet());
390   }

20  package org.sonar.java.checks;
21  
22  import com.google.common.collect.ImmutableList;
23  
24  import org.sonar.check.Rule;
25  import org.sonar.java.model.declaration.MethodTreeImpl;
26  import org.sonar.plugins.java.api.IssuableSubscriptionVisitor;
27  import org.sonar.plugins.java.api.semantic.Type;
28  import org.sonar.plugins.java.api.tree.ClassTree;
29  import org.sonar.plugins.java.api.tree.MethodTree;
30  import org.sonar.plugins.java.api.tree.Tree;
31  import org.sonar.plugins.java.api.tree.TypeTree;
32  
33  import java.util.List;
34  
35  @Rule(key = "S1210")
36  public class EqualsNotOverridenWithCompareToCheck extends IssuableSubscriptionVisitor {
37  
38    @Override
39    public List<Tree.Kind> nodesToVisit() {
40      return ImmutableList.of(Tree.Kind.CLASS, Tree.Kind.ENUM);
41    }
42  
43    @Override
44    public void visitNode(Tree tree) {
45      ClassTree classTree = (ClassTree) tree;
46      if (!isComparable(classTree)) {
47        return;
48      }
49      boolean hasEquals = false;
50      MethodTree compare = null;
51  
52      for (Tree member : classTree.members()) {
53        if (member.is(Tree.Kind.METHOD)) {
54          MethodTree method = (MethodTree) member;
55  
56          if (isEqualsMethod(method)) {
57            hasEquals = true;
58          } else if (isCompareToMethod(method)) {
59            compare = method;
60          }
61        }
62      }
63  
64      if (compare != null && !hasEquals) {
65        reportIssue(compare.simpleName(), "Override \"equals(Object obj)\" to comply with the contract of the \"compareTo(T o)\" method.");
66      }
67    }
68  
69    private static boolean isCompareToMethod(MethodTree method) {
70      String name = method.simpleName().name();
71      return "compareTo".equals(name) && returnsInt(method) && method.parameters().size() == 1;
72    }
73  
74    private static boolean isEqualsMethod(MethodTree method) {
75      return ((MethodTreeImpl) method).isEqualsMethod();
76    }
77  
78    private static boolean isComparable(ClassTree tree) {
79      for (Type type : tree.symbol().interfaces()) {
80        if (type.is("java.lang.Comparable")) {
81          return true;
82        }
83      }
84      return false;
85    }
86  
87    private static boolean returnsInt(MethodTree tree) {
88      TypeTree typeTree = tree.returnType();
89      return typeTree != null && typeTree.symbolType().isPrimitive(Type.Primitives.INT);
90    }
91  
92  }

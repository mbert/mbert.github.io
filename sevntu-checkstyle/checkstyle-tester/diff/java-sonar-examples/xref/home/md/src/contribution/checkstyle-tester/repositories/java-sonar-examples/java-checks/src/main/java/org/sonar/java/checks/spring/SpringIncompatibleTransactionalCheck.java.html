48  @Rule(key = "S2229")
49  public class SpringIncompatibleTransactionalCheck extends IssuableSubscriptionVisitor {
50  
51    private static final String SPRING_TRANSACTIONAL_ANNOTATION = "org.springframework.transaction.annotation.Transactional";
52    private static final String JAVAX_TRANSACTIONAL_ANNOTATION = "javax.transaction.Transactional";
53  
54    private static final String MANDATORY = "MANDATORY";
55    private static final String NESTED = "NESTED";
56    private static final String NEVER = "NEVER";
57    private static final String NOT_SUPPORTED = "NOT_SUPPORTED";
58    private static final String REQUIRED = "REQUIRED";
59    private static final String REQUIRES_NEW = "REQUIRES_NEW";
60    private static final String SUPPORTS = "SUPPORTS";
61  
62    private static final Map<String, Set<String>> INCOMPATIBLE_PROPAGATION_MAP = buildIncompatiblePropagationMap();
63  
64    private static Map<String, Set<String>> buildIncompatiblePropagationMap() {
65      Map<String, Set<String>> map = new HashMap<>();
66      map.put(null, new HashSet<>(Arrays.asList(MANDATORY, NESTED, REQUIRED, REQUIRES_NEW)));
67      map.put(MANDATORY, new HashSet<>(Arrays.asList(NESTED, NEVER, NOT_SUPPORTED, REQUIRES_NEW)));
68      map.put(NESTED, new HashSet<>(Arrays.asList(NESTED, NEVER, NOT_SUPPORTED, REQUIRES_NEW)));
69      map.put(NEVER, new HashSet<>(Arrays.asList(MANDATORY, NESTED, REQUIRED, REQUIRES_NEW)));
70      map.put(NOT_SUPPORTED, new HashSet<>(Arrays.asList(MANDATORY, NESTED, REQUIRED, REQUIRES_NEW)));
71      map.put(REQUIRED, new HashSet<>(Arrays.asList(NESTED, NEVER, NOT_SUPPORTED, REQUIRES_NEW)));
72      map.put(REQUIRES_NEW, new HashSet<>(Arrays.asList(NESTED, NEVER, NOT_SUPPORTED, REQUIRES_NEW)));
73      map.put(SUPPORTS, new HashSet<>(Arrays.asList(MANDATORY, NESTED, NEVER, NOT_SUPPORTED, REQUIRED, REQUIRES_NEW)));
74      return map;
75    }
76  
77    @Override
78    public List<Tree.Kind> nodesToVisit() {
79      return Collections.singletonList(Tree.Kind.CLASS);
80    }
81  
82    @Override
83    public void visitNode(Tree tree) {
84      if (!hasSemantic()) {
85        return;
86      }
87      ClassTree classTree = (ClassTree) tree;
88      Map<Symbol, String> methodsPropagationMap = collectMethodsPropagation(classTree);
89      if (hasSameValues(methodsPropagationMap.values())) {
90        return;
91      }
92      methodsPropagationMap
93        .forEach((symbol, propagation) -> checkMethodInvocations((MethodTree) symbol.declaration(), propagation, methodsPropagationMap));
94    }
95  
96    private void checkMethodInvocations(MethodTree method, @Nullable String callerPropagation, Map<Symbol, String> methodsPropagationMap) {
97      BlockTree methodBody = method.block();
98      if (methodBody == null) {
99        return;
100     }
101     methodBody.accept(new BaseTreeVisitor() {
102       @Override
103       public void visitMethodInvocation(MethodInvocationTree methodInvocation) {
104         super.visitMethodInvocation(methodInvocation);
105         Symbol calleeMethodSymbol = methodInvocation.symbol();
106         if (methodsPropagationMap.containsKey(calleeMethodSymbol) && methodInvocationOnThisInstance(methodInvocation)) {
107           String calleePropagation = methodsPropagationMap.get(calleeMethodSymbol);
108           checkIncompatiblePropagation(methodInvocation, callerPropagation, calleeMethodSymbol, calleePropagation);
109         }
110       }
111     });
112   }
113 
114   private static boolean methodInvocationOnThisInstance(MethodInvocationTree methodInvocation) {
115     if (methodInvocation.symbol().isStatic()) {
116       return false;
117     }
118     ExpressionTree expression = methodInvocation.methodSelect();
119     if (expression.is(Tree.Kind.MEMBER_SELECT)) {
120       expression = ((MemberSelectExpressionTree) expression).expression();
121       return expression.is(Tree.Kind.IDENTIFIER) && ((IdentifierTree) expression).name().equals("this");
122     }
123     return expression.is(Tree.Kind.IDENTIFIER);
124   }
125 
126   private void checkIncompatiblePropagation(MethodInvocationTree methodInvocation, @Nullable String callerPropagation, Symbol calleeMethodSymbol, String calleePropagation) {
127     Set<String> incompatiblePropagation = INCOMPATIBLE_PROPAGATION_MAP.getOrDefault(callerPropagation, Collections.emptySet());
128     if (incompatiblePropagation.contains(calleePropagation)) {
129       String message = "\"" + calleeMethodSymbol.name() + "'s\" @Transactional requirement is incompatible with the one for this method.";
130       List<JavaFileScannerContext.Location> secondaryLocations = Collections.singletonList(
131         new JavaFileScannerContext.Location("", ((MethodTree) calleeMethodSymbol.declaration()).simpleName()));
132       reportIssue(ExpressionUtils.methodName(methodInvocation), message, secondaryLocations, null);
133     }
134   }
135 
136   private static Map<Symbol, String> collectMethodsPropagation(ClassTree classTree) {
137     Map<Symbol, String> methodPropagationMap = new HashMap<>();
138     String classPropagation = getPropagation(classTree.symbol(), null);
139     for (Tree member : classTree.members()) {
140       if (member.is(Tree.Kind.METHOD)) {
141         MethodTree method = (MethodTree) member;
142         if (method.symbol().isPublic()) {
143           methodPropagationMap.put(method.symbol(), getPropagation(method.symbol(), classPropagation));
144         }
145       }
146     }
147     return methodPropagationMap;
148   }
149 
150   private static boolean hasSameValues(Collection<String> methodsPropagationList) {
151     return methodsPropagationList.stream().distinct().count() <= 1;
152   }
153 
154   @CheckForNull
155   private static String getPropagation(Symbol symbol, @Nullable String inheritedPropagation) {
156     String defaultValue = inheritedPropagation != null ? inheritedPropagation : REQUIRED;
157     List<AnnotationValue> values = symbol.metadata().valuesForAnnotation(SPRING_TRANSACTIONAL_ANNOTATION);
158     if (values != null) {
159       return getAnnotationAttributeAsString(values, "propagation", defaultValue);
160     } else {
161       values = symbol.metadata().valuesForAnnotation(JAVAX_TRANSACTIONAL_ANNOTATION);
162       if (values != null) {
163         return getAnnotationAttributeAsString(values, "value", defaultValue);
164       } else {
165         return inheritedPropagation;
166       }
167     }
168   }
169 
170   private static String getAnnotationAttributeAsString(List<AnnotationValue> values, String attributeName, String defaultValue) {
171     return values.stream()
172       .filter(annotationValue -> annotationValue.name().equals(attributeName))
173       .map(AnnotationValue::value)
174       .filter(Tree.class::isInstance)
175       .map(Tree.class::cast)
176       .map(tree -> tree.is(Tree.Kind.MEMBER_SELECT) ? ((MemberSelectExpressionTree) tree).identifier() : tree)
177       .filter(tree -> tree.is(Tree.Kind.IDENTIFIER))
178       .map(IdentifierTree.class::cast)
179       .map(IdentifierTree::name)
180       .findFirst()
181       .orElse(defaultValue);
182   }

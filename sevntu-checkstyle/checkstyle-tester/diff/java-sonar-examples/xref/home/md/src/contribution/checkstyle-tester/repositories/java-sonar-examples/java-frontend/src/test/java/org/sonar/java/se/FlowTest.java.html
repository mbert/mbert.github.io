20  package org.sonar.java.se;
21  
22  import org.junit.Test;
23  import org.sonar.plugins.java.api.JavaFileScannerContext;
24  import org.sonar.plugins.java.api.tree.Tree;
25  
26  import java.util.List;
27  import java.util.stream.Collectors;
28  import java.util.stream.Stream;
29  
30  import static org.assertj.core.api.Assertions.assertThat;
31  import static org.mockito.Mockito.mock;
32  
33  public class FlowTest {
34  
35    @Test
36    public void test_first_flow_location() {
37      Flow flow1 = Flow.builder()
38        .add(locationWithMockTree("last"))
39        .add(locationWithMockTree("first"))
40        .build();
41      List<JavaFileScannerContext.Location> collect = flow1.firstFlowLocation().collect(Collectors.toList());
42      assertThat(collect).hasSize(1);
43      assertThat(collect.get(0).msg).isEqualTo("first");
44  
45      Stream<JavaFileScannerContext.Location> empty = Flow.empty().firstFlowLocation();
46      assertThat(empty).isEmpty();
47    }
48  
49    @Test
50    public void testEquals() {
51  
52      Flow flow = Flow.builder()
53        .add(locationWithMockTree("first"))
54        .add(locationWithMockTree("second"))
55        .build();
56  
57      Flow flow1 = Flow.builder().addAll(flow).build();
58      Flow flow2 = Flow.builder().addAll(flow).setAsExceptional().build();
59  
60      assertThat(flow1.equals(null)).isFalse();
61      assertThat(flow1.equals(new Object())).isFalse();
62  
63      assertThat(flow1.equals(Flow.empty())).isFalse();
64      assertThat(flow1.equals(flow2)).isFalse();
65  
66      assertThat(flow1.equals(flow1)).isTrue();
67      assertThat(flow1.equals(Flow.of(flow1))).isTrue();
68    }
69  
70    private static JavaFileScannerContext.Location locationWithMockTree(String message) {
71      return new JavaFileScannerContext.Location(message, mock(Tree.class));
72    }
73  
74  }

20  package org.sonar.java.ast.parser;
21  
22  import com.sonar.sslr.api.typed.ActionParser;
23  import org.sonar.java.model.JavaTree;
24  import org.sonar.plugins.java.api.tree.Tree;
25  import org.sonar.sslr.grammar.LexerlessGrammarBuilder;
26  
27  import java.io.File;
28  import java.nio.charset.StandardCharsets;
29  import java.util.Deque;
30  import java.util.LinkedList;
31  
32  public class JavaParser extends ActionParser<Tree> {
33    private Deque<JavaTree> parentList = new LinkedList<>();
34  
35    private JavaParser(LexerlessGrammarBuilder grammarBuilder, Class<JavaGrammar> javaGrammarClass,
36      TreeFactory treeFactory, JavaNodeBuilder javaNodeBuilder, JavaLexer compilationUnit) {
37      super(StandardCharsets.UTF_8, grammarBuilder, javaGrammarClass, treeFactory, javaNodeBuilder, compilationUnit);
38    }
39  
40    public static ActionParser<Tree> createParser() {
41      return new JavaParser(JavaLexer.createGrammarBuilder(),
42        JavaGrammar.class,
43        new TreeFactory(),
44        new JavaNodeBuilder(),
45        JavaLexer.COMPILATION_UNIT);
46    }
47  
48    @Override
49    public Tree parse(File file) {
50      return createParentLink((JavaTree) super.parse(file));
51    }
52  
53    @Override
54    public Tree parse(String source) {
55      return createParentLink((JavaTree) super.parse(source));
56    }
57  
58    private Tree createParentLink(JavaTree topParent) {
59      parentList.push(topParent);
60      while (!parentList.isEmpty()) {
61        JavaTree parent = parentList.pop();
62        if (!parent.isLeaf()) {
63          for (Tree nextTree : parent.getChildren()) {
64            JavaTree next = (JavaTree) nextTree;
65            if (next != null) {
66              next.setParent(parent);
67              parentList.push(next);
68            }
69          }
70        }
71      }
72      return topParent;
73    }
74  }

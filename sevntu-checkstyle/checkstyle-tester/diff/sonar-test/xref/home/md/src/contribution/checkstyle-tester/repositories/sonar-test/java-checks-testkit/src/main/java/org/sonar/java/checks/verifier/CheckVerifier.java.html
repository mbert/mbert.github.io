105   protected void collectExpectedIssues(String comment, int line) {
106     String expectedStart = getExpectedIssueTrigger();
107     if (comment.startsWith(expectedStart)) {
108       String cleanedComment = StringUtils.remove(comment, expectedStart);
109 
110       EnumMap<IssueAttribute, String> attr = new EnumMap<>(IssueAttribute.class);
111       String expectedMessage = StringUtils.substringBetween(cleanedComment, "{{", "}}");
112       if (StringUtils.isNotEmpty(expectedMessage)) {
113         attr.put(IssueAttribute.MESSAGE, expectedMessage);
114       }
115       int expectedLine = line;
116       String attributesSubstr = extractAttributes(comment, attr);
117 
118       cleanedComment = StringUtils.stripEnd(StringUtils.remove(StringUtils.remove(cleanedComment, "[[" + attributesSubstr + "]]"), "{{" + expectedMessage + "}}"), " \t");
119       if (StringUtils.startsWith(cleanedComment, "@")) {
120         final int lineAdjustment;
121         final char firstChar = cleanedComment.charAt(1);
122         final int endIndex = cleanedComment.indexOf(' ');
123         if (endIndex == -1) {
124           lineAdjustment = Integer.parseInt(cleanedComment.substring(2));
125         } else {
126           lineAdjustment = Integer.parseInt(cleanedComment.substring(2, endIndex));
127         }
128         if (firstChar == '+') {
129           expectedLine += lineAdjustment;
130         } else if (firstChar == '-') {
131           expectedLine -= lineAdjustment;
132         } else {
133           Fail.fail("Use only '@+N' or '@-N' to shifts messages.");
134         }
135       }
136       updateEndLine(expectedLine, attr);
137       expected.put(expectedLine, attr);
138     }
139   }
140 
141   private static String extractAttributes(String comment, Map<IssueAttribute, String> attr) {
142     String attributesSubstr = StringUtils.substringBetween(comment, "[[", "]]");
143     if (!StringUtils.isEmpty(attributesSubstr)) {
144       Iterable<String> attributes = Splitter.on(";").split(attributesSubstr);
145       for (String attribute : attributes) {
146         String[] split = StringUtils.split(attribute, '=');
147         if (split.length == 2 && CheckVerifier.ATTRIBUTE_MAP.containsKey(split[0])) {
148           attr.put(CheckVerifier.ATTRIBUTE_MAP.get(split[0]), split[1]);
149         } else {
150           Fail.fail("// Noncompliant attributes not valid: " + attributesSubstr);
151         }
152       }
153     }
154     return attributesSubstr;
155   }
156 
157   private static void updateEndLine(int expectedLine, EnumMap<IssueAttribute, String> attr) {
158     if (attr.containsKey(IssueAttribute.END_LINE)) {
159       String endLineStr = attr.get(IssueAttribute.END_LINE);
160       if (endLineStr.charAt(0) == '+') {
161         int endLine = Integer.parseInt(endLineStr);
162         attr.put(IssueAttribute.END_LINE, Integer.toString(expectedLine + endLine));
163       } else {
164         Fail.fail("endLine attribute should be relative to the line and must be +N with N integer");
165       }
166     }
167   }

173   public void checkIssues(Set<AnalyzerMessage> issues, boolean bypassNoIssue) {
174     if (expectNoIssues) {
175       assertNoIssues(issues, bypassNoIssue);
176     } else if (StringUtils.isNotEmpty(expectFileIssue)) {
177       assertSingleIssue(issues);
178     } else {
179       assertMultipleIssue(issues);
180     }
181   }

183   static SonarComponents sonarComponents(File file) {
184     SensorContextTester context = SensorContextTester.create(new File("")).setRuntime(SonarRuntimeImpl.forSonarLint(Version.create(6, 7)));
185     context.setSettings(new MapSettings().setProperty("sonar.java.failOnException", true));
186     SonarComponents sonarComponents = new SonarComponents(null, context.fileSystem(), null, null, null) {
187       @Override
188       public boolean reportAnalysisError(RecognitionException re, File file) {
189         return false;
190       }
191     };
192     sonarComponents.setSensorContext(context);
193     context.fileSystem().add(new TestInputFileBuilder("", file.getPath()).setCharset(StandardCharsets.UTF_8).build());
194     return sonarComponents;
195   }

197   private void assertMultipleIssue(Set<AnalyzerMessage> issues) throws AssertionError {
198     Preconditions.checkState(!issues.isEmpty(), "At least one issue expected");
199     List<Integer> unexpectedLines = Lists.newLinkedList();
200     RemediationFunction remediationFunction = remediationFunction(issues.iterator().next());
201     for (AnalyzerMessage issue : issues) {
202       validateIssue(expected, unexpectedLines, issue, remediationFunction);
203     }
204     if (!expected.isEmpty() || !unexpectedLines.isEmpty()) {
205       Collections.sort(unexpectedLines);
206       String expectedMsg = expectedMessage();
207       String unexpectedMsg = unexpectedMessage(unexpectedLines);
208       Fail.fail(expectedMsg + unexpectedMsg);
209     }
210   }
211 
212   private String expectedMessage() {
213     return !expected.isEmpty() ? ("Expected " + expected) : "";
214   }
215 
216   private String unexpectedMessage(List<Integer> unexpectedLines) {
217     if (unexpectedLines.isEmpty()) {
218       return "";
219     }
220     return (expected.isEmpty() ? "" : ", ") + "Unexpected at " + unexpectedLines;
221   }

223   enum RemediationFunction {
224     LINEAR, CONST
225   }
226 
227   static class RuleJSON {
228     static class Remediation {
229       String func;
230     }
231     Remediation remediation;
232   }
233 
234   @CheckForNull
235   private static RemediationFunction remediationFunction(AnalyzerMessage issue) {
236     String ruleKey = ruleKey(issue);
237     try {
238       RuleJSON rule = getRuleJSON(ruleKey);
239       if (rule.remediation == null) {
240         return null;
241       }
242       switch (rule.remediation.func) {
243         case "Linear":
244           return RemediationFunction.LINEAR;
245         case "Constant/Issue":
246           return RemediationFunction.CONST;
247         default:
248           return null;
249       }
250     } catch (IOException | JsonParseException e) {
251       // Failed to open json file, as this is not part of API yet, we should not fail because of this
252       LOG.debug("Remediation function and cost not provided, \"constant\" is assumed.");
253       return null;
254     }
255   }
256 
257   private static RuleJSON getRuleJSON(String ruleKey) throws IOException {
258     String ruleJson = "/org/sonar/l10n/java/rules/squid/" + ruleKey + "_java.json";
259     URL resource = CheckVerifier.class.getResource(ruleJson);
260     if(resource == null) {
261       throw new IOException(ruleJson + " not found");
262     }
263     Gson gson = new Gson();
264     return gson.fromJson(new InputStreamReader(resource.openStream(), "UTF-8"), RuleJSON.class);
265   }
266 
267   private static String ruleKey(AnalyzerMessage issue) {
268     String ruleKey;
269     RspecKey rspecKeyAnnotation = AnnotationUtils.getAnnotation(issue.getCheck().getClass(), RspecKey.class);
270     if(rspecKeyAnnotation != null) {
271       ruleKey = rspecKeyAnnotation.value();
272     } else {
273       Rule ruleAnnotation = AnnotationUtils.getAnnotation(issue.getCheck().getClass(), Rule.class);
274       if (ruleAnnotation != null) {
275         ruleKey = ruleAnnotation.key();
276       } else {
277         Fail.fail("Rules should be annotated with '@Rule(key = \"...\")' annotation (org.sonar.check.Rule).");
278         // unreachable
279         return null;
280       }
281     }
282     return ruleKey;
283   }

285   private static void validateIssue(Multimap<Integer, Map<IssueAttribute, String>> expected, List<Integer> unexpectedLines, AnalyzerMessage issue,
286     @Nullable RemediationFunction remediationFunction) {
287     int line = issue.getLine();
288     if (expected.containsKey(line)) {
289       Map<IssueAttribute, String> attrs = Iterables.getLast(expected.get(line));
290       assertEquals(issue.getMessage(), attrs, IssueAttribute.MESSAGE);
291       Double cost = issue.getCost();
292       if (cost != null) {
293         Preconditions.checkState(remediationFunction != RemediationFunction.CONST, "Rule with constant remediation function shall not provide cost");
294         assertEquals(Integer.toString(cost.intValue()), attrs, IssueAttribute.EFFORT_TO_FIX);
295       } else if(remediationFunction == RemediationFunction.LINEAR){
296         Fail.fail("A cost should be provided for a rule with linear remediation function");
297       }
298       validateAnalyzerMessage(attrs, issue);
299       expected.remove(line, attrs);
300     } else {
301       unexpectedLines.add(line);
302     }
303   }
304 
305   private static void validateAnalyzerMessage(Map<IssueAttribute, String> attrs, AnalyzerMessage analyzerMessage) {
306     Double effortToFix = analyzerMessage.getCost();
307     if (effortToFix != null) {
308       assertEquals(Integer.toString(effortToFix.intValue()), attrs, IssueAttribute.EFFORT_TO_FIX);
309     }
310     AnalyzerMessage.TextSpan textSpan = analyzerMessage.primaryLocation();
311     assertEquals(normalizeColumn(textSpan.startCharacter), attrs, IssueAttribute.START_COLUMN);
312     assertEquals(Integer.toString(textSpan.endLine), attrs, IssueAttribute.END_LINE);
313     assertEquals(normalizeColumn(textSpan.endCharacter), attrs, IssueAttribute.END_COLUMN);
314     if (attrs.containsKey(IssueAttribute.SECONDARY_LOCATIONS)) {
315       List<AnalyzerMessage> secondaryLocations = analyzerMessage.flows.stream().map(l -> l.get(0)).collect(Collectors.toList());
316       Multiset<String> actualLines = HashMultiset.create();
317       for (AnalyzerMessage secondaryLocation : secondaryLocations) {
318         actualLines.add(Integer.toString(secondaryLocation.getLine()));
319       }
320       List<String> expected = Lists.newArrayList(Splitter.on(",").omitEmptyStrings().trimResults().split(attrs.get(IssueAttribute.SECONDARY_LOCATIONS)));
321       List<String> unexpected = new ArrayList<>();
322       for (String actualLine : actualLines) {
323         if (expected.contains(actualLine)) {
324           expected.remove(actualLine);
325         } else {
326           unexpected.add(actualLine);
327         }
328       }
329       if (!expected.isEmpty() || !unexpected.isEmpty()) {
330         // Line is not covered by JaCoCo because of thrown exception but effectively covered in UT.
331         Fail.fail(String.format("Secondary locations: expected: %s unexpected:%s. In %s:%d", expected, unexpected, normalizedFilePath(analyzerMessage), analyzerMessage.getLine()));
332       }
333     }
334   }

336   private static String normalizedFilePath(AnalyzerMessage analyzerMessage) {
337     String absolutePath = analyzerMessage.getFile().getPath();
338     return absolutePath.replace("\\", "/");
339   }
340 
341   private static String normalizeColumn(int startCharacter) {
342     return Integer.toString(startCharacter + 1);
343   }

345   private static void assertEquals(String value, Map<IssueAttribute, String> attributes, IssueAttribute attribute) {
346     if (attributes.containsKey(attribute)) {
347       assertThat(value).as("attribute mismatch for " + attribute + ": " + attributes).isEqualTo(attributes.get(attribute));
348     }
349   }

351   private void assertSingleIssue(Set<AnalyzerMessage> issues) {
352     Preconditions.checkState(issues.size() == 1, "A single issue is expected on the file");
353     AnalyzerMessage issue = Iterables.getFirst(issues, null);
354     assertThat(issue.getLine()).isNull();
355     assertThat(issue.getMessage()).isEqualTo(expectFileIssue);
356   }

358   private void assertNoIssues(Set<AnalyzerMessage> issues, boolean bypass) {
359     assertThat(issues).overridingErrorMessage("No issues expected but got: " + issues).isEmpty();
360     if (!bypass) {
361       // make sure we do not copy&paste verifyNoIssue call when we intend to call verify
362       assertThat(expected.isEmpty()).overridingErrorMessage("The file should not declare noncompliants when no issues are expected").isTrue();
363     }

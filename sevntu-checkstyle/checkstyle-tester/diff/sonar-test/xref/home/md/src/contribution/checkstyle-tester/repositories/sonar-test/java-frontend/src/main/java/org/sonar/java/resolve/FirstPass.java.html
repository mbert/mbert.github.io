82    public FirstPass(SemanticModel semanticModel, Symbols symbols, Resolve resolve, ParametrizedTypeCache parametrizedTypeCache, TypeAndReferenceSolver typeAndReferenceSolver) {
83      this.semanticModel = semanticModel;
84      this.resolve = resolve;
85      this.completer = new SecondPass(semanticModel, symbols, parametrizedTypeCache, typeAndReferenceSolver);
86      this.symbols = symbols;
87      this.parametrizedTypeCache = parametrizedTypeCache;
88    }

90    private void restoreEnvironment(Tree tree) {
91      if (env.next == null) {
92        // Invariant: env.next == null for CompilationUnit
93        Preconditions.checkState(tree.is(Tree.Kind.COMPILATION_UNIT));
94      } else {
95        env = env.next;
96      }
97    }

99    public void completeSymbols() {
100     for (JavaSymbol symbol : uncompleted) {
101       symbol.complete();
102     }
103     uncompleted.clear();
104   }

107   @Override
108   public void visitCompilationUnit(CompilationUnitTree tree) {
109     JavaSymbol.PackageJavaSymbol compilationUnitPackage = symbols.defaultPackage;
110 
111     PackageDeclarationTree packageDeclaration = tree.packageDeclaration();
112     if (packageDeclaration != null) {
113       ExpressionTree packageName = packageDeclaration.packageName();
114       PackageResolverVisitor packageResolver = new PackageResolverVisitor();
115       packageName.accept(packageResolver);
116       compilationUnitPackage = (JavaSymbol.PackageJavaSymbol) resolve.findIdentInPackage(compilationUnitPackage, packageResolver.packageName, JavaSymbol.PCK);
117       semanticModel.associateSymbol(packageName, compilationUnitPackage);
118     }
119     compilationUnitPackage.members = new Scope(compilationUnitPackage);
120 
121     env = new Resolve.Env();
122     env.packge = compilationUnitPackage;
123     env.scope = compilationUnitPackage.members;
124     env.namedImports = new Scope.ImportScope(compilationUnitPackage);
125     env.starImports = resolve.createStarImportScope(compilationUnitPackage);
126     env.staticStarImports = resolve.createStaticStarImportScope(compilationUnitPackage);
127     semanticModel.associateEnv(tree, env);
128 
129     scan(tree.types());
130     restoreEnvironment(tree);
131     resolveImports(tree.imports());
132     completeSymbols();
133   }

135   private static class PackageResolverVisitor extends BaseTreeVisitor {
136     private String packageName;
137 
138     public PackageResolverVisitor() {
139       packageName = "";
140     }
141 
142     @Override
143     public void visitIdentifier(IdentifierTree tree) {
144       if (!packageName.isEmpty()) {
145         packageName += ".";
146       }
147       packageName += tree.name();
148     }
149   }

151   private void resolveImports(List<ImportClauseTree> imports) {
152     ImportResolverVisitor importResolverVisitor = new ImportResolverVisitor();
153     for (ImportClauseTree importClauseTree : imports) {
154       importClauseTree.accept(importResolverVisitor);
155     }
156   }

158   private class ImportResolverVisitor extends BaseTreeVisitor {
159     private JavaSymbol currentSymbol;
160     private List<JavaSymbol> resolved;
161     private boolean isStatic;
162 
163     @Override
164     public void visitImport(ImportTree tree) {
165       //reset currentSymbol to default package
166       currentSymbol = symbols.defaultPackage;
167       isStatic = tree.isStatic();
168       tree.qualifiedIdentifier().accept(this);
169       //Associate symbol only if found.
170       if (currentSymbol.kind < JavaSymbol.ERRONEOUS) {
171         enterSymbol(currentSymbol, tree);
172       } else if (isStatic) {
173         resolved.stream()
174           //add only static fields
175           //TODO accessibility should be checked : package/public
176           .filter(symbol -> Flags.isFlagged(symbol.flags, Flags.STATIC))
177           //TODO only the first symbol found will be associated with the tree.
178           .forEach(symbol -> enterSymbol(symbol, tree));
179       }
180     }
181 
182     private void enterSymbol(JavaSymbol symbol, ImportTree tree) {
183       env.namedImports.enter(symbol);
184       //FIXME We add all symbols to named Imports for static methods, but only the first one will be resolved as we don't handle arguments.
185       //FIXME That is why we only add the first symbol so we resolve references at best for now.
186       //add to semantic model only the first symbol.
187       //twice the same import : ignore the duplication JLS8 7.5.1.
188       if (semanticModel.getSymbol(tree) == null) {
189         semanticModel.associateSymbol(tree, symbol);
190       }
191     }
192 
193     @Override
194     public void visitIdentifier(IdentifierTree tree) {
195       if (JavaPunctuator.STAR.getValue().equals(tree.name())) {
196         //star import : we save the current symbol
197         if (isStatic) {
198           env.staticStarImports.enter(currentSymbol);
199         } else {
200           env.starImports.enter(currentSymbol);
201         }
202         //we set current symbol to not found to do not put it in named import scope.
203         currentSymbol = new Resolve.JavaSymbolNotFound();
204       } else {
205         if (currentSymbol.kind == JavaSymbol.PCK) {
206           currentSymbol = resolve.findIdentInPackage(currentSymbol, tree.name(), JavaSymbol.PCK | JavaSymbol.TYP);
207           resolved = Collections.emptyList();
208         } else if (currentSymbol.kind == JavaSymbol.TYP) {
209           resolved = ((JavaSymbol.TypeJavaSymbol) currentSymbol).members().lookup(tree.name());
210           currentSymbol = resolve.findIdentInType(env, (JavaSymbol.TypeJavaSymbol) currentSymbol, tree.name(), JavaSymbol.TYP | JavaSymbol.VAR).symbol();
211         } else {
212           //Site symbol is not found so we won't be able to resolve the import.
213           currentSymbol = new Resolve.JavaSymbolNotFound();
214           resolved = Collections.emptyList();
215         }
216       }
217     }

221   @Override
222   public void visitClass(ClassTree tree) {
223     int flag = 0;
224     boolean anonymousClass = tree.simpleName() == null;
225     String name = "";
226     if (!anonymousClass) {
227       name = tree.simpleName().name();
228       flag = computeClassFlags(tree);
229     }
230     JavaSymbol.TypeJavaSymbol symbol = new JavaSymbol.TypeJavaSymbol(flag, name, env.scope.owner);
231     symbol.declaration = tree;
232     ((ClassTreeImpl) tree).setSymbol(symbol);
233     //Only register classes that can be accessible, so classes owned by a method are not registered.
234     //TODO : register also based on flags ?
235     if (!anonymousClass) {
236       if (env.scope.owner.kind == JavaSymbol.TYP || env.scope.owner.kind == JavaSymbol.PCK) {
237         resolve.registerClass(symbol);
238       }
239       enterSymbol(tree, symbol);
240     }
241     symbol.members = new Scope(symbol);
242     symbol.completer = completer;
243     uncompleted.add(symbol);
244 
245     //Define type parameters:
246     TypeParameters typeParameterTrees = tree.typeParameters();
247     createNewEnvironment(typeParameterTrees);
248     // Save current environment to be able to complete class later
249     semanticModel.saveEnv(symbol, env);
250     for (TypeParameterTree typeParameterTree : typeParameterTrees) {
251       JavaSymbol.TypeVariableJavaSymbol typeVariableSymbol = new JavaSymbol.TypeVariableJavaSymbol(typeParameterTree.identifier().name(), symbol);
252       symbol.addTypeParameter((TypeVariableJavaType) typeVariableSymbol.type);
253       enterSymbol(typeParameterTree, typeVariableSymbol);
254     }
255     if(!typeParameterTrees.isEmpty()) {
256       symbol.type = parametrizedTypeCache.getParametrizedTypeType(symbol, new TypeSubstitution());
257     }
258     symbol.typeParameters = env.scope;
259     Resolve.Env classEnv = env.dup();
260     classEnv.outer = env;
261     classEnv.enclosingClass = symbol;
262     classEnv.scope = symbol.members;
263     env = classEnv;
264     semanticModel.associateEnv(tree, env);
265     scan(tree.modifiers());
266     //skip type parameters
267     scan(tree.superClass());
268     scan(tree.superInterfaces());
269     scan(tree.members());
270 
271     if (tree.is(Tree.Kind.ENUM)) {
272       // implicit methods from enum: JLS8 : 8.9.2
273       // add 'public static E[] values()'
274       JavaSymbol.MethodJavaSymbol valuesMethod = new JavaSymbol.MethodJavaSymbol((symbol.flags & Flags.ACCESS_FLAGS) | Flags.STATIC, "values", symbol);
275       ArrayJavaType enumArrayType = new ArrayJavaType(symbol.type, symbols.arrayClass);
276       MethodJavaType valuesMethodType = new MethodJavaType(ImmutableList.<JavaType>of(), enumArrayType, ImmutableList.<JavaType>of(), symbol);
277       valuesMethod.setMethodType(valuesMethodType);
278       valuesMethod.parameters = new Scope(valuesMethod);
279       classEnv.scope.enter(valuesMethod);
280 
281       // add 'public static E valueOf(String name)'
282       JavaSymbol.MethodJavaSymbol valueOfMethod = new JavaSymbol.MethodJavaSymbol((symbol.flags & Flags.ACCESS_FLAGS) | Flags.STATIC, "valueOf", symbol);
283       MethodJavaType valueOfMethodType = new MethodJavaType(ImmutableList.<JavaType>of(symbols.stringType), symbol.type, ImmutableList.<JavaType>of(), symbol);
284       valueOfMethod.setMethodType(valueOfMethodType);
285       valueOfMethod.parameters = new Scope(valueOfMethod);
286       valueOfMethod.parameters.enter(new JavaSymbol.VariableJavaSymbol(0, "name", symbols.stringType, valueOfMethod));
287       classEnv.scope.enter(valueOfMethod);
288     }
289     restoreEnvironment(tree);
290     restoreEnvironment(tree);
291   }

293   private int computeClassFlags(ClassTree tree) {
294     int flags = computeFlags(tree.modifiers(), tree);
295     if (tree.is(Tree.Kind.INTERFACE)) {
296       flags |= Flags.INTERFACE;
297     }else if (tree.is(Tree.Kind.ENUM)) {
298       flags |= Flags.ENUM;
299     }else if (tree.is(Tree.Kind.ANNOTATION_TYPE)) {
300       flags |= Flags.INTERFACE | Flags.ANNOTATION;
301     }
302     return flags;
303   }

305   @Override
306   public void visitMethod(MethodTree tree) {
307     String name = tree.returnType() == null ? CONSTRUCTOR_NAME : tree.simpleName().name();
308     JavaSymbol.MethodJavaSymbol symbol = new JavaSymbol.MethodJavaSymbol(computeFlags(tree.modifiers(), tree), name, env.scope.owner);
309     symbol.declaration = tree;
310     if (Flags.isFlagged(env.scope.owner.flags, Flags.ENUM) && tree.returnType() == null) {
311       //enum constructors are private.
312       symbol.flags |= Flags.PRIVATE;
313     }
314     enterSymbol(tree, symbol);
315     symbol.parameters = new Scope(symbol);
316     symbol.completer = completer;
317     uncompleted.add(symbol);
318 
319     ((MethodTreeImpl) tree).setSymbol(symbol);
320     createNewEnvironment(tree.typeParameters());
321     for (TypeParameterTree typeParameterTree : tree.typeParameters()) {
322       JavaSymbol.TypeVariableJavaSymbol typeVariableSymbol = new JavaSymbol.TypeVariableJavaSymbol(typeParameterTree.identifier().name(), symbol);
323       symbol.addTypeParameter((TypeVariableJavaType) typeVariableSymbol.type);
324       enterSymbol(typeParameterTree, typeVariableSymbol);
325     }
326     // Save current environment to be able to complete method later
327     semanticModel.saveEnv(symbol, env);
328 
329     symbol.typeParameters = env.scope;
330     // Create new environment - this is required, because new scope is created
331     Resolve.Env methodEnv = env.dup();
332     methodEnv.scope = symbol.parameters;
333     methodEnv.outer = env;
334     env = methodEnv;
335     scan(tree.modifiers());
336     //skip type parameters.
337     scan(tree.returnType());
338     scan(tree.parameters());
339     scan(tree.throwsClauses());
340     scan(tree.defaultValue());
341     symbol.defaultValue = getDefaultValueFromTree(tree.defaultValue());
342     scan(tree.block());
343     restoreEnvironment(tree);
344     restoreEnvironment(tree);
345   }

347   @Nullable
348   private static Object getDefaultValueFromTree(@Nullable ExpressionTree expressionTree) {
349     if(expressionTree == null) {
350       return null;
351     }
352     if (expressionTree.is(Tree.Kind.STRING_LITERAL)) {
353       return LiteralUtils.trimQuotes(((LiteralTree) expressionTree).value());
354     }
355     if (expressionTree.is(Tree.Kind.INT_LITERAL)) {
356       return LiteralUtils.intLiteralValue(expressionTree);
357     }
358     if (expressionTree.is(Tree.Kind.LONG_LITERAL)) {
359       return LiteralUtils.longLiteralValue(expressionTree);
360     }
361     return null;
362   }

364   @Override
365   public void visitEnumConstant(EnumConstantTree tree) {
366     // JLS-8.9.3
367     int flags = Flags.PUBLIC | Flags.STATIC | Flags.FINAL | Flags.ENUM;
368     if (hasDeprecatedAnnotation(tree.modifiers().annotations())) {
369       flags |= Flags.DEPRECATED;
370     }
371     declareVariable(flags, tree.simpleName(), (VariableTreeImpl) tree);
372     super.visitEnumConstant(tree);
373   }

375   @Override
376   public void visitVariable(VariableTree tree) {
377     declareVariable(computeFlags(tree.modifiers(), tree), tree.simpleName(), (VariableTreeImpl) tree);
378     super.visitVariable(tree);
379   }

381   private int computeFlags(ModifiersTree modifiers, Tree tree) {
382     int result = 0;
383     if (Flags.isFlagged(env.scope.owner.flags, Flags.INTERFACE)) {
384       result = computeFlagsForInterfaceMember(tree);
385     }
386     for (ModifierKeywordTree modifier : modifiers.modifiers()) {
387       result |= Flags.flagForModifier(modifier.modifier());
388     }
389     if(hasDeprecatedAnnotation(modifiers.annotations())) {
390       result |= Flags.DEPRECATED;
391     }
392     return result;
393   }
394 
395   private static int computeFlagsForInterfaceMember(Tree tree) {
396     int result;
397     if (tree.is(Tree.Kind.METHOD)) {
398       MethodTree methodTree = (MethodTree) tree;
399       // JLS9 9.4 A method in the body of an interface may be declared public or private
400       if (ModifiersUtils.hasModifier(methodTree.modifiers(), Modifier.PRIVATE)) {
401         result = Flags.PRIVATE;
402       } else {
403         result = Flags.PUBLIC;
404       }
405       if (methodTree.block() == null) {
406         // JLS8 9.4: methods lacking a block are implicitly abstract
407         result |= Flags.ABSTRACT;
408       }
409     } else {
410       // JLS7 9.5: member type declarations are implicitly static and public
411       result = Flags.PUBLIC | Flags.STATIC;
412       if (tree.is(Tree.Kind.VARIABLE)) {
413         // JLS7 9.3: fields are implicitly public, static and final
414         result |= Flags.FINAL;
415       }
416     }
417     return result;
418   }

420   private static boolean hasDeprecatedAnnotation(Iterable<AnnotationTree> annotations) {
421     for (AnnotationTree annotationTree : annotations) {
422       if (isDeprecated(annotationTree)) {
423         return true;
424       }
425     }
426     return false;
427   }
428 
429   private static boolean isDeprecated(AnnotationTree tree) {
430     return tree.annotationType().is(Tree.Kind.IDENTIFIER) &&
431         "Deprecated".equals(((IdentifierTree) tree.annotationType()).name());
432   }

434   private void declareVariable(int flags, IdentifierTree identifierTree, VariableTreeImpl tree) {
435     String name = identifierTree.name();
436     Object constantValue = semanticModel.constantValue(env.scope.owner, name);
437     JavaSymbol.VariableJavaSymbol symbol = new JavaSymbol.VariableJavaSymbol(flags, name, env.scope.owner, constantValue);
438     symbol.declaration = tree;
439     enterSymbol(tree, symbol);
440     symbol.completer = completer;
441     uncompleted.add(symbol);
442 
443     tree.setSymbol(symbol);
444 
445     // Save current environment to be able to complete variable declaration later
446     semanticModel.saveEnv(symbol, env);
447   }

449   @Override
450   public void visitBlock(BlockTree tree) {
451     // Create new environment - this is required, because block can declare types
452     createNewEnvironment(tree);
453     super.visitBlock(tree);
454     restoreEnvironment(tree);
455   }

457   @Override
458   public void visitTryStatement(TryStatementTree tree) {
459     if (!declaresResourceAsVariable(tree.resourceList())) {
460       super.visitTryStatement(tree);
461     } else {
462       //Declare scope for resources
463       createNewEnvironment(tree.resourceList());
464       scan(tree.resourceList());
465       scan(tree.block());
466       restoreEnvironment(tree.resourceList());
467       scan(tree.catches());
468       scan(tree.finallyBlock());
469     }
470   }
471 
472   private static boolean declaresResourceAsVariable(ListTree<Tree> resources) {
473     return resources.stream().anyMatch(r -> r.is(Tree.Kind.VARIABLE));
474   }

516   private void createNewEnvironment(Tree tree) {
517     Scope scope = new Scope(env.scope);
518     Resolve.Env newEnv = env.dup();
519     newEnv.scope = scope;
520     env = newEnv;
521     semanticModel.associateEnv(tree, env);
522   }

524   private void enterSymbol(Tree tree, JavaSymbol symbol) {
525     env.scope.enter(symbol);
526     semanticModel.associateSymbol(tree, symbol);
